# version python 3.8
# composition d'applications
# composition of maps

# compos√©e de deux applications
# composite of two mappings
def compose(f :callable, g :callable)->callable:
    def fonc(x):
        return g(f(x))
    return fonc

# pour exemple - just for an example
def f(x :float)->float:
    return 1 / x
# pour exemple - just for an example
def g(x :float)->float:
    return 1 + x * x

def main(): # function principale - main function
    print(compose(f, g)(1.0))
    print(compose(g, f)(1.0)) # composition non commutative

if __name__ == '__main__':
    main()
