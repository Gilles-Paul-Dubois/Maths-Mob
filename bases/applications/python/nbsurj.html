# Calcul des Sp,n - computing the Sp,n

# calcule les Sp,n à partir des Sp,n-1
# computes Sp,n from the Sp,n-1
def NextStirling(L :list)->list:
    Next = []
    Next.append(1)
    for i in range(1, len(L)):
        Next.append(L[i - 1] + (i + 1) * L[i])
    Next.append(1)
    return Next

# donne le coefficient de Stirling de seconde espèce
# gives the second kind Stirling coefficient
def Stirling2(p :int, n :int)->int:
    L = [1]
    for i in range(0, n - 1):
        L = NextStirling(L)
    return L[p - 1]

# comme son nom l'indique
# the factorial
def factorielle(n :int)->int:
    f = 1
    for i in range(1, n + 1):
        f *= i
    return f

# nombre de surjections de n dans p
# number of surjections from n to p
def Sur(n :int, p :int)->int:
    return Stirling2(p, n) * factorielle(p)

def main(): # fonction principale - main function
    print(Sur(7, 6))

if __name__ == '__main__':
    main()
