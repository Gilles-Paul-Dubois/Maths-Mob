# Test d'involution
# checking involutory mappings

# a dictionary - un dictionnaire
assoc = {'a': 'b', 'b': 'a', 'c': 'd', 'd': 'c'}

# la fonction associée à ce dictionnaire
# function associated with this dictionary
def f(c :str)->str:
    return assoc.get(c)

# la réciproque de f 
# the inverse of f
def rec(f : callable)->callable:
    def fonrec(x):
        L = [k for k in assoc.keys() if assoc.get(k) == x]  # liste des antécédents de x - list of preimages of x
        return L[0]  # le premier car il n'y en a qu'un - the first because there is only one
    return fonrec

# the test f involutory ?
# test d'involution
def EstInvolutive(f :callable)->bool:
    for k in assoc.keys():
        if f(k) != rec(f)(k):  # les images par f et f^-1 sont-elles identiques ?- are the images by f and f^-1 the same ?
            return False  # si non, perdu ! - if not failure !
    return True  # gagné ! success !

def main(): # fonction principale - main function
    print(EstInvolutive(f))  # affichera True dans ce cas - will print True in this case

if __name__ == '__main__':
    main()
