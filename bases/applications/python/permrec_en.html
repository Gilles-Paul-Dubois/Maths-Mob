# version python 3.4 
 
import copy 
 
 
# set of all permutations of the list L 
def permutations (L): 
    if len(L)==1: # case of a single element 
        return [L] # only one permutation 
    X=L[0] # otherwise, catch the first element 
    L1=L[1:len(L)] # eliminate it from L  
    L2=permutations(L1) # recursive call 
    L4=[] # to receive the result 
    for L3 in L2: # take all permutations of the tail of the list 
        for i in range(0,len(L3)+1): # circulate now the first element  
            L5=copy.deepcopy(L3) #L5 is a clone of L3 
            L5.insert(i,X) # we place X at the right place 
            L4.insert(0,L5)# and we enrich L4 
    return L4 
 
def main(): 
    print (permutations([0, 1, 2,3])) 
 
if __name__ == '__main__': 
    main()
