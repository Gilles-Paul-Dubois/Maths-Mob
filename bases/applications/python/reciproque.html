# version python 3.8
# bijection réciproque
# inverse bijection

# un dictionnaire - a dictionary
assoc = {'a': 'v', 'b': 'x', 'c': 'z', 'd': 't'}

# the function associated with the dictionary
# la fonction associée au dictionnaire
def f(c :str)->str:
    return assoc.get(c)

# la réciproque de f
# the inverse of f
def rec(f):
    def fonrec(x):
        L = [k for k in assoc.keys() if assoc.get(k) == x]  # liste des antécédents de x - list of preimages of x
        return L[0]  # le premier car il n'y en a qu'un - first of them because there's only one
    return fonrec

def main(): # fonction principale - main function
    print(rec(f)('v'))

if __name__ == '__main__':
    main()
