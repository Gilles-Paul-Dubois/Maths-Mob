# -*- coding: utf-8 -*-
# python version 3.4 
 
# generation of permutations is a standard tools of versions >3 provided by itertools 
from itertools import permutations 
 
 
def main(): 
    for p in permutations(range(3)): 
        print(p) 
    for p in permutations('abc'): 
        print(p) 
 
 
if __name__ == '__main__': 
    main()
