import copy

# liste des permutations de la liste L
# list of permutations of list L
def permutations(L :list)->list:
    if len(L) == 1:  # cas où il n'y a qu'un seul élément - case when there is only one element
        return [L]  # une seule permutation - only one permutation
    X = L[0]  # sinon en récupère le premier élément - if not let's take the first element
    L1 = L[1:len(L)]  # et on l'élimine de L - and we remove it from L
    L2 = permutations(L1)  # appel récursif - recursive call
    L4 = []  # pour recevoir le résultat - will receive final result
    for L3 in L2:  # on parcourt toutes les permutations de la queue de liste - let consider all permutation of the tail of the list
        for i in range(0, len(L3) + 1):  # on fait circuler le premier élément - circulate the first element
            L5 = copy.deepcopy(L3)  # L5 est un clone de L3 - L5 is a clone of L3
            L5.insert(i, X)  # on place X au bon endroit  - we place w at the right place
            L4.insert(0, L5)  # et on enrichit L4 - and we enrich L4
    return L4

def main(): # fonction principale - main function
    print(permutations([0, 1, 2, 3]))

if __name__ == '__main__':
    main()
