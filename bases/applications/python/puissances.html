#puissances d'applications - powers of mappings 
 
def compose (f,g): 
    def fonc(x): 
        return g(f(x)) 
    return fonc 
 
def powern (f,n): 
    def p0(x): 
        return x 
    pn=p0 
    for i in range (0,n): 
        pn=compose(f,pn) 
    return pn 
 
def f(x): 
    return 1/x 
 
def main(): 
    print (powern(f,0)(2.0)) 
    print (powern(f,1)(2.0)) 
    print (powern(f,2)(2.0)) 
 
if __name__ == '__main__': 
    main()
