# version python 3.8
# test de bijectivité pour une application définie par un dictionnaire
# test of bijectivity for a map defined with a dictionary

# les dictionnaires - the dictionaries
assoc1 = {'a': 'v', 'b': 'x', 'c': 'z', 'd': 't'}
assoc2 = {'a': 'v', 'b': 'x', 'c': 'z', 'd': 'x'}

# fonction associée à un dictionaire D
#function associated with dictionary D
def f(D :dict )->callable :
    return lambda x : D[x]

#fonctions définie au moyen des dictionnaires
# functions defined using  dictionaries
fonc1=f(assoc1)
fonc2=f(assoc2)

# test d'injectivité
#injectivity checking
def estInjective(f :callable, E :set)->bool: # utilise la définition - using definition
    return not any (f(x)==f(y) for x in E for y in E if x!=y)

# test de surjectivité
#surjectivity checking
def estSurjective(f :callable, E :set,F :set)->bool:
    return len(F) == len([f(x) for x in E])

# F est l'ensemble but contenant au moins toutes les valeurs de assoc
#E est la source c'est à dire l'ensemble des clés
# E is domain that is the set of keys
# F is the codomain containing at least all the values of dictionary assoc
def estBijective(f :callable, E :set,F :set)->bool:
    return estInjective(f,E) and estSurjective(f,E, F)

def main(): #fonction principale - main function
    E=set("abcd")
    F1 = set("vxzt")
    F2=set("vxz")
    print(estBijective(fonc1, E, F1))
    print(estBijective(fonc2,E,F2))

if __name__ == '__main__':
    main()
