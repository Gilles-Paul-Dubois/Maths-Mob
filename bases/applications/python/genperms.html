# version python 3.8
# génération des permutations d'un ensemble fini
# generation of permutations of a finite set
# par un procédé itératif utilisant les générateurs de python
# by an iterative process using a generator

from functools import reduce

#fonction d'insertion - insert function
def ins(X :int, i :int, L :list)->list:
    R = L[:]  # copie de L - copy of L
    R.insert(i, X)  # insertion de X à l'index i dans R - insert X at index i in L
    return R

# fait tourner X dans L à toutes les positions
# rotate x in L at all possible positions
def turn(X :int, L : list)->list:
    return [ins(X, i, L) for i in range(0, len(L) + 1)]

# le générateur - the generator
def permutations()->list:
    i, P = 0, [[]]
    yield P  # stoppe l'exécution pour la reprendre à cet endroit précis- stop execution to restart at this precise point
    while True:  # fait passer d'une génération à la suivante - allows to pass from one generation to next one
        i = i + 1
        Q = [turn(i, X) for X in P]
        P = reduce(lambda x, y: x + y, Q)  # concaténation à répétition - repeated concatenation
        yield P  # stoppe l'exécution pour la reprendre à cet endroit précis - stops executions and return result

def main(): #fonction principale - main function
    Perm = permutations()
    for i in range(0, 6):  # permutations d'un ensemble à 5 éléments - permutations of a set with 6 elements
        X = next(Perm)
    print(len(X))
    print(X)

if __name__ == '__main__':
    main()
