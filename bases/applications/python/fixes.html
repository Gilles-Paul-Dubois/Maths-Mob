# version python 3.8
# points fixes - fixed points

# a dictionary - un dictionnaire
assoc = {'a': 'c', 'b': 'a', 'c': 'c', 'd': 'b'}

# fonction associée au dictionnaire précédent
# function associated with the previous dictionary
def fonc(c :str)->str:
    return assoc.get(c)

# l'ensemble' des points fixes de f
# the set of fixed points of f
def Fixes(f :callable)->set:
    return {x for x in assoc.keys() if x == f(x)}

# le nombre des points fixes de f
# the number of fixed points of f
def NbFixes(f :callable)->int:
    return len(Fixes(f))

def main():#fonction principale - main function
    print(Fixes(fonc))
    print(NbFixes(fonc))

if __name__ == '__main__':
    main()
