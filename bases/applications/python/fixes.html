# version python 3.4 
 
# points fixes 
 
assoc={ 'a':'c', 'b':'a', 'c':'c','d':'b'} 
 
def fonc(c): 
    return assoc.get(c) 
 
# la liste des points fixes de f 
def Fixes (f): 
    return [x for x in assoc.keys() if x==f(x) ] 
 
# le nombre des points fixes de f 
def NbFixes(f): 
    return len(Fixes(f)) 
 
def main(): 
    print (Fixes(fonc)) 
    print (NbFixes(fonc)) 
 
if __name__ == '__main__': 
    main()
