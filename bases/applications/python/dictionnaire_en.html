# version python 3.4 
 
# non numerical function 
 
 
# Python's data type called 'dictionary' 
# In other languages 'map' or 'mapping' or A-list or P-list 
# It is an associative list  key->valuer 
assoc={ 'a':'v', 'b':'x', 'c':'z','d':'x'} 
 
#function taking as argument a character and returning a character 
#using a dictionary 
def fonc(c): 
    return assoc.get(c) 
 
# direct image of a subset 
def ImDir(X): 
    return set([fonc(c) for c in X]) 
 
# preimage of a subset 
def ImRec(Y): 
    return set([x for x in assoc.keys() if fonc(x)in Y ]) 
 
def main(): 
    print (set(assoc.keys())) #The domain 
    print (set(assoc.values())) # The codomain 
    print (fonc('b')) # image of element b 
    print (ImDir(['a','b','d'])) # Direct image of {a,b,d} 
    print (ImRec(['z','x'])) # preimage of {x,z} 
 
if __name__ == '__main__': 
    main()
