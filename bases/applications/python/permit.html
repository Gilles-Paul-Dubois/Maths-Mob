# une méthode itérative pour générer le groupe symétrique
# an iterative method to generate the symmetric group

# cycle opérant sur les p derniers éléments
#cycle operating on yhe p last elements
def cycle(n :int, p :int)->list:
    L1 = [i for i in range(0, n - p)]
    L2 = [i + 1 for i in range(n - p, n - 1)]
    return L1 + L2 + [n - p]

# composée de 2 permutations
# composite of two permutations
def compose(P1 :list, P2 :list)->list:
    return [P2[P1[i]] for i in range(0, len(P1))]

# puissance d'une permutation
# powsers of a permutation
def puissance(P : list, n : int)->list:
    PUISS = range(0, len(P))
    for i in range(0, n):
        PUISS = compose(PUISS, P)
    return PUISS

# générer toutes les permutations
# generate all permutations
def permutations(n :int)->list:
    CYCLES = [cycle(n, 2 + i) for i in range(0, n - 1)]
    PERM1 = [cycle(n - 1, 0)]
    for i in range(0, n - 1):
        PERM2 = [compose(PERM1[k], puissance(CYCLES[i], j)) for j in range(0, i + 2) for k in range(0, len(PERM1))]
        PERM1 = PERM2
    return PERM2

def main():#fonction principale - mainfunction
    print(permutations(5))

main()
