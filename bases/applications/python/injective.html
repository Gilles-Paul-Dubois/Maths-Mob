# version python 3.8
# test d'injectivité
# checking injectivity

# Deux dictionnaires - two dictionaries
assoc1 = {'a': 'v', 'b': 'x', 'c': 'z', 'd': 'x'}
assoc2 = {'a': 'v', 'b': 'x', 'c': 'z', 'd': 'u'}

# fonction associée à un dictionaire D
#function associated with dictionary D
def f(D :dict )->callable :
    return lambda x : D[x]

#fonctions définie au moyen des dictionnaires
# functions defined using 'assoc' dictionaries
fonc1=f(assoc1) # non injective - not injective fonc1(b)=fonc1(d)
fonc2=f(assoc2) # injective

# teste l'injectivité de fonc
# testing if fonc is injective
def estInjective(f :callable, E :set)->bool: # utilise la définition - using definition
    return not any (f(x)==f(y) for x in E for y in E if x!=y)

#fonction principale - main function
def main():
    E=set(assoc1.keys())
    print(estInjective(fonc1, E))
    F=set(assoc2.keys())
    print(estInjective(fonc2,F))

if __name__ == '__main__':
    main()
