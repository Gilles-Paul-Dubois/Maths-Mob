# inverse function 
assoc={ 'a':'v', 'b':'x', 'c':'z','d':'t'} 
 
def f(c): 
    return assoc.get(c) 
 
# the inverse of f 
def rec(f): 
    def fonrec(x): 
        L=[k for k in assoc.keys() if assoc.get(k)==x]# list of preimages of x 
        return L[0] # The first one because there's only one 
    return fonrec 
 
def main(): 
    print (rec(f)('v')) 
 
 
if __name__ == '__main__': 
    main()
