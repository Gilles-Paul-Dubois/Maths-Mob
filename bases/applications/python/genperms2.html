# python version 3.8
# la génération des permutations est un outil standard des versions >3 
# generation of permutations is a standard tool for Python versions >3
from itertools import permutations

def main(): #fonction principale - main function
    for p in permutations(range(3)):
        print(p)
    for p in permutations('abc'):
        print(p)

if __name__ == '__main__':
    main()
