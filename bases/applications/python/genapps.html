#générer toutes les applications de E dans F 
#generate all mappings from E to F
 
def power(n,F): 
    G=[[]] 
    for i in range (0,n): 
        G= [y+[x] for y in G for x in F] 
    return G 
 
def main(): 
    E=[1,2,3] 
    F=['a','b'] 
    print (power(len(E),F)) 
 
if __name__ == '__main__': 
    main()
