# Structure Python appelée 'dictionnaire'
# Python structure named 'dictonnary'
# dans d'autres langages 'map' ou 'mapping' ou A-list ou P-list
# in other languages known as 'map' or 'mapping' or A-list or P-list
# Il s'agit d'une liste associative clé->valeur
# It's an associative list key->value
assoc = {'a': 'v', 'b': 'x', 'c': 'z', 'd': 'x'}

# fonction prenant en argument un caractère et retournant un caractère
# function ahaving set of strings as domain and set of strings as codomain
# utilisant un dictionnaire - using a dictionnary
def fonc(c :str)->str:
    return assoc.get(c)

# image directe d'une partie - direct image of a subset
def ImDir(X :set)->set:
    return set((fonc(c) for c in X))

# image réciproque d'une partie - preimage of a subset
def ImRec(Y :set)->set:
    return set((x for x in assoc.keys() if fonc(x) in Y))

def main(): #fonction principale - mainfunction
    print(set(assoc.keys()))  # l'ensemble source
    print(set(assoc.values()))  # l'ensemble but
    print(fonc('b'))  # l'image de b
    print(ImDir(['a', 'b', 'd']))  # l'image directe de {a,b,d}
    print(ImRec(['z', 'x']))  # l'image réciproque de {x,z}

if __name__ == '__main__':
    main()
