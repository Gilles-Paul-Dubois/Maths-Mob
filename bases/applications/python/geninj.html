# génération d'injections
# generating injections
from copy import copy, deepcopy
from itertools import permutations

# générer les injections de {1,2,..,p} dans E représenté comme liste
# generates injections from {1,2,...,n} into E represented as a list
def GenInj(p :int, E :list)->list:
    if p == 1:
        return [[x] for x in E]  # cas où p=1 - case p=1
    else:
        R = []
        for x in E:
            F = copy(E)  # recopier E - copy E in F
            F.remove(x)  # enlever x - remove x
            PGI = GenInj(p - 1, F)  # appel récursif - rceursive call
            Cx = deepcopy(PGI)  # recopie du résultat - copy of result
            for A in Cx:
                A.append(x)  # recombiner avec x tous les éléments - recombine all elements with x
            R += Cx  # Ajouter les résultats obtenus - add the results obtained
    return R

def main(): #fonction principale - main function
    E = ['a', 'b', 'c', 'd']
    print(GenInj(2, E))
    print(list(permutations(E,2)))# avec le module itertools - with itertool module (library)


if __name__ == '__main__':
    main()
