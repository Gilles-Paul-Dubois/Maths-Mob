# fonctions numériques d'une variable entière et à valeurs réelles
# numerical functions of an integral variable
# Deux façons de coder Stirling(n) =(n^n*e^-n*racine(2*pi*n))/n!
# Two ways of coding Stirling(n)
# on utilise les fonctions et les constantes de la bibliothèque (module math)
# we use fuctions and constant from the library (math module)

import math
pi = math.pi
e = math.e

# provoque rapidement un débordement
# quickly causes an overflow
def Factorielle(n :int)->int:
    r = 1
    for i in range(1, n + 1):
        r *= i
    return r

# Même chose encore plus vite
# the same even quicker
def Stirling1(n :int)->float:
    return (n ** n) * math.exp(-n) * math.sqrt(2 * pi * n) / Factorielle(n)

# ne provoque pas de débordement
# doesn't generate any overflow
def Stirling2(n :int)->float:
    r = math.sqrt(2 * pi * n)
    for i in range(1, n + 1):
        r *= n / (e * i)
    return r

def main(): #fonction principale - main function
    print(Stirling2(100))
    print(Stirling1(100))

if __name__ == '__main__':
    main()
