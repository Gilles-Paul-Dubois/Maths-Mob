function power(x,n)
    r=big(1) #pour éviter les débordements forçage de type
    for i in 1:n
        r*=x #les claculs sont faits dans les bignums
    end
    return r
end

#version naïve utilisant la formule mathématique
function factorielle(n)
    r=big(1) #on travaille avec de très grands nombres
    for i in 2:n
        r*=i
    end
    return r
end
#la formule de Stirling
stirling1(n)=(power(n,n)/factorielle(n))*exp(-n)*sqrt(2*pi*n)

#autre version n'utilisant pas les entiers longs'
function stirling2(n)
    e=exp(1)
    p(i)=n/(i*e)
    r=1
    for i in 1:n
        r*=p(i)
    end
    return r*sqrt(2*pi*n)
end

println(stirling1(100))
println(stirling2(100))
