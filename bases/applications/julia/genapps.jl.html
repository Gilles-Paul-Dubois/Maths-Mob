# julia 1.6
# génération d'applications
# generation of mappings
using Base.Iterators
E=[1,2,3]
F=['a','b']

# fabrication d'un produit de n facteurs tous égaux à X
# building a product of n fctors all of them equal to X
PowerX_n(X, n) = Iterators.product(ntuple(i->X, n)...)

# possibilités pour les images
# possibilities for images
E3=Set(PowerX_n(F,3))

# afficher les applications E->F sous forme de dictionaires
# print maps E->F as dictionaries
while !isempty(E3)
    println(Dict(E .=> pop!(E3)))
end
