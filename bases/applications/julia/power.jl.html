compose(f,g)=x->g(f(x))

function powerr(f,n) #version récursive - recursive version
    if n==0
        return x->x
     else
         return compose(f,powerr(f,n-1))
     end
end

f(x)=2x

function poweri(f,n) #version itérative - iterative version
    g=x->x
    for i in 1:n
       g=compose(g,f)
    end
    return g
end
