function power(x,n)
    r=big(1) #type forcing to avoid overflow
    for i in 1:n
        r*=x #Computations are done with bignums 
    end
    return r
end

#naive form using the mathematical formula
function factorielle(n)
    r=big(1) #on travaille avec de tr√®s grands nombres
    for i in 2:n
        r*=i
    end
    return r
end
#Stirling's formula
stirling1(n)=(power(n,n)/factorielle(n))*exp(-n)*sqrt(2*pi*n)

#other version not using long integers'
function stirling2(n)
    e=exp(1)
    p(i)=n/(i*e)
    r=1
    for i in 1:n
        r*=p(i)
    end
    return r*sqrt(2*pi*n)
end

println(stirling1(100))
println(stirling2(100))
