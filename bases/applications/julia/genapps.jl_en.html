using Base.Iterators
E=[1,2,3]
F=['a','b']
# generating a cartesian product of n factors identical to X X
PowerX_n(X, n) = Iterators.product(ntuple(i->X, n)...)

#possibilities for images
E3=Set(PowerX_n(F,3))

#display maps E->F as dictionaries
while !isempty(E3)
    println(Dict(E .=> pop!(E3)))
end
