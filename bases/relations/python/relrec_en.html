# Illustrate the link between R and its conversee 
 
# the relation 
def R(x,y): 
    return x<y; # alphabetic order 
 
def Reciproque(x,y): 
    return R(y,x) 
 
def Antecedents(y,E,Rel): 
    return [x for x in E if Rel(x,y)] 
 
def Images(x,F,Rel): 
    return [y for y in F if R(x,y)] 
 
def main(): 
    E=set('abuvy') # departure 
    F=set('bdaz') #  destinationt 
    for x in E:# standard display as a list 
        print (x,end=':') 
        print(Images(x,F,R),end="") 
        print(Antecedents(x,F,Reciproque) ) 
 
if __name__ == '__main__': 
    main()
