# test for symmetry 
 
#non symmetric relation 
def R1(x,y): 
    return x<y; #strict alphabetical ordert 
 
#symmetric relation 
def R2(x,y): 
    return x==y; #equality 
 
#test function 
def EstSymetrique (Rel,E): 
    return all([Rel(y,x) or (not (Rel(x,y))) for x in E for y in E]) 
 
def main(): 
    E=set('abuvy') # the set 
    print (EstSymetrique(R1,E)) 
    print (EstSymetrique(R2,E)) 
 
if __name__ == '__main__': 
    main()
