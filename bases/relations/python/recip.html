# exemple d'une fonction prenant en argument une relation et retournant sa réciproque
# exemple of a function taking a relation as an argument and returning a relation

# return the inverse of relation R
def Reciproque(Rel :callable)->callable:
    return lambda x, y: Rel(y, x)

def Inf(x :int, y :int)->bool: # inégalité usuelle - usual inequality between numbers
    return x <= y

def main():
    Sup = Reciproque(Inf)
    print(Sup(2, 1))
    print(Sup(1, 2))

if __name__ == '__main__':
    main()
