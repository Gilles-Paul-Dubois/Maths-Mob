# Vérifier la reflexivité
#checking reflexivity

# la relation non réflexive -the one not reflexive
def R1(x :str, y :str)->bool:
    return x < y;  # ordre alphabétique des caractères -alpahabetic order

# la relation réflexive - the one which is reflexive
def R2(x :str, y :str)->bool:
    return x <= y;  # ordre alphabétique des caractères - alphabetic order

def EstReflexive(Rel :callable, E :set)->bool: # prediacte about reflexivity
    return all((Rel(x, x) for x in E)) # every element connected to itself

def main():
    E = set('abuvy')  # l'ensemble - the set for example
    print(EstReflexive(R1, E))
    print(EstReflexive(R2, E))

if __name__ == '__main__':
    main()
