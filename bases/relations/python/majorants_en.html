# version python 3.4 
 
#partial order 
def R(x,y): 
    return x<=y; #alphanumeric comparison 
 
# test y majorant of X for the relation Rel 
def EstMajorant(y,X,Rel): 
    return all([Rel(x,y) for x in X]) 
#return the set of all majorants of the subset X of E for the relation Rel 
def Majorants(X,E,Rel): 
    return set([y for y in E if EstMajorant(y,X,Rel)]) 
 
def main(): 
    E=set('abuvy') # the set 
    X=set('bu') 
    print (Majorants(X,E,R)) 
 
if __name__ == '__main__': 
    main()
