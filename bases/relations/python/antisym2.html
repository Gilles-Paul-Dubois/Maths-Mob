# Vérifier l'antisymétrie
# checking antisymmetry

# la relation antisymétrique-antisymmtric relation
def R1(x :str, y :str)->bool:
    return x < y;  # ordre alphabétique strict - alphabetic order

# la relation non antisymétrique - the non antisymmetric one
def R2(x :str, y :str)->bool:
    return True;  # relation pleine - total relation

# fonction de test - test function
def EstAntiSymetrique(Rel :callable, E :set)->bool:
    return all(((x == y) or not (Rel(x, y) and Rel(y, x)) for x in E for y in E))

def main(): #fonction principale - main function
    E = set('abuvy')  # l'ensemble
    print(EstAntiSymetrique(R1, E))
    print(EstAntiSymetrique(R2, E))

if __name__ == '__main__':
    main()
