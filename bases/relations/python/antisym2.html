#Vérifier l'antisymétrie 
 
#la relation antisymétrique 
def R1(x,y): 
    return x<y; #ordre alphabétique strict 
 
#la relation non antisymétrique 
def R2(x,y): 
    return True; #relation pleine 
 
#fonction de test 
def EstAntiSymetrique (Rel,E): 
    return all([(x==y) or not(Rel(x,y)and Rel(y,x)) for x in E for y in E]) 
 
def main(): 
    E=set('abuvy') #l'ensemble 
    print (EstAntiSymetrique(R1,E)) 
    print (EstAntiSymetrique(R2,E)) 
 
if __name__ == '__main__': 
    main()
