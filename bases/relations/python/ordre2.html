# Tester une relation d'ordre
# checking for partial order

# relation non antisymétrique - relation non antisymmetric
def R1(x :str, y :str)->bool:
    return True;  # relation totale 

# Relation d'ordre - partial order
def R2(x :str, y :str)->bool:
    return x <= y;  # comparaison alpha - alphabetic order

def EstTransitive(Rel :callable, E :set)->bool: #transitivité - transitivity
    return all((Rel(x, z) or not (Rel(x, y) and Rel(y, z)) for x in E for y in E for z in E))

def EstAntiSymetrique(Rel :callable, E :set)->bool: #antisymétrie - antisymmery
    return all(((x == y) or not (Rel(x, y) and Rel(y, x)) for x in E for y in E))

def EstReflexive(Rel :callable, E :set)->bool: #reflexivité - reflexivity
    return all((Rel(x, x) for x in E))

# fonction de test d'ordre - test for artial order
def EstOrdre(Rel :callable, E :set):
    return EstReflexive(Rel, E) and EstAntiSymetrique(Rel, E) and EstTransitive(Rel, E)

def main(): #fonction principale - main function
    E = set('abuvy')  # l'ensemble 
    print(EstOrdre(R1, E))
    print(EstOrdre(R2, E))

if __name__ == '__main__':
    main()
