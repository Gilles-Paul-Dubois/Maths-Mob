# version python 3.8
# Vérifier la transitivité
# checking transitivity

# la relation non transitive - the non transitive relation
def R1(x :str, y :str)->bool:
    return x != y;  # est différent de - is different from

# la relation transitive - the transitive one
def R2(x :str, y :str)->bool:
    return x <= y;  # comparaison alpha  - alphabetic order


# fonction de test - test function
def EstTransitive(Rel :callable, E :set)->bool:
    return all((Rel(x, z) or not (Rel(x, y) and Rel(y, z)) for x in E for y in E for z in E))

def main(): # fonction principale - main function
    E = set('abuvy')  # l'ensemble - the set
    print(EstTransitive(R1, E))
    print(EstTransitive(R2, E))

if __name__ == '__main__':
    main()
