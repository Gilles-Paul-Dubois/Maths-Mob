# Tester une relation d'ordre total

E = set([1, 2, 3, 6, 10, 12, 15, 30, 60, 120, 180])  # l'ensemble
A = set([10, 12, 30])
B = set([2, 3, 6, 10, 30])
C = set([3, 120])


def R(x, y):
    return y % x == 0;


def EstTransitive(Rel, E):
    for x in E:
        for y in E:
            for z in E:
                if Rel(x, y) and Rel(y, z) and (not Rel(x, z)):
                    return False
    return True


def EstAntiSymetrique(Rel, E):
    for x in E:
        for y in E:
            if (Rel(x, y) and Rel(y, x) and x != y):
                return False
    return True


def EstReflexive(Rel, E):
    for x in E:
        if (not Rel(x, x)):
            return False
        return True


def EstOrdre(Rel, E):
    return EstReflexive(Rel, E) and EstAntiSymetrique(Rel, E) and EstTransitive(Rel, E)


def EstTotal(Rel, E):
    for x in E:
        for y in E:
            if ((not R(x, y)) and (not R(y, x))):
                return False
    return True


def EstMajorant(y, X):
    for x in X:
        if not R(x, y):
            return False
    return True


def Majorants(X, E):
    return set([y for y in E if EstMajorant(y, X)])


def EstMinorant(y, X):
    for x in X:
        if not R(y, x):
            return False
    return True


def Minorants(X, E):
    return set([y for y in E if EstMinorant(y, X)])


def PlusGrand(X, E):
    M = Majorants(X, E)
    PGL = [x for x in M if x in X]
    if PGL == []:
        return 'Inexistant'
    else:
        return PGL[0]


def PlusPetit(X, E):
    M = Minorants(X, E)
    PGL = [x for x in M if x in X]
    if PGL == []:
        return 'Inexistant'
    else:
        return PGL[0]


def BorneSup(X, E):
    return PlusPetit(Majorants(X, E), E)


def BorneInf(X, E):
    return PlusGrand(Minorants(X, E), E)


def Test1():
    print("Ordre: ", EstOrdre(R, E))
    print("Total: ", EstTotal(R, E))
    print("Majorants ", Majorants(E, E))
    print("Minorants ", Minorants(E, E))
    print("Plus Grand: ", PlusGrand(E, E))
    print("Plus petit: ", PlusPetit(E, E))
    return


def Test2(X):
    print("Majorants ", Majorants(X, E))
    print("Minorants ", Minorants(X, E))
    print("Plus Grand: ", PlusGrand(X, E))
    print("Plus petit: ", PlusPetit(X, E))
    print("Borne supérieure: ", BorneSup(X, E))
    print("Borne inférieure: ", BorneInf(X, E))


def main():
    print("Pour l'ensemble E:")
    Test1()
    print("Pour la partie A:")
    Test2(A)
    print("Pour la partie B:")
    Test2(B)
    print("Pour la partie C:")
    Test2(C)


if __name__ == '__main__':
    main()
