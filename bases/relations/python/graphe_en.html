#display of an ordered pair in maths style 
def printc (Couple): 
    print ('(',end="") 
    print (Couple[0],end=",") 
    print (Couple[1],end=")") 
 
#printing of all pairs 
def printC (Produit): 
    print ('{',end="") 
    n=len(Produit) 
    for i in range(0,n-1): 
        printc (Produit[i]) 
        print (',',end='') 
    if n>0: 
        printc(Produit[n-1]) 
    print ('}',end="") 
 
#the relation 
def R(x,y): 
    return x<y; #alphabetic order of characters 
 
#building of the graph using a 'comprehension list' 
def Graphe (E,F,Rel): 
    return [(x,y) for x in E for y in F if Rel(x,y)] 
 
#computes the graph and displays it 
def main(): 
    E=set('abuvy') #set of departure 
    F=set('bdaz') # set of destination 
    G=Graphe(E,F,R) 
    printC (G) 
 
if __name__ == '__main__': 
    main()
