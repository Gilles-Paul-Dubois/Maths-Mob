# Trouver les antécédents d'un élément
# finds preimages of an element

# la relation- the relation
def R(x :str, y :str)->bool:
    return x < y;  # ordre alphabétique des caractères

# retourne l'ensemble des antécédents de y par Rel dans Ee
#computes the set of preimages of an element of the codomain
def Antecedents(y :str, E :set, Rel : callable)->set:
    return {x for x in E if Rel(x, y)}

# Afficher les antécédents pour chaque élément du but
# displays  preimages of  every element of codomain
def main():
    E = set('abuvy')  # la source - the domain
    F = set('bdaz')  # le but - the codomain
    for y in F:  
        print("antécédents de",y, end='=')
        print(Antecedents(y, E, R))


if __name__ == '__main__':
    main()
