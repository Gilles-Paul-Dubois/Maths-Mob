# Trouver l'image directe d'une partie
# Find the direct image of a subset of a domain

# la relation
def R(x :str, y :str)-> bool:
    return x < y;  # ordre alphabétique des caractères - alphabetic order

# Calcule l'image directe de la partie X de E
# pour la relation Rel de E dans F
#computes the direct image of a subset for any relation
def ImDir(X :set, F :set, Rel : callable)->set:
    L={(x,y) for x in X for y in F if Rel(x,y)}
    return {z[1] for z in L}

def main():
    E = set('abuvy')  # la source -domain
    X = set('bu') # la partie -subset
    F = set('bdaz')  # le but - codomain
    print("Imdir ", X, end="=")
    print(ImDir(X, F, R))  # affichage de l'image directe - displays direct image

if __name__ == '__main__':
    main()
