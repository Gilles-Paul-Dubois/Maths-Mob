#find the preimages of an element 
 
#the relation 
def R(x,y): 
    return x<y; #alphabetical order 
 
#for simplicity we just return a list without transforming into set 
def Antecedents(y,E,Rel): 
    return [x for x in E if Rel(x,y)] 
 
#Adisplay the preimages for each element of destination 
def main(): 
    E=set('abuvy') #departure 
    F=set('bdaz') # destination 
    for y in F:#standard display as a list 
        print (y,end=':') 
        print(Antecedents(y,E,R)) 
 
 
if __name__ == '__main__': 
    main()
