# Trouver les images d'un élément
# find the images of an element

# la relation - the relation
def R(x :str, y :str) ->bool:
    return x < y;  # ordre alphabétique des caractères- alphabetic order of characters

# calcule les images d'un élément de la source
# computes the iages of an element of the domain
def Images(x :str, F :set, Rel :callable) ->set:
    return {y for y in F if Rel(x, y)}

# Afficher les images pour chaque élément de la source
# print images foar each element of the domain
def main():
    E = set('abuvy')  # la source - domain
    F = set('bdaz')  # le but - codomain
    for x in E:
        print("Images de ",x, end=':')
        print(Images(x, F, R))


if __name__ == '__main__':
    main()
