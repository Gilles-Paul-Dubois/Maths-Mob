#check out reflexivity 
 
#non reflexive relation 
def R1(x,y): 
    return x<y; #strict alphabetic order 
 
#la relation réflexive 
def R2(x,y): 
    return x<=y; #before or equal 
 
def EstReflexive (Rel,E): 
    return all([Rel(x,x) for x in E]) 
 
def main(): 
    E=set('abuvy') #the set 
    print (EstReflexive(R1,E)) 
    print (EstReflexive(R2,E)) 
if __name__ == '__main__': 
    main()
