#construction du produit cartésien par une 'comprehension list' 
def cartesien (E,F): 
    return [[x,y] for x in E for y in F] 
 
#affichage d'un couple à la mode maths 
def printc (Couple): 
    print ('(',end="") 
    print (Couple[0],end=",") 
    print (Couple[1],end=")") 
 
 
#affichage de tous les couples 
def printC (Produit): 
    print ('{',end="") 
    n=len(Produit) 
    for i in range(0,n-1): 
        printc (Produit[i]) 
        print (',',end='') 
    if n>0: 
        printc(Produit[n-1]) 
    print ('}',end="") 
 
def main(): 
    E=set('ab') #facteur gauche = {a,b} 
    F=set('123') # facteur droit = {1,2,3} 
    P=cartesien(E,F) 
    printC (P) 
 
if __name__ == '__main__': 
    main()
