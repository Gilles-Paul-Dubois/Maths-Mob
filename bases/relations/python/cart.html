# construction of a Cartesian product with a comprehension list
def cartesien(E :set , F :set) ->set:
    return {(x, y) for x in E for y in F}

def main():
    E = set('ab')  # = {a,b}
    F = set('123')  # = {1,2,3}
    P = cartesien(E, F)
    print(P)

if __name__ == '__main__':
    main()
