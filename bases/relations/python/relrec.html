# Mettre en relief le lien entre R et sa réciproque
# show link between relation and its inverse
# la relation
def R(x :str, y :str)->bool:
    return x < y;  # ordre alphabétique des caractères - alphabetic order

def Reciproque(x :str, y :str)->bool: # inverse of R
    return R(y, x)

def Antecedents(y :str, E :set, Rel: callable): #preimages of y
    return {x for x in E if Rel(x, y)}

def Images(x :str, F :set, Rel :callable): #images of x
    return [y for y in F if R(x, y)]


def main():
    E = set('abuvy')  # la source -domain
    F = set('bdaz')  # le but-codomain
    for x in E:
        print(x, end=':')
        print(Images(x, F, R), end="")
        print(Antecedents(x, F, Reciproque))


if __name__ == '__main__':
    main()
