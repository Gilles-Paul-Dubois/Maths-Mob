#find the direct image of a subset 
 
#the relation 
def R(x,y): 
    return x<y; #alphabetical order 
 
# computes the direct image of the subset X of E 
# for the relation R from E to F 
def ImDir(X,F,Rel): 
    return set([y for y in F for x in X if Rel(x,y)]) 
 
#displays images for every element of departure 
def main(): 
    E=set('abuvy') #departure 
    X=set('bu') 
    F=set('bdaz') # destination 
    print (X,end=":") 
    print (ImDir(X,F,R))#standard display as Set object 
      
 
if __name__ == '__main__': 
    main()
