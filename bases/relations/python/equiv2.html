# version python 3.8
# Tester une relation d'équivalence
# checking equivalence

def Classe(x :str)->int: # seulement pour dfinir R1 - just to define R1
    if x < 'm':
        return 1
    else:
        return 2

def R1(x :str, y :str)->bool: # candidat pour une equivalence - candidate for an equivalence
    return Classe(x) == Classe(y);  # partition en deux classes - partition into two classes

# Relation non symétrique - non symmetric relation
def R2(x :str, y :str)->bool:
    return x <= y;  # comparaison alpha - alphabetic order

def EstTransitive(Rel :callable, E :set)->bool: # test de transitivité - transitivity test
    return all((Rel(x, z) or not (Rel(x, y) and Rel(y, z)) for x in E for y in E for z in E))

def EstSymetrique(Rel :callable, E :set)->bool: # test de symétrie - symmetry test
    return all((Rel(x, y) or not Rel(y, x) for x in E for y in E))

def EstReflexive(Rel :callable, E :set)->bool: #test de reflexivité - reflexivity test
    return all((Rel(x, x) for x in E))

# fonction de test d'équivalence - testing equivalence
def EstEquivalence(Rel :callable, E :set):
    return EstReflexive(Rel, E) and EstSymetrique(Rel, E) and EstTransitive(Rel, E)

def main(): # fonction principale - main function
    E = set('abuvy')  # l'ensemble
    print(EstEquivalence(R1, E))
    print(EstEquivalence(R2, E))

if __name__ == '__main__':
    main()
