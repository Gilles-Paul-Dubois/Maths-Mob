#Tester une relation d'équivalence 
 
def Classe (x): 
    if x<'m': 
        return 1 
    else: 
        return 2 
 
#relation d'équivalence 
def R1(x,y): 
    return Classe(x)==Classe(y); # 
 
#Relation non symétrique 
def R2(x,y): 
    return x<=y; #comparaison alpha 
 
def EstTransitive (Rel,E): 
    return all([Rel(x,z) or not(Rel(x,y) and Rel(y,z)) for x in E for y in E for z in E]) 
 
def EstSymetrique (Rel,E): 
    return all([Rel(x,y) or not Rel(y,x) for x in E for y in E]) 
 
def EstReflexive (Rel,E): 
    return all([Rel(x,x) for x in E]) 
 
# fonction de test d'équivalence 
def EstEquivalence(Rel,E): 
    return EstReflexive(Rel,E) and EstSymetrique(Rel,E) and EstTransitive(Rel,E) 
 
def main(): 
    E=set('abuvy') #l'ensemble 
    print (EstEquivalence(R1,E)) 
    print (EstEquivalence(R2,E)) 
 
if __name__ == '__main__': 
    main()
