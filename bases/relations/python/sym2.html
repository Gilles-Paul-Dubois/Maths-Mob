# version python 3.8
# Vérifier la symétrie
# check symmetry

# la relation non symétrique - the non-symmetric relation
def R1(x :str, y :str)->bool:
    return x < y;  # ordre alphabétique strict- strict alphabetic order

# la relation symétrique - the symmetric relation
def R2(x :str, y :str)->bool:
    return x == y;  # égalité - equality

# fonction de test- test function
def EstSymetrique(Rel :callable, E :set)->bool:
    return all((Rel(y, x) or (not (Rel(x, y))) for x in E for y in E))

def main(): #fonction principale - main function
    E = set('abuvy')  # l'ensemble -the set
    print(EstSymetrique(R1, E))
    print(EstSymetrique(R2, E))

if __name__ == '__main__':
    main()
