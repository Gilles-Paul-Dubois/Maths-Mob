# version python3.8
# produits cartésiens généralisés avec les 'comprehension lists'
#generalized Cartesian products

def main():
    Tailles = [30, 32]
    Couleurs = ['jaune', 'rouge', 'bleu']
    Textures = ['Jeans', 'Cuir']
    print({(X, Y, Z)for X in Tailles for Y in Couleurs for Z in Textures})

if __name__ == '__main__':
    main()
