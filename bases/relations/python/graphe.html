# version python3.8

# la relation -the relation
def R(x :str, y :str) ->bool:
    return x < y;  # ordre alphabétique des caractères

# construction du graphe en 'comprehension'
# constructing the graph in comprehension
def Graphe(E :set, F :set, Rel :callable) ->set:
    return {(x, y) for x in E for y in F if Rel(x, y)}

# calcule un graphe et l'affiche
#computes a graph and print it
def main():
    E = set('abuvy')  # la source - domain
    F = set('bdaz')  # le but - codomain
    G = Graphe(E, F, R)
    print(G)

if __name__ == '__main__':
    main()
