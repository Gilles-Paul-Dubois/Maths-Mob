#affichage d'un couple à la mode maths 
def printc (Couple): 
    print ('(',end="") 
    print (Couple[0],end=",") 
    print (Couple[1],end=")") 
 
#affichage de tous les couples 
def printC (Produit): 
    print ('{',end="") 
    n=len(Produit) 
    for i in range(0,n-1): 
        printc (Produit[i]) 
        print (',',end='') 
    if n>0: 
        printc(Produit[n-1]) 
    print ('}',end="") 
 
#la relation 
def R(x,y): 
    return x<y; #ordre alphabétique des caractères 
 
#construction du graphe par une 'comprehension list' 
def Graphe (E,F,Rel): 
    return [(x,y) for x in E for y in F if Rel(x,y)] 
 
#calcule un graphe et l'affiche 
def main(): 
    E=set('abuvy') #la source 
    F=set('bdaz') # le but 
    G=Graphe(E,F,R) 
    printC (G) 
 
if __name__ == '__main__': 
    main()
