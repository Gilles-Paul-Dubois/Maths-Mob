# Trouver l'image réciproque d'une partie du but
# find preimage of a subset of the codomain

# la relation - the relation
def R(x :str, y :str)->bool:
    return x < y;  # ordre alphabétique des caractères- alphabetic ordzer


# Calcule l'image réciproque de la partie Y de F - compute the preimage of subset y
# pour la relation Rel de E dans F - for relation Rel from E to F
def ImRec(Y :set, E :set, Rel :callable) ->set:
    L= {(x,y) for x in E for y in Y if Rel(x,y)}
    return {z[0] for z in L}


# Afficher l'image réciproque de Y- print preimage of Y
def main():
    E = set('abuvy')  # la source
    Y = set('bd')
