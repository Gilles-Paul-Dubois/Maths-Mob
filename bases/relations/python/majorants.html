# Relation d'ordre
# partial order

#the relation to test
# la relation Ã  tester
def R(x :str, y :str)->bool:
    return x <= y;  # comparaison alpha 

# test y majorant de X pour la relation Rel
# checks y majorant of X for Rel on E
def EstMajorant(y :str, X :set, Rel :callable)->bool:
    return all((Rel(x, y) for x in X))

# retourne l'ensemble des majorants de la partie X de E pour la relation Rel
# return the set of majorants
def Majorants(X :set, E :set, Rel :callable)->set :
    return set((y for y in E if EstMajorant(y, X, Rel)))

#main function - fonction principale
def main():
    E = set('abuvy')  # l'ensemble 
    X = set('bu')
    print(Majorants(X, E, R))


if __name__ == '__main__':
    main()
