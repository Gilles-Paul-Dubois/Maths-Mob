# version python 3.4 
 
#Check transitivity 
 
#non transitive relation 
def R1(x,y): 
    return x!=y; # is different from 
 
#the transitive relation 
def R2(x,y): 
    return x<=y; #alphanumerical order 
 
#test function 
def EstTransitive (Rel,E): 
    return all([Rel(x,z) or not(Rel(x,y) and Rel(y,z)) for x in E for y in E for z in E]) 
 
def main(): 
    E=set('abuvy') #the set 
    print (EstTransitive(R1,E)) 
    print (EstTransitive(R2,E)) 
if __name__ == '__main__': 
    main()
