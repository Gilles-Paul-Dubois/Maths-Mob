# julia 1.6
# produits cartésiens - Cartesian products

# deux ensembles pour test
# two sample sets
A=Set("abc")
B=Set([1,2,3])

# fabrication des produits
# building products
P1=Set(collect(Base.product(A,B)))
P2=Set(collect(Base.product(A,B,A)))

# écriture classique des maths
# usual writing in maths
function imprime(X)
  println("")
  print('{') #début de l'écriture -beginning of display
  n = length(X)
  i = 0
  for x in X
    i = i + 1
    print(x)
    if i < n
      print(',') #seulement si ce n'est pas le dernier - only if not last
    end
  end
  println('}') #fin de l'écriture - end of display
end

imprime(P1)
imprime(P2)
