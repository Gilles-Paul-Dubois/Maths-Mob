A=Set("abc")
B=Set([1,2,3])
P1=Set(collect(Base.product(A,B)))
P2=Set(collect(Base.product(A,B,A)))

function imprime(X) #écriture classique des maths
  println("")
  print('{') #début de l'écriture
  n = length(X)
  i = 0
  for x in X
    i = i + 1
    print(x)
    if i < n
      print(',') #seulement si ce n'est pas le dernier
    end
  end
  println('}') #fin de l'écriture
end

imprime(P1)
imprime(P2)
