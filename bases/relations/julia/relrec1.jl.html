#la relation- the relation
R(x,y)=x<y
#Sa r√©ciproque - its converse
Reciproque(x,y)=R(y,x)

Antecedents(y,E,Rel) = [x for x in E if Rel(x,y)]
Images(x,F,Rel)=[y for y in F if Rel(x,y)]

function main()
    E="abuvy"
    F="bdaz"
    for x in E
        print("x=")
        print(x)
        print(Images(x,F,R))
        println(Antecedents(x,F,Reciproque))
    end
end

main()
