#la relation
R(x,y)=x<y
#fonction qui calcule les antécédents
Antecedents(y,X,Rel)=[x for x in X if Rel(x,y)]
#fonction principale
function main()
    E="abuvy"
    F="bdaz"
    for y in F
        print("y=")
        print(y)
        print(" antécédents : ")
        println(Antecedents(y,E,R))
    end
 end

main()
