#la relation - the relation
R(x,y)=x<y

#fonction qui calcule les images
# function whic calculates the images
Images(x,Y,Rel)=[y for y in Y if Rel(x,y)]

#fonction principale
#main function
function main()
    E="abuvy"
    F="bdaz"
    for x in E
        print("x=")
        print(x)
        print(" images : ")
        println(Images(x,F,R))
    end
 end

main()
