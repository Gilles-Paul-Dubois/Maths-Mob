#predicat unaire - unary predicate
Pair(x) = x%2==0

#predicat binaire - binary predicate
MemeParite(x,y)= x%2==y%2

#predicat ternaire - ternary predicate
Ordonnes(x,y,z) = x<=y<=z

function main()
    println(Pair(2))
    println(Pair(3))
    println(MemeParite(2,4))
    println(Ordonnes(2,3,5))
end
    
main()
