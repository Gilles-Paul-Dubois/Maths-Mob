using Pkg
Pkg.add("IterTools")
using IterTools
#ensemble de toutes les parties de {1,2,3,4}- set of all subsets of {1,2,3,4}
E = subsets([1, 2, 3, 4])
#propriété P : être un nombre premier - Property P : to be a prime number
P(x)= length([i for i in 1:x if x%i==0])==2

# Tous les éléments de X sont premiers - all elements of X are prime
#proposition avec quantificateur universel - proposition with universal quantifier
A(X)=all((P(x) for x in X))
#sa négation - its negation
B(X)= any((!P(x) for x in X))

for X in E
    print(X)
    print(" : ")
    print(A(X))
    print(" - ")
    print(B(X))
    println("")
end
