x=1
function fonc1()
    x=2
    print("x local of fonc1 has value  ")
    println(x)
    function fonc2()
        x=3
        print("x local of fonc2 has value  : ")
        println(x)
    end
    fonc2()
    print(" x local of fonc1 has now value  : ")
    println(x)
end

function fonc3()
    global x
    print(" global x  has still  value  : ")
    println(x)
end

function fonc4(x)
    print("My parameter x has value  : ")
    println(x)
end

function main()
    print("global variable x has value  : ")
    println(x)
    fonc1()
    fonc3()
    fonc4(5)
end

main()
