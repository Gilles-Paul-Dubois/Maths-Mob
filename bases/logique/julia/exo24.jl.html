constP(f) = length(f) == 1 && f[1] in "VF"

varP(f) = length(f) == 1 && f[1] in "ABCDEGHIJKLMNOPQRSTUWXYZ"

inparP(f) = f[1] == '(' && f[length(f)] == ')'

betweenpar(f) = f[2:length(f)-1]

vars(f) = Set(x for x in f if x in "ABCDEGHIJKLMNOPQRSTUWXYZ")
nbvars(f) = length(vars(f))

PowerX_n(X, n) = Iterators.product(ntuple(i -> X, n)...)
LV = [true, false]
Fn(f) = collect(PowerX_n(LV, nbvars(f)))
Env(f, k) = Dict(vars(f) .=> Fn(f)[k])
function parseP(f)
    if constP(f)
        return f[1]
    end
    if varP(f)
        return f[1]
    end
    if inparP(f)
        a = parseP(betweenpar(f))
        if a != nothing
            return a
        end
    end
    if f[1] == '!'
        a = parseP(f[2:end])
        if a != nothing
            return ['!', a]
        end
    end
    if '&' in f
        for i = 1:length(f)
            if f[i] == '&'
                a = parseP(f[1:i-1])
                b = parseP(f[i+1:end])
                if (a != nothing) && (b != nothing)
                    return ['&', a, b]
                end
            end
        end
    end
    if '|' in f
        for i = 1:length(f)
            if f[i] == '|'
                a = parseP(f[1:i-1])
                b = parseP(f[i+1:end])
                if (a != nothing) && (b != nothing)
                    return ['|', a, b]
                end
            end
        end
    end
    if '>' in f
    for i = 1:length(f)
        if f[i] == '>'
            a = parseP(f[1:i-1])
            b = parseP(f[i+1:end])
            if (a != nothing) && (b != nothing)
                return ['>', a, b]
            end
        end
    end
    end
    if '#' in f
        for i = 1:length(f)
            if f[i] == '#'
                a = parseP(f[1:i-1])
                b = parseP(f[i+1:end])
                if (a != nothing) && (b != nothing)
                    return ['#', a, b]
                end
            end
        end
    end
    return nothing
end

function evalP(Tree, Env)
    if Tree == 'V'
        return true
    elseif Tree == 'F'
        return false
    elseif Tree in "ABCDEGHIJKLMNOPQRSTUWXYZ"
        return Env[Tree]
    elseif Tree[1] == '!'
        a = evalP(Tree[2], Env)
        return !a
    elseif Tree[1] == '&'
        a = evalP(Tree[2], Env)
        b = evalP(Tree[3], Env)
        return a && b
    elseif Tree[1] == '|'
        a = evalP(Tree[2], Env)
        b = evalP(Tree[3], Env)
        return a || b
    elseif Tree[1] == '>'
        a = evalP(Tree[2], Env)
        b = evalP(Tree[3], Env)
        return b||(!a)
    elseif Tree[1] == '#'
        a = evalP(Tree[2], Env)
        b = evalP(Tree[3], Env)
        return (b||(!a))&&(a||(!b))
    else
    end
end

valeur(formule, k) = evalP(parseP(formule), Env(formule, k))

function table(f)
    print("formule: ")
    println(f)
    analyse = parseP(f)
    print("analys√©e comme :")
    println(analyse)
    print("variables de la formule: ")
    println(vars(f))
    print("Nombre de ses variables : ")
    println(nbvars(f))
    for k in 1:2^nbvars(f)
        print(Env(f,k))
        print(":")
        println(valeur(f,k))
    end
    if all([valeur(f,k)==true for k in 1:2^nbvars(f)])
        println("Tautologie: oui")
    else
        println("Tautologie: non")
    end
end
