function ContreExemple(SP,C)
    VC=VARS(SP)
    ENVS=[Env(VC,k) for k in 1:2^NBVARS(SP)]
    EVLF=Valide(SP,ENVS)
    for E in EVLF
        if Valeur(C,E)==false
            return E
        end
     end
 end


function main()
    F1="A|!(B&C)"
    F2="!A"
    SP=[F1,F2]
    C="!B"
    println(ContreExemple(SP,C))
end

main()
