# Boolean type variables

import random

def main():
    x = random.randint(0, 9)
    print(x > 5)  # une chance sur deux - one chance on two
    y = random.randint(0, 9)
    print(x < y)  # une chance sur deux aussi - the same
    z = random.randint(0, 9)
    print(x <= y <= z)  # 25% des cas - 25% of cases only

if __name__ == '__main__':
    main()
