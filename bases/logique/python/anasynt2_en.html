# Python version 3.4 
#parsing program 
#for propositional logic 
# with connectors not, and, or, implication and equivalence  
# represented by symbols - , &, |, >,# 
# 
 
 
# logical constants 0= False 1=True 
def ConstantP(X): 
    return X=='0' or X=='1' 
#variables 
def VariableP(X): 
    return  len(X)==1 and 'A'<=X <='Z' 
#atoms 
def AtomP(X): 
    return ConstantP(X) or VariableP(X) 
 
#checks if X is a valid proposition 
# recursive function 
def ProP (X): 
    if AtomP(X): # atoms are propositions 
        return True 
    if X[0]=='(': #case of parentheses 
        n=1 
        for i in range(1,len(X)): 
            if X[i]=='(': 
                n=n+1 #increments opening par. 
            if X[i]==')': 
                n=n-1 #decrements closing par. 
                if n==0: # we have found the closing par. matching the first opening one. 
                    #we replace the group between parentheses by constant 1 
                    return ProP("1"+X[i+1:]) 
        return False 
    # looking for a negation (symbol -) 
    if X[0]=='-' and ProP(X[1:]): 
        return True 
    # looking for a conjunction (symbol &) 
    for i in range(0,len(X)): 
        if X[i]=='&' and ProP(X[0:i]) and ProP(X[i+1:]): 
            return True 
    # looking for a disjunvtion (symbol |) 
    for i in range(0,len(X)): 
        if X[i]=='|' and ProP(X[0:i]) and ProP(X[i+1:]): 
            return True 
    # looking for an implication (symbol >) 
    for i in range(0,len(X)): 
        if X[i]=='>' and ProP(X[0:i]) and ProP(X[i+1:]): 
            return True 
    # looking for an equivalence (symbol #) 
    for i in range(0,len(X)): 
        if X[i]=='#' and ProP(X[0:i]) and ProP(X[i+1:]): 
            return True 
def main(): 
    print (ProP("((-B)&A)|(C>B)&0")) 
 
if __name__ == '__main__': 
    main()
