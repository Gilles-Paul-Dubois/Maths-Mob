# Python version 3.4 
# exemples de prédicats 
# examples of predicates 
import random 
 
# prédicat unaire - unary predicate 
def Pair(x): 
    return x%2==0 
 
# prédicat binaire - binary predicate 
def MemeParite(x,y): 
    return x%2==y%2 
 
# prédicat ternaire - ternary predicate 
def Ordonnes(x,y,z): 
    return (x <=y) and (y<=z) 
 
def main(): 
    x = random.randint(0,9) 
    print (Pair(x)) # une chance sur deux - probability 50% 
    y= random.randint(0,9) 
    print (MemeParite(x,y)) # une chance sur deux aussi - the same 
    z=random.randint(0,9) 
    print (Ordonnes(x,y,z)) # 25% des cas - 25% of cases 
 
 
if __name__ == '__main__': 
    main()
