# Python version 3.4 
#quantificateur existentiel 
#une utilisation de la fonction any(...) 
#existential quantifier 
#use of function any(...) 
 
L1=range(8,12) 
L2=range(10,13) 
 
P=range(1,4) 
 
#vérifie si L contient le carré d'un nombre entre 1 et 3 
# donc s'il existe x dans P tel que x² dans L 
#checks if L contains the square of a number between 1 and 3  
# so if it exists x in P such that x² in L 
def test (L): 
    return any ([x*x in L for x in P]) 
 
def main(): 
    print (test(L1)) #vrai - true 
    print (test(L2)) #faux - false 
 
if __name__ == '__main__': 
    main()
