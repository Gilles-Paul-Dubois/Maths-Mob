# Python version 3.4 
#quantificateur universel 
#une utilisation de la fonction all(...) 
#universal quantifier 
#use of the function all(...) 
 
 
L1=range(8,12) 
L2=range(-5,5) 
 
 
#vérifie si L ne contient que des nombres positifs 
# donc si quelque soit x dans L on a: x>0 
#checks if L contains only positive numbers 
# so if for any x in L we have : x>0 
 
def test (L): 
    return all ([x>0  for x in L]) 
 
def main(): 
    print (test(L1)) #vrai - true 
    print (test(L2)) #faux - false 
 
if __name__ == '__main__': 
    main()
