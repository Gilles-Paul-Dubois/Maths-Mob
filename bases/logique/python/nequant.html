# négation d'une formule quantifiée 
# negation of a quantified formula 
# Ceci ne constitue nullement une preuve 
# This is not a proof 
# mais une simple vérification sur des cas particuliers aléatoires.
# but a simple checking on random cases.  
import random 
 
L= [random.randint(1,2) for x in range(0,3)] # suite de 3 pile ou face - sequence of three head and tail 
 
def main(): 
    P= all([y%2 for y in L]) # tous impairs - all of them odd  
    R= any([y%2==0 for y in L]) # il existe un pair - It exists an even one 
    Q=not P 
    print (P) 
    print (Q) 
    print (R) 
    print("-------------------") 
    P=any([y%2 for y in L]) #il existe un impair - It exists one odd  
    R=all([y%2==0 for y in L]) # Tous pairs - all of them even 
    Q=not P 
    print (P) 
    print (Q) 
    print (R) 
 
 
if __name__ == '__main__': 
    main()
