# Python version 3.8
# variables locales et globales - local and global variables
# portée des variables - scope of variables

x = 1  # x global initialisé au niveau fichier - x is global initialization at file level

def fonc1()->None:
    x = 2  # x locale de fonc1 initialisée dans le corps de fonc1 (autre variable homonyme)
    # x local variable of fonc1 initialized in the body of fonc1
    print("x local de fonc1 vaut: ", x)
    def fonc2()->None:
        x = 3  # x local de fonc2 initialisé dans le corps de fonc2 (encore un x différent)
        # local variable of fonc2 initialized in the body of fonc2
        print("x local de fonc2 vaut: ", x)
    fonc2()  # appel de fonc2 sous-fonction de fonc1 - call to fonc2 local (child) function of fonc1
    print("x local de fonc1 vaut toujours: ", x)  # variable inchangée - unchanged value

def fonc3()->None:
    global x  # On précise: c'est bien la variable globale du début que l'on veut lire puis modifier
    # We specify that it's the global variable which we want to read and edit
    print("x global vaut toujours: ", x)  # accès à la variable globale - access to the global variable
    x = 4  # modification de la variable globale

def fonc4(x :int)->None:  # le paramètre formel d'une fonction se comporte comme une variable locale et masque les globales
    # a formal parameter of a function  behaves like a local vaiable and hides the global ones
    print("Mon paramètre x vaut: ", x)

def main(): # fonction principale - main function
    print("x global vaut: ", x)  # simple vérification - simple checking
    fonc1()  # création de variables locales homonymes - creation of local variables with the same name
    fonc3()  # accès en lecture écriture à une variable globale - access read-write to a global variable
    print("x global changé par fonc3: ", x)  # vérification de l'effet de fonc3 sur le x global  - checking the effect of fonc3 on global x
    fonc4(5) # no comment

if __name__ == '__main__':
    main()
