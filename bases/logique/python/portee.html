# Python version 3.4 
# variables locales et globales 
# portée des variables 
 
x=1 # x global initialisé au niveau fichier 
 
def fonc1 (): 
    x=2 # x local de fonc1 initialisé dans le corps de fonc1 (autre variable homonyme) 
    print ("x local de fonc1 vaut: ",x) 
    def fonc2 (): 
        x=3 # x local de fonc2 initialisé dans le corps de fonc2 (encore un x différent) 
        print ("x local de fonc2 vaut: ",x) 
    fonc2() # appel de fonc2 sous-fonction de fonc1 
    print ("x local de fonc1 vaut toujours: ",x) # variable inchangée 
 
def fonc3(): 
    global x # On précise: c'est bien la variable globale du début que l'on veut lire puis modifier 
    print ("x global vaut toujours: ",x) # accès à la variable globale 
    x=4 #modification de la variable globale 
 
def fonc4(x): # le paramètre formel d'une fonction se comporte comme une variable locale et masque les globales 
    print ("Mon paramètre x vaut: ",x) 
 
def main(): 
    print ("x global vaut: ",x) # simple vérification 
    fonc1() # création de variables locales homonymes 
    fonc3() # accès en lecture écriture à une variable globale 
    print ("x global changé par fonc3: ",x) # vérification de l'effet de fonc3 sur le x global 
    fonc4(5) 
 
if __name__ == '__main__': 
    main()
