import random 
 
def main(): 
    x = random.randint(0,9) 
    print (x>5) # One chance on two

    print (x<y) # One chance on two again 
    z=random.randint(0,9) 
    print (x <=y <=z) # 25% of all cases 
 
 
if __name__ == '__main__': 
    main()
