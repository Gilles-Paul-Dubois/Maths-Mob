# Python version 3.4


def ConstantP(X):
    return X == '0' or X == '1'


def VariableP(X):
    return len(X) == 1 and 'A' <= X <= 'Z'


def AtomP(X):
    return ConstantP(X) or VariableP(X)


def ProP(X):
    if AtomP(X):
        return True
    if X[0] == '(':
        n = 1
        for i in range(1, len(X)):
            if X[i] == '(':
                n = n + 1
            if X[i] == ')':
                n = n - 1
                if n == 0:
                    return ProP("1" + X[i + 1:])
        return False
    if X[0] == '-' and ProP(X[1:]):
        return True
    for i in range(0, len(X)):
        if X[i] == '&' and ProP(X[0:i]) and ProP(X[i + 1:]):
            return True
    for i in range(0, len(X)):
        if X[i] == '|' and ProP(X[0:i]) and ProP(X[i + 1:]):
            return True
    for i in range(0, len(X)):
        if X[i] == '>' and ProP(X[0:i]) and ProP(X[i + 1:]):
            return True
    for i in range(0, len(X)):
        if X[i] == '#' and ProP(X[0:i]) and ProP(X[i + 1:]):  # Ã©quivalence
            return True


def Eval_Constant(X):
    if X == '0':
        return 0
    else:
        return 1


def Eval_Variable(X, Env):
    return Env[X]


def Eval_Atom(X, Env):
    if ConstantP(X):
        return Eval_Constant(X)
    else:
        return Eval_Variable(X, Env)


def Eval_Pro(X, Env):
    if AtomP(X):
        return Eval_Atom(X, Env)
    if X[0] == '(':
        n = 1
        for i in range(1, len(X)):
            if X[i] == '(':
                n = n + 1
            if X[i] == ')':
                n = n - 1
                if n == 0:
                    return Eval_Pro(str(Eval_Pro(X[1:i], Env)) + X[i + 1:], Env)
        return False
    if X[0] == '(' and X[len(X) - 1] == ')':
        return Eval_Pro(X[1:len(X) - 1], Env)
    if X[0] == '-' and ProP(X[1:]):
        if Eval_Pro(X[1:], Env) == 0:
            return 1
        if Eval_Pro(X[1:], Env) == 1:
            return 0
    for i in range(0, len(X)):
        if X[i] == '&' and ProP(X[0:i]) and ProP(X[i + 1:]):
            return Eval_Pro(X[0:i], Env) * Eval_Pro(X[i + 1:], Env)
    for i in range(0, len(X)):
        if X[i] == '|' and ProP(X[0:i]) and ProP(X[i + 1:]):
            return max(Eval_Pro(X[0:i], Env), Eval_Pro(X[i + 1:], Env))
    for i in range(0, len(X)):  # implication
        if X[i] == '>' and ProP(X[0:i]) and ProP(X[i + 1:]):
            if Eval_Pro(X[0:i], Env) == 1 and Eval_Pro(X[i + 1:], Env) == 0:
                return 0
            else:
                return 1
    for i in range(0, len(X)):
        if X[i] == '#' and ProP(X[0:i]) and ProP(X[i + 1:]):
            if Eval_Pro(X[0:i], Env) != Eval_Pro(X[i + 1:], Env) == 0:
                return 0
            else:
                return 1


def VariablesDans(F):
    if ConstantP(F):
        return set([])
    if VariableP(F):
        return set([F[0]])
    if F[0] == '(' and F[len(F) - 1] == ')' and ProP(F[1:len(F) - 1]):
        return VariablesDans(F[1:len(F) - 1])
    if F[0] == '-' and ProP(F[1:]):
        return VariablesDans(F[1:])
    for i in range(0, len(F)):
        if F[i] == '&' and ProP(F[0:i]) and ProP(F[i + 1:]):
            return VariablesDans(F[0:i]) | VariablesDans(F[i + 1:])
    for i in range(0, len(F)):
        if F[i] == '|' and ProP(F[0:i]) and ProP(F[i + 1:]):
            return VariablesDans(F[0:i]) | VariablesDans(F[i + 1:])
    for i in range(0, len(F)):
        if F[i] == '>' and ProP(F[0:i]) and ProP(F[i + 1:]):
            return VariablesDans(F[0:i]) | VariablesDans(F[i + 1:])
    for i in range(0, len(F)):
        if F[i] == '#' and ProP(F[0:i]) and ProP(F[i + 1:]):
            return VariablesDans(F[0:i]) | VariablesDans(F[i + 1:])


def power(n, F):
    G = [[x] for x in F]
    for i in range(0, n - 1):
        G = [y + [x] for y in G for x in F]
    return G


def ListEnv(S):
    R = []
    K = list(S)
    V = power(len(S), [0, 1])
    for LV in V:
        A = dict()
        for i in range(0, len(S)):
            A[K[i]] = LV[i]
        R.append(A)
    return R


def Table(F):
    S = VariablesDans(F)
    ENVS = ListEnv(S)
    return [[E, Eval_Pro(F, E)] for E in ENVS]


def Valeurs(F):
    return [X[1] for X in Table(F)]


def Tautologie(F):
    return Valeurs(F) == [1] * len(Valeurs(F))


def main():
    F = "R>(S>R)"
    print(Tautologie(F))
    F = "(R>S)>((S>T)>(R>T))"
    print(Tautologie(F))
    F = "(R|S)#((R>S)>S)"
    print(Tautologie(F))
    F = "(R>(S|-T))#((T&R)>S)"
    print(Tautologie(F))


if __name__ == '__main__':
    main()
