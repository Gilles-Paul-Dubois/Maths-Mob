# disjonction logique
# logical disjunction
# l'op√©rateur 'or' - the 'or' operator

def main():
    print((2 < 3) or (4 < 3))  # True
    print((3 < 2) or (4 < 3))  # False
    A, B = set([1, 2]), set([2, 1])
    print((A <= B) or (B <= A))  # True
    print((A < B) or (B < A))  # False

if __name__ == '__main__':
    main()
