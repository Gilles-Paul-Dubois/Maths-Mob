# Translation of constants 
def t_CONST(t): 
    r'V|F' 
    t.value = True if t.value=='V' else False   
    return t 
 
# Access to truth values of variables 
def t_VAR(t): 
    r'[P-U]' 
    if t.value in globals().keys(): 
        t.value = globals()[t.value] 
    return t
