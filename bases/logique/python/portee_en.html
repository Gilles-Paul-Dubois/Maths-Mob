# Python version 3.4 
# local and global variables 
# scope of variables 
 
x=1 # global x with initialization at the file level 
 
def fonc1 (): 
    x=2 # x local of fonc1 initialized inside the body of fonc1 (a new x))  
    print ("x local of fonc1 has value : ",x) 
    def fonc2 (): 
        x=3 # x local of fonc2 initiualized inside the body of fonc2 (still another distinct x) 
        print ("x local of fonc2 has value : ",x) 
    fonc2() # call of fonc2 sub-function of fonc1 
    print ("x local of fonc1 still has value : ",x) #unchanged variable 
 
def fonc3(): 
    global x # We specify that it is really the first x encountered that we want to change 
    print ("x global vaut toujours: ",x) # access to the global variable 
    x=4 #changing the value of the global variable 
 
def fonc4(x): # the formal parameter of a function has the behaviour of a local variable and hides any global with the same name 
    print ("My parameter x has value : ",x) 
 
def main(): 
    print ("x global has value : ",x) # simple checking 
    fonc1() # creation of homonymous local variables 
    fonc3() # full access (reading - writing) to a global variable  
    print ("global x changed by fonc3: ",x) # checking the effect of fonc3 on global x 
    fonc4(5) 
 
if __name__ == '__main__': 
    main()
