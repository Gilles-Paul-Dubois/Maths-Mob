#creating a random operation on [1,2,3,4,5]
n=5
E=collect(1:n)
E2=Set(collect(Base.product(E,E)))
Table=Dict()
#generating results
for i in E2
   Table[i]=rand(E)
end
# Display the results
for i in E
   for j in E
       print("(",i)
       print('*')
       print(j,")=")
       print(Table[(i,j)])
       println("")
   end
end
