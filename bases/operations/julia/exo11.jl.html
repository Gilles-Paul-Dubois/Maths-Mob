E="eabxyz"
R1=["eabxyz","abezxy","beayzx","xyzeba","yzxbae","zxyaeb"]
R2=["eabxyz","abezxy","beayzx","xyzbea","yzxeab","zxyabe"]
R3=["eabxyz","abezxy","beayzx","xyzeab","yzxbea","zxyabe"]
Map1,Map2,Map3=Dict(),Dict(),Dict()
for i in 1:6
    for j in 1:6
        Map1[(E[i],E[j])]=(R1[i])[j]
    end
end
for i in 1:6
    for j in 1:6
        Map2[(E[i],E[j])]=(R2[i])[j]
    end
end
for i in 1:6
    for j in 1:6
        Map3[(E[i],E[j])]=(R3[i])[j]
    end
end
fonc1(x,y)=Map1[(x,y)]
fonc2(x,y)=Map2[(x,y)]
fonc3(x,y)=Map3[(x,y)]
associative(f)=all((f(x,f(y,z))==f(f(x,y),z) for x in E for y in E for z in E))
commutative(f)=all((f(x,y)==f(y,x) for x in E for y in E))

println(associative(fonc1))
println(associative(fonc2))
println(associative(fonc3))
println(commutative(fonc3))
