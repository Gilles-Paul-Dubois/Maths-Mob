# julia 1.6
#fabrication d'une loi associative  sur [0,1,2,3,4,5]
# creating an associative operation on [0,1,2,3,4,5]
n=5
E=collect(0:n)
E2=Set(collect(Base.product(E,E)))

# génération d'une'loi aléatoire
# generating a random operation
function genere1()
    Table=Dict()
    for i in E2
       Table[i]=rand(E)
    end
    return Table
end

# génération d'une loi associative
# generating an associative operation
function genere2()
    Table=Dict()
    for i in 0:n
        for j in 0:n
            Table[(i,j)]=(i+j)%n
        end
    end
    return Table
end
Table1=genere1()
Table2=genere2()

associative(T)= all((T[(T[(i,j)],k)]==T[(i,T[(j,k)])] for i in 0:n for j in 0:n for k in 0:n))

println(associative(Table1))
println(associative(Table2))
