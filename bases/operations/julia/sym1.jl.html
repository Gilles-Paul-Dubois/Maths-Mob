# julia 1.6
# neutre - éléments symétriques
# neutral element - inverses

n=4
E=collect(0:n)
E2=Set(collect(Base.product(E,E)))

# génération d'une loi avec élément neutre
# generation of an operation with neutral element
function genere2()
    Table=Dict()
    for i in 0:n
        for j in 0:n
            Table[(i,j)]=(i+j)%(n+1)
        end
    end
    return Table
end

Table2=genere2()

neutre(x,T)= all([T[(x,i)]==i && T[(i,x)]==i for i in 0:n])

function TrouveNeutre(T) #finds neutral
    for x in 0:n
        if neutre(x,T)
            println(x)
         end
     end
end

function main() #fonction principale - main function
    println("Neutre de Table2")
    TrouveNeutre(Table2) # affiche 0 -displays 0
    #recherche des symétriques - looking for inverses
    println("couples de symétriques")
    S= ((i,j) for i in 0:n for j in 0:n if Table2[(i,j)]==0 && Table2[(j,i)]==0 && i<=j)
    for c in S
       println(c)
    end
end

main()
