n=4
E=collect(0:n)
E2=Set(collect(Base.product(E,E)))

#génération d'une'loi aléatoire-generating a random operation
function genere1()
    Table=Dict()
    for i in E2
       Table[i]=rand(E)
    end
    return Table
end

#-génération d'une loi avec neutre - generating an operation with neutral
function genere2()
    Table=Dict()
    for i in 0:n
        for j in 0:n
            Table[(i,j)]=(i+j)%(n+1)
        end
    end
    return Table
end
Table1=genere1()
Table2=genere2()

neutre(x,T)= all([T[(x,i)]==i && T[(i,x)]==i for i in 0:n])

function TrouveNeutre(T)
    for x in 0:n
        if neutre(x,T)
            println(x)
         end
     end
end

function main()
    println("Neutre de Table1")
    TrouveNeutre(Table1)
    println("Neutre de Table2")
    TrouveNeutre(Table2)
end

main()
