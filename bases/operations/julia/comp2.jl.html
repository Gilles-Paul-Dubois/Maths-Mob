using Base.Iterators
#réunion d'ensembles'
#union of sets'
A=[[1,2,3],[1,3,4,5],[1,6]]
reunion(X)=Set(flatten(X)) #'Set' enlève les doublons - 'Set' removes multiple occurrences

#intersection de deux ensembles
#intersection of two sets
inter(E,F)= [x for x in E if x in F]
#intersection d'une famille quelconque
#intersection of any family
intersection(X)= reduce((E,F)->inter(E,F),X)

function main()
    println(reunion(A))
    println(intersection(A))
end

main()
