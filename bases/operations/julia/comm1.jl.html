#fabrication d'une loi aléatoire  sur [1,2,3,4,5]
n=5
E=collect(1:n)
E2=Set(collect(Base.product(E,E)))

#génération d'une'loi aléatoire
function genere1()
    Table=Dict()
    for i in E2
       Table[i]=rand(E)
    end
    return Table
end

#génération d'une loi commutative
function genere2()
    Table=Dict()
    for i in 1:n
        for j in 1:n
            if i<=j
                Table[(i,j)]= rand(E)
             else
                 Table[(i,j)]=Table[(j,i)]
             end
        end
    end
    return Table
end
Table1=genere1()
Table2=genere2()

commutative(T)= all((T[(i,j)]==T[(j,i)] for i in 1:n for j in 1:n))

println(commutative(Table1))
println(commutative(Table2))
