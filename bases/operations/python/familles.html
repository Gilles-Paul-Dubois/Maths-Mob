# intersection et réunion de familles d'ensembles 
# intersection and union of families of sets 
# présentées comme liste de listes 
# given as lists of lists  
from functools import reduce 
 
def Inter(L): 
    if len(L)==1: 
        return set(L[0]) 
    I=[X for X in L[0] if all([X in B for B in L[1:]])] 
    return set(I) 
 
def Union (L): 
    if L==[]: 
        return set([]) 
    if len(L)==1: 
        return set(L[0]) 
    U=reduce(lambda x,y:x+y,L) 
    return set(U) 
 
def main(): 
    print (Union([])) 
    print (Union([[1,2]])) 
    print (Union([[1,2], [1,2,3],[2,3]])) 
    print (Inter([[1,2]])) 
    print (Inter ([[1,2], [1,2,3],[2,3]])) 
 
if __name__ == '__main__': 
    main()
