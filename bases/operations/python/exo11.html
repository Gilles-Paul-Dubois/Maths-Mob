E="eabxyz"
R1=["eabxyz","abezxy","beayzx","xyzeba","yzxbae","zxyaeb"]
R2=["eabxyz","abezxy","beayzx","xyzbea","yzxeab","zxyabe"]
R3=["eabxyz","abezxy","beayzx","xyzeab","yzxbea","zxyabe"]
Map1,Map2,Map3={},{},{}
for i in range(0,6):
    for j in range(0,6):
        Map1[(E[i],E[j])]=(R1[i]) [j]
for i in range(0,6):
    for j in range(0,6):
        Map2[(E[i],E[j])]=(R2[i]) [j]
for i in range(0,6):
    for j in range(0,6):
        Map3[(E[i],E[j])]=(R3[i]) [j]

def fonc1(x, y):
    return Map1.get((x, y))

def fonc2(x, y):
    return Map2.get((x, y))

def fonc3(x, y):
    return Map3.get((x, y))

def associative(f):
    for x in E:
        for y in E:
            for z in E:
                if f(f(x, y), z) != f(x, f(y, z)):
                    print(x, y, z)
                    return False
    return True

def commutative(f):
    for x in E:
        for y in E:
            if f(x, y) != f(y, x):
                print(x, y)
                return False
    return True


def main():
    print(associative(fonc1))
    print(associative(fonc2))
    print(associative(fonc3))
    print(commutative(fonc3))

if __name__ == '__main__':
    main()
