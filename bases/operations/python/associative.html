# version python 3.8
# tester si une loi est associative
# checking associativity
import random

Table = []

# génère une loi totalement aléatoire sur un ensemble à n éléments
# generates a totally random operation on a set with n elements
def genere(n :int)->list:
    return [random.randint(0, n - 1) for i in range(0, n) for j in range(0, n)]

# génère une loi associative
# generates an associative operation
def genere2(n :int)->list:
    return [(i + j) % n for i in range(0, n) for j in range(0, n)]

# calcule le composé de i et j selon la table générée
# computes image of pair(i,j) according generated operation
def compose(i :int, j :int, n :int)->int:
    return Table[i * n + j]

# teste si la loi est associative
def test_associative(n :int)->bool:
    return all(
        [compose(i, compose(j, k, n), n) == compose(compose(i, j, n), k, n) for i in range(0, n) for j in range(0, n)
         for k in range(0, n)])

# fonction principale - main function
def main():
    global Table
    Table = genere(9)
    print(test_associative(9))  # échec presque sûr
    Table = genere2(9)
    print(test_associative(9))  # succès

if __name__ == '__main__':
    main()
