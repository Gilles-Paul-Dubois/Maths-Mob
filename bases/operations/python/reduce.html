# Somme et produit de familles finies de nombres 
# Sum and product of finite families of numbers 
# présentées comme listes 
# given as lists
# au moyen de 'reduce' 
# by mean of 'reduce' 
 
from functools import reduce 
 
def Somme(L): 
    if len(L)==0: 
        return 0 
    if len(L)==1: 
        return L[0] 
    return reduce (lambda x,y:x+y,L) 
 
def Produit(L): 
    if len(L)==0: 
        return 1 
    if len(L)==1: 
        return L[0] 
    return reduce (lambda x,y:x*y,L) 
 
def main(): 
    print (Somme([])) 
    print (Somme([1])) 
    print (Somme(range(1,11))) #10*11/2 
    print (Produit([])) 
    print (Produit([0])) 
    print (Produit(range(1,11))) # factorielle 10 
 
if __name__ == '__main__': 
    main()
