# version python 3.8
# générer une loi aléatoire sur l'ensemble {0,1,2, ...,n-1}
# generate a random operation on set {0,....n-1}
import random

Table = []  #global variable used by some functions

# génère une loi totalement aléatoire sur un ensemble à n éléments
# generates a totally random operation on a set with n elements
def genere(n :int)->list:
    return [random.randint(0, n - 1) for i in range(0, n) for j in range(0, n)]


# calcule le composé de i et j selon la table générée
# computes the image of pair (i,j) (result)
def compose(i :int, j :int, n :int)->int:
    return Table[i * n + j]

# affiche la table de la loi générée - displays table of generated operation
# alignement vertical correct seulement si n<=10 - vertical align OK only if n<=10
# sinon il faut formater - in other case you must reformat
def Affiche_Table(n :int)->None:
    global Table
    for i in range(n - 1, -1, -1):
        print(i, end="|")
        for j in range(0, n):
            print(compose(i, j, n), end=" ")
        print("")
    print('-' * (2 * n + 3))
    print(' ' * 2, end="")
    for i in range(0, n):
        print(i, end=" ")

def main(): #fonction principale - main function
    global Table
    Table = genere(9)
    Affiche_Table(9)

if __name__ == '__main__':
    main()
