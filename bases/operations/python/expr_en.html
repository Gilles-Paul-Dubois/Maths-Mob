# version python 3.4 
 
# Parsing program 
# for a numerical expression with all parentheses 
# with the four operators +,-,*,/ 
# The function parses a string and returns a tree under the form of a list 
# Letter W is reserved and should not be used as a variable 
 
# Custom exception which will be raised in case of error 
class ExpException(Exception): 
    def __init__(self, message): 
        self.mess=message 
 
 
# numerical constants 
def ConstantP(X): 
    try: 
        x=float(X) 
    except: 
        return False 
    return True 
 
#numerical variables 
def VariableP(X): 
    return len(X)==1 and 'A'<=X <='Z' 
 
 
#atoms 
def AtomP(X): 
    return ConstantP(X) or VariableP(X) 
 
# replace X by E in F 
def subst(E,X,F): 
    if type(F)!=list: 
        if F==X: 
            return E 
        else: 
            return F 
    else: 
        return [subst(E,X,Y) for Y in F]#recursive call 
 
#checks if X is an expression 
# recursive function 
def Exp (X): 
    if AtomP(X): # atoms are expressions 
        return X 
    else: 
        if X[0]=='(': #case of parentheses 
            n=1 
            for i in range(1,len(X)): 
                if X[i]=='(': 
                    n=n+1 #increases level of opening parentheses 
                if X[i]==')': 
                    n=n-1 #decrease the number of closing parenthses 
                if n==0: # we have found the closing par. of the opening first 
                    return subst(Exp(X[1:i]),'W',Exp("W"+X[i+1:])) #special role of w 
            raise ExpException ("Expression mal formée") 
        else: 
            # looking for an opposite(sign -) 
            if X[0]=='-': 
                if Exp(X[1:]): 
                    return ['OPP',Exp(X[1:])] 
                else: 
                    raise ExpException ("Expression mal formée") 
            # searching for an addition addition (sign +) 
            for i in range(0,len(X)): 
                if X[i]=='+': 
                    if Exp(X[0:i]) and Exp(X[i+1:]): 
                        return ['ADD',Exp(X[0:i]),Exp(X[i+1:])] 
                    else: 
                        raise ExpException ("Expression mal formée") 
            # searching for a product (sign *) 
            for i in range(0,len(X)): 
                if X[i]=='*': 
                    if Exp(X[0:i]) and Exp(X[i+1:]): 
                        return ['MUL',Exp(X[0:i]),Exp(X[i+1:])] 
                    else: 
                        raise ExpException ("Expression mal formée") 
            # searching for a quotient (sign /) 
            for i in range(0,len(X)): 
                if X[i]=='/': 
                    if Exp(X[0:i]) and Exp(X[i+1:]): 
                        return ['DIV',Exp(X[0:i]),Exp(X[i+1:])] 
                    else: 
                        raise ExpException ("Expression mal formée") 
            # searching for a difference (sign -) 
            for i in range(1,len(X)): 
                if X[i]=='-': 
                    if Exp(X[0:i]) and Exp(X[i+1:]): 
                        return ['SUB',Exp(X[0:i]),Exp(X[i+1:])] 
                    else: 
                        raise ExpException ("Expression mal formée") 
            raise ExpException("Expression mal formée") 
def main(): 
    X="-((Y-T)*(3/Z))" 
    try: # attempt to evaluate 
        print (Exp(X)) 
    except ExpException as E: # en cas d'échec 
        print (E.mess) 
if __name__ == '__main__': 
     main()
