# generate a random operation on the set {0,1,2, ...,n-1} 
import random 
 
Table=[] 
 
#create a totally random operation on a set with n elements 
def genere(n): 
    return [random.randint(0,n-1) for i in range(0,n) for j in range(0,n)] 
 
# computes the composite of  i and j according the generated table 
def compose (i,j,n): 
    return Table[i*n+j] 
 
#prints the table 
# vertical alignment correct only if n<=10 
# otherwise formatting is needed 
def Affiche_Table(n): 
    global Table 
    for i in range (n-1,-1,-1): 
        print (i ,end="|") 
        for j in range(0,n): 
            print (compose(i,j,n),end=" ") 
        print("") 
    print ('-'*(2*n+3)) 
    print (' '*2,end="") 
    for i in range(0,n): 
        print (i,end=" ") 
 
def main(): 
    global Table 
    Table=genere(9) 
    Affiche_Table(9) 
 
if __name__ == '__main__': 
    main()
