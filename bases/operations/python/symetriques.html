# tester deux éléments symétriques
# test two inverse elements
import random

Table = []

# génère une loi avec élément neutre 0
# generates an operation with neutral 0
def genere2(n :int)->list:
    return [(i + j) % n for i in range(0, n) for j in range(0, n)]

# calcule le composé de i et j selon la table générée
# computes the image of the pair (i,j) according generated table
def compose(i :int, j :int, n :int)->int:
    return Table[i * n + j]

# teste si iet j sont symétriques (avec 0 neutre)
# checks if i and j are mutula inverses (with neutral 0)
def teste_symetriques(i, j, n):
    return compose(i, j, n) == 0

def main(): # fonction principale - main function
    global Table
    Table = genere2(9)
    print(teste_symetriques(4, 5, 9))  # succès

if __name__ == '__main__':
    main()
