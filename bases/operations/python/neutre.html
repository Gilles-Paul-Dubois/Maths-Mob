# version python 3.8
# tester si un élément est neutre
# checking for neutral element
import random

Table = []

# génère une loi totalement aléatoire sur un ensemble à n éléments
# generates a totally random operation on a set with n elements
def genere(n :int)->list:
    return [random.randint(0, n - 1) for i in range(0, n) for j in range(0, n)]

# génère une loi avec 0 pour élément neutre
# generates an operation with 0 as a neutral element
def genere2(n :int)->list:
    return [(i + j) % n for i in range(0, n) for j in range(0, n)]

# calcule le composé de i et j selon la table générée
# computes the image of pair (i,j) according generated table
def compose(i :int, j :int, n :int)->int:
    return Table[i * n + j]

# teste si i est neutre
# checks if i is neutral
def test_neutre(i, n):
    for j in range(0, n):
        if compose(i, j, n) != j:
            return False
        if compose(j, i, n) != j:
            return False
    return True

def main(): #fonction principale - main function
    global Table
    Table = genere(9)
    print(test_neutre(0, 9))  # almost certain failure - échec presque sûr.
    Table = genere2(9)
    print(test_neutre(0, 9))  # succès - success


if __name__ == '__main__':
    main()
