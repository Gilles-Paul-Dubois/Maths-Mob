#checks if two elements are inverses from each other 
import random 
 
Table=[] 
 
#generates an operation with neutral 0 
def genere2(n): 
    return [(i+j)%n for i in range(0,n) for j in range(0,n)] 
 
# computes the image of ordered pair (i,j) according table 
def compose (i,j,n): 
    return Table[i*n+j] 
 
#checks if i and j are mutual inverses (with neutral 0) 
def teste_symetriques(i,j,n): 
    return compose(i,j,n)==0 
 
# main function 
def main(): 
    global Table 
    Table=genere2(9) 
    print (teste_symetriques(4,5,9)) # success 
 
if __name__ == '__main__': 
    main()
