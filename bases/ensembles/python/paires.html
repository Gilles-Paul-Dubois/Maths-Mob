import itertools  # pour combinations

# un objet frozenset est comme un objet set mais immuable
# a frozenset is imumtable
# les frozensets sont introduits ici parce que en Python les ensembles ne peuvent être éléments d'autres ensembles que s'ils sont immuables
# We use frozensets here because Python sets can contain sets as elements only if they are immutable
# itertools fournit une fonctions 'combinations' qui génère toutes les combinaisons d'un ensemble à k éléments
# the itertools modu provides a 'combinations' function which generates all combinations with k elements of of set
# notre fonction combinaisons ne fait que constituer la liste engendrée.
# Our own 'combinaisons' juste create the list of these combinations
def combinaisons(E :set, k :int) ->list:
    return list(itertools.combinations(E, k))

# la fonction suivante calcule l'ensemble des parties de E ayant exactement k éléments
#the following function lists all pairs of a given set
def paires(E :set) ->set:
    return {frozenset(x) for x in combinaisons(E, 2)}

def main():
    S = {1, 2, 3, 4}
    print(paires(S))  # parties de S à deux éléments

main()
