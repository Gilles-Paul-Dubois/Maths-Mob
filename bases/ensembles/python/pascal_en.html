#creating Pascal's triangle with a generator 
def Pascal(): 
    n,C=0,[1] 
    yield n,C # return for the first call 
    while True: # later calls 
        n+=1 
        D=C[:] 
        D.insert(0,1) 
        for i in range(1,len(C)): 
            D[i]=C[i-1]+C[i] # Pascal's triangle induction formula 
        C=D 
        yield n,C 
 
# coefficients Cp,n 
def Cpn(p,n): 
    TP=Pascal() 
    for i in range(0,n+1): 
        L=next(TP)[1] 
    return L[p] 
 
def main(): 
    TP=Pascal() 
    #See the beginning of Pascal's triangle 
    for i in range(0,11): 
        print (next(TP)) 
    # Compute one coefficient 
    x=Cpn(5,10) 
    print (x) 
 
if __name__ == '__main__': 
    main()
