# tester la vacuité avec Python - test emptyness with Python
# Pour de nombreux conteneurs (listes, ensembles, dictionnaires, chaînes, n-uples, etc...
# for a lot of containers (lists, sets, dictionnaries, strings,tuples, etc)
# il peut être important de faire un test pour savoir si l'objet est vide
# it can be important to make test to know if the object is empty or not
# En Python cela se fait au moyen d'un simple test if(), comme suit
# With Python a simple if conditional suffices

def is_empty(any_structure :any) -> bool:
    """decide si un objet est vide - cehcks if a container is empty whatever the type"""
    if any_structure:
        print('le conteneur est non vide.')
        return False
    else:
        print('Le conteneur est vide.')
        return True

def main():
    is_empty([])  # liste vide - empty list
    is_empty(set())  # ensemble vide - empty set
    is_empty({})  # dictionnaire sans entrée - empty dictionary
    is_empty("")  # chaîne vide -empty string
    is_empty(())  # n-uple vide - empty tuple
    is_empty([1])  # liste non  vide - not empty list
    is_empty(set("a"))  # ensemble non vide - not empty set
    is_empty({'a': 1})  # dictionnaire avec une entrée - dictionary  with one key
    is_empty("ab")  # chaîne non vide - non empty string
    is_empty((1, 2))  # n-uple non vide - non empty tuple

if __name__ == '__main__':
    main()
