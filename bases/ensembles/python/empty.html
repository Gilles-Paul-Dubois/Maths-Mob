# version python 3.4 
 
# tester la vacuité avec Python 
# Pour de nombreux conteneurs (listes, ensembles, dictionnaires, chaînes, n-uples, etc... 
# il peut être important de faire un test pour savoir si l'objet est vide 
# En Python cela se fait au moyen d'un simple test if(), comme suit 
 
def is_empty(any_structure): 
    if any_structure: 
        print('le conteneur est non vide.') 
        return False 
    else: 
        print('Le conteneur est vide.') 
        return True 
 
def main(): 
   is_empty([]) #liste vide 
   is_empty(set()) #ensemble vide 
   is_empty({}) #dictionnaire sans entrée 
   is_empty("") #chaîne vide 
   is_empty(()) #n-uple vide 
   is_empty([1]) #liste non  vide 
   is_empty(set("a")) #ensemble non vide 
   is_empty({'a':1}) #dictionnaire avec une entrée 
   is_empty("ab") #chaîne non vide 
   is_empty((1,2)) #n-uple non vide 
 
 
if __name__ == '__main__': 
    main()
