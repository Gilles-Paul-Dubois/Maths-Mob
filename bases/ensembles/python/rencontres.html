import copy

def pprint(L :list)->None:  # seulement pour un bel affichage - just for a pretty print
    for i in range(0, len(L)):
        print(L[i])


def supprime(L :list, i :int, j :int)->list:  # enlève les éléments d'indice i et j de la liste L - removes elements with indexes i and j from the list
    R = []
    for k in range(0, len(L)):
        if ((k != i) and (k != j)):
            R.append(L[k])
    return R


def Rencontres(L :list)->list:  # renvoie la liste des possibilités de rencontres - returns the list of possibilities of encounters
    S = []
    if len(L) == 2:
        return [[L]]  # terminaison de la récursivité deux joueurs= une seule rencontre- end of recursivity two players only one possible match
    else:
        for i in range(1, len(L)):
            R = Rencontres(supprime(L, 0, i))  # appel récursif - recursive call
            for j in range(0, len(R)):
                T = R[j]
                T.append([L[0], L[i]])
                S.append(T)
    return S


# fonction principale - main function
def main():
    pprint(Rencontres([1, 2, 3, 4, 5, 6, 7, 8]))


if __name__ == '__main__':
    main()
