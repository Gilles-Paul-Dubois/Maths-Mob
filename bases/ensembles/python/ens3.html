# sous-ensembles et sur-ensembles - subsets and supersets
# vider un ensemble - empty a set
# faire une copie d'un ensemble - make a copy of a set
# Tester l'égalité de deux ensembles - test equality of two sets

# ensemble de nombres - sets of numbers
V = set([])
E = set([1, 2, 3, 4])
F = set([2, 3])
G = set([1, 2, 3, 4, 5, 5, 7])

# fonction de test - test function
def test1():
    print(V <= E)  # Vrai True 
    print(F <= E)  # Vrai True 
    print(E <= E)  # Vrai  True
    print(E < E)  # Faux (inclusion stricte)  False
    print(E <= G)  # Oui Yes
    print(G <= E)  # Non  No
    print(G)  # Affiche G comme défini au dessus  displays G as dfined above
    G.clear()  # vide G to empty G
    print(G)  # pour voir just to see

def test2(): #other test
    G = E.copy()  # G reçoit maintenant une copie de E - G is now a copy of E
    print(G)  # pour voir to see
    print(E == G)  # teste l'égalité (True) to test equality
    print(E == F)  # Non (False) No !

# fonction principale - main function
def main():
    test1()
    test2()

if __name__ == '__main__':
    main()
