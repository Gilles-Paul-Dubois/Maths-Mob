# faire une copie d'un ensemble 
# Tester l'égalité de deux ensembles 
 
 
 
# ensemble de nombres 
V=set([]) 
E=set([1,2,3,4]) 
F=set([2,3]) 
G=set([1,2,3,4,5,5,7]) 
 
# fonction d'impression personnalisée d'un ensemble 
# proche de l'écriture mathématique 
def Imprime (S): 
    L=list(S) #on repasse en mode liste pour indexation 
    n=len(L) #cardinal de l'ensemble 
    if n==0: #cas du vide 
        print("{ }") 
    else: 
        print ('{', end='') 
        for i in range(0, len(L) - 1): 
            print(L[i], end=',') # tous les éléments sauf le dernier suivis d'une virgule 
        print(L[n - 1], end='}\n') # accolade fermante puis saut de ligne après le dern 
 
# fonction de test 
def test1(): 
    print (V<=E) # Vrai True 
    print (F<=E)  # Vrai True 
    print (E<=E) # Vrai 
    print (E<E) # Faux (inclusion stricte) 
    print (E<=G) # Oui 
    print (G<=E) # Non 
    Imprime(G) # Affiche G comme défini au dessus 
    G.clear()  # vide G 
    Imprime(G) # pour voir 
 
def test2(): 
    G=E.copy() # G reçoit maintenant une copie de E 
    Imprime(G) # pour voir 
    print (E==G) # teste l'égalité (True) 
    print (E==F) # Non (False) 
 
#programme principal 
def main(): 
    test1() 
    test2() 
 
if __name__ == '__main__': 
    main()
