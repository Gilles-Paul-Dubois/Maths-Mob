# calcule l'ensemble des parties de E si E est donné sous forme de liste sans répétition
#computes the set of all parts of E if E is given as a list without repetition
def PartsOf(E :list) ->list:
    P = [[]]
    while E:
        a = E[0]
        E = E[1:]
        Q = [x + [a] for x in P] + P
        P = Q
    return P

# parties à p éléments seulement  - parts with p elements only
def PartspOf(E :list, p :int)->list:
    return [X for X in PartsOf(E) if len(X) == p]

#fonction principale - main function
def main():
    print(PartspOf([1, 2, 3, 4], 2))


if __name__ == '__main__':
    main()
