# Calcule les nombres de Bell 
#B(n)= Nombre de partitions d'un ensemble à n éléments 
 
from functools import reduce 
 
def Bell(n): 
    def add(x,y): 
        return x+y 
    return reduce(add,[Stirling2(p,n) for p in range(1,n+1)],0) 
 
# calcule les Sp,n à partir des Sp,n-1 
def NextStirling(L): 
    Next=[] 
    Next.append(1) 
    for i in range(1,len(L)): 
        Next.append(L[i-1]+(i+1)*L[i]) 
    Next.append(1) 
    return Next 
 
#donne le coefficient de Stirling de seconde espèce 
def Stirling2(p,n): 
    L=[1] 
    for i in range (0,n-1): 
        L=NextStirling(L) 
    return L[p-1]
