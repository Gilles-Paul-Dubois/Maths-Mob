# building sets

# set of numbers 
E=set([1,2,3,4]) 
 
# custom printing of a set 
# close to maths habits 
def Imprime (S): 
    L=list(S) #shift to list mode for indexing 
    n=len(L) #cardinal number of the set 
    if n==0: #special case of empty set 
        print("{ }") 
    else: 
        print ('{', end='') 
        for i in range(0, len(L) - 1): 
            print(L[i], end=',') # all elements followed by comma, except last one 
        print(L[n - 1], end='}\n') # closing brace then line jump after last element 
 
# ftest function 
def test(): 
    print (len(E)) # cardinality of the set, that is  4 
    print (5 in E) # should print 'False' 
    E.add(5) # add element 5 
    print (len(E)) # cardinality of the set or 5 
    print (5 in E) # should print 'True' 
    E.discard(1) # remove the first element 
    Imprime(E) # just to check

    test()
