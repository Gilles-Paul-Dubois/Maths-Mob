# building of sets 
# vshowing sets  
 
V = set([])  # empty set 
C1 = set(['a', 'b', 'c'])  #set of chars 
C2 = set('abc')  #other possibility giving the same result 
N1 = set([0, 1, 2, 3, 4, 5])  # set of numbers 
N2 = set(range(6))  # other possibility with the same result 
M = set([1, 2, 'a', 'B'])  # mixed set 
 
# custom printing of a set  
# close to mathematical habits  
def Imprime(S): 
    L = list(S)  #shift to list mode for indexing 
    n = len(L)  #cardinal number of the set 
    if n == 0:  #case of empty set 
        print("{ }") 
    else: 
        print('{', end='') 
        for i in range(0, len(L) - 1): 
            print(L[i], end=',')  # all elemnts followed by comma, except last one 
        print(L[n - 1], end='}\n')  # closing brace then line jump after last 
 
# standard printing  
def test1(): 
    print(V) 
    print(C1) 
    print(C2) 
    print(N1) 
    print(N2) 
    print(M) 
 
 
#custom printing  
def test2(): 
    Imprime(V) 
    Imprime(C1) 
    Imprime(C2) 
    Imprime(N1) 
    Imprime(N2) 
    Imprime(M) 
 
 
#main function  
def main(): 
    print('standard printing') 
    test1() 
    print('custom printing') 
    test2() 
 
 
if __name__ == '__main__': 
    main()
