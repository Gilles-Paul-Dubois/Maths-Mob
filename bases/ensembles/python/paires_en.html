# lists pairs of a set 
 
 
def Paires(E): 
    L=[set([x,y]) for x in E for y in E if y !=x] 
    R=[] 
    for z in L: #elimination of doubles 
        if z not in R: 
            R.append(z) 
    return R 
 
def main(): 
    E=set([1,2,3,4]) 
    print (Paires(E)) 
 
if __name__ == '__main__': 
    main()
