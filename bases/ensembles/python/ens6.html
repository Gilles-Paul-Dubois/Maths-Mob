# mapping et filtrage - mapping and filtering
# on peut ainsi définir un ensemble 'en compréhension'- this is one way to define a set 'in comprehension'
# ensemble des carrés des nombres impairs entre 0 et 20
# set of squares of odd numbers

# fonction principale - main function
def main():
    # les 20 premiers entiers - the first 20 naturals
    L = range(0, 20)
    print(L)
    # les nombres impairs parmi ceux ci - among them the odd ones (filtering)
    I = [i for i in L if i % 2 == 1]
    EI = set(I)  # transforme liste en ensemble - builds set from list
    print(EI)
    # leurs carres - their squares
    C = map(lambda x: x * x, I) # apply x->x*x to all elements (mapping)
    EC = set(C)  # transforme liste en set - transforms list into set
    print(EC)


if __name__ == '__main__':
    main()
