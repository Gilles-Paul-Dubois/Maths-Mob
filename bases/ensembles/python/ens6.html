# mapping et filtrage 
# visualisation 
# on peut ainsi définir un ensemble 'en compréhension' 
# ensemble des carrés des nombres impairs entre 0 et 20 
 
def Imprime (S): 
    L=list(S) #on repasse en mode liste pour indexation 
    n=len(L) #cardinal de l'ensemble 
    if n==0: #cas du vide 
        print("{ }") 
    else: 
        print ('{', end='') 
        for i in range(0, len(L) - 1): 
            print(L[i], end=',') # tous les éléments sauf le dernier suivis d'une virgule 
        print(L[n - 1], end='}\n') # accolade fermante puis saut de ligne après le dernier 
 
#programme principal 
def main(): 
    # les 20 premiers entiers 
    L=range(0,20) 
    Imprime(L) 
    #les nombres impairs parmi ceux ci 
    I=[i for i in L if i%2==1] 
    EI=set(I) #transforme liste en ensemble 
    Imprime(EI) 
    # leurs carres 
    C=map(lambda x: x*x, I) 
    EC=set(C) #transforme liste en Set 
    Imprime(EC) 
 
 
if __name__ == '__main__': 
    main()
