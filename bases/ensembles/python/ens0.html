# fabrication d'ensembles 
# visualisation  
 
V = set([])  # ensemble vide 
C1 = set(['a', 'b', 'c'])  #ensemble de caractères 
C2 = set('abc')  #autre possibilité donnant le même résultat 
N1 = set([0, 1, 2, 3, 4, 5])  # ensemble de nombres 
N2 = set(range(6))  # autre possibilité donnant le même résultat 
M = set([1, 2, 'a', 'B'])  # ensemble mixte 
 
# fonction d'impression personnalisée d'un ensemble  
# proche de l'écriture mathématique  
def Imprime(S): 
    L = list(S)  #on repasse en mode liste pour indexation 
    n = len(L)  #cardinal de l'ensemble 
    if n == 0:  #cas du vide 
        print("{ }") 
    else: 
        print('{', end='') 
        for i in range(0, len(L) - 1): 
            print(L[i], end=',')  # tous les éléments sauf le dernier suivis d'une virgule 
        print(L[n - 1], end='}\n')  # accolade fermante puis saut de ligne après le dernier 
 
# impression standard  
def test1(): 
    print(V) 
    print(C1) 
    print(C2) 
    print(N1) 
    print(N2) 
    print(M) 
 
 
#impression personnalisée  
def test2(): 
    Imprime(V) 
    Imprime(C1) 
    Imprime(C2) 
    Imprime(N1) 
    Imprime(N2) 
    Imprime(M) 
 
 
#programme principal  
def main(): 
    print('impression standard') 
    test1() 
    print('impression personnalisee') 
    test2() 
 
 
if __name__ == '__main__': 
    main()
