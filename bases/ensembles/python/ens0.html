# -*- coding: utf_8 -*-
# python  version 3.8
# fabrication d'ensembles - building sets
# visualisation - showing them

V = set()  # ensemble vide - empty set
C1 = {'a','b','c'}  # ensemble de caractères - set of characters
C2 = set('abc')  # autre possibilité donnant le même résultat - other possibility from string
N1 = set([0, 1, 2, 3, 4, 5])  # ensemble de nombres - set of numbers
N2 = set(range(6))  # autre possibilité donnant le même résultat- other possibility giving the same result
M = {1, 2, 'a', 'B'}  # ensemble mixte - mixed set

# fonction principale - main function
#affichage des ensembles - displaying sets
def main():
    print(V)
    print(C1)
    print(C2)
    print(N1)
    print(N2)
    print(M)

if __name__ == '__main__':
    main()
