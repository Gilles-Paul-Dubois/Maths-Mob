# the generator 
def PartiesDe(E): 
    P=[[]] 
    yield P # return [[]] for the first call 
    while E: # later calls 
        P=[x+[a] for x in P for a in E if all(a>y for y in x) ] # works only if there is an order on E 
        yield P 
    return 
 
def main(): 
    E=[1,2,3] 
    P=PartiesDe(E) # nothing is computed yet ! 
    for i in range(0,4): #displays parts with 0,1,2,3 elements line by line 
        print (next(P)) # it's a true iteration 
 
if __name__ == '__main__': 
    main()
