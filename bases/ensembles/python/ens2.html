# ensemble de nombres 
E=set([1,2,3,4]) 
 
# fonction d'impression personnalisée d'un ensemble 
# proche de l'écriture mathématique 
def Imprime (S): 
    L=list(S) #on repasse en mode liste pour indexation 
    n=len(L) #cardinal de l'ensemble 
    if n==0: #cas du vide 
        print("{ }") 
    else: 
        print ('{', end='') 
        for i in range(0, len(L) - 1): 
            print(L[i], end=',') # tous les éléments sauf le dernier suivis d'une virgule 
        print(L[n - 1], end='}\n') # accolade fermante puis saut de ligne après le dernier 
 
# fonction de test 
def test(): 
    print (len(E)) # cardinal de l'ensemble soit 4 
    print (5 in E) # doit afficher 'False' 
    E.add(5) # ajoute l'élément 5 
    print (len(E)) # cardinal de l'ensemble soit 5 
    print (5 in E) # doit afficher 'True' 
    E.discard(1) # enlève le premier élément 
    Imprime(E) # pour vérifier 
 
#programme principal 
def main(): 
    test() 
 
if __name__ == '__main__': 
    main()
