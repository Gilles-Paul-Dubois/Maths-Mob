# fabrication d'ensembles - adding and removing elements
# visualisation - with display

# ensemble de nombres - set of numbers
E = set([1,2,3,4])

# fonction principale - main function
def main():
    global E
    print(len(E))  # cardinal de l'ensemble soit 4 - cardinality of E means 4
    print(5 in E)  # doit afficher 'False' - should print 'False'
    E.add(5)  # ajoute l'élément 5 - add an element
    print(len(E))  # cardinal de l'ensemble soit 5 - cardinality increases
    print(5 in E)  # doit afficher 'True' - Should print 'True' now
    E.discard(1)  # enlève le premier élément - remove element 1
    print(E)  # pour vérifier - just to check

if __name__ == '__main__':
    main()
