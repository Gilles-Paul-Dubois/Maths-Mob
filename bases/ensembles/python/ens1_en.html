E=set([]) # empty set 
 
# custom printing for a set 
# close to mathematical habits 
def Imprime (S): 
    L=list(S) #list mode again for indexing 
    n=len(L) #cardinal number 
    if n==0: #case of empty set 
        print("{ }") 
    else: 
        print ('{', end='') 
        for i in range(0, len(L) - 1): 
            print(L[i], end=',') # all elements followed by comma except last one 
        print(L[n - 1], end='}\n') # closing brace then line jump after last one  
 
# keyboard input 
def lire_ensemble(): 
    while 1: #infinite loop 
        S=input() 
        if(S=="-1"): #exit from loop by entering value -1 
            break 
        else: 
            E.add(S) #add inputi S to set E 
 
#main function 
#imput of chars between single quotes ex: 'a' 
#for numbers without anything 
def main(): 
    print ('Enter elements, code -1 to end process') 
    lire_ensemble() 
    Imprime(E) 
 
if __name__ == '__main__': 
    main()
