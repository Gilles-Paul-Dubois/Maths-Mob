# operations with sets 
# union and intyersection 
# difference and symmetric difference 
 
 
#sets of numbers 
V=set([]) 
E=set([1,2,3,4]) 
F=set([0,2,3]) 
 
# custom display for a set 
# close to usual habits 
def Imprime (S): 
    L=list(S) #back to list mode for indexing 
    n=len(L) #cardinal number of the set 
    if n==0: #case of empty set 
        print("{ }") 
    else: 
        print ('{', end='') 
        for i in range(0, len(L) - 1): 
            print(L[i], end=',') # all elements followed by coma, except last one 
        print(L[n - 1], end='}\n') # closing brace then linefeed after last one 
 
 
# intersection 
def test1 (): 
    Imprime (V&E) # should be empty 
    Imprime(E&E) # should display E E 
    Imprime(E&F) # or {2 3} 
 
# la réunion 
def test2(): 
    Imprime(V|E) # should show E 
    Imprime(E|E) # same thing 
    Imprime(E|F) # 5 elements 
 
# difference 
def test3(): 
    Imprime (E-V) # should show E E 
    Imprime(V-E) # should be empty 
    Imprime (E-F) # two elements left 
 
# symmetric diference 
def test4(): 
    Imprime(E^F) 
    print (E^F == ((E|F) - (E&F))) # should be true 
 
 
def main(): 
    print ('Test intersection') 
    test1() 
    print ('Test union') 
    test2() 
    print ('test difference') 
    test3() 
    print ('test symmetric difference') 
    test4() 
 
if __name__ == '__main__': 
    main()
