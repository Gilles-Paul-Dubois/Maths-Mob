# fabrication d'ensembles
# visualisation 
 
E=set([]) # ensemble vide 
 
# fonction d'impression personnalisée d'un ensemble 
# proche de l'écriture mathématique 
def Imprime (S): 
    L=list(S) #on repasse en mode liste pour indexation 
    n=len(L) #cardinal de l'ensemble 
    if n==0: #cas du vide 
        print("{ }") 
    else: 
        print ('{', end='') 
        for i in range(0, len(L) - 1): 
            print(L[i], end=',') # tous les éléments sauf le dernier suivis d'une virgule 
        print(L[n - 1], end='}\n') # accolade fermante puis saut de ligne après le dernier 
 
# lecture au clavier 
def lire_ensemble(): 
    while 1: #boucle infinie 
        S=input() 
        if(S=="-1"): #sortie de boucle si valeur entrée est -1 
            break 
        else: 
            E.add(S) #ajout de l'élément saisi S à l'ensemble E 
 
#programme principal 
#saisir les caractères entre simples cotes ex: 'a' 
#pour les nombres sans apostrophes 
def main(): 
    print ('Saisir des elements - taper -1 pour finir') 
    lire_ensemble() 
    Imprime(E) 
 
if __name__ == '__main__': 
    main()
