# opérations avec les ensembles - operations with sets
# réunion et intersection - union and intersection
# complémentaire et différence symétrique - difference and symmetric difference

# ensembles de nombres - sets of numbers
V = set([])
E = set([1, 2, 3, 4])
F = set([0, 2, 3])

# l'intersection - intersection
def test1():
    print(V & E)  # doit être vide - should be empty
    print(E & E)  # doit donner E - should retrun E
    print(E & F)  # soit {2 3}

# la réunion - union
def test2():
    print(V | E)  # doit donner E  - should give E
    print(E | E)  # même chose - same thing
    print(E | F)  # 5 éléments - 5 elements

# complémentaire
def test3():
    print(E - V)  # doit donner E - should give E
    print(V - E)  # doit être vide should be empty
    print(E - F)  # reste deux éléments - two elements left

# différence symétrique - symmetric difference
def test4():
    print(E ^ F)
    print(E ^ F == ((E | F) - (E & F)))  # doit être vrai  should be true

def main():
    print('Test intersection')
    test1()
    print('Test reunion')
    test2()
    print('test complementaire')
    test3()
    print('test difference symetrique')
    test4()

if __name__ == '__main__':
    main()
