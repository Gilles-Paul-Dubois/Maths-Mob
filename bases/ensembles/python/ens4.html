# opérations avec les ensembles 
# réunion et intersection 
# complémentaire et différence symétrique 
 
 
# ensembles de nombres 
V=set([]) 
E=set([1,2,3,4]) 
F=set([0,2,3]) 
 
# fonction d'impression personnalisée d'un ensemble 
# proche de l'écriture mathématique 
def Imprime (S): 
    L=list(S) #on repasse en mode liste pour indexation 
    n=len(L) #cardinal de l'ensemble 
    if n==0: #cas du vide 
        print("{ }") 
    else: 
        print ('{', end='') 
        for i in range(0, len(L) - 1): 
            print(L[i], end=',') # tous les éléments sauf le dernier suivis d'une virgule 
        print(L[n - 1], end='}\n') # accolade fermante puis saut de ligne après le dernier 
 
 
# l'intersection 
def test1 (): 
    Imprime (V&E) # doit être vide 
    Imprime(E&E) # doit donner E 
    Imprime(E&F) # soit {2 3} 
 
# la réunion 
def test2(): 
    Imprime(V|E) # doit donner E 
    Imprime(E|E) # même chose 
    Imprime(E|F) # 5 éléments 
 
# complémentaire 
def test3(): 
    Imprime (E-V) # doit donner E 
    Imprime(V-E) # doit être vide 
    Imprime (E-F) # reste deux éléments 
 
# différence symétrique 
def test4(): 
    Imprime(E^F) 
    print (E^F == ((E|F) - (E&F))) # doit être vrai 
 
 
def main(): 
    print ('Test intersection') 
    test1() 
    print ('Test reunion') 
    test2() 
    print ('test complementaire') 
    test3() 
    print ('test difference symetrique') 
    test4() 
 
if __name__ == '__main__': 
    main()
