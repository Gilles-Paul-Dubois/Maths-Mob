# version python 3.4 
 
# test emtyness with python 
# for a lot of containers (lists, sets, dictionnaries, strings,tuples, etc... 
# It can be useful to make a test to check if the object is empty or not 
# In python you use only the conditionnal 'if' as follows 
 
def is_empty(any_structure): 
    if any_structure: 
        print('Container is not empty.') 
        return False 
    else: 
        print('Contaner is empty.') 
        return True 
 
def main(): 
   is_empty([]) #empty list 
   is_empty(set()) #empty set 
   is_empty({}) #dictionnary without any key 
   is_empty("") #empty string 
   is_empty(()) #empty tuple 
   is_empty([1]) #non empty list 
   is_empty(set("a")) #non empty set 
   is_empty({'a':1}) #dictionnary with one key 
   is_empty("ab") #not empty string 
   is_empty((1,2)) #not empty tuple 
 
 
if __name__ == '__main__': 
    main()
