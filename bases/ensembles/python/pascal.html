# génération du triangle de Pascal au moyen d'un générateur
# generates Pascal triangle
def Pascal():
    n, C = 0, [1]
    yield n, C  # retour du premier appel - return value for first call
    while True:  # exécution des appels suivants - next calls
        n += 1
        D = C[:]
        D.insert(0, 1)
        for i in range(1, len(C)):
            D[i] = C[i - 1] + C[i]  # relation de récurrence du Triangle de Pascal- recurrence relation of the triangle
        C = D
        yield n, C

# coefficients Cp,n
def Cpn(p :int, n :int) ->int:
    TP = Pascal()
    for i in range(0, n + 1):
        L = next(TP)[1]
    return L[p]

def main():
    TP = Pascal()
    # voir le début du triangle de Pascal - display beginning of the traingle
    for i in range(0, 11):
        print(next(TP))
        # calculer un coefficient - compute one coefficient
    x = Cpn(5, 10)
    print(x)


if __name__ == '__main__':
    main()
