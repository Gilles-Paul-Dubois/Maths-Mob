# le générateur - the generator
def PartiesDe(E :list)->list:
    P = [[]]
    yield P  # retourner [[]] au premier appel - return [[]] for first call
    while E:  # appels suivants - next calls
        P = [x + [a] for x in P for a in E if all(a > y for y in x)]  # marche si E est pourvu d'un ordre - works only if there is an order on E
        yield P
    return

def main():
    E = [1, 2, 3]
    P = PartiesDe(E)  # rien n'est encore calculé ! - nothing is computed yet
    for i in range(0, 4):  # affiche les parties à 0,1,2,3 éléments ligne par ligne - display parts with 0,1,2,3 elemnts line by line
        print(next(P))  # il s'agit bien d'une itération - it's an iterative process

# fonction principale - main function
if __name__ == '__main__':
    main()
