def disjoints(A,B): 
    return not any (x in A for x in B) 
  
#programme principal  
def main():  
    A=set("ab") 
    B=set("cd") 
    C=set("bcd") 
    print (disjoints(A,B)) 
    print (disjoints(A,C)) 
 
 
if __name__ == '__main__':  
    main()
