#python 3.8

import itertools  # pour combinations

# un objet frozenset est comme un objet set mais immuable
# a frozenset is just like a set but immutable
# on ne peut pas changer son contenu par add, discard, pop, clear, etc... - impossible to change content through add, discard, etc.
# les frozensets sont introduits ici parce que en Python les ensembles ne peuvent être éléments d'autres ensembles que s'ils sont immuables
# We use frozenset because python sets cannot have mutable sets as elements

# itertools fournit une fonction  'combinations' qui génère toutes les combinaisons d'un ensemble à k éléments
# itertools modules provides a 'combinations' function which generates all combinations with k elemnts from any set
# notre fonction combinaisons ne fait que constituer la liste engendrée.
# Our function just generates the corresponding list
def combinaisons(E :set, k :int) ->list :
    return list(itertools.combinations(E, k))

# la fonction suivante calcule l'ensemble des parties de E ayant exactement k éléments
# This function computes the sets of all subsets of E with k elements
def parties_a_k_elements(E :set, k :int) ->set :
    return {frozenset(x) for x in combinaisons(E, k)}

# maintenant on va calculer toutes les parties de E ayant 0,1,2,... jusqu'à n éléments où n est le nombre d'éléemnts de E
# now we will computes ALL subsets of E whatever their cardinality
def parties(E :set) ->set :
    result = set()  # accumulateur vide au départ - accumulateur empty to begin with
    for k in range(0, len(E) + 1):
        result = result | parties_a_k_elements(E, k)  # on ajoute toutes les parties de E à k éléments - we had all subsets with k elements
    return result

def main():
    S = {1, 2, 3, 4}
    print(S)
    print(parties_a_k_elements(S, 2))  # parties de S à deux éléments - subsets of S with 2 elements
    print(parties(S))  # toutes les parties de S - all subsets of S

main()
