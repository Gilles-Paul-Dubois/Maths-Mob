# mapping and filtering 
# visualization 
# with such techniques it is possible to define sets in 'comprehension' 
# set of squeres of odd numbers between 0 and 20 
 
def Imprime (S): 
    L=list(S) #list mode again for indexing 
    n=len(L) #cardinal number of the set 
    if n==0: #case of empty set 
        print("{ }") 
    else: 
        print ('{', end='') 
        for i in range(0, len(L) - 1): 
            print(L[i], end=',') # tall elements followed by separator coma except last one 
        print(L[n - 1], end='}\n') # closing brace and line-feed to finish 
 
#main function 
def main(): 
    # the 20 first naturals 
    L=range(0,20) 
    Imprime(L) 
    #among them odd ones 
    I=[i for i in L if i%2==1] 
    EI=set(I) #transformation into set 
    Imprime(EI) 
    # their squares 
    C=map(lambda x: x*x, I) 
    EC=set(C) #transform list into a set 
    Imprime(EC) 
 
 
if __name__ == '__main__': 
    main()
