# julia 1.6
# calcul des Cp,n par les formules
# computation of the Cp,n using formulas

# version itérative de la factorielle
# iterative version for the factorial
function fact(n)
    r=1
    for i in 1:n
        r*=i
    end
    return r
end

#calcul des coefficients binomiaux
# computation of binomial coefficients
function cpn(p,n)
    return fact(n)÷(fact(n-p)*fact(p))
end

#fonction de test (triangle de Pascal)
# test function (Pascal triangle)
function test(n)
    for i in 0:n
        for j in 0:i
            print(cpn(j,i), ' ')
        end
        println("")
    end
end

test(6)
