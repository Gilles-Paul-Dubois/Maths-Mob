# one can this way define a set in 'comprehension'
# set of squeres of odd numbers between 0 and 20

function imprime(X) #usual display in maths
  println("")
  print('{') #beginning of writing
  n = length(X)
  i = 0
  for x in X
    i = i + 1
    print(x)
    if i < n
      print(',') #only if it's not last
    end
  end
  println('}') #end of writing
end

#main function
function main()
  # les 20 premiers entiers
  L = Set(0:20)
  imprime(L)
  I=[i for i in L if i%2==1]
  EI=Set(I)
  imprime(EI)
  C=map(x->x^2,I)
  #anonymous functions in julia do not need the key-word 'lambda'
  EC=Set(C)
  imprime(EC)
end

main()
