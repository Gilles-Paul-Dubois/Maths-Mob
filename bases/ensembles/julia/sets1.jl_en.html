V = Set([]) #empty set
C1 = Set(['a', 'b', 'c']) #set of chars generated by a list
C2 = Set("abc") #the same with a string
N1 = Set([0, 1, 2, 3, 4, 5])  # set of numbers
N2 = Set(0:5)  #other possibility
M = Set([1, 2, 'a', 'B'])  # mixed set

function test1() #standard printing
  println("\njulia standard printing")
  println(V)
  println(C1)
  println(C2)
  println(N1)
  println(N2)
  println(M)
end

function test2() #custom printing
  println("\nusual maths habit")
  imprime(V)
  imprime(C1)
  imprime(C2)
  imprime(N1)
  imprime(N2)
  imprime(M)
end

function imprime(S) #classical writing in maths
  println("")
  print('{') #beginning of writing
  n = length(S)
  i = 0
  for x in S
    i = i + 1
    print(x)
    if i < n
      print(',') #only if not last one
    end
  end
  println('}') #end of process
end



function main()
  test1()
  test2()
end


main()
