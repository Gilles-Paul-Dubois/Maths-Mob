# -*- coding: utf_8 -*-
# python  version 3.8
# fabrication d'ensembles - constructing sets from keyboard
# visualisation  -displaying them
E=set()
# lecture au clavier- reading from keyboard (as strings)
def lire_ensemble():
    global E
    while 1:  # boucle infinie - infinite loop
        S = input()
        if (S == "-1"):  # sortie de boucle si valeur entrée est -1 - break if value entered=-1
            break
        else:
            E.add(S)  # ajout de l'élément saisi S à l'ensemble E - add element entered from keyboard

# programme principal
def main():
    print('Saisir des elements - taper -1 pour finir') # instruction for user
    lire_ensemble() # filling the set
    print(E) # displaying it



if __name__ == '__main__':
    main()
