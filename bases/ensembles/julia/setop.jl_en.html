#julia  version 1.6
# operations with sets
# union and intersection
# difference and symmetric difference


# sets of numbers
V=Set([])
E=Set([1,2,3,4])
F=Set([0,2,3])

function imprime(X) #classical maths habits
  println("")
  print('{') #beginning of display
  n = length(X)
  i = 0
  for x in X
    i = i + 1
    print(x)
    if i < n
      print(',') #sonly if not last one
    end
  end
  println('}') #end of display
end

# intersection
function test1()
  println("intersection")
  imprime(intersect(V, E)) # should be empty
  imprime(intersect(E, E)) # should show E
  imprime(intersect(E, F)) # or {2 3}
end

# union
function test2()
  println("union")
  imprime(union(V, E)) #should be E
  imprime(union(E, E)) # should give V
  imprime(union(E, F)) # or {0,4,2,3,1}
end

# le complémentaire
function test3()
  println("difference")
  imprime(setdiff(V, E)) # should be empty
  imprime(setdiff(E, V)) # should give V
  imprime(setdiff(E, F)) # 2 elements left
end


# différence sysmétrique
function test4()
  println("symmetric difference")
  imprime(symdiff(V, E)) # should be empty
  imprime(symdiff(E, V)) # should be V
  imprime(symdiff(E, F)) # 2 elements left
end
#main function
function main()
  test1()
  test2()
  test3()
  test4()
end

main()
