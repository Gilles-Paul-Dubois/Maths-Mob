# visualisation - display
# on peut ainsi définir un ensemble 'en compréhension' - this way we can define a set 'in comprehension'''
# ensemble des carrés des nombres impairs entre 0 et 20 - set of squares of odd numbers between 0 and 20

function imprime(X) #écriture classique des maths - usual maths writing
  println("")
  print('{') #début de l'écriture - beginning of display
  n = length(X)
  i = 0
  for x in X
    i = i + 1
    print(x)
    if i < n
      print(',') #seulement si ce n'est pas le dernier - only if not last
    end
  end
  println('}') #fin de l'écriture  end of display
end

# fonction principale - main function
function main()
  # les 20 premiers entiers - first 20 integers
  L = Set(0:20)
  imprime(L)
  I=[i for i in L if i%2==1]
  EI=Set(I)
  imprime(EI)
  C=map(x->x^2,I)
  #les fonctions anonymes n'ont pas besoin du mot-clef 'lambda'
  #anonymous functions doesn't need key word 'lambda'
  EC=Set(C)
  imprime(EC)
end

main()
