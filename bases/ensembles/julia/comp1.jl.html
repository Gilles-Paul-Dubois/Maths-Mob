# mapping et filtrage
# visualisation
# on peut ainsi définir un ensemble 'en compréhension'
# ensemble des carrés des nombres impairs entre 0 et 20

function imprime(X) #écriture classique des maths
  println("")
  print('{') #début de l'écriture
  n = length(X)
  i = 0
  for x in X
    i = i + 1
    print(x)
    if i < n
      print(',') #seulement si ce n'est pas le dernier
    end
  end
  println('}') #fin de l'écriture
end

#programme principal
function main()
  # les 20 premiers entiers
  L = Set(0:20)
  imprime(L)
  I=[i for i in L if i%2==1]
  EI=Set(I)
  imprime(EI)
  C=map(x->x^2,I)
  #les fonctions anonymes n'ont pas besoin du mot-clef 'lambda'
  EC=Set(C)
  imprime(EC)
end

main()
