function main()
    println(isempty([])) #empty list
    println(isempty(Set())) #empty set
    println(isempty(Dict())) #dictionnary without any key
    println(isempty("")) #empty string
    println(isempty(())) #tuple vide
    println(isempty([1])) #not empty list
    println(isempty(Set("a"))) #not empty set
    println(isempty(Dict("a"=>1))) #dictionnary with one key
    println(isempty("ab")) #not empty string
    println(isempty((1,2))) #not empty tuple
end

main()
