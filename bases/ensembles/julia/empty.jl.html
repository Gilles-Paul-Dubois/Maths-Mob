#julia 1.6

function main()
    println(isempty([])) #liste vide - empty list
    println(isempty(Set())) #ensemble vide - empty set
    println(isempty(Dict())) #dictionnaire sans entrée -dictionary without any entry
    println(isempty("")) #chaîne vide -empty string
    println(isempty(())) #tuple vide - empty tuple
    println(isempty([1])) #liste non  vide - non empty list
    println(isempty(Set("a"))) #ensemble non vide - non empty set
    println(isempty(Dict("a"=>1))) #dictionnaire avec une entrée - dictionary with at least one entry
    println(isempty("ab")) #chaîne non vide - non empty string
    println(isempty((1,2))) #tuple non vide - non empty tuple
end

main()
