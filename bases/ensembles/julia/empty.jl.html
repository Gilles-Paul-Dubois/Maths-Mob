function main()
    println(isempty([])) #liste vide
    println(isempty(Set())) #ensemble vide
    println(isempty(Dict())) #dictionnaire sans entrée
    println(isempty("")) #chaîne vide
    println(isempty(())) #tuple vide
    println(isempty([1])) #liste non  vide
    println(isempty(Set("a"))) #ensemble non vide
    println(isempty(Dict("a"=>1))) #dictionnaire avec une entrée
    println(isempty("ab")) #chaîne non vide
    println(isempty((1,2))) #tple non vide
end

main()
