# calcule l'ensemble des parties de E à p éléments si E est donné sous forme de liste sans répétition

function partsof(E)
    P = [[]]
    while !isempty(E)
        R = deepcopy(P)
        a = pop!(E)
        Q = [push!(x, a) for x in P]
        P = append!(R, Q)
    end
    return P
end

function partsofsizep(E,p)
    P=partsof(E)
    return [x for x in P if length(x)==p]
end


function main()
    println(partsofsizep([1, 2, 3],0))
    println(partsofsizep([1, 2, 3],1))
    println(partsofsizep([1, 2, 3],2))
    println(partsofsizep([1, 2, 3],3))
end

main()
