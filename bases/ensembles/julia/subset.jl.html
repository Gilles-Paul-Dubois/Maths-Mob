# sous-ensembles et sur-ensembles -subsets and supersets
# vider un ensemble -empty a set
# faire une copie d'un ensemble -make a copy of a set
# Tester l'égalité de deux ensembles -checking equality of two sets

# ensembles de nombres sets of numbers
V=Set([])
E=Set([1,2,3,4])
F=Set([2,3])
G=Set([1,2,3,4,5,5,7])

function imprime(X) #écriture classique des maths - classical display in maths
  println("")
  print('{') #début de l'écriture - beginning of display
  n = length(X)
  i = 0
  for x in X
    i = i + 1
    print(x)
    if i < n
      print(',') #seulement si ce n'est pas le dernier - only if not last one
    end
  end
  println('}') #fin de l'écriture - end of display
end

# check if proper subset - sous-ensemble au sens strict
function isstrictsubset(X,Y)
  return issubset(X,Y) && length(X)<length(Y)
end

function test1()
  println(issubset(V,E)) #le vide est contenu dans tout -empty set is subset of any set
  println(issubset(F,E))
  println(issubset(E,E))
  println(isstrictsubset(E,E))
end

function clear(X) #vide la collection - empty the collection
  while !isempty(X)
    pop!(X)
  end
end

function test2()
  H=copy(E)
  imprime(E)
  imprime(H)
  println(H==E)
  println(E==F)
  clear(H)
  imprime(H)
end

function main()
  imprime(G) #le doublon 5 est éliminé - double 5 is eliminated
  test1()
  test2()
end

main()
