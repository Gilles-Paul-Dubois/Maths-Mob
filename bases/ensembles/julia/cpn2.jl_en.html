#builds next line of Pascal's triangle
function next(L)
    R=[1]
    for i in 1:length(L)-1
        push!(R,L[i]+L[i+1])
    end
    push!(R,1)
    return R
end

#givs by iterative way n-th line of triangle
function Pascal(n)
    L=[1]
    for i in 1:n
        L=next(L)
    end
    return L
end

#test function
function test(k)
    for i in 1:k
        println(Pascal(i))
    end
end


test(6)
