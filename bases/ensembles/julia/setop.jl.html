# opérations avec les ensembles- operations with sets
# réunion et intersection - union and intersection
# complémentaire et différence symétrique - difference and symmetric difference


# ensembles de nombres -sets of numbers
V=Set([])
E=Set([1,2,3,4])
F=Set([0,2,3])

function imprime(X) #écriture classique des maths - classical writing in maths
  println("")
  print('{') #début de l'écriture - beginning of display
  n = length(X)
  i = 0
  for x in X
    i = i + 1
    print(x)
    if i < n
      print(',') #seulement si ce n'est pas le dernier - only if not last
    end
  end
  println('}') #fin de l'écriture - end of display
end

# l'intersection - intersection
function test1()
  println("intersection")
  imprime(intersect(V, E)) # doit être vide - shoud be empty
  imprime(intersect(E, E)) # doit donner E - should give E
  imprime(intersect(E, F)) # soit {2 3} - will return {2,3}
end

# la réunion - union
function test2()
  println("réunion")
  imprime(union(V, E)) # doit être E - should be E
  imprime(union(E, E)) # doit donner V -should be V
  imprime(union(E, F)) # soit {0,4,2,3,1} - gives {0,1,2,3,4}
end

# le complémentaire - difference
function test3()
  println("complémentaire")
  imprime(setdiff(V, E)) # doit être vide - should be empty
  imprime(setdiff(E, V)) # doit donner V - should give V
  imprime(setdiff(E, F)) # reste deux éléments - two elements left
end

# différence sysmétrique - symmetric difference
function test4()
  println("différence symétrique")
  imprime(symdiff(V, E)) # doit être vide - should be empty
  imprime(symdiff(E, V)) # doit donner V - should give V
  imprime(symdiff(E, F)) # reste deux éléments - two elements left
end

#fonction principale - main function
function main()
  test1()
  test2()
  test3()
  test4()
end

main()
