# opérations avec les ensembles
# réunion et intersection
# complémentaire et différence symétrique


# ensembles de nombres
V=Set([])
E=Set([1,2,3,4])
F=Set([0,2,3])

function imprime(X) #écriture classique des maths
  println("")
  print('{') #début de l'écriture
  n = length(X)
  i = 0
  for x in X
    i = i + 1
    print(x)
    if i < n
      print(',') #seulement si ce n'est pas le dernier
    end
  end
  println('}') #fin de l'écriture
end

# l'intersection
function test1()
  println("intersection")
  imprime(intersect(V, E)) # doit être vide
  imprime(intersect(E, E)) # doit donner E
  imprime(intersect(E, F)) # soit {2 3}
end

# la réunion
function test2()
  println("réunion")
  imprime(union(V, E)) # doit être E
  imprime(union(E, E)) # doit donner V
  imprime(union(E, F)) # soit {0,4,2,3,1}
end

# le complémentaire
function test3()
  println("complémentaire")
  imprime(setdiff(V, E)) # doit être vide
  imprime(setdiff(E, V)) # doit donner V
  imprime(setdiff(E, F)) # reste deux éléments
end


# différence sysmétrique
function test4()
  println("différence symétrique")
  imprime(symdiff(V, E)) # doit être vide
  imprime(symdiff(E, V)) # doit donner V
  imprime(symdiff(E, F)) # reste deux éléments
end
#fonction principale
function main()
  test1()
  test2()
  test3()
  test4()
end

main()
