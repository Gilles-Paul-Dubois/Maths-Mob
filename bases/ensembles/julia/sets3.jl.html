global E=Set([1,2,3,4])
function test()
    println(length(E)) #affiche le cardinal - prints the cardinality
    println(in(5,E)) #test d'appartenance  (échec) - test of membership (failure)
    global E=push!(E,5) #ajout d'un élément - adding an element
    println(in(5,E)) #test d'appartenance (succès) - test of membership (success)
    global E=delete!(E,1) #suppression d'un élément - removing an element
    show(E) #résultat - show result
end

test()
