function Paires(E)
    L=[Set([x,y]) for x in E for y in E if y!=x]
    R=[] #liste vide au départ - empty list to begin with
    for p in L #élimination des doubles -elimination of doubles
        if !(p in R)
            push!(R,p)
        end
    end
    return R # retourne la liste des paires - return list of pairs
end
