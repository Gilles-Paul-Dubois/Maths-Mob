#construit la ligne suivante du triangle de Pascal
# build next row from Pascal triangle
function next(L)
    R=[1]
    for i in 1:length(L)-1
        push!(R,L[i]+L[i+1])
    end
    push!(R,1)
    return R
end

#donne par it√©ration la ligne d'ordre n du triangle de Pascal
# gives by iteration n-th row of Pascal triangle
function Pascal(n)
    L=[1]
    for i in 1:n
        L=next(L)
    end
    return L
end

#fonction de test - test function
function test(k)
    for i in 1:k
        println(Pascal(i))
    end
end

test(6)
