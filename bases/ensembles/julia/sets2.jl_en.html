function imprime(X) #classical maths habit
  println("")
  print('{') #beginning of process
  n = length(X)
  i = 0
  for x in X
    i = i + 1
    print(x)
    if i < n
      print(',') #only if not last one
    end
  end
  println('}') #end of process
end
function lire_ensemble()
  E=Set([])
    while true
        S=readline()
        if S=="-1"
            break
        else
            E=push!(E,S)
        end
    end
    return E
end

function main()
    println("enter elements one after another - type -1 to end")
    E=lire_ensemble()
    imprime(E)
  end

main()
