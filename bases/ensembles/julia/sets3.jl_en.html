global E=Set([1,2,3,4])
function test()
    println(length(E)) #print the cardinal number
    println(in(5,E)) #test of belonging - failure
    global E=push!(E,5) # adding one element
    println(in(5,E)) #test of belonging - success
    global E=delete!(E,1) #suppression d'un élément
    show(E) #result
end

test()
