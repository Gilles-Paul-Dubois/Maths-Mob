# -*- coding: utf-8 -*-
# python version 3.4 
global c 
c = 1024 * 1024 * 1024 * 2 
 
 
def GCL(a, b, c, x0): 
    """le générateur - the generator""" 
    x = x0 
    yield x 
    while True: 
        x = (a * x + b) % c 
        yield x 
 
 
def main(): 
    """fonction principale - main function""" 
    RANDU = GCL(65539, 0, c, 1) 
    PASCAL = GCL(129, 907633385, 2 * c, 0) 
    UNIX = GCL(1103515245, 12345, c, 0) 
    for i in range(0, 10): 
        print(next(RANDU), end=" ") 
    print("\n-------------------------")  # séparateur - separator 
    for i in range(0, 10): 
        print(next(PASCAL), end=" ") 
    print("\n-------------------------")  # séparateur - separator 
    for i in range(0, 10): 
        print(next(UNIX), end=" ") 
    print("\n-------------------------")  # séparateur - separator 
 
 
if __name__ == '__main__': 
    main()
