# -*- coding: utf-8 -*-

import random 
 
Chevaux = range(1, 21)  # la liste des numéros des chevaux - list of horses numbers 
 
 
def arrivee(): 
    """une arrivée aléatoire - random arrival""" 
    return random.sample(Chevaux, 3) 
 
 
def main(): 
    """Fonction principale - main function""" 
    print(arrivee()) 
 
 
if __name__ == '__main__': 
    main()
