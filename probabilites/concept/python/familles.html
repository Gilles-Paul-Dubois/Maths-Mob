import random 
 
""" Probleme des familles avec deux enfants  problem of families with two children""" 
 
global F  # les 5000 familles - the 5000 families 
 
 
def initF(): 
    """initialisation des familles - initiaization of families""" 
    global F 
    F = [(random.randint(0, 1), random.randint(0, 1)) for i in range(0, 5000)] 
    # 0=garcon, 1=fille - 0=boy , 1=girl 
 
 
def simul(m): 
    """m simulations """ 
    boys = 0.0 
    girls = 0 
    for i in range(0, m): 
        n = random.randint(0, 4999) 
        Famille = F[n] 
        if 0 in Famille: 
            boys += 1 
            if 1 in Famille: 
                girls += 1 
    return girls / boys 
 
 
def main(): 
    initF() 
    print(simul(2000)) 
 
 
if __name__ == '__main__': 
    main()
