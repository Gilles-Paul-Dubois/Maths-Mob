"""Knuth et Lewis - Knoth and Lewis method """ 
global c 
c = 1024 * 1024 * 1024 * 4 
 
def GCL(a, b, c, x0): 
    """le générateur congruence linéaire général the general LCG generator""" 
    x = x0 
    yield x 
    while True: 
        x = (a * x + b) % c 
        yield x 
 
def KLfloat(): 
    """générateur de flottants utilisant Knuth-Lewis - float generator using KL""" 
    KL = GCL(1664525, 1013904223, c, 1)#Knuth-Lewis 
    fc=c*1.0 #pour assurer la division dans les flottants  - to ensure float division 
    while True: 
        yield next(KL)/fc 
 
def test(m): 
    """Fonction de test - test function""" 
    Comptes=[0]*10 
    M=m*1.0 
    KLF = KLfloat() 
    for i in range(0,m): 
        x=next(KLF)*10 
        Comptes[int (x)]+=1 
    return [x/M for x in Comptes]#pour les fréquences - for frequencies 
 
 
def main(): 
    """fonction principale - main function""" 
    print (test(2000)) 
 
if __name__ == '__main__': 
    main()
