"""Méthode du Middle-square de Von Neumann avec 4 chiffres""" 
"""Von Neumann Middle-Square with 4 digits"""  
 
def NombreDeChiffres(n): 
    """Trouve le nombre de chiffres d'un entier - finds the number of digits of an integer""" 
    if (n == 0): 
        return 0 
    else: 
        return 1 + NombreDeChiffres(n // 10) 
 
 
def std(n, p): 
    """complete avec des zéros devant le nombre n pour en faire un nombre à p chiffres sous forme de chaîne de caractères""" 
    """completes with leading zeros the number n to build a string representing a number with p digits """ 
    k = NombreDeChiffres(n) 
    return "0" * (p - k) + str(n) 
 
 
def MiddleSquare(n): 
    """Calcule le middle-square d'un entier - computes the middle-square of an integer""" 
    k = n * n 
    ms = std(k, 8) 
    msn = ms[2:6] 
    return int(msn) 
 
 
def Longueur(n): 
    """Calcule le nombre d'entiers distincts générés pour la méthode de Von Neumann avec le terme initial n""" 
    """Computes the number of distincts integers  generated by the method of  Von Neumann with initial seed n""" 
    L = [] 
    count = 0 
    while True: 
        L += [n] 
        count += 1 
        n = MiddleSquare(n) 
        if n in L: 
            return count 
 
 
def main(): 
    """fonction principale - main function""" 
    max = 1 
    N = 0 
    for i in range(1, 10000): 
        if Longueur(i) > max: 
            max = Longueur(i) 
            N = i 
    print(N, max) 
 
 
if __name__ == '__main__': 
    main()
