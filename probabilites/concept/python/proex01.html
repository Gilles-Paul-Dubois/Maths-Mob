import math 
import random 
 
""" Problème de Buffon - simulation pour l=a=1 - Buffon problem simulation for l=a=1""" 
 
 
def epingle(): 
    """épreuve aléatoire - random test""" 
    return (random.random(), math.pi * random.random()) 
 
 
def condition(e): 
    """la condition - the condition""" 
    return e[0] <= math.sin(e[1]) 
 
 
def lancers(n): 
    """simule n lancers - simulates n rolls""" 
    succes = 0.0 
    for i in range(0, n): 
        if condition(epingle()): 
            succes += 1 
    return succes / n 
 
 
def main(): 
    """fonction principale - main function""" 
    print(lancers(10000)) 
    print(2 / math.pi) 
 
 
if __name__ == '__main__': 
    main()
