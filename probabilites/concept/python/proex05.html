#génération du triangle de Pascal au moyen d'un générateur- generating binomial coefficients with a generator 
def Pascal(): 
    n, C = 0, [1] 
    yield n, C # retour du premier appel - return for the first call 
    while True: # exécution des appels suivants - executing other calls 
        n += 1 
        D = C[:] 
        D.insert(0, 1) 
        for i in range(1, len(C)): 
            D[i] = C[i-1] + C[i] # relation de récurrence du Triangle de Pascal - induction relation for Pascal triangle 
        C = D 
        yield n, C 
 
# coefficients Cp,n - binomial coefficients 
def Cpn(p, n): 
    TP = Pascal() 
    for i in range(0, n + 1): 
        L = next(TP)[1] 
    return L[p] 
 
 
def main(): 
    TP = Pascal() 
    for i in range(0, 14): 
        L = next(TP) 
    X = L[1]# les coefficients de la treizième ligne du triangle de Pascal. - 13th row of Pascal triangle 
    S = [X[i] * X[13-i] for i in range(1, 13)] 
    F = 6 * sum(S) 
    P = Cpn(13, 52) 
    print ("Nombre de cas favorables: - favorable cases", F) 
    print ("Nombre de cas possibles: - possible cases", P) 
    print (float(F) / P) 
 
if __name__ == '__main__': 
    main()
