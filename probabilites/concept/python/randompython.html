import random 
 
"""test des fonctions élémentaires du module générateur de nombres pseudo-aléatoires""" 
"""test of standard functions of the module which generates (pseudo) random numbers"""
X = [i for i in range(1, 21)]  # la séquence qui sera battue - sequence which will be shuffled 
 
 
def EntreBornes(a, b): 
    """ Tire au hasard un entier entre a et b bornes comprises - random integer between a and b bounds included""" 
    for i in range(0, 10): 
        print(random.randint(a, b), end=" ") 
 
 
def ChoisisParmi(L): 
    """Choisis au hasard un élément dans une séquence - take a random element in a sequence""" 
    for i in range(0, 10): 
        print(random.choice(L), end=" ") 
 
 
def Battre(L): 
    """Mélanger - shuffle""" 
    return random.shuffle(L) 
 
 
def ReelAlea(): 
    """réel aléatoire dans l'intervalle [0,1[ random real number in [0,1[""" 
    for i in range(0, 10): 
        print(random.random(), end=" ") 
 
 
def ReelEntreBornes(a, b): 
    """réel uniforméùment distribué dans l'intervalle [a,b[ - real nomber with uniform distribution in [a,b[""" 
    for i in range(0, 10): 
        print(random.uniform(a, b), end=" ") 
 
 
def main(): 
    """Fonction principale - main function""" 
    EntreBornes(0, 5) 
    print("\n.................")  # saut de ligne marqué - line jump with symbol 
    ChoisisParmi([4, 5, 8, 10]) 
    print("\n.................")  # saut de ligne marqué 
    Battre(X) 
    print(X, "\n.................") 
    ReelAlea() 
    print("\n.................")  # saut de ligne marqué 
    ReelEntreBornes(2, 6) 
 
 
if __name__ == '__main__': 
    main()
