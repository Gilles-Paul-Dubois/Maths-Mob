import random 
 
""" Probleme de Chicago""" 
 
 
def simul(n, m): 
    L = [i for i in range(0, n)] 
    success = 0.0 
    for i in range(0, m): 
        random.shuffle(L) 
        if any([L[i] == i for i in range(0, n)]): 
            success += 1 
    return success / m 
 
 
def main(): 
    e = 2.71828 
    print(simul(100, 8000)) 
    print(1 - 1 / e) 
 
 
if __name__ == '__main__': 
    main()
