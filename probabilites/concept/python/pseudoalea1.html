"""Standard minimal de Park & Miller """ 
"""générateur du type congruence linéaire""" 
global M 
M = 1024 * 1024 * 1024 * 2 - 1 
# ce nombre est 2^31-1 
 
def alea(n): 
    """le générateur""" 
    x = n  # le 'seed' 
    yield x 
    while True: 
        x = (x * 16807) % M 
        yield x 
 
 
def main(): 
    """Fonction principale""" 
    A = alea(1) 
    for i in range(0, 20): 
        print(next(A)) 
 
 
if __name__ == '__main__': 
    main()
