def GCL(a, b, c, x0): 
    """le générateur - the generator""" 
    x = x0 
    yield x 
    while True: 
        x = (a * x + b) % c 
        yield x 
 
 
def main(): 
    """fonction principale - main function""" 
    Gen1 = GCL(31415821, 1, 100000000, 0) 
    for i in range(0, 20): 
        print(next(Gen1), end=" ") 
 
 
if __name__ == '__main__': 
    main()
