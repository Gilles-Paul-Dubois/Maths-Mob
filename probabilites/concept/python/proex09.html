import random 
 
""" Probleme des dates de naissances - problem of dates of birth""" 
 
 
def test(): 
    L = [random.randint(1, 365) for i in range(0, 25)] 
    C = [L.count(x) for x in L] 
    return any([x > 1 for x in C]) 
 
 
def simul(m): 
    """m simulations """ 
    success = 0.0 
    for i in range(0, m): 
        if test(): 
            success += 1 
    return success / m 
 
 
def calprob(): 
    """calcul de la probabilité - computation of probability""" 
    N = D = 1 
    x = 365 
    for i in range(0, 25): 
        N *= x 
        x -= 1 
        D *= 365 
    return 1 - float(N) / D 
 
 
def main(): 
    print(simul(1000)) 
    print(calprob()) 
 
 
if __name__ == '__main__': 
    main()
