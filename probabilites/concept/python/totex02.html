import random 
"""exercice n°2 exercise #2""" 
 
global U #les urnes - the urns 
U = [[], [], [], [], []] #vides au départ- empty to begin with 
#remplissage 0 = boule blanche 1=boule noire - filling 0=white ball, 1=black ball 
U[0] = [0, 0, 1] 
U[1] = U[0] 
U[2] = [1] * 10 
U[3] = [0, 0, 0, 1] 
U[4] = U[3] 
 
def epreuve(): 
    """l'épreuve aléatoire - the random test""" 
    V = U[random.randint(0, 4)]# choix de l'urne - choice of urn 
    n = len(V) #nombre de boules dans V - number of balls in V 
    i = random.randint(0, n-1) 
    return V[i] 
 
def simul(m): 
    """ la simulation - the simulation """ 
    count = 0.0 
    for i in range(0,m): 
        if epreuve() == 0: 
            count += 1 
    return count / m 
 
def main(): 
    """fonction principale - main function""" 
    print (simul(5000)) 
    print (17 / 30.0) 
 
if __name__ == '__main__': 
    main()
