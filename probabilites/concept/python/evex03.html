# génération des permutations d'un ensemble fini - generation of permutations of a finite set 
# comptage des permutations ayant au moins un point fixe - counting permutations with at least one fixed point 
 
from functools import reduce 
 
# fonction d'insertion - inserting function 
def ins(X, i, L): 
    R = L[:]  # copie de L - copy of L 
    R.insert(i, X)  # insertion de X à l'index i dans R - inserting X with index i in R 
    return R 
 
 
# fait tourner X dans L à toutes les positions - rotate X in L in all possible posistions 
def turn(X, L): 
    return [ins(X, i, L) for i in range(0, len(L) + 1)] 
 
 
# un générateur - a generator 
def permutations(): 
    i, P = 0, [[]] 
    yield P  # stoppe l'exécution pour la reprendre à cet endroit précis - stps running to restart to this precise point 
    while True:  # fait passer d'une génération à la suivante - passes from one generation to next one 
        i = i + 1 
        Q = [turn(i, X) for X in P] 
        P = reduce(lambda x, y: x + y, Q)  # concaténation à répétition - repeated concatenation  
        yield P  # stoppe l'exécution pour la reprendre à cet endroit précis - stops running to respatrt a this special point 
 
 
def PossedeUnInvariant(x): 
    """au moins un invariant - at least one fixed point""" 
    return any([x[i] == i + 1 for i in range(0, len(x))]) 
 
 
def main(): 
    """ fonction principale - main function""" 
    Perm = permutations() 
    for i in range(0, 6):  # permutations d'un ensemble à 5 éléments - permutations of a set with 5 elements 
        X = next(Perm) 
    Y = [x for x in X if PossedeUnInvariant(x)] 
    print(len(Y))  # combien de permutations de {1,2,3,4,5} ont au moins un point fixe - how many permutations with at least one fixed point 
    print(Y)  # afficher les toutes - display all of them 
 
 
if __name__ == '__main__': 
    main()
