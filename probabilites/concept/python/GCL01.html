def GCL(a, b, c, x0): 
    """le générateur - the generator""" 
    x = x0 
    yield x 
    while True: 
        x = (a * x + b) % c 
        yield x 
 
 
def main(): 
    """fonction principale - maain function""" 
    Gen1 = GCL(25, 16, 256, 12) 
    Gen2 = GCL(25, 16, 256, 11) 
    Gen3 = GCL(25, 16, 256, 10) 
    for i in range(0, 10): 
        print(next(Gen1), end=" ") 
    print("\n-------------------")  # séparateur -separator 
    for i in range(0, 10): 
        print(next(Gen2), end=" ") 
    print("\n-------------------")  # séparateur - separator 
    for i in range(0, 10): 
        print(next(Gen3), end=" ") 
 
 
if __name__ == '__main__': 
    main()
