import random 
 
""" Probleme des 3 portes - problem of the 3 doors""" 
 
 
def choice(): 
    """la bonne porte est toujours la 1 - good door will always be #1""" 
    x = random.randint(1, 3)  # tirage d'une porte au hasard - choosing one door randomly 
    if x == 1: 
        # le choix est bon on indique alors la porte 2 comme mauvaise - choiceis good we indicate door #2 as bad one 
        return 3  #on choisit l'autre (donc perdu) - we choose the other so it's lost 
    if x == 2: 
        # le choix est mauvais on indique alors la porte 3 comme mauvaise - choice is bad so we indicate #3 as bad one 
        return 1  #on choisit l'autre (donc gagné)- we choose the other one (so success) 
    if x == 3: 
        # le choix est mauvais on indique alors la porte 2 comma mauvaise - choice is bad we indicate door #2 as bad one 
        return 1  #on choisit l'autre (donc gagné) 
 
 
def simul(m): 
    """m simulations du choix de la porte - simulating choice of the door""" 
    success = 0.0 
    for i in range(0, m): 
        if choice() == 1: 
            success += 1 
    return success / m 
 
 
def main(): 
    print(simul(1000)) 
 
 
if __name__ == '__main__': 
    main()
