"""Mitchell-Moore """ 
import random 
 
global c 
c = 1024 * 1024 * 1024 
 
 
def GCA(): 
    """le générateur - the generator""" 
    L = [random.randint(1, 100) for i in range(0, 56)] 
    while True: 
        x = (L[0] + L[31]) % c  # nouveau nombre généré - new generated number 
        L = L + [x]  # ajouter le nouveau nombre à la fin - add the new number to the end 
        L = L[1:]  # enlever la tête - remove head 
        yield x 
 
 
def main(): 
    """fonction principale - main function""" 
    MM = GCA() 
    for i in range(0, 20): 
        print(next(MM), end=" ") 
    print("\n-------------------------")  # séparateur - separator 
 
 
if __name__ == '__main__': 
    main()
