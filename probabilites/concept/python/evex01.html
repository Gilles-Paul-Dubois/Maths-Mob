"""Lister tous les 64 évènements liés au lancer unique d'un dé""" 
"""List all the 64 events connected with rolling a unique dice"""
 
 
def PartiesDe(E): 
    P = [[]] 
    yield P  # retourner [[]] au premier appel - return [[]] for the first call 
    while E:  # appels suivants - next calls 
        P = [x + [a] for x in P for a in E if all(a > y for y in x)]  # marche si E est pourvu d'un ordre - works only if E endowed with an order 
        yield P 
    return 
 
 
def main(): 
    """Fonction principal - main functione""" 
    Omega = range(1, 7) 
    Y = PartiesDe(Omega) 
    Evenements = [] 
    for i in range(0, 7): 
        Evenements += next(Y) 
    print(Evenements) 
 
 
if __name__ == "__main__": 
    main()
