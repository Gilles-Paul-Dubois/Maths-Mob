import math 
import random 
 
""" Exercice n°4 - exercise #4""" 
 
 
def epreuve(n): 
    """épreuve aléatoire - random test""" 
    L = n * [1] + n * [-1] 
    random.shuffle(L) 
    return L 
 
 
def condition(L): 
    """la condition - the condition""" 
    S = [sum(L[0:i]) for i in range(1, len(L) + 1)] 
    return all([x >= 0 for x in S]) 
 
 
def repeter(m, n): 
    """répéter m fois et faire les stats - repeat m times and compute stats""" 
    Count = 0.0 
    for i in range(0, m): 
        if condition(epreuve(n)): 
            Count += 1 
    return Count / m 
 
 
def main(): 
    """fonction principale - main function""" 
    n = 10 
    for m in [1000, 2000, 5000, 10000]: 
        print(repeter(m, n)) 
    print('------------------') 
    print(1.0 / (n + 1)) 
 
 
if __name__ == '__main__': 
    main()
