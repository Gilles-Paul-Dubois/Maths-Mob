import random 
 
 
class CarteBridge(): 
    couleurs = ['Trèfle', 'Carreau', 'Coeur', 'Pique'] 
    niveaux = ['Deux', 'Trois', 'Quatre', 'Cinq', 'Six', 'Sept', 'Huit', 'Neuf', 'Dix', 'Valet', 'Dame', 'Roi', 'As'] 
 
    def __init__(self, n): 
        self.code = n 
        self.code_couleur = n // 13 
        self.code_niveau = n % 13 
        self.couleur = CarteBridge.couleurs[self.code_couleur] 
        self.niveau = CarteBridge.niveaux[self.code_niveau] 
        self.PH = self.code_niveau - 8 if self.code_niveau >= 9 else 0  # points d'honneur - honnor points 
 
    def __str__(self): 
        return self.niveau + " de " + self.couleur 
 
 
def Carte_au_hasard(): 
    """Tirage d'une carte dans le jeu - random drawing of a card in the deck""" 
    return CarteBridge(random.randint(0, 52)) 
 
 
def Main_au_hasard(): 
    """13 cartes au hasard dans le jeu - 13 random cards in the deck""" 
    Liste_codes = random.sample(range(0, 52), 13) 
    return [CarteBridge(i) for i in Liste_codes] 
 
 
def Points(M): 
    """calcul des points d'honneur d'une main - computation of honor points in a hand""" 
    return sum([c.PH for c in M]) 
 
 
def main(): 
    """Fonction principale - main function""" 
    CB = Carte_au_hasard() 
    print(CB) 
    print('--------------') 
    M = Main_au_hasard() 
    for c in M: 
        print(c) 
    print('--------------') 
    print(Points(M)) 
 
 
if __name__ == "__main__": 
    main()
