nombre_de_chiffres(n)= n==0 ? 0 : 1+nombre_de_chiffres(n√∑10)

std(n,p)= string(repeat("0",p-nombre_de_chiffres(n)),string(n))

function middle_square(n)
    k=n*n
    ms=std(k,8)
    msm=ms[3:6]
    return parse(Int64,msm)
end

function longueur(n)
    L,count=[],0
    while true
        push!(L,n)
        count+=1
        n=middle_square(n)
        if n in L
            return count
        end
    end
end

function main()
    max,N=1,0
    for i in 1:9999
        if longueur(i)>max
            max=longueur(i)
            N=i
        end
    end
    print(N,"  ",max)
end

main()
