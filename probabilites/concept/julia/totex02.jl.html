U=repeat([[]],5)
U[1]=[0,0,1]
U[2]=[0,0,1]
U[3]=repeat([1],10)
U[4]=[0,0,0,1]
U[5]=[0,0,0,1]

function epreuve()
    V=U[rand(1:5)]
    n=length(V)
    i=rand(1:n)
    return V[i]
end

function simul(m)
    count=0
    for i in 1:m
        if epreuve()==0
            count+=1
        end
    end
    return count/m
end

println(simul(10000))
