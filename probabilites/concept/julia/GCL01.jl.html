function gcl(a,b,c,x0,channel)
    x=x0
    put!(channel,x0)
    while true
        x=(a*x+b)%c
        put!(channel,x)
    end
end

function main()
    Gen1=Channel((channel_arg)->gcl(25,16,256,12,channel_arg))
    for i in 1:10
        print(take!(Gen1)," , ")
    end
    print("\n-------------------------\n")
    Gen2=Channel((channel_arg)->gcl(25,16,256,11,channel_arg))
    for i in 1:10
        print(take!(Gen2)," , ")
    end
    print("\n-------------------------\n")
    Gen3=Channel((channel_arg)->gcl(25,16,256,10,channel_arg))
    for i in 1:10
        print(take!(Gen3)," , ")
    end
    print("\n-------------------------")    
end


main()
