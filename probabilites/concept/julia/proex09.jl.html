function test()
    L=[rand(1:365) for i in 1:25]
    C=[count(y->y==x,L) for x in L]
    return any((x>1 for x in C))
end

function simul(m)
    success=0
    for i in 1:m
        if test()
            success+=1
        end
    end
    return success/m
end

function calprob()
    N,D,x=big(1),big(1),365
    for i in 1:25
        N=N*x
        x=x-1
        D=D*365
    end
    return 1-N/D
end

function main()
    println(simul(1000))
    println(calprob())
end

main()
