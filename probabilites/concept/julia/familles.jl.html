F=[(rand(0:1),rand(0:1)) for i in 1:5000]

function simul(m)
    boys,girls=0,0
    for i in 1:m
        n=rand(1:5000)
        Famille=F[n]
        if 0 in Famille
            boys+=1
            if 1 in Famille
                girls += 1
            end
        end
    end
    return girls/boys
end

println(simul(2000))
