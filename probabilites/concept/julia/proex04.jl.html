using Random

function epreuve(n)
    L1=repeat([1],n)
    L2=repeat([-1],n)
    append!(L1,L2)
    return shuffle!(L1)
end

function condition(L)
    M=[L[1:i] for i in 1:length(L)]
    S=[reduce(+,x) for x in M]
    return all((x>=0 for x in S))
end


function repeter(m,n)
    count=0
    for i in 1:m
        if condition(epreuve(n))
            count+=1
        end
    end
    return count/m
end

function main()
    for m in [1000,2000,5000,10000]
        println(repeter(m,10))
    end
    println("-------------------")
    println(1/11)
end

main()
