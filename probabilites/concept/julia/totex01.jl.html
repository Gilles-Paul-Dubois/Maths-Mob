U=repeat([[]],5)
U[1]=[0,0,1,1,1]
U[2]=[0,0,1,1,1]
U[3]=[0,1,1,1,1]
U[4]=[0,1,1,1,1]
U[5]=[0,0,0,0,1]

function epreuve()
    i=rand(1:5)
    V=U[i]
    j=rand(1:5)
    return(i,V[j])
end

function simul(m)
    count,success=0,0
    for i in 1:m
        E=epreuve()
        if E[2]==0
            count+=1
            if E[1]==5
                success+=1
            end
        end
    end
    return success/count
end

println(simul(10000))
