using Random
entier_entre_bornes(a,b) =rand(a:b)
choisis_parmi(L)=L[rand(1:end)]
battre(L)=shuffle(L)
reel_alea()=rand()
reel_entre_bornes(a,b)=a+rand()*(b-a)

function main()
    for i in 1:5
        print(entier_entre_bornes(3,5)," : ")
    end
    println("\n---------------------")
    L=['a','b','c','d','e']
    for i in 1:5
        print(choisis_parmi(L)," : ")
    end 
    println("\n---------------------")
    println(shuffle(L))
    println("---------------------")
    for i in 1:5
        println(rand())
    end
    println("--------------------")
    for i in 1:5
        println(reel_entre_bornes(1,5))
    end    
end

main()
