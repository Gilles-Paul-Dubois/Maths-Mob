c = 1024 * 1024 * 1024  

function gca(channel)
    L=[rand(1:100) for i in 1:56]
    while true
        x=(L[1]+L[32])%c
        push!(L,x)
        L=L[2:end]
        put!(channel,x)
    end
end

function main()
    MM=Channel((channel_arg)->gca(channel_arg))
    for i in 1:20
        print(take!(MM)," ")
    end    
end

main()
