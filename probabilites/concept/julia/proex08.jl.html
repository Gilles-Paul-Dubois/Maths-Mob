function choice()
    x=rand(1:3)
    if x==1
        return 3
    end
    if x==2
        return 1
    end
    if x==3
        return 1
    end
end

function simul(m)
    success=0
    for i in 1:m
        if choice()==1
            success+=1
        end
    end
    return success/m
end

println(simul(1000))
