/* Programme C affichant 10 entiers aléatoires - C program displaying 10 random integers */

#include <stdlib.h>
#include <stdio.h>
#include <time.h>

int main(int argc, char** argv) {
    int i;
    srand (time (NULL));// réinitialise le générateur - initialize again the generator
    for (i = 0; i < 10; i++) {
        printf("%d\n", rand());
    }
    return (EXIT_SUCCESS);
}
