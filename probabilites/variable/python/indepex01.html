import itertools as it 
import random 
# jeu de bridge - game of bridge 
 
class carte(): 
    couleurs = ['trèfle', 'carreau', 'coeur', 'pique'] 
    figures = ['deux', 'trois', 'quatre', 'cinq', 'six', 'sept', 'huit', 'neuf', 'dix', 'valet', 'dame', 'roi', 'as'] 
 
    def __init__(self, n): 
        self.n = n 
        self.couleur = n // 13 
        self.figure = n % 13 
 
    def valeur(self): 
        if (self.figure) < 9: 
            return 0 
        else: 
            return self.figure - 8 
 
    def __str__(self): 
        return carte.figures[self.figure] + " de " + carte.couleurs[self.couleur] + " valeur:" + str(self.valeur()) 
 
 
def random_combination(p, r): 
    "Random selection from itertools.combinations(iterable, r)" 
    iterable = range(p) 
    pool = tuple(iterable) 
    n = len(pool) 
    indices = sorted(random.sample(range(n), r)) 
    return tuple(pool[i] for i in indices) 
 
 
class mainB():  # modélisation d'une main de 13 cartes - model for a hand of 13 cards 
    def __init__(self): 
        L = random_combination(52, 13) 
        self.cartes = [carte(i) for i in L] 
 
    def difCol(self): #nombre de couleurs différentes dans la main - number of distinct suits in the hand 
        L=[c.couleur for c in self.cartes] 
        S=set(L) 
        return len(S) 
 
    def strlist(self): 
        return [str(c) for c in self.cartes] 
 
    def __str__(self): 
        return "\n".join(self.strlist()) 
 
    def valMain(self): 
        return sum(c.valeur() for c in self.cartes) 
 
 
def calFreqVal (k1,k2, repeat): 
    count=0 
    for i in range(repeat): 
        uneMain=mainB() 
        if uneMain.valMain()>=k1 and uneMain.valMain()<=k2 : 
            count+=1 
    return count/repeat 
 
def calFreqDifCol(k1,k2, repeat): 
    count=0 
    for i in range(repeat): 
        uneMain=mainB() 
        if uneMain.difCol()>=k1 and uneMain.difCol()<=k2: 
            count+=1 
    return count/repeat 
 
def calFreqValEtDifCol(v1,v2,c1,c2,repeat): 
    count=0 
    for i in range(repeat): 
        uneMain=mainB() 
        if uneMain.valMain()>=v1 and uneMain.valMain()<=v2  and uneMain.difCol()>=c1 and uneMain.difCol()<=c2: 
            count+=1 
    return count/repeat
