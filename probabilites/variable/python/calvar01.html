import math 
 
 
class UniversFini: 
    def __init__(self, L): 
        """initialisation avec une liste de probabilités de somme 1 - initialization with a list of probabilities with total sum 1""" 
        self.P = L 
        self.n = len(L) 
 
 
class Variable: 
    def __init__(self, L): 
        """Initialisation à partir d'une liste de valeurs - constructor from a list of values""" 
        self.X = L 
 
    def Esperance(self, U): 
        """Calcule l'espérance de X - computes the mathematical expectation of X """ 
        return sum([self.X[i] * U.P[i] for i in range(0, U.n)]) 
 
    def Variance(self, U): 
        """Calcule la variance de X - computes the variance of X""" 
        E = self.Esperance(U) 
        Q = [(E - x) * (E - x) for x in self.X] 
        V = Variable(Q) 
        return V.Esperance(U) 
 
 
def main(): 
    """Fonction principale - main function""" 
    m = 36.0 
    L1 = [i / m for i in range(1, 7)] 
    L2 = [i / m for i in range(5, 0, -1)] 
    L = L1 + L2 
    U = UniversFini(L) 
    V = range(2, 13) 
    X = Variable(V) 
    print(X.Esperance(U)) 
    print(X.Variance(U)) 
 
 
if __name__ == '__main__': 
    main()
