""" Problème du Chevalier de Méré  - Problem of Chevalier de Méré
"""
import random


def epreuve():  # on lance 4 dés - we roll 4 dices
    lancer = (random.randint(1, 6), random.randint(1, 6), random.randint(1, 6), random.randint(1, 6))
    return 6 in lancer  # succès si on a un 6 sur les 4 dés, échec dans le cas contraire - success if we have a 6 failure if contrary


def repepreuve(n):
    count = 0
    for i in range(0, n):
        if epreuve():
            count += 1
    return (count / n)


def main():
    compteur = 0  # on estime la fréquence des cas où sur 1000 lancer la fréquence d'apparition du 6 est moindre que 1/2 - we stimate the frequency of cases when on 1000 rolls the frequency of appearence of 6 is less than 1/2
    for i in range(0, 1000):
        if repepreuve(1000) < 0.5:
            compteur += 1
    print(compteur / 1000)


if __name__ == '__main__':
    main()
