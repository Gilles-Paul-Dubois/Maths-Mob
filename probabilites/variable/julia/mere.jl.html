function epreuve()
    lancer=[rand(1:6),rand(1:6),rand(1:6),rand(1:6)]
    return 6 in lancer
end

function repepreuve(n)
    count=0
    for i in 1:n
        if epreuve()
            count+=1
        end
    end
    return count/n
end

function main()
    compteur=0
    for i in 1:1000
        if repepreuve(1000)<0.5
            compteur += 1
        end
    end
    println(compteur/1000)
end
        
main()
