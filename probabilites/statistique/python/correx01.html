import numpy as np 
import random 
 
 
def main(): 
    N1 = [random.randint(0, 20) for i in range(0, 30)] 
    N2 = [random.randint(0, 20) for i in range(0, 30)] 
    c = np.cov(N1, N2, bias=1)[1][0] 
    print(c) 
    S = [N1[i] * N2[i] for i in range(0, 30)] 
    print(np.mean(S) - np.mean(N1) * np.mean(N2)) 
 
 
if __name__ == '__main__': 
    main()
