# -*- coding: utf-8 -*-
# python version 3.4 
import random 
 
 
def GenereSerieAlea(n, p): 
    """génère une série statistique aléatoire de n entiers entre 0 et p""" 
    """generates a random data series of n integers between 0 and p"""    
    return [random.randint(0, p) for i in range(0, n)] 
 
 
def EcAbs(L): 
    """Calcul de l'écart absolu moyen """ 
    """Computes absolute mean deviation """    
    m = sum(L, 0.0) / len(L) 
    M = [abs(x - m) for x in L] 
    return sum(M, 0.0) / len(M) 
 
 
def main(): 
    """Fonction principale - main function""" 
    L = GenereSerieAlea(30, 20) 
    print(L) 
    print(EcAbs(L)) 
 
 
if __name__ == "__main__": 
    main()
