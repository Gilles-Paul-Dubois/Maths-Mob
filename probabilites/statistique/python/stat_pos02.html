import random 
 
"""Calcul de valeurs médianes""" 
"""Computation of median values""" 
 
 
def GenereSerieAlea(n, p): 
    """génère une série statistique aléatoire de n entiers entre 0 et p""" 
	 """generates a random series of n integers between 0 and p"""    
    return [random.randint(0, p) for i in range(0, n)] 
 
 
def Mediane(L): 
    """Calcul de la médiane sur liste triée - Computation of the median on sorted list""" 
    n = len(L) 
    if n % 2:  # n impair - if n odd 
        return L[n // 2 + 1] 
    else: 
        return (L[n // 2] + L[n // 2 + 1]) / 2.0 
 
 
def main(): 
    """Fonction principale - Main function""" 
    n = 31 
    L = GenereSerieAlea(n, 20) 
    L.sort() 
    print(L) 
    print(Mediane(L)) 
    n = 30 
    L = GenereSerieAlea(n, 20) 
    L.sort() 
    print(L) 
    print(Mediane(L)) 
 
 
if __name__ == "__main__": 
    main()
