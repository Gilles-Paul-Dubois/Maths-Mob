import random 
 
 
def GenereSerieAlea(n, p): 
    """génère une série statistique aléatoire de n entiers entre 0 et p""" 
    """generates random data of n integer values between 0 and p"""    
    return [random.randint(0, p) for i in range(0, n)] 
 
 
def CalculeEffClass(L, a, b): 
    """Calcule l'effectif de la classe correspondant à l'intervalle [a,b[""" 
    """Ccomputes the tally of the class corresponding to the interval [a,b["""    
    return len([x for x in L if (x >= a and x < b)]) 
 
 
def main(): 
    """Fonction principale - main function""" 
    L = GenereSerieAlea(30, 40) 
    print(L) 
    print(CalculeEffClass(L, 15, 25)) 
 
 
if __name__ == '__main__': 
    main()
