import random 
 
 
def GenereSerieAlea(n, p): 
    """génère une série statistique aléatoire de n entiers entre 0 et p""" 
    """generates a random sequence of n integers between 0 and p"""     
    return [random.randint(0, p) for i in range(0, n)] 
 
 
def ElimineDoublons(L): 
    """Comme son nom l'indique""" 
    """Eliminates doubles"""     
    return list(set(L)) 
 
 
def CalculeEffectifs(L): 
    """Calcule l'effectif de chaque modalité""" 
    """Computes the count of each value-point"""    
    M = ElimineDoublons(L) 
    return [[x, L.count(x)] for x in M] 
 
 
def main(): 
    """Fonction principale - main function""" 
    L = GenereSerieAlea(30, 20) 
    print(L) 
    print(CalculeEffectifs(L)) 
 
 
if __name__ == "__main__": 
    main()
