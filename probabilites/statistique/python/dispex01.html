import random 
import math 
 
"""Calcul de variance, ecart-type""" 
"""Calculation of variance, standard deviation""" 
 
 
def GenereSerieAlea(n, p): 
    """génère une série statistique aléatoire de n entiers entre 0 et p""" 
    """generates a random statistical series of n integers between 0 and p"""    
    return [random.randint(0, p) for i in range(0, n)] 
 
 
def Moyenne(L): 
    """Calcul de la moyenne - Computation of mean """ 
    return sum(L, 0.0) / len(L) 
 
 
def Variance(L): 
    """Calcul de la variance - computation of variance""" 
    M = [x * x for x in L] 
    m = Moyenne(L) 
    k = Moyenne(M) 
    return k - m * m 
 
 
def EcartType(L): 
    """ecart-type - standard deviation""" 
    return math.sqrt(Variance(L)) 
 
 
def EcartTypeRelatif(L): 
    """ La moyenne doit être non nulle - mean should not be zero""" 
    return EcartType(L) / Moyenne(L) 
 
 
def main(): 
    """Fonction principale - main function""" 
    L = GenereSerieAlea(30, 20) 
    print(L) 
    print(EcartType(L)) 
    print(EcartTypeRelatif(L)) 
 
 
if __name__ == '__main__': 
    main()
