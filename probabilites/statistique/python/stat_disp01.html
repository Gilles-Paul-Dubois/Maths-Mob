import random 
 
 
def GenereSerieAlea(n, p): 
    """génère une série statistique aléatoire de n entiers entre 0 et p""" 
    """generates a random data series of n integers between 0 and p"""    
    return [random.randint(0, p) for i in range(0, n)] 
 
 
def CalculeMax(L): 
    """Calcule la plus grande modalité - computes the greatest modality""" 
    Max = L[0] 
    for x in L[1:]: 
        if x > Max: 
            Max = x 
    return Max 
 
 
def CalculeMin(L): 
    """Calcule la plus petite modalité - computes the samllest modality""" 
    Min = L[0] 
    for x in L[1:]: 
        if x < Min: 
            Min = x 
    return Min 
 
 
def CalculeEtendue(L):  # premiere version - first version 
    """Calcule l'étendue ou l'amplitude de L - computes the range of L""" 
    return CalculeMax(L) - CalculeMin(L) 
 
 
def CalculeAmplitude(L):  # seconde version plus efficace - second version with better efficiency 
    """Calcule l'étendue ou l'amplitude en un seul balayage de la liste - computes the range in only one path""" 
    Min = Max = L[0] 
    for x in L[1:]: 
        if x < Min: 
            Min = x 
        if x > Max: 
            Max = x 
    return Max - Min 
 
 
def main(): 
    """Fonction principale- main function""" 
    L = GenereSerieAlea(30, 40) 
    print(L) 
    print(CalculeMax(L)) 
    print(CalculeMin(L)) 
    print(CalculeEtendue(L)) 
    print(CalculeAmplitude(L)) 
 
 
if __name__ == "__main__": 
    main()
