import random 
 
"""Calcul de moyenne - computation of mean value""" 
 
 
def GenereSerieAlea(n, p): 
    """génère une série statistique aléatoire de n entiers entre 0 et p""" 
    """generates a random statistical series of n integhers between 0 and p"""    
    return [random.randint(0, p) for i in range(0, n)] 
 
 
def Moyenne(L): 
    """Calcul de la moyenne - compuation of the mean valuez""" 
    n = len(L) 
    s = 0.0 
    for x in L: 
        s += x 
    return s / n 
 
 
def main(): 
    """Fonction principale - main function""" 
    L = GenereSerieAlea(30, 20) 
    print(L) 
    print(Moyenne(L)) 
 
 
if __name__ == "__main__": 
    main()
