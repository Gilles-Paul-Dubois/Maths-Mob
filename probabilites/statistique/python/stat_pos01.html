import random 
 
"""Calcul de valeurs modales - calculation of modal values""" 
 
 
def GenereSerieAlea(n, p): 
    """génère une série statistique aléatoire de n entiers entre 0 et p""" 
    """generates a random statistical series of n integers between  0 and p"""     
    return [random.randint(0, p) for i in range(0, n)] 
 
 
def ElimineDoublons(L): 
    """Comme son nom l'indique - eliminates multiple occurences""" 
    return list(set(L)) 
 
 
def CalculeEffectifs(L): 
    """Calcule l'effectif de chaque modalité - computes the tally for every modality""" 
    M = ElimineDoublons(L) 
    return [[x, L.count(x)] for x in M] 
 
 
def main(): 
    """Fonction principale - main function""" 
    L = GenereSerieAlea(30, 20) 
    print(L) 
    M = CalculeEffectifs(L) 
    Triee = sorted(M, key=lambda x: -x[1])  # trie le résultat suivant les effectifs décroissants - sorts the result according decreasng tallies 
    print(Triee)  # les valeurs modales apparaissent en tête - modal values appear first 
 
 
if __name__ == "__main__": 
    main()
