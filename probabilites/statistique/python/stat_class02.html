"Valeurs d'une fonction de répartition - values of a cumulative distribution function" 
 
 
def F(x, L): 
    """Fonction de répartition - cumulative distribution function""" 
    n = len(L) 
    s = 0.0 
    for u in L: 
        if u <= x: 
            s += 1 
    return s / n 
 
 
def main(): 
    """Fonction principale - main function""" 
    L = [11, 11.5, 8, 11, 8.5, 7.5, 14.5, 9, 13, 6.5, 6.5, 11, 9, 12, 8.5, 5.5, 8, 15, 12.5, 11, 10.5, 11.5, 10.5, 7, 8, 
         9.5, 8.5, 14, 9, 12, 12, 15, 10, 3.5, 11, 11.5, 7.5, 10.5, 6.5, 4] 
    M = list(set(L))  # élimination des doublons - elimination of doubles 
    M.sort()  # tri croissant 
    print(M) 
    for x in M: 
        print(x, F(x, L)) 
 
 
if __name__ == "__main__": 
    main()
