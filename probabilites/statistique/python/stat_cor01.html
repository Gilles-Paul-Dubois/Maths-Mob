import random 
 
"""Calcul de covariance - calculation of covariance""" 
 
 
def GenereSerieAlea(n, p): 
    """génère une série statistique aléatoire de n entiers entre 0 et p""" 
    """generates a random series of n integers between 0 and p"""    
    return [random.randint(0, p) for i in range(0, n)] 
 
 
def Moyenne(L): 
    """Calcul de la moyenne- computation of the average """ 
    return sum(L, 0.0) / len(L) 
 
 
def Covariance(L1, L2): 
    m1 = Moyenne(L1) 
    m2 = Moyenne(L2) 
    M = [(L1[i] - m1) * (L2[i] - m2) for i in range(0, len(L1))] 
    return Moyenne(M) 
 
 
def main(): 
    """Fonction principale- main function""" 
    L1 = GenereSerieAlea(30, 20) 
    L2 = GenereSerieAlea(30, 20) 
    print(Covariance(L1, L2)) 
 
 
if __name__ == "__main__": 
    main()
