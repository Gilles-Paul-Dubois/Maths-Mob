genere_serie_alea(n,p)=[rand(0:p) for i in 1:n]

moyenne(L)=reduce(+,L)/length(L)

function covariance(L1,L2)
    m1,m2=moyenne(L1),moyenne(L2)
    M=[(L1[i]-m1)*(L2[i]-m2) for i in 1:length(L1)]
    return moyenne(M)
end

function variance(L)
    M=[x*x for x in L]
    return moyenne(M) - moyenne(L)*moyenne(L)
end

ecart_type(L)=sqrt(variance(L))

pearson(L1,L2)=covariance(L1,L2)/(ecart_type(L1)*ecart_type(L2))

L1=genere_serie_alea(30,20)
L2=genere_serie_alea(30,20)
println(pearson(L1,L2))
