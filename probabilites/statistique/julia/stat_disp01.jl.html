using Statistics
genere_serie_alea(n,p)=[rand(0:p) for i in 1:n]

function calcule_max(L)
    Max=L[1]
    for x in L
        if x>Max
            Max=x
        end
    end
    return Max
end

function calcule_min(L)
    Min=L[1]
    for x in L
        if x<Min
            Min=x
        end
    end
    return Min
end

E1(L)=calcule_max(L)-calcule_min(L)     

function E2(L)
    Min=Max=L[1]
    for x in L
        if x<Min
            Min=x
        end
        if x>Max
            Max=x
        end
    end
    return Min,Max,Max-Min
end

L=genere_serie_alea(30,40)
println(L)
println(calcule_max(L))
println(calcule_min(L))
println(E1(L))
println(E2(L))
