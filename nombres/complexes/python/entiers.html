# division euclidienne d'un entier de Gauss par un autre 
# Euclidean division of a Gauss integer by another one
 
import math 
 
def Norme(z): 
    """Il ne s'agit pas d'une norme euclidienne ici""" 
    """It's not an Euclidean norm here"""     
    return z.real*z.real+z.imag*z.imag 
 
def FourCorners (z1,z2): 
    """Détermine les 4 solutions possibles pour le quotient euclidien""" 
    """Finds out the four possible solutions for the Euclidean quotient"""    
    q=z1/z2 
    x=q.real 
    y=q.imag 
    n=math.floor(x) 
    m=math.floor(y) 
    return complex(n,m),complex(n+1,m),complex(n+1,m+1),complex(n,m+1) 
 
def main(): 
    u=-36+242j 
    v=50+50j 
    L=FourCorners(u,v) 
    S=[z for z in L if Norme(u-z*v)<Norme(v)] 
    print(S) 
 
if __name__ == '__main__': 
    main()
