# Solution de l'équation du second degré 
# Solving quadratics 
import math 
 
class MonComplexe (): 
    """ modélisation des nombres complexes - class to represent complex numbers """ 
    def __init__(self,a=0,b=0): 
        """le constructeur  - the constructor""" 
        self.x=a 
        self.y=b 
 
    def __str__(self): 
        """ représentation externe pour print et str - conversion to string """ 
        if self.y==0: 
            return str(self.x) 
        if self.x==0 and self.y==1: 
            return "i" 
        if self.x==0 and self.y==-1: 
            return "-i" 
        if self.x==0: 
            return str(self.y)+"i" 
        if self.y==1: 
            return str(self.x)+"+i" 
        if self.y==-1: 
            return str(self.x)+"-i" 
        if self.y>0: 
            return str(self.x)+"+"+str(self.y)+"i" 
        else: 
            return str(self.x)+str(self.y)+"i" 
 
    def __add__(self,other): 
        """somme de deux complexes - sum""" 
        a=self.x+other.x 
        b=self.y+other.y 
        return MonComplexe(a,b) 
 
    def __sub__(self,other): 
        """différence de deux complexes - difference""" 
        a=self.x-other.x 
        b=self.y-other.y 
        return MonComplexe(a,b) 
 
    def null(self): 
        """test de nullité""" 
        return self.x==0 and self.y==0 
 
    def __mul__(self,other): 
        """produit de deux complexes - product""" 
        a=self.x*other.x-self.y*other.y 
        b=self.x*other.y+self.y*other.x 
        return MonComplexe(a,b) 
 
    def __truediv__(self,other): 
        """quotient de deux complexes - quotient""" 
        return self*(~other) 
 
    def conj(self): 
         """conjugué d'un complexe- conjugate""" 
         return MonComplexe(self.x,-self.y) 
    def module(self): 
        """module d'un complexe - modulus""" 
        return math.sqrt(self.x*self.x+self.y*self.y) 
    def argument(self): 
        """argument d'un complexe - angle""" 
        if self.x==0 and self.y==0: 
            return 0 
        if self.x==0 and self.y>0: 
            return math.pi/2 
        if self.x==0 and self.y<0: 
            return -math.pi/2 
        if self.x >0: 
            return math.atan(self.y/self.x) 
        return math.pi-math.atan(self.y/(-self.x)) 
 
    def __invert__(self): 
        """inverse d'un complexe - inverse""" 
        if self.null(): 
            raise ZeroDivisionError 
        return MonComplexe(self.x/(self.x*self.x+self.y*self.y),-self.y/(self.x*self.x+self.y*self.y)) 
 
    def __pow__(self,n): 
        """puissances d'un complexe - powers""" 
        if n<=0 and self.null(): 
            raise ZeroDivisionError 
        if n >=0: 
            R=MonComplexe(1,0) 
            while n: 
                R=R*self 
                n=n-1 
            return R 
        if n<0: 
            return (~R)**(-n) 
 
    def racine(self): 
        """racine d'un nombre complexe (une seule sur deux)- root of a complex""" 
        a=self.argument() 
        r=self.module() 
        x=math.sqrt(r)* math.cos(a/2.0) 
        y=math.sqrt(r)* math.sin(a/2.0) 
        return MonComplexe(x,y) 
 
    def __neg__(self): 
        """opposé d'un complexe - opposite""" 
        return MonComplexe(-self.x,-self.y) 
 
def Delta (a,b,c): 
    """discriminant d'une équation - discriminant """ 
    return b*b-MonComplexe(4,0)*a*c 
 
def Solutions(a,b,c): 
    """Calcule les deux racines de l'équation du second degré - solves the quadratic""" 
    D=Delta(a,b,c) 
    z1=D.racine() 
    n1=(-b-z1)/(a*MonComplexe(2.0,0)) 
    n2=(-b+z1)/(a*MonComplexe(2.0,0)) 
    return n1,n2 
 
# fonction de test - test function 
def main(): 
    a,b,c=MonComplexe(1,0),MonComplexe(2,0),MonComplexe(3,0) 
    S1,S2= Solutions(a,b,c) 
    print(S1) 
    print(S2) 
    print(a*S1*S1+b*S1+c) # exact à 10^-16 près - acuurate 10^-16 close  
    print(a*S2*S2+b*S2+c) 
 
if __name__ == '__main__': 
    main()
