# ébauche d'un objet pour la représentation des nombres complexes 
# draft of an object to represent complex numbers
import math 
 
class MonComplexe (): 
    """ modélisation des nombres complexes """ 
    """ modelization of complex numbers """    
    def __init__(self,a=0,b=0): 
        """le constructeur soit en cartésiennes soit en polaires""" 
        """constructor either in Cartesian form or in polar form"""         
        self.x=a 
        self.y=b 
 
    def __str__(self): 
        """ représentation externe pour print et str """ 
        """ external representation as a string """         
        if self.y==0: 
            return str(self.x) 
        if self.x==0 and self.y==1: 
            return "i" 
        if self.x==0 and self.y==-1: 
            return "-i" 
        if self.x==0: 
            return str(self.y)+"i" 
        if self.y==1: 
            return str(self.x)+"+i" 
        if self.y==-1: 
            return str(self.x)+"-i" 
        if self.y>0: 
            return str(self.x)+"+"+str(self.y)+"i" 
        else: 
            return str(self.x)+str(self.y)+"i" 
 
    def __add__(self,other): 
        """somme de deux complexes - sum of two complexes""" 
        a=self.x+other.x 
        b=self.y+other.y 
        return MonComplexe(a,b) 
 
    def __sub__(self,other): 
        """différence de deux complexes - difference of ..""" 
        a=self.x-other.x 
        b=self.y-other.y 
        return MonComplexe(a,b) 
 
    def __neg__(self): 
        """opposé d'un complexe - opposite""" 
        return MonComplexe(-self.x,-self.y) 
 
    def null(self): 
        """test de nullité - nullity test""" 
        return self.x==0 and self.y==0 
 
    def __mul__(self,other): 
        """produit de deux complexes - product""" 
        a=self.x*other.x-self.y*other.y 
        b=self.x*other.y+self.y*other.x 
        return MonComplexe(a,b) 
 
    def __truediv__(self,other): 
        """quotient de deux complexes - quotient""" 
        return self*(~other) 
 
    def conj(self): 
         """conjugué d'un complexe - conjugate""" 
         return MonComplexe(self.x,-self.y) 
 
    def module(self): 
        """module d'un complexe - modulus""" 
        return math.sqrt(self.x*self.x+self.y*self.y) 
    def argument(self): 
        """argument d'un complexe - argument""" 
        if self.x==0 and self.y==0: 
            return 0 
        if self.x==0 and self.y>0: 
            return math.pi/2 
        if self.x==0 and self.y<0: 
            return -math.pi/2 
        if self.x >0: 
            return math.atan(self.y/self.x) 
        return math.pi-math.atan(self.y/(-self.x)) 
 
    def __invert__(self): 
        """inverse d'un complexe - inverse""" 
        if self.null(): 
            raise ZeroDivisionError 
        return MonComplexe(self.x/(self.x*self.x+self.y*self.y),-self.y/(self.x*self.x+self.y*self.y)) 
 
    def __pow__(self,n): 
        """puissances d'un complexe - powers""" 
        if n<=0 and self.null(): 
            raise ZeroDivisionError 
        if n >=0: 
            R=MonComplexe(1,0) 
            while n: 
                R=R*self 
                n=n-1 
            return R 
        if n<0: 
            return (~R)**(-n)
