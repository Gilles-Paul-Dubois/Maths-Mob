z1=2+im
z2=3-2im
#opérations classiques - usual operations
println(-z1)
println(z1+z2)
println(z1-z2)
println(z1*z2)
println(inv(z1))
println(z1^2)
#forme polaire - polar form
println(abs(z1)) #module - modulus
println(angle(z1)) #argument
#accès aux composantes - access to real and imaginary parts
println(real(z1))
println(imag(z1))
# conjugué - conjugate
println(conj(z1))
