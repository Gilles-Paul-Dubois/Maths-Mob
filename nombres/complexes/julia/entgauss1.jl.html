norme(z::Complex)=real(z)*real(z)+imag(z)*imag(z)

function fourcorners(z1::Complex,z2::Complex)
    q=z1/z2
    x=real(q)
    y=imag(q)
    n=Int(floor(x))
    m=Int(floor(y))
    return n+m*im,n+1+m*im, n+(m+1)*im,(n+1)+(m+1)*im
end

function diveuc(z1::Complex, z2::Complex)
    L=fourcorners(z1,z2)
    S=[z for z in L if norme(z)<norme(z2)]
    return S
end

function main() #fonction principale
    z1=36+242im
    z2=20+30im
    println(diveuc(z1,z2))
end

main() #appel de main
