# génération des suites [1,2,...,2] 
# generation of sequences [1,2,...,2] 
def suite (n): 
    return [1]+[2]*(n-1) 
 
#décomposition de la fraction p/q en fraction continue 
#decomposition of fraction p/q in continued fraction 
def fracontinue (p,q): 
    sol=[] 
    p0=p 
    p1=q 
    while p1: 
        a=p0//p1 
        sol.append(a) 
        p=p1 
        p1=p0-a*p1 
        p0=p 
    return sol 
 
#restitution d'une fraction à partir de sa décomposition 
#restitution of a fraction from its decomposition 
def restitue(L): 
    if len(L)==1: 
        return [L[0],1] 
    f=restitue(L[1:]) 
    g=[f[1],f[0]] #inverse de f 
    return [L[0]*g[1]+ g[0], g[1]] 
 
#approximation de racine(2) par des rationnels. 
#approximation of square root of 2 by rationals. 
def main(): 
    for i in range (2,20): 
        f=restitue(suite(i)) 
        x=float(f[0]) 
        y=float(f[1]) 
        print((f,x/y)) 
 
if __name__ == '__main__': 
    main()
