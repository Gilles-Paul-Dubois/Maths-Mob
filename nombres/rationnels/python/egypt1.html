# décomposition de f fraction inférieure à 1 en somme de fractions egyptiennes
# decomposition of f fraction less than 1 into a sum of Egyptian rfactions
# par un algorithme 'glouton' - with greedy algorithm of Fibonacci-Sylvester
# le résultat ne fait apparaître que les dénominateurs des fractions egyptiennes
# Displays only denominators of Egyptian fractions
def egypt(x, y):
    L = []
    k = 1
    while x:
        if y % x == 0:
            k = y // x
        else:
            k = y // x + 1
        x = x * k - y
        y = y * k
        L.append([k])
    return L


print(egypt(5, 121))
