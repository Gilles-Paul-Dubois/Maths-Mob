# écriture des fractions (positives) comme développements décimaux périodiques 
# Expression of positive fractions as periodical decimal expansion 
import os 
 
def periode (p,q): 
    e=p//q # partie entière - integral part 
    p=p%q 
    R=[p] # on ajoute le premier reste - we add the first remainder 
    Q=[] 
    r=p 
    while (True): # boucle interrompue par 'break'-Infinite loop with interruption by 'break' 
        d=(10*r)//q 
        r=(10*r)%q 
        if (r in R): # si le reste est déja apparu - If the remainder already appeared  
            x=R.index(r) # on repère à quel endroit - we check in which position 
            Q.append(d) # on ajoute le dernier diviseur -we add the last divisor 
            break # on s'arrête -we stop 
        else: 
            R.append(r) # on ajoute le nouveau reste - we add the new remainder 
            Q.append(d) # idem pour le nouveau quotient - same for the new quotient 
    # on fabique maintenant la chaîne formatée - we now build the formatted string 
    S=str(e)+'.' # partie entière - integral part 
    for i in Q: # début du développement décimal- beginning of the decimal expanson 
        S=S+str(i) 
    for j in Q[x:]:# on répète une fois la période - we repeat the period one time 
        S=S+str(j) 
    for j in range(x,len(Q)): # répétition de la période à l'infini - Infinite repetition of the period 
        S=S+'-'        
    return S 
 
#programme principal - main function 
# développement décimal périodique du rationnel 17125/9990 - periodical decimal expansion of 17125/9990 
print(periode(17125,9990))
