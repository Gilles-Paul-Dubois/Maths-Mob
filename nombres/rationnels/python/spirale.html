def squarodd(k): 
    return (2*k+1)*(2*k+1) 
 
#détermine sur quel carré le point d'indice n se trouve 
#determines on which square point of index n is located  
def Trouve_k(n): 
    i=0 
    while squarodd(i) <=n: 
        i+=1 
    return i 
 
#détermine l'abscisse absolue sur le carré à partir du coin inférieur gauche 
#determines absolute abscissa on the square starting from left lower corner 
def Trouve_h(n): 
    k=Trouve_k(n) 
    return (8*k+n-squarodd(k)+1)%(8*k) 
 
# calcule le décalage par rapport à ce même point 
# computes offset relative to this same point
def Offset(k,h): 
    q=h//2*k 
    r=h-q*2*k 
    if q==0: 
        return [0,r] 
    if q==1: 
        return [r,2*k] 
    if q==2: 
        return [2*k, 2*k-r] 
    else: 
        return [2*k-r,0] 

# Calcule les coordonnées du point P d'indice n de la spirale
# computes coordinates of point P with index n on the spiral
def P(n): 
    k,h=Trouve_k(n),Trouve_h(n) 
    O=Offset(k,h) 
    return [-k+O[0],-k+O[1]] 
 
#fonction principale - main function 
def main(): 
    for n in range(1,25): 
        print(P(n)) 
 
if __name__ == '__main__': 
    main()
