# Ce programme restitue une fraction à partir de son développement décimal périodique 
# This program finds a fraction starting from its decimal expansion 
 
 
# calcule le pgcd de deux nombres 
# computes the gcd of two integers 
def pgcd (m,n): 
    while (m%n): 
        t=m 
        m=n 
        n=t%n 
    return n 
 
# nombres de chiffres de n en base 10 
# number of digits of n using base 10 
def nbc(n): 
    if n<10: 
        return 1 
    else: 
        return 1+nbc(n//10) 
 
#restitue la fraction ayant le développement e,(p) 
#rebuilds the fraction with expansion e,(p)
# e partie entière 
# e integral part 
# (p) période 
# (p) period 
def restitue1(e, p): 
    b = nbc(p) 
    m = e * (10 ** b - 1) + p 
    n = (10 ** b - 1) 
    D = pgcd(m, n) 
    return [m//D, n//D] 
 
 
# restitue la fraction ayant le dévelopement e,d(p)
# rebuilds the fraction with expansion e,d(p) 
#e partie entière - integral part  
# d : décimales avant la période - decimals before period 
#(p) période - period 
#exemple 2,78141414  e=2 d=78 p=14 
def restitue2(e,d,p): 
    a=nbc(d) 
    b=nbc(p) 
    m=e*(10**(2*a))*(10**b-1)+d*(10**b-1)*(10**a)+p*(10**a) 
    n=(10**(2*a))*(10**b-1) 
    D=pgcd(m,n) 
    return[m//D,n//D] 
 
 
def main(): 
    print(restitue1(1,1234)) 
    print(restitue2(2,27,45)) 
 
if __name__ == '__main__': 
    main()
