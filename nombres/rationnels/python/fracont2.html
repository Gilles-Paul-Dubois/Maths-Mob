#décomposition de la fraction p/q en fraction continue 
#decomposition of fraction p/q i continued fraction  
def fracontinue (p,q): 
    sol=[] 
    p0=p 
    p1=q 
    while p1: 
        a=p0//p1 
        sol.append(a) 
        p=p1 
        p1=p0-a*p1 
        p0=p 
    return sol 
 
#restitution d'une fraction à partir de sa décomposition 
#rebuilding of fraction from decomposition 
def restitue(L): 
    if len(L)==1: 
        return [L[0],1] 
    f=restitue(L[1:]) 
    g=[f[1],f[0]] #inverse de f 
    return [L[0]*g[1]+ g[0], g[1]] 
 
def main(): 
    p,q=18,31 
    L= fracontinue(p,q) 
    print(L) 
    print(restitue(L)) 
 
if __name__ == '__main__': 
    main()
