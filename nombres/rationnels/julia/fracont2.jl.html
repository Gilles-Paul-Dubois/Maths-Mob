function fracontinue(p,q)
    sol,p0,p1=[],p,q
    while p1!=0
        a=p0Ã·p1
        push!(sol,a)
        p=p1
        p1=p0-a*p1
        p0=p
    end
    return sol
end

function restitue(L)
    if length(L)==1
        return[L[1],1]
    end
    f=restitue(L[2:end])
    g=[f[2],f[1]]
    return [L[1]*g[2]+g[1],g[2]]
end
