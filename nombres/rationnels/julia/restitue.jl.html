function pgcd(m,n)
    while m%n!=0
        t=m
        m=n
        n=t%n
    end
    return n
end

nbc(n)=n<10 ? 1 : 1+nbc(n÷10)

function restitue1(e,p)
    b=nbc(p)
    n=10^b-1
    m=e*n+p
    D=pgcd(m,n)
    return m÷D,n÷D
end

function restitue2(e,d,p)
    a,b=nbc(d),nbc(p)
    n=(10^(2a))*(10^b-1)
    m=n+d*(10^b-1)*(10^a)+p*(10^a)
    D=pgcd(m,n)
    return m÷D,n÷D
end

println(restitue1(1,1234))
println(restitue2(2,27,45))
