function periode(p,q)
    e=p÷q
    p=p%q
    R,Q,r,x=[p],[],p,0
    while true
        d=(10*r)÷q
        r=(10*r)%q
        if r in R
            x=indexin(r,R)[1]
            push!(Q,d)
            break
        else
            push!(R,r)
            push!(Q,d)
        end
    end
    S=string(string(e),".")
    for i in Q
        S=string(S,string(i))
    end
    for j in Q[x:end]
        S=string(S,string(j))
    end
    for j in x:length(Q)
        S=string(S,"-")
    end
    return S
end

println(periode(17125,9990))
