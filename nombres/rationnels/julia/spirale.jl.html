squarodd(k)=(2k+1)*(2k+1)

function trouve_k(n)
    i=0
    while squarodd(i)<=n
        i+=1
    end
    return i
end

function trouve_h(n)
    k=trouve_k(n)
    return (8k+n-squarodd(k)+1)%(8k)
end

function offset(k,h)
    q=h√∑(2*k)
    r=h-2q*k
    if q==0
        return 0,r
    elseif q==1
        return r,2k
    elseif q==2
        return 2k,2k-r
    else
        return 2k-r,0
    end
end

function P(n)
    k,h=trouve_k(n),trouve_h(n)
    o=offset(k,h)
    return -k+o[1],-k+o[2]
end

for n in 1:24
    println(P(n))
end
