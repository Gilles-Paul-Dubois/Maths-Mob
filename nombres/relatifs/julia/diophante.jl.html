reste(a,b)= a%b>=0 ? a%b : b+a%b 
quot(a,b) =(a-reste(a,b))÷b 

function Bezout(a,b)
    s,t,u,v=1,0,0,1
    while b!=0
        q=quot(a,b)
        a, s, t, b, u, v = b, u, v, a - q * b, s - q * u, t - q * v
    end
    return a>0 ? (a,s,t) : (-a, -s, -t)
end

function Diophante(a,b,c,n)
    s=Bezout(a,b)
    d=s[1] 
    if c%d!=0
        return nothing
    end
    sp=[s[2]*c÷d,s[3]*c÷d]
    sph=[b÷d,-a÷d]
    S=[(sp[1]+k*sph[1],sp[2]+k*sph[2]) for k in -n:n]
    return S
end


println(Diophante(29,-11,5,10))
