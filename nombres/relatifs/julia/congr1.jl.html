somme(x,y,n)=(x+y)%n
produit(x,y,n)=(x*y)%n

function TableSomme(n)
    T=zeros(Int,n,n)
    for i in 0:n-1
        for j in 0:n-1
            T[i+1,j+1]=somme(i,j,n)
        end
    end
    return T
end

function TableProduit(n)
    T=zeros(Int,n,n)
    for i in 0:n-1
        for j in 0:n-1
            T[i+1,j+1]=produit(i,j,n)
        end
    end
    return T
end

inversible(x,n)=any((produit(x,y,n)==1 for y in 1:n-1))

corps(n)=all((inversible(x,n) for x in 1:n-1))

println( [n for n in 2:20 if corps(n)])
