# rectifier les définitions de julia
# change Julia's definitions of quotient and remainder
reste(a,b)= a%b>=0 ? a%b : b+a%b # définition mathématique- mathematical definition
quot(a,b) =(a-reste(a,b))÷b #idem - the same

function Bezout(a,b)
    s,t,u,v=1,0,0,1
    while b!=0
        q=quot(a,b)
        a, s, t, b, u, v = b, u, v, a - q * b, s - q * u, t - q * v
    end
    return a>0 ? (a,s,t) : (-a, -s, -t)
end

println(Bezout(21,8))
