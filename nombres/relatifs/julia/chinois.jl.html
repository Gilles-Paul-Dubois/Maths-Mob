function bezout(a,b)
    s,t,u,v=1,0,0,1
    while b!=0
        q=a÷b
        a, s, t, b, u, v = b, u, v, a - q * b, s - q * u, t - q * v
    end
    return a>0 ? (a,s,t) : (-a, -s, -t)
end

function solution(Restes,Modulos)
    N=reduce(*, Modulos)
    M=[N÷n for n in Modulos]
    B=[bezout(Modulos[i],M[i]) for i in 1:length(Modulos)]
    V=[s[3] for s in B]
    E=[V[i]*M[i] for i in 1:length(Modulos)]
    P=[Restes[i]*E[i] for i in 1:length(Modulos)]
    x=reduce(+,P)
    if x<0
        while x<0
            x+=N
        end
        return x-N
    else
        while x>0
            x-=N
        end
        return x+N
    end
end
