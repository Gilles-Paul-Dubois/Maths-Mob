# Trouve les éléments inversibles dans Z/nZ 
# Finds units in Z/nZ
 
# teste si x est inversible dans l'anneau Z/nZ 
# checks if x is a unit in the ring Z/nZ 
def inversible(x,n): 
    for y in range(1,n): 
        if (x*y)%n==1: 
            return True 
    return False 
 
def main(): 
    n=18 
    print([x for x in range(1,n) if inversible(x,n) ]) 
 
if __name__ == '__main__': 
    main()
