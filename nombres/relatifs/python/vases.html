#opérations licites 
#authorized operations 
def ViderA(N): 
    return [0,N[1]] 
 
def ViderB(N): 
    return [N[0],0] 
 
def RemplirA(N): 
    return [13,N[1]] 
 
def RemplirB(N): 
    return [N[0],31] 
 
def TransvaserAdansB(N): 
    if N[0]+N[1]<=31: 
        return [0,N[1]+N[0]] 
    else: 
        return [N[0]-(31-N[1]),31] 
 
def TransvaserBdansA(N): 
    if N[0]+N[1]<=13: 
        return [N[1]+N[0],0] 
    else: 
        return [13,N[1]-(13-N[0])] 
 
#états possibles suivant N en appliquant les opérations ci-dessus 
#possible sates following N applying the above operations 
def next (N): 
    Result=[ViderA(N),ViderB(N),RemplirA(N),RemplirB(N),TransvaserAdansB(N),TransvaserBdansA(N)] 
    return [x for x in Result if x !=N] 
 
#états possibles après n opérations élémentaires 
#possible states after n elementary operations 
def step(n,N): 
    if n==1: 
        return next(N) 
    Result=[] 
    for x in step(n-1,N): 
        for y in next(x): 
            if y not in Result: 
                Result.append(y) 
    return Result 
 
#état gagnant 
#winning state 
def winsN(N): 
    return N[0]==1 or N[1]==1 
#liste gagnante: 
#winning list: 
def winsL(L): 
    for x in L: 
        if winsN(x): 
            return True 
    return False 
 
# preuve qu'on ne peut parvenir au résultat avec moins de 32 manipulations. 
# Proof that you cannot reach the result with less than 32 operations.
def main(): 
    for n in range(1,33): 
        if winsL(step(n,[0,0])): 
            print(n) 
 
 
if __name__ == '__main__': 
    main()
