from functools import reduce


# Théorème chinois - Chinese theorem

# Trouve la solution - finds the solution 
def Solution(Restes, Modulos):
    N = reduce(lambda x, y: x * y, Modulos)  # produit des ni - product of ni
    M = [N // n for n in Modulos]  # liste des mi - list of mi
    B = [bezout(Modulos[i], M[i]) for i in range(0, len(Modulos))]
    V = [s[2] for s in B]  # coefficients vi
    E = [V[i] * M[i] for i in range(0, len(Modulos))]  # coefficients ei
    P = [Restes[i] * E[i] for i in range(0, len(Modulos))]  # produits des ai*ei products ai*ei 
    x = reduce(lambda x, y: x + y, P)  # solution particulière - special solution
    # détermine la solution positive minimale - finds the minimal positive solution 
    if x < 0:
        while x < 0:
            x += N
        return x - N
    else:
        while x > 0:
            x -= N
        return x + N


def bezout(a, b):
    s, t, u, v = 1, 0, 0, 1
    while b != 0:
        q = a // b
        a, s, t, b, u, v = b, u, v, a - q * b, s - q * u, t - q * v
    return (a, s, t) if a > 0 else (-a, -s, -t)


def main():
    print(Solution([3, 4, 5], [17, 11, 6]))


if __name__ == '__main__':
    main()
