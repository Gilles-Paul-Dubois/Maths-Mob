# identité remarquable dans Z/pZ (p premier) 
# Algebraical ide,tity in Z/pZ (p prime)
 
# la somme de x et y dans Z/nZ 
# sum of x and y in Z/nZ 
def somme(x,y,n): 
    return (x+y)%n 
 
# product of x by y in Z/nZ 
def produit(x,y,n): 
    return (x*y)%n 
 
# puissance p de x 
# p-th power of x 
def puissance (x,p,n): 
    if p==0: 
        return 1 
    r=1 
    for i in range (1,p+1): 
        r=produit(x,r,n) 
    return r 
 
def main(): 
    n=7 
    L1= [somme(puissance(x,n,n),puissance(y,n,n),n) for x in range(0,n) for y in range(0,n)] 
    L2= [puissance(x+y,n,n) for x in range(0,n) for y in range(0,n)] 
    print(L1) 
    print(L2) 
if __name__ == '__main__': 
    main()
