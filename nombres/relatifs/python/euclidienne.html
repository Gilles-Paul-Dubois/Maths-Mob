# ce programme met en évidence le fait que l'opérateur python / en version 3.4 
# ne donne plus par défaut le quotient entiers de deux entiers 
# ce qui était le cas avec les versions 2 
# c'est maintenant l'opérateur // 
# This program shows that python's operator / in version 3.4 + 
# doesn't give by default the integer quatient  
# which was case with versions 2.7
# You must now use operator //
 
def main(): 
    x1 = 11 
    x2 = -11 
    y = 3 
    print(x1 // y, x1 % y) 
    print(x2 // y, x2 % y) 
    print(x1 / y) 
   
if __name__ == '__main__': 
    main()
