# résolution d'une diophantine:   
# solving a diophantine:   

def Diophante(a, b, c, n):
    s = bezout(a, b)
    d = s[0]
    if c % d:
        print("Equation sans solution")
        return
    sp = [s[1] * c / d, s[2] * c / d]  # solution particulière -special solution 
    sph = [b / d, -a / d]  # solution générique de l'équation homogène - generic sol. for homogeneous equation  
    # pour obtenir 2n solutions - to get 2n distinct solutions  
    return [[int(sp[0] + k * sph[0]), int(sp[1] + k * sph[1])] for k in range(-n, n + 1)]

# affiche le pgcd et les entiers de Bezout correspondant   
# displays gcd and corresponding Bezout's integers   

def bezout(a, b):
    s, t, u, v = 1, 0, 0, 1
    while b != 0:
        q = a // b
        a, s, t, b, u, v = b, u, v, a - q * b, s - q * u, t - q * v
    return (a, s, t) if a > 0 else (-a, -s, -t)


def main():
    print(Diophante(29, -11, 5, 10))


if __name__ == '__main__':
    main()
