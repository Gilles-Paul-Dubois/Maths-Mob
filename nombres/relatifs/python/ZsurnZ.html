#anneau Z/nZ 
#ring Z/nZ 
 
# addition dans Z/nZ 
# addition ins Z/nZ 
def somme(x,y,n): 
    return (x+y)%n 
#multiplication dans Z/nZ 
#multiplication in Z/nZ 
def produit(x,y,n): 
    return (x*y)%n 
 
# ligne de x dans la table d'addition de Z/nZ 
# Row of x in the addition table of Z/nZ 
def addligne(x,n): 
    return [somme(x,y,n) for y in range(0,n)] 
 
# ligne de x dans la table de multiplication de Z/nZ 
# row of x in the multiplication table of Z/nZ 
def multligne(x,n): 
    return [produit(x,y,n) for y in range(0,n)] 
 
# affiche la table d'addition de Z/nZ 
# Displays the addition table of Z/nZ 
def tableadd(n): 
    for x in range(0,n): 
        print(addligne(x,n)) 
 
# affiche la table de multiplication de Z/nZ 
# Displays the multiplication table of Z/nZ 
def tablemult(n): 
    for x in range(0,n): 
        print(multligne(x,n)) 
 
# teste si x est inversible dans l'anneau Z/nZ 
# checks if x is a unit in the ring Z/nZ 
def inversible(x,n): 
    return 1 in multligne(x,n) 
 
# checks if Z/nZ is a field 
def corps(n): 
    for x in range(1,n): 
        if not inversible(x,n): 
            return False 
    return True 
 
def main(): 
    tableadd(7) 
    print("-"*20) 
    tablemult(7) 
    print("-"*20) 
    print(inversible(0,7)) 
    print(inversible(4,7)) 
    #vérification du théorème du cours 
    print([n for n in range(2,10) if corps(n)]) 
 
if __name__ == '__main__': 
    main()
