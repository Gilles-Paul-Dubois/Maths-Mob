#include <stdio.h>
#include <stdlib.h>

// Ce programme a bour but de mettre en relief la représentation interne des entiers courts sur 2 octets
// dans le langage C
// le type entier_court est défini comme un champ de bits
// pour accéder à tous les bits de la donnée
typedef struct
{
unsigned int chiffre00:
    1;
unsigned int chiffre01:
    1;
unsigned int chiffre02:
    1;
unsigned int chiffre03:
    1;
unsigned int chiffre04:
    1;
unsigned int chiffre05:
    1;
unsigned int chiffre06:
    1;
unsigned int chiffre07:
    1;
unsigned int chiffre08:
    1;
unsigned int chiffre09:
    1;
unsigned int chiffre10:
    1;
unsigned int chiffre11:
    1;
unsigned int chiffre12:
    1;
unsigned int chiffre13:
    1;
unsigned int chiffre14:
    1;
unsigned int chiffre15:
    1;
}entier_court;

// on stocke au même endroit un short int et un champ de bits
typedef union
{
    unsigned short int n;
    short int m;
    entier_court ec;
}test;

// pour voir les chiffres binaires du short int
void affiche (entier_court x)
{
    printf("%d",x.chiffre15);
    printf("%d",x.chiffre14);
    printf("%d",x.chiffre13);
    printf("%d",x.chiffre12);
    printf("%d",x.chiffre11);
    printf("%d",x.chiffre10);
    printf("%d",x.chiffre09);
    printf("%d",x.chiffre08);
    printf("%d",x.chiffre07);
    printf("%d",x.chiffre06);
    printf("%d",x.chiffre05);
    printf("%d",x.chiffre04);
    printf("%d",x.chiffre03);
    printf("%d",x.chiffre02);
    printf("%d",x.chiffre01);
    printf("%d",x.chiffre00);
    printf("\n");
}
test T;

int main()
{
    T.n=65535;
    affiche(T.ec);
    T.m=-1;
    affiche(T.ec);
    T.n=32767;
    affiche(T.ec);
    T.n=32768;
    affiche(T.ec);
    T.m=-32768;
    affiche(T.ec);
    return 0;
}
