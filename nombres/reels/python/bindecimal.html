def binaire(n):
    """conversion d'un entier d√©cimal ->binaire - converts integers dec->binary"""
    if n <= 1:
        return n
    else:
        return binaire(n // 2) * 10 + n % 2


def binaire_int(n):
    return str(binaire(n))


def binaire_float(x, p):
    e = int(x)
    d = x - e
    E = binaire_int(e)
    r = [0] * p
    for i in range(1, p + 1):
        y = 1 / (2 ** i)
        if d >= y:
            r[i - 1] = 1
            d = d - y
    D = ""
    for x in r:
        D += str(x)
    S = E + "," + D
    return S


def main():
    print(binaire_int(10))
    print(binaire_float(1.800, 3))
    print(binaire_float(1.800, 5))


if __name__ == '__main__':
    main()
