from functools import reduce

# Calcul de P(racine(n)) où P est un polynome quelconque représenté par la liste de ses coefficients 
# Computation of P(√n) where P is any polynomial represented by the list of its (integral) coefficients 
def PowerRac (n,i,a): 
    """puissance i-ième de racine(n) multiplié par a""" 
    """i-th power of √n multiplied by a"""     
    if not i%2: 
        return [a*n**(i//2),0] #pairs 
    else: 
        return [a*n**((i-1)//2),1] #impairs 
 
def EvalRac(P,n): 
    """Evaluation de P(racine(n))-evaluation of P(√n)""" 
    S= [PowerRac(n,i,P[i]) for i in range(0,len(P))] 
    R=[ x[0] for x in S if x[1]==0] 
    I= [ x[0] for x in S if x[1]==1] 
    D= [reduce(lambda x,y:x+y,R),reduce(lambda x,y:x+y,I)] 
    return str(D[0])+"+"+str(D[1])+"rac("+str(n)+")" 
 
def main(): 
    P=[1,2,3] 
    print(EvalRac(P,3)) 
 
if __name__ == '__main__': 
    main()
