#La méthode des approximations successives de Newton 
#Newton's method of successive approximations 
# fournit une suite de Cauchy alternée représentant (très vite) le nombre racine(2) 
# provides an aletrnate Cauchy sequence converging very quickly to √2 
 
 
def Cauchy(): 
    R=Decimal(1) 
    yield  R 
    while True : # détermination des décimales successives - determination of successive decimals 
        N=R*R + 2 
        D=R*Decimal(2) 
        R=N/D 
        yield R 
 
def main(): 
    C=Cauchy() 
    L=[next(C) for i in range(0,10)] 
    for x in L: 
        getcontext().prec=100 #fixe la précision - determines precision 
        print(x) 
 
if __name__ == '__main__': 
    main()
