from fractions import Fraction


def valp(n, p):
    """valuation p-adique de l'entier n"""
    """p-adic valuation of integer n"""    
    if n % p:
        return 1
    else:
        return 1 + valp(n // p, p)


def normp(r, p):
    """norme p-adique d'un rationnel"""
    """p-adic norm of a rational"""    
    if r == 0:
        return 0
    else:
        return Fraction(p ** valp(r.denominator, p), p ** valp(r.numerator, p))


def main():
    print(normp(Fraction(0), 2))
    for p in [2, 3, 5, 7, 11, 13]:
        print(normp(Fraction(63, 550), p))


if __name__ == '__main__':
    main()
