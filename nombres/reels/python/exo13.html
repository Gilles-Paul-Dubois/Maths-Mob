def pow(a, b, p, n):
    """ Calcule a+b*rac(p) Ã  la puissance n"""
    """ Computes nth power of a+b*√p """    
    if n == 0:
        return 1, 0
    else:
        P = pow(a, b, p, n - 1)
        return P[0] * a + P[1] * b * p, P[0] * b + P[1] * a


def main():
    p = 3
    x = pow(2, 2, p, 3)
    print(str(x[0]) + "+" + str(x[1]) + "*rac(" + str(p) + ")")


if __name__ == '__main__':
    main()
