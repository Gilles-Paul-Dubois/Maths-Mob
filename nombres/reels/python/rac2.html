#calcul de racine de 2 avec une précision donnée 
#computation of √2 with given precision
import math 
 
# le générateur - the generator 
def rac2 (): 
    pow=1 
    app=1 
    yield app # première approximation entière - first integral approx. 
    while True : # détermination des décimales successives - determining decimals in sequence 
        pow=pow*100 
        u=0 
        while (app*10+u)*(app*10+u) <= 2*pow: 
            u+=1 
        app=app*10+u-1 
        yield app 
 
def main(): 
    racine=rac2() # objet généré - generated object 
    print(math.sqrt(2)) # juste pour vérifier - just to check 
    for i in range(1,50): 
        root=str(next(racine)) #pour un affichage propre - for clean display 
        print(root[0]+','+root[1:]) 
 
if __name__ == '__main__': 
    main()
