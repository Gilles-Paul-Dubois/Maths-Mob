// l'exposant en décimal
unsigned int exposant (flottant f)
{
    unsigned int e=0;
    e+=f.chiffre23;
    e+=f.chiffre24*2;
    e+=f.chiffre25*4;
    e+=f.chiffre26*8;
    e+=f.chiffre27*16;
    e+=f.chiffre28*32;
    e+=f.chiffre29*64;
    e+=f.chiffre30*128;
    return e-127;
}

// la mantisse en décimal
float mantisse (flottant f)
{
    float m =1;
    m+=f.chiffre22*(1.0/2);
    m+=f.chiffre21*(1.0/4);
    m+=f.chiffre20*(1.0/8);
    m+=f.chiffre19*(1.0/16);
    m+=f.chiffre18*(1.0/32);
    m+=f.chiffre17*(1.0/64);
    m+=f.chiffre16*(1.0/128);
    m+=f.chiffre15*(1.0/256);
    m+=f.chiffre14*(1.0/512);
    m+=f.chiffre13*(1.0/1024);
    m+=f.chiffre12*(1.0/2048);
    m+=f.chiffre11*(1.0/4096);
    m+=f.chiffre10*(1.0/8192);
    m+=f.chiffre09*(1.0/16384);
    m+=f.chiffre08*(1.0/32768);
    m+=f.chiffre07*(1.0/65536);
    m+=f.chiffre06*(1.0/131072);
    m+=f.chiffre05*(1.0/262144);
    m+=f.chiffre04*(1.0/524288);
    m+=f.chiffre03*(1.0/1048576);
    m+=f.chiffre02*(1.0/2097152);
    m+=f.chiffre01*(1.0/4194304);
    m+=f.chiffre00*(1.0/8388608);
    return m;
}

// affiche l'exposant en décimal

{

}
test T;

//affiche la mantisse en décimal
void affmantisse(flottant f)
{
    printf("mantisse: %22.20f\n",mantisse(f));
}

//affiche le flottant
void affiche (test t, char * s)
{
    printf("\nAffichage du flottant %s sur 32 bits IEEE standard 754\n",s);
    affbinaire(t.f);
    affexpo(t.f);
    affmantisse(t.f);
    printf("verification: %32.30f\n",t.x);
}

int main()
{
    T.x=3.1416;// un nombre positif (presque) quelconque
    affiche(T,"3.1416");
    T.x=-3.1416; // son opposé
    affiche(T,"-3.1416");
    T.x=0.000000000000000000001; // un nombre très petit
    affiche(T,"0.000000000000000000001");
    T.x=10000000000000.0; // un nombre très grand.
    affiche(T,"10000000000000");

    return 0;
}
