using Decimals
function cauchy()
    Channel() do c
        R=decimal("1")
        push!(c,R)
        while true
            N=R*R+2
            D=R*decimal("2")
            R=N/D
            push!(c,R)
        end

    end
end

RAC2=cauchy()

for i in 1:10
    println(take!(RAC2))
end
