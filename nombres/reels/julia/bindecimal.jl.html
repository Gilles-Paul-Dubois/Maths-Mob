function binaire_float(x,p)
    e=Int(floor(x))
    E=binaire_int(e)
    d=x-e
    r=repeat([0],p)
    for i in 1:p
        y=1/(2^i)
        if d>=y
            r[i]=1
            d=d-y
        end
    end
    D=""
    for x in r
        D=string(D,x)
    end
    return string(E,string(","),D)
end

println(binaire_float(1.8,3))
println(binaire_float(1.8,5))
