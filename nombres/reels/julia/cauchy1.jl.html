function racine(p)
    Channel() do c
        pow=1
        num=1
        den=1
        push!(c,num//den)
        while true
            pow=pow*100
            u=0
            while (num*10+u)*(num*10+u)<=p*pow
                u+=1
            end
            num=num*10+u-1
            den=den*10
            push!(c,num//den)
        end
    end
end

RAC3=racine(3)

for i in 1:10
    println(take!(RAC3))
end
