PowerRac(n,i,a)= i%2==0 ? (a*n^(i÷2),0) : (a*n^((i-1)÷2),1)

function EvalRac(P,n)
    S=[PowerRac(n,i,P[i+1]) for i in 0:length(P)-1]
    R,I=[x[1] for x in S if x[2]==0],[x[1] for x in S if x[2]==1]
    D=(reduce(+,R),reduce(+,I))
    return string(string(D[1]),string("+"),string(D[2]),string("√"),string(n)) 
end

P=[1,2,3]    
println(EvalRac(P,3))
