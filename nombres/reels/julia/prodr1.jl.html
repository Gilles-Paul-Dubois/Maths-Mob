using Decimals

function racine(p)
    Channel() do c
        pow=1
        num=1
        den=1
        push!(c,num//den)
        while true
            pow=pow*100
            u=0
            while (num*10+u)*(num*10+u)<=p*pow
                u+=1
            end
            num=num*10+u-1
            den=den*10
            push!(c,num//den)
        end
    end
end

#transformation en decimal d'un rationnel
#transformation into decimalof a rational
dec(r)= decimal(numerator(r))/decimal(denominator(r))

RAC2=racine(2)
RAC3=racine(3)
RAC6=racine(6)

L2=[take!(RAC2) for i in 1:10]
L3=[take!(RAC3) for i in 1:10]
L6=[take!(RAC6) for i in 1:10]
PROD=[L2[i]*L3[i] for i in 1:10]
FL6=map(x->dec(x),L6)
FPROD=map(x->dec(x),PROD)

for i in 1:10
    print(FL6[i])
    print("---")
    print(FPROD[i])
    println(" ")
end
