import math 
#version récursive - recursive version 
def convertir1 (n,b): 
    if n<b: 
        print(n%b, end=' ') 
    else: 
        r=n%b 
        q=n//b 
        convertir1(q,b) 
        print(n%b, end=' ') 
 
#version itérative - iterative version 
def convertir2(n,b): 
    if n==0: 
        print(0, end=' ') 
        return 
    else: 
        L=[] 
        while n >1: 
            r=n%b 
            L.append(r) 
            n=n//b 
        L.append(1) 
        L.reverse() 
        for x in L: 
            print(x, end=' ') 
 
 
def main(): 
    convertir1(33,2) 
    print("\n") 
    convertir2 (27,3) 
 
if __name__ == '__main__': 
    main()
