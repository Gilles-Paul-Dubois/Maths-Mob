# donne le chiffre hexa correspondant à n 
# gives hexa digit corresponding to n 
def chiffre_hexa(n): 
    if n<10: 
        return str(n) 
    else: 
        return chr(ord('A')+int(n-10)) 
 
# convertir le nombre n en hexa 
# converts number n in hexa 
def conversion_hexa(n): 
    if n<16: 
        return chiffre_hexa(n) 
    else: 
        r=n%16 
        return conversion_hexa((n-r)/16)+ chiffre_hexa(r) 
 
def main(): 
    print(conversion_hexa(255)) 
 
if __name__ == '__main__': 
    main()
