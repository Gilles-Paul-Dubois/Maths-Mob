# Cube de Nikomakhos - Cube of Nikomakhos  
# Toutes les solutions - all solutions 
 
# sommes à p termes - sums with p terms 
def Sol(p,n): 
    if (p*p >n) or ((n-p*p)%(2*p)): 
        return False 
    return [((n-p*p)/p)+1+2*k for k in range(0,p)] 
 
# Toutes les possibilités pour le cube de m - all possibilities for the cube of m 
def Solutions (m): 
    n=m*m*m 
    return [Sol(p,n) for p in range(2,n) if Sol(p,n)] 
 
def main(): 
    print(Solutions(4)) 
 
if __name__ == '__main__': 
    main()
