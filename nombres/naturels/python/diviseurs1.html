#lister les diviseurs d'un nombre - find all divisors of some number 
 
import math 
 
#version naïve (et simple) - naive and simple version 
def diviseurs1(n): 
    return [i for i in range(1,n+1) if n%i==0] 
 
#plus élaborée - more elaborated 
def diviseurs2(n): 
    r=int(math.sqrt(n))+1 # par sécurité +1 - for safety 
    E1=[i for i in range(1,r+1) if n%i==0] # les diviseurs jusqu'à la racine - divisors till square root 
    E2= [n/i for i in E1] # les autres en ordre inverse - others in reverse order 
    if E1[len(E1)-1]==E2[len(E2)-1]: #cas des carrés parfaits - case of perfect squares 
        return sorted(E1+E2[0:len(E2)-1]) #élimination doublon et on trie - elimination of double and sorting 
    else: 
        return sorted(E1+E2) #on trie - sorting in any case 
 
def main(): 
    print(diviseurs1(60)) 
    print(diviseurs2(60)) 
 
if __name__ == '__main__': 
    main()
