import math 
n=10000 
premiers=[True]*n #a priori tous premiers - give a chance to everybody to start 
premiers[0]=False #0 n'est pas premier - 0 not prime  
premiers[1]=False # 1 non plus - neither 1 
racine=math.sqrt(n) #la borne pour les tests- bound for tests 
 
# la méthode du crible - the sieve method 
def eratosthene (): 
    i=2 
    while i <= racine: 
        k=2 
        h=(n-1)/i #pour ne pas recalculer un produit à chaque tour de boucle - not to compute a product for every pass in the loop 
        while k<=h: 
            premiers[k*i]=False 
            k+=1 
        j=i+1 if i<3 else i+2 #après 2 et 3 plus de premiers consécutifs - after (2,3) no couple of consecutive primes 
        while not premiers[j]: 
            j+=1 
        i=j 
 
def main(): 
    eratosthene() 
    print([i for i in range(0,n) if premiers[i]])#affiche les premiers entre 1 et n - displays primes between 1 and n 
 
if __name__ == '__main__': 
    main()
