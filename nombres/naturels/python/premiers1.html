#recherche de nombres premiers - finding prime numbers 
import math 
   
# un nombre est premier s'il est égal à 2 ou à 3 ou bien s'il n'est pas multiple ni de 2 ni de trois  
# a number is prime if it is noeiteur 2 nor 3 nor a multiple of 2 or 3  
 
# et s'il ne possède aucun diviseur impair entre 3 et sa racine  
# and has no odd divisor between 3 and its own square root 
 
   
def premier (n):   
    return (n in [2,3]) or ((n%2 and n%3) and (not any([n%x==0 for x in range(3,int(math.sqrt(n)+1))])))   
   
def main():   
    print([n for n in range(2,100) if premier(n)])   
   
if __name__ == '__main__':   
    main()
