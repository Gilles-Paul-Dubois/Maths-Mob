#recherche de triplets pythagoriciens primitifs 
#looking for primitive Pythagorean triples 
import math 
# computation of ppt 
 
def Power5(n): 
    if n==1: 
        return True 
    else: 
        if n%5: 
            return False 
        else: 
            return Power5(n/5) 
 
# borne supérieure de p et q pour z <k 
# upper bound of p and q for z <k 
def bsup (k): 
    return int(math.sqrt(k))+1 
 
def pgcd(m,n): 
    while m%n: 
        r=m%n 
        m=n 
        n=r 
    return n 
 
# postulat de parité différente 
# different parity postulate 
def pardif(p,q): 
    return p%2 != q%2 

#liste des p possibles pour q donné avec z <k 
#list of possible p for given q with z <k 
def possp(q,k): 
    return [p for p in range(q,bsup(k)) if p>q and pardif(p,q) and pgcd(p,q)==1] 
 
# donne tous les tpp possibles pour q donné avec z <k 
# gives all possible ppt for given  q with z <k
def tppq(q,k): 
    S= [[p*p-q*q, 2*p*q,p*p+q*q] for p in possp(q,k) if p*p+q*q<k] 
    U=[sorted(t) for t in S] 
    return U 
 
# donne tous les tpp possibles avec z <k 
# gives all possible ppt with z <k
def tpp(k): 
    R=[] 
    for q in range (1,bsup(k)): 
        R=R+tppq(q,k) 
    return R 
 
# fonction principale - main function 
def main(): 
    L = tpp (126) 
    P=[t for t in L if Power5(t[2])] 
    print(P) # liste les résultats - lists all results 
 
if __name__ == '__main__': 
    main()
