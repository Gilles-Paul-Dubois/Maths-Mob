# calcul de ppcm 
# computation of lcm
 
import math 
def pgcd(m,n): 
    while m%n: 
        r=m%n 
        m=n 
        n=r 
    return n 
 
def ppcm(m,n): 
    return m*n//pgcd(m,n) 
 
def main(): 
    print(ppcm(12,15)) 
    print(ppcm(12,7)) 
 
if __name__ == '__main__': 
    main()
