#calcule des premiers et les sauve dans un fichier 
#Computes prime numbers and saves them into a file
from math import sqrt 
 
#teste si un nombre n est premier 
#checks if number n is prime 
def Est_Premier(n): 
    if n<=1: # 0 et 1 non premiers - 0 and 1 not prime 
        return False 
    elif n<=3: # 2 et 3 premiers - 2 and 3 primes 
        return True 
    elif n%2 == 0: # on ne teste pas les pairs >2 - no need to check even numbers >2 
        return False 
    else: 
        r=int(sqrt(n))+1 # borne supérieure pour les diviseurs possibles- upper bound for possible divisors 
        for i in range(3,r,2): # on ne teste que les diviseurs impairs- we test only odd numbers 
            if not (n%i): 
                return False # on a trouvé un diviseur, perdu !- We found a divisor, failure ! 
        return True  #on n'a rien trouvé, gagné !- we didn't find anythong, success ! 
 
# trouve les premiers entre 2 et 65535 
# finds prime numbers between 2 and 65535 
def Premiers(): 
    return [p for p in range(2,65535) if Est_Premier(p)] 
 
#Ecrire dans un fichier la liste L 
#Write list L in a file 
def Sauve (L): #sauver sur fichier - save on file 
    f=open("premiers.txt", 'w') 
    for p in L: 
        f.write(str(p)) 
        f.write("\n") 
    f.close() 
    return 
 
def main(): 
    L=Premiers() 
    Sauve(L) 
 
if __name__ == '__main__': 
    main()
