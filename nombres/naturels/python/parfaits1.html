#Trouver les nombres parfaits entre 1 et 100000 
#find perfect numbers between 1 and 100000 
#programme semi-naïf 
#half-naive program 
 
import math 
from functools import reduce 
 
#liste des diviseurs de n 
#list of divisors of n
def diviseurs2(n): 
    r=int(math.sqrt(n))+1 # par sécurité +1- for safety +1 
    E1=[i for i in range(1,r+1) if n%i==0] # les diviseurs jusqu'à la racine - divisors till root 
    E2= [n/i for i in E1] # les autres en ordre inverse - others in reverse order 
    if E1[len(E1)-1]==E2[len(E2)-1]: #cas des carrés parfaits- case of perfect squares 
        return sorted(E1+E2[0:len(E2)-1]) #élimination doublon et on trie- elimination of double and sorting 
    else: 
        return sorted(E1+E2) #on trie 
 
# Calcule la somme des diviseurs de n autres que n 
# Compute the sum of aliquots
def SommeAliquotes(n): 
    def add(x,y): #fonction locale pour reduce- local function for reduce 
        return x+y 
    lis=diviseurs2(n) 
    lis=lis[0:len(lis)-1] # enlever le dernier diviseur - remove the last divisor (n) 
    return reduce( add,lis) #ajouter tous les éléments de la liste_- add all elements of the list 
 
#décide si un nombre est parfait 
#decides if a number is perfect 
def Parfait(n): 
    return n==SommeAliquotes(n) 
 
# Trouve les parfaits entre 2 et n 
# finds perfects between 2 and n 
def TrouveLes (n): 
    return [i for i in range(2,n+1) if Parfait(i)] 
 
def main(): 
    print(TrouveLes(100000)) 
 
if __name__ == '__main__': 
    main()
