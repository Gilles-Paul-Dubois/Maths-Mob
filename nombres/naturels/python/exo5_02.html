#utiliser les nombres premiers dans un fichier pour tester la primalité d'un nombre 
#use prime numbers in a file to test primality of a number 
   
L=[]   
   
# récupère dans une variable globale de type liste les nombres premiers du fichier   
# Restores in a global variable of type 'list' the prime numbers of a file   
def Restaure():   
    f=open("premiers.txt", 'r')   
    for line in f:   
        L.append(int(line))   
    f.close   
    return L   
   
#nouveau test - new test   
def Est_Premier(N):   
    global L   
    for i in L:   
        if N%i==0:   
            return False   
        if i*i >N:   
            return True   
   
#fonction principale - main function   
def main():   
    L=Restaure()   
    print(Est_Premier(4000000007))   
   
if __name__ == '__main__':   
    main()
