from math import sqrt 
 
premiers=[] # recevra la suite des nombres premiers- will receive list of prime numbers 
 
#teste si un nombre n est premier 
#checks if n is prime 
def Est_Premier(n): 
    if n<=1: # 0 et 1 non premiers - 0 and 1 not prime 
        return False 
    elif n<=3: # 2 et 3 premiers - 2 and 3 prime 
        return True 
    elif n%2 == 0: # on ne teste pas les pairs >2 
        return False 
    else: 
        r=int(sqrt(n))+1 # borne supérieure pour les diviseurs possibles - upper bound for divisor candidates 
        for i in range(3,r,2): # on ne teste que les diviseurs impairs - test only for odd numbers 
            if not (n%i): 
                return False # on a trouvé un diviseur, perdu ! - divisor found, failure ! 
        return True  #on n'a rien trouvé, gagné ! - nothing found, success ! 
 
# remplit la liste premiers globale avec les n premiers nombres premiers 
# fills the global list with the first n prime numbers  
def RemplirPremiers(n): 
    i=0 
    j=2 
    while i<n: 
        if Est_Premier(j): 
            premiers.append(j) 
            j=j+1 
            i=i+1 
        else: 
            j=j+1 
 
# nombre de Mersenne d'ordre p 
# Mersenne number of order p p
def Mersenne(p): 
    return 2**premiers[p]-1 
 
# affiche les 10 premiers nombres de Mersenne 
# prints the first 10 Mersenne numbers 
# puis, parmi ceux-ci, ceux qui sont premiers 
# Then, among these, those which are prime
def main(): 
    RemplirPremiers(100) 
    M= [Mersenne(p) for p in range(0,10)] 
    MP=[n for n in M if Est_Premier(n)] 
    print(M) 
    print(MP) 
 
 
if __name__ == '__main__': 
    main()
