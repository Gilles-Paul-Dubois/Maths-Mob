import math 
 
#donne la liste de tous les triangulaires entre 1 et n 
#gives the list of all triangular numbers between 1 and n 
def triangulaires(n): 
    return [i*(i+1)/2 for i in range(1,n+1)] 
 
#teste si un nombre est triangulaire 
#checks if n is triangular 
def est_triangulaire(n): 
    r=int(math.sqrt(2*n)) 
    if n==r*(r+1)/2 or n==(r+1)*(r+2)/2: 
        return True 
    else: 
        return False 
 
def main(): 
    print(triangulaires(10)) 
    print([i for i in range(1,56) if est_triangulaire(i)]) 
 
if __name__ == '__main__': 
    main()
