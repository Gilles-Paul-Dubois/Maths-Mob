# un exemple de 'generator expression' 
# An example of 'generator expression'
 
from itertools import islice 
 
# générateur avec condition d'arrêt 
# generator with stop condition 
def premiers(n): 
    i=1 
    yield 2 #premier appel - first call 
    i+=1 
    yield 3 # second appel - second call 
    P,p=[2,3],3 
    while 1: 
        p+=2 # à partir de 3 on saute de 2 en 2 - from 3 step 2 
        if all([p%x for x in P]): 
            P.append(p) 
            i+=1 
            if i>n: 
                raise StopIteration  
            yield p 
 
def main(): 
    P=premiers(20) 
    for i in P: 
        print(i) 
    # nous avons ci-après une "generator expression". 
    # We have hereafter a "generator expression".     
    # La liste n'est pas évaluée a priori 
    # The list is not evaluated automatically     
    # Elle ne l'est qu'en cas de besoin 
    # It is only if needed     
    # C'est un cas particulier d'évaluation "paresseuse" 
    # It's a typical case of 'lazy evaluation'     
    # noter la syntaxe voisine des 'comprehension lists' 
    # Please note the syntax very close to that of 'comprehension lists'     
    # Brackets are replaced by parentheses. 
    Q=(p for p in premiers(100000000)) 
    premier_rang_2000 = next(islice(Q, 2000, 2001)) 
    print(premier_rang_2000)  
     
if __name__ == '__main__': 
    main()
