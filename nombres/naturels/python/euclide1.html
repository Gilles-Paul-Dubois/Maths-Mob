#version récursive de l'algorithme d'Euclide 
#recursive version of Euclidean algorithm 
def pgcd1(m,n): 
    if m%n==0: 
        return n 
    else: 
        return pgcd1(n,m%n) 
 
#version itérative du même 
#iterative version of the same 
def pgcd2(m,n): 
    while m%n: 
        r=m%n 
        m=n 
        n=r 
    return n 
 
def main(): 
    print(pgcd1(120,150)) 
    print(pgcd2(150,120)) 
if __name__ == '__main__': 
    main()
