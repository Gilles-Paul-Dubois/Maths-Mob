# décomposition en facteurs premiers 
# decomposition into prime factors 
# contiendra la liste des facteurs premiers  
#will contain the list of prime factors 
facteurs=[] 
#contiendra les exposants correspondants 
#will contain the list of exponents 
exposants=[] 
 
#le travail de décomposition 
#decomposition job 
def decompose (n): 
    i=2 
    while n>1: 
        if n%i==0: 
            facteurs.append(i) 
            e=0 
            while n%i==0: 
                e=e+1 
                n=n/i 
            exposants.append(e) 
        i=i+1 
 
def main(): 
    decompose(120) 
    print(facteurs) 
    print(exposants) 
 
if __name__ == '__main__': 
    main()
