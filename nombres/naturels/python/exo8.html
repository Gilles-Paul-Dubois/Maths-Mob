# vérifier l'hypothèse de Goldbach jusqu'à 1000 
# Check Goldbach's conjecture till 1000 
   
L=[]   
   
#écrire n comme somme de deux premiers de la liste L   
#write n as a sum of two primes of list L   
def Goldbach(n):   
    for x in L:   
        for y in L:   
            if n==x+y:   
                return [x,y] #succès - success   
    return [] #échec final - final failure   
   
# récupère dans une variable globale la liste des nombres premiers du fichier   
# restores in a global variable the list of prime numbers contained in the file   
def Restaure():   
    f=open("premiers.txt", 'r')   
    for line in f:   
        L.append(int(line))   
    f.close   
    return L   
   
   
#fonction principale - main function   
def main():   
    L=Restaure()   
    print([Goldbach(i) for i in range(4,1001,2)])   
   
if __name__ == '__main__':   
    main()
