# un exemple de 'generator' 
# an example of 'generator' 
# l'instruction yield equivaut à un 'return' mais l'exécution est suspendue 
# instruction yield has value of 'return' but execution is paused 
# A l'appel suivant elle redémarre exactement 
# At the next call it restarts exactly 
# au point où elle a été abandonnée.
# at the point where it was abandoned   
# l'environnement (variables locales et leurs valeurs) est conservé. 
# Environment (local variables and their values) is kept unchanged 
 
def premiers(): 
    count=1 
    yield count,2 #premier appel - first call 
    count+=1 
    yield 2,3 # second appel - second call 
    P,p=[2,3],3 
    while 1: 
        p+=2 # à partir de 3 on saute de 2 en 2 - from 3 we jump with step 2 
        if all([p%x for x in P]): 
            P.append(p) 
            count+=1 
            yield count,p 
 
def main(): 
    P=premiers() 
    for i in range(0,20): # donne les 20 premiers nombres premiers avec leur rang- gives the first 20 prime numbers together with their rank 
        print(next(P)) # C'est ainsi qu'on invoque un générateur - this is the way to call a generator 
 
if __name__ == '__main__': 
    main()
