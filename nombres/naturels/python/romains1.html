# les caractères qui correspondent à des chiffres 
# Characters used as numerical symbols 
Romains="IVXLCDM" 
 
# teste si un caractère est un chiffre romain 
# Checks if a letter is a numerical symbol 
def IsRomanNumeral(lettre): 
    return lettre in "IVXLCDM" 
 
# compare deux chiffres 
# compares two symbols 
def IsLessThan(lettre1,lettre2): 
    i1=Romains.index(lettre1) 
    i2=Romains.index(lettre2) 
    return i1<i2 
 
# definit la valeur d'un chiffre 
# defines the value of a symbol 
def Value(lettre): 
    if lettre=='I': 
        return 1 
    elif lettre=='V': 
        return 5 
    elif lettre=='X': 
        return 10 
    elif lettre=='L': 
        return 50 
    elif lettre=='C': 
        return 100 
    elif lettre=='D': 
        return 500 
    elif lettre=='M': 
        return 1000 
    else: 
        return 0 
 
# romains non puissances de 10
# romans which are not powers of 10  
def Quinqua(X): 
    return X in "VLD" 
 
#chiffre précédent 
#previous digit 
def Previous(C): 
    i=Romains.index(C) 
    return Romains[i-1] 
 
# met une expression sous forme nomale sans soustraction 
#puts an expression under 'normal' form without subtraction
def Normalize(N): 
    M="" 
    for i in range(0,len(N)-1): 
        if IsLessThan(N[i],N[i+1]): 
            if Quinqua(N[i+1]): 
                NewNum=4*N[i] 
            else: 
                NewNum=Previous(N[i+1])+4*N[i] 
            M+=NewNum 
        else: 
            M+=N[i] 
            if i==len(N)-2: 
                M+=N[i+1] 
    return M 
 
# évaluation d'un nombre sous forme normale 
# evaluation of a number under normal form
def EvalNorm(N): 
    s=0 
    for x in N: 
        s+=Value(x) 
    return s 
 
#évaluation d'un numéral romain 
#évaluation of a roman numeral
def Eval(N): 
    return EvalNorm(Normalize(N)) 
 
#fonction principale 
#main function 
def main(): 
    print(Eval("MCDLIX")) 
 
if __name__ == '__main__': 
    main()
