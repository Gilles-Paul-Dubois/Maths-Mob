function decompose(n)
    facteurs, exposants=[],[]
    i=2
    while n>1
        if n%i==0
            push!(facteurs,i)
            e=0
            while n%i==0
                e+=1
                n=n√∑i
            end
            push!(exposants,e)
        end
        i+=1
    end
    return facteurs, exposants
end

println(decompose(120))
