function diviseurs1(n)
    return [i for i in 1:n if n%i==0]
end

function diviseurs2(n)
    r=round(Int32,sqrt(n))
    E1=[i for i in 1:r if n%i==0]
    E2=[n√∑i for i in E1]
    if E1[end]==E2[end]
        return sort(append!(E1[1:end-1],E2))
    else
        return sort(append!(E1,E2))
    end
end

function main()
    println(diviseurs1(12))
    println(diviseurs2(12))
    println(diviseurs1(16))
    println(diviseurs2(16))
end

main()
