function Restaure()
    L=[]
    open("premiers.txt") do f
        L=readlines(f)
    end
    L=[parse(Int64,x) for x in L]
    return L
end

function Est_Premier(N)
    L=Restaure()
    for i in L
        if N%i==0
            return False
        end
        if i*i>N
            return true
        end
    end
end


println(Est_Premier(4000000007))
