function premier(n)
    if n in 2:3
        return true
    else
        borne=Int(floor(sqrt(n)))+1
        if n%2!=0 && n%3!=0 &&!(any(n%x==0 for x in 3:borne))
            return true
        end
    end
    return false
end

function compte(m)
    result=0
    for i in 1:m
        if premier(i)
            result+=1
        end
    end
    return result,m
end


for n in [100, 1000, 10000,100000,1000000,10000000]
    x=1/log(n)
    y= compte(n)[1]/compte(n)[2]
    println(x/y)
end
