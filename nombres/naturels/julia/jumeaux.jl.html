function premier(n)
    if n in 2:3
        return true
    else
        borne=Int(floor(sqrt(n)))+1
        if n%2!=0 && n%3!=0 &&!(any(n%x==0 for x in 3:borne))
            return true
        end
    end
    return false
end

jumeaux(n)= premier(n)&& premier(n+2)

function main(n)
    result=(0,0)
    i=3
    while i<n
        if jumeaux(i)
           result=(i,i+2)
        end
        i+=2
    end
    return result
end

main(20000000)
