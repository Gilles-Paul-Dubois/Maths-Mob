function premier(n)
    if n in 2:3
        return true
    else
        borne=Int(floor(sqrt(n)))+1
        if n%2!=0 && n%3!=0 &&!(any(n%x==0 for x in 3:borne))
            return true
        end
    end
    return false
end

Premiers()=(p for p in 2:65635 if premier(p))

function Sauve(L)
    open("premiers.txt","w") do io
        for p in L
            write(io, string(p))
            write(io,"\n")
        end
    end
end

Sauve(Premiers())
