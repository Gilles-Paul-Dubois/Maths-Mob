function premier(n)
    if n in 2:3
        return true
    else
        borne=Int(floor(sqrt(n)))+1
        if n%2!=0 && n%3!=0 &&!(any(n%x==0 for x in 3:borne))
            return true
        end
    end
    return false
end

L=[i for i in 2:1000 if premier(i)]

function Sol(n)
    for p in L
        if n-p in L
            return p,n-p
        end
    end
    return nothing
end

function main()
    i=4
    while i<=1000
        print(i); print("="); print(Sol(i)[1]); print("+"); println(Sol(i)[2])
        i=i+2
    end
end

main()
