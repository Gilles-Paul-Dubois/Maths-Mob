function diviseurs2(n)
    r=round(Int32,sqrt(n))
    E1=[i for i in 1:r if n%i==0]
    E2=[n√∑i for i in E1]
    if E1[end]==E2[end]
        return sort(append!(E1[1:end-1],E2))
    else
        return sort(append!(E1,E2))
    end
end

function SommeAliquotes(n)
    L=diviseurs2(n)
    pop!(L)
    return reduce(+,L)
end

function PossedeAmi(n)
    m=SommeAliquotes(n)
    return m >n && SommeAliquotes(m)==n
end

TrouveLesAmis(n)=[(x,SommeAliquotes(x)) for x in 2:n if PossedeAmi(x)]
println(TrouveLesAmis(10000))
