bsup(k)=Int(floor(sqrt(k)))

function pgcd(m,n)
    while m%n !=0
        r=m%n
        m=n
        n=r
    end
    return n
end

pythagoricien(t)=t[3]*t[3]==t[1]*t[1]+t[2]*t[2]

TousOK(L)=all((pythagoricien(t) for t in L))

pardif(p,q)=p%2!=q%2

possp(q,k)=[p for p in q:bsup(k) if p>q && pardif(p,q) && pgcd(p,q)==1]

tppq(q,k)=sort([[p*p-q*q, 2*p*q,p*p+q*q] for p in possp(q,k) if p*p+q*q<k])

function tpp(k)
    R=[]
    for q in 1:bsup(k)
        append!(R,tppq(q,k))
    end
    return R
end

L=tpp(200)
println(L)
println(TousOK(L))
