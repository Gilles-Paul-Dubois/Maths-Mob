n = 10000
premiers = repeat([true], n)
racine = Int(floor(sqrt(n)))
premiers[1] = false

function eratosthene()
    i = 2
    while i <= racine
        h = (n - 1) / i
        k = 2
        while k < h
            #println(k*i)
            premiers[k*i] = false
            k += 1
        end
        j = (i < 3) ? i + 1 : i + 2
        if j > n
            return
        end
        while !(premiers[j])
            j += 1
        end
        i = j
    end
end

function main()
    eratosthene()
    println([i for i in 1:100 if premiers[i]])
end

main()
