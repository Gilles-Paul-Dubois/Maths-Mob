function factorielle1(n)
    print("début de l'exécution de factorielle1 avec le paramètre :")
    println(n)
    result = 1
    println(typeof(result)) # le type des entiers par défaut Int64 - default type for integers
    for i = 1:n
        result *= i
    end
    print("fin de l'exécution de factorielle1 : ")
    return result
end

function factorielle2(n)
    print("début de l'exécution de factorielle2 avec le paramètre :")
    println(n)
    result::UInt128 = 1
    println(typeof(result)) # affiche le type forcé - displays forced type
    for i = 1:n
        result *= i
    end
    print("fin de l'exécution de factorielle2 : ")

end

function factorielle3(n)

function main()
