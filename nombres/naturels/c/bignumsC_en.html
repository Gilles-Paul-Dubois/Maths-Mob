#include <stdio.h>
#include <stdlib.h>
#include <string.h>


typedef int Bignum [100]; // Custom data-type 'bignum'




void Saisir (Bignum A)
{ char s[401]; //to enter digits as a string
  static char CA[100][5]; // to create slices of 4 digits
  // NB we end with a NULL (C language convention)
  int i,j,k,ls; // local variables
  for(i=0;i<100;i++) for(j=0;j<5;j++) CA[i][j]=0; // initialisation à 0
  printf("Five a Bignum: ");
  gets(s); // enter the number
  ls=strlen(s); // picking up length of the string
  for(j=ls-4,k=0;j>=0;j=j-4,k++) for(i=0;i<4;i++)CA[k][i]=s[i+j];
  // picking up all slices beginning by the end
  // To finish : last slice, 'digit' with strongest weight
  j=j+4;
  for(i=0;i<j;i++) CA[k][i]=s[i];
  // Finally conversion of slice in integer type
  for(i=0;i<100;i++) A[i]=atoi(CA[i]); // done !
  }
  
 void Afficher (Bignum A)
 {int i=99; while(A[i]==0) i--; // we go directly to the digit with biggest weight
  printf("\n"); // line-feed to begin
  printf("%d",A[i]); // we display 'as is'
  i--;
  while(i>=0)
  {printf("%04d",A[i]);i--;}
   // For other we eventually add zeros in front as needed
  } 
  
// The process of 'carry'
void report(Bignum A)
{int i,q,r;
 for(i=0;i<99;i++)
 {q=A[i]/10000;
  r=A[i]%10000;
  A[i]=r;
  A[i+1]+=q;
  }
 } 
 // addition
 void Ajouter(Bignum A, Bignum B, Bignum C)
 { int i;
   for(i=0;i<100;i++) C[i]=A[i]+B[i]; // addition digit by digit
 report(C);
 // result is in C
 }

// left-shift one digit
void Decaler (Bignum A, int n)
{ int i,j;
  if (n==0) return ;
  for(i=1; i<=n; i++)
  {for(j=99;j>0;j--)
   A[j]=A[j-1]; 
     A[0]=0;
     }
  }  
// Multiplication
void Multiplier(Bignum A, Bignum B, Bignum C)
{int i, j;Bignum D;
   for(i=0;i<100;i++) C[i]=0; // initialization with 0
    for(i=0;i<100;i++)
    {for(j=0;j<100;j++)
    D[j]=A[i]*B[j];
    report(D); // partial product
    Decaler(D,i); // shift according factor
    Ajouter(D,C,C);// add to C
    }
}
