// crible.cpp : définit le point d'entrée pour l'application console.
#include "stdafx.h"
#include <stdio.h>
#include <stdlib.h>
#include <math.h>

#define max 30000000

int racine=(int)sqrt((double)max)+1;

char Crible[max];

//initialisation du crible - initialization of sieve
void init ()
{
	int i;
	for (i=0;i<max;i++)
		Crible[i]=1;
	Crible[0]=Crible[1]=0;
}

// cribler - to sieve
void go()
{
	int k=2;
	int j;
	while (k<=racine)
	{
		for (j=k;j*k<max;j++)
			Crible[j*k]=0;
		k++;
		while (! Crible[k])
			k++;
	}
}

// compter les premiers - count the primes
unsigned int compte()
{
	int i;
	unsigned int s;
	for (i=0,s=0;i<max;i++)
		if (Crible[i])
			s++;
	return s;
}

int _tmain(int argc, _TCHAR* argv[])
{
	init();
	go();
	unsigned int n=compte();
	printf ("%.9lf\n",((double)n)/(max/log((double)max)));
	system("Pause");
	return 0;
}
