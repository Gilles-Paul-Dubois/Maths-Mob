# -*- coding: utf-8 -*-
# python version 3.4 
import math 
 
 
class DevLim: 
    def __init__(self, n, a, f, C): 
        """constructeur admettant en paramètre l'ordre, le point, la fonction et la suite des coefficients""" 
        """constructor taking as parameters theorder, the point, the function, and the sequence of coefficients"""         
        self.n = n 
        self.a = a 
        self.f = f 
        self.C = C 
 
    def approx(self, x): 
        """Approximation polynomiale de f - Polynomial approximation of f""" 
        return sum([self.C[k] * x ** k for k in range(self.n + 1)]) 
 
    def reste(self, x): 
        """erreur commise - the error""" 
        return self.f(x) - self.approx(x) 
 
    def epsilon(self, x): 
        """relativisation par rapport à (x-a)^n - comparison with(x-a)^n  """ 
        return self.reste(x) / ((x - self.a) ** self.n) 
 
 
def main(): 
    # Développement limité d'ordre 4 de la fonction sinus au voisinage de 0 
    # F.T.E. 4th order for  sine function in the neighborhood of 0      
    DL = DevLim(4, 0, math.sin, [0, 1, 0, -1 / 6.0, 0]) 
    S = [0.1, 0.01, 0.001] 
    for x in S: 
        print(DL.approx(x), DL.reste(x), DL.epsilon(x)) 
 
 
if __name__ == '__main__': 
    main()
