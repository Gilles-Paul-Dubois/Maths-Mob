# calcul symbolique en Python (développement limité d'ordre 6) 
# symbolic computation with Python  (F.T.E. order 6)
 
from sympy import * 
 
 
def main(): 
    x = Symbol('x') 
    print(series(log(3 * exp(x) + exp(-x)), x)) 
 
 
if __name__ == '__main__': 
    main()
