"""approximation des fonctions par des sommes partielles de séries entières""" 
"""approximation of functions by partial sums of power series""" 
 
import math 
 
 
def u1(x, k): 
    """ terme général de la série exponentielle - general term of exponential series""" 
    d = 1.0 
    for i in range(1, k + 1): 
        d *= i 
    return x ** k / d 
 
 
def u2(x, k): 
    """terme général de la série sinus - general term of sine series""" 
    if not k % 2: 
        return 0 
    else: 
        d = 1.0 
        for i in range(1, k + 1): 
            d *= i 
        v = x ** k / d 
        if ((k - 1) / 2) % 2: 
            v = -v 
    return v 
 
 
def u3(x, k): 
    """ terme général série 1/rac(1-x²) - general term of series 1/rac(1-x²)""" 
    n = 1.0 
    for i in range(1, 2 * k, 2): 
        n *= i 
    d = 1.0 
    for i in range(2, 2 * k + 1, 2): 
        d *= i 
    return ((x ** (2 * k)) * n) / d 
 
 
def S(u, x, start, end): 
    """Somme des éléments de la suite u lorsque i va de start à end compris - sum of elements when i goes from start to end included""" 
    return sum((u(x, k) for k in range(start, end + 1)))  # fait la somme de tous les u(k) pour k entre start et end 
 
 
def main(): 
    """fonction principale - main function""" 
    print(S(u1, 1, 0, 10), math.exp(1)) 
    print(S(u2, 1, 0, 10), math.sin(1)) 
    print(S(u3, 0.5, 0, 10), 1.0 / math.sqrt(1 - 0.5 * 0.5)) 
 
 
if __name__ == '__main__': 
    main()
