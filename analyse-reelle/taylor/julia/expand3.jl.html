using Polynomials
using MTH229

x=symbols("x")
f(x)=sin(x)
g(x)=cos(x)
sum(x)=f(x)+g(x)
prod(x)=f(x)*g(x)
# DL en h avec h=(x-a) - F.T.E. in h with h=(x-a)
function DL(g,a,n)
    fact(n)=reduce(*,1:n)
    cnf(g,a,i)=diff(g,i)(a)/fact(i)
    C=[cnf(g,a,i) for i in 0:n]
    return Polynomial(C,:h)
end

function produit(DL1,DL2,n)
    p(n)=Polynomial([0,1],:h)^n
    return rem(DL1*DL2,p(n+1))
end

println("somme de deux D.L.")
println(DL(f,0,5)+DL(g,0,5)) 
println(DL(sum,0,5))
println("produit de deux D.L.")
println(produit(DL(f,0,5),DL(g,0,5),5))
println(DL(prod,0,5))
