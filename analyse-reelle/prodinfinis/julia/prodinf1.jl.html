function prod0(L) #itératif simple
    P=1
    for f in L
        P*=f
    end
    return P
end

prod1(L)= isempty(L) ? 1 : reduce(*,L) #itératif via reduce

prod2(L)= isempty(L) ? 1 : pop!(L)*prod2(L) #récursif

function main()
    println("test-prod0")
    println(prod0([]))  # liste vide
    println(prod0([1]))  # un seul facteur
    L = [i for i in 1:4]  # factorielle 4
    println(prod0(L))
    println("test-prod1")
    println(prod1([]))  # liste vide
    println(prod1([1]))  # un seul facteur
    L = [i for i in 1:4]  # factorielle 4
    println(prod1(L))
    println("test-prod2")
    println(prod2([]))  # liste vide
    println(prod2([1]))  # un seul facteur
    L = [i for i in 1:4]  # factorielle 4
    println(prod2(L))
end

main()
