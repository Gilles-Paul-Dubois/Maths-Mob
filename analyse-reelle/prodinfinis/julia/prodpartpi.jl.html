function liste_2000_premiers()
    P, p,i = [2, 3], 3,2
    while i < 2000
        p += 2 #inutile de tester les nombres pairs - not necessary to check even numbers
        if all((p % x != 0 for x in P))
            push!(P, p)
            i+=1
         end
    end
    return P
end

L=liste_2000_premiers()

premier(n)=L[n]

function v(n)
    p=premier(n)
    return (p-1)%4==0 ? 1+1/p : 1 -1/p
end

prod(n)=reduce(*,(v(i) for i in 2:n))

for i in [100,500,1000,1800]
    println(2/prod(i))
end
