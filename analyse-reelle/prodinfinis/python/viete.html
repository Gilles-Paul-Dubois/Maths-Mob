# -*- coding: utf-8 -*-
import math
from functools import reduce 

def v(n):
    """facteur général de la suite de Viète - general factore of Viete(s sequence"""
    if n==1:
        return math.sqrt(2)/2
    return math.sqrt((1+v(n-1))/2)

def p(n):
    """produit partiel d'ordre n - partial product of order n"""
    return reduce(lambda x,y:x*y,(v(i) for i in range(1,n+1)))

#fonction principale - main function
print (p(19))
print (2/math.pi)
