"""Exemple de produit infini - Formule de Wallis""" 
"""Example of infinite product- Wallis formula"""  
 
def genw(): 
    """générateur du produit de Wallis - generator of Wallis product""" 
    k = 0 
    p = 2 
    yield p 
    while True: 
        k += 1 
        v = 1 - 1.0 / ((2 * k + 1) * (2 * k + 1)) 
        p *= v 
        yield p 
 
 
def main(): 
    """fonction principale - main function""" 
    Wallis = genw() 
    P = 2 
    for i in range(0, 5000): 
        P = next(Wallis) 
    print(P * 2) 
 
 
if __name__ == "__main__": 
    main()
