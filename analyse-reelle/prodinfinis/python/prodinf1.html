from functools import reduce 
import itertools as it 
import operator as op 
 
 
def prod0(L): 
    """produit itératif simple - simple iterative product""" 
    P = 1 
    while L: 
        P *= L[0] 
        L = L[1:] 
    return P 
 
 
def prod1(L): 
    """produit itératif des éléments d'une liste avec la fonction reduce - iterative product using reduce function""" 
    if len(L) == 0: 
        return 1 
    if len(L) == 1: 
        return L[0] 
    return reduce(lambda x, y: x * y, L) 
 
 
def prod2(L): 
    """produit récursif des éléments d'une liste - recursive product of elements of a list""" 
    if len(L) == 0: 
        return 1 
    if len(L) == 1: 
        return L[0] 
    return L[0] * prod2(L[1:]) 
 
 
def main(): 
    """fonction principale - main function""" 
    print("test prod0") 
    print(prod0([]))  # liste vide - empty list 
    print(prod0([1]))  # un seul facteur - only one factor 
    L = [i for i in range(1, 4)]  # factorielle 4 - 4! factorial 4 
    print(prod0(L)) 
    print("test prod1") 
    print(prod1([]))  # liste vide - empty list 
    print(prod1([1]))  # un seul facteur - only one factor 
    L = [i for i in range(1, 4)]  # factorielle 4 - factorial 4 
    print(prod1(L)) 
    print("test prod2") 
    print(prod2([]))  # liste vide - empty list 
    print(prod2([1]))  # un seul facteur - only one factor 
    L = [i for i in range(1, 4)]  # factorielle 4 - factorial 4 
    print(prod2(L)) 
    print("test prod3") 
    # nouveau avec python 3.4 - new with Python 3.4 
    # ramène la suite des produits partiels - computes the sequence of partial products 
    L = (it.accumulate([1, 2, 3, 4, 5], op.mul)) 
    for x in L: 
        print(x) 
 
 
if __name__ == '__main__': 
    main()
