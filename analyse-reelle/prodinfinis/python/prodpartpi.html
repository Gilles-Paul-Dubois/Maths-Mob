import math
# calcul de produits partiels - computation of partial products

def premgen():
    """génère la suite des nombres premiers - generates the sequence of prime numbers"""
    L = [2]
    p = L[-1]
    yield p
    while True:
        k = p + 1
        while any(k % x == 0 for x in L):
            k += 1
        p = k
        L.append(p)
        yield p

def premier(n):
    """retourne le n-ième nombre premier - returns the n-th prime number"""
    P = premgen()
    Premiers = [next(P) for i in range(0, n)]
    return Premiers[-1]  # dernier élément de la liste

def v(n):
    """facteur général du produit - general factor of the product"""
    p = premier(n)
    if ((p - 1) / 2) % 2 == 0:
        return 1 + 1.0 / p
    else:
        return 1 - 1.0 / p

def prod(n):
    """produit partiel d'ordre n - partial product of order n"""
    k = 1
    for i in range(2, n + 1):
        k *= v(i)
    return k

def main():
    """fonction principale - main function"""
    n = input("Donner l'entier n<1000: ")  # saisie d'un entier sous forme de chaîne
    p = prod(int(n))  # calcul du produit partiel- computation of the partial product
    print(2/p)

if __name__ == "__main__":
    main()
