# vérification du développement du sinus en produit infini (Euler) 
# checking for expansion of sinus as an infinite productn (Euler) 
 
import math 
 
 
def v(x, n): 
    """facteur général du produit d'Euler- general factor of Euler's product""" 
    return 1 - x * x / (n * n * math.pi * math.pi) 
 
 
def p(n, x): 
    """produit partiel - partial product""" 
    R = 1 
    for i in range(1, n + 1): 
        R *= v(x, i) 
    return R 
 
 
def main(): 
    """fonction principale - main function""" 
    L = [math.pi / k for k in [6, 4, 3]] 
    for x in L: 
        print(p(100, x), math.sin(x) / x) 
 
 
if __name__ == '__main__': 
    main()
