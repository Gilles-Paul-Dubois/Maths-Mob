import math 
 
 
def PrimApprox(f, x, n, F0): 
    """Approximation de la primitive de f (à supposer qu'elle existe)  
    prenant la valeur F0 en 0 
    en allant de 0 à x par un step de x/n""" 
    """Approximating the anti-derivative of f (supposing it exists)  
    taking the value F0 in 0
    going from 0 to x with a step of x/n""" 
    epsilon = x / n 
    F = F0 
    
    for i in range(1, n + 1): 
        F = F + f(i * epsilon) * epsilon 
    return F 
 
 
def f1(x): 
    """ une fonction dont on connait la primitive = atan juste pour test de la méthode - a function the anti-derivative of which is known just for checking""" 
    return 1 / (1 + x * x) 
 
 
def f2(x): 
    """la fonction donnée en exemple ci-dessus, primitive inconnue - function giv en above as an example integral unknown""" 
    return math.sqrt(1 + math.sin(x) * math.sin(x)) 
 
 
def main(): 
    """ on teste la méthode avec 100,200,300, ...,1000 itérations 
    pour calculer une approximation de la primitive en x=1""" 
    """ We check the method with 100,200,300, ...,1000 iterations 
    to compute an approximation of the integral in x=1"""     
    for i in range(1, 11): 
        n = i * 100 
        print(PrimApprox(f1, 1.0, n, 0), math.pi / 4)  # pour vérifier - to check 
        print(PrimApprox(f2, 1.0, n, 0)) 
 
 
if __name__ == '__main__': 
    main()
