import math 
 
# ## Calcul de ln(x) à 0.00001 près au moyen de la série ln(1+x) 
# ## Computation of ln(x) with precision 0.00001 using the series ln(1+x) 
def u(n, x): 
    """terme général de la série ln(1+x)- general term of the series ln(1+x)""" 
    r = 1 
    for k in range(0, n): 
        r *= x 
    if n % 2 == 0: 
        return -r / float(n) 
    else: 
        return r / float(n) 
 
 
def s(n, x): 
    """somme partielle de ln(1+x) - partial sum of ln(1+x)""" 
    s = 0 
    for k in range(1, n + 1): 
        s += u(k, x) 
    return s 
 
 
def ordre(x): 
    """jusqu'où pousser la sommation pour une erreur inférieure à 0.00001 dans ln(1+x) si -1<x<0""" 
	 """Till which rank should we go in partial sums to get an error less than 0.00001 in ln(1+x) if -1<x<0"""    
    n = 1 
    e = 1 
    while True: 
        e *= (-x) 
        if abs(e) / (n * (1 + x)) < 0.00001: 
            return n 
        n += 1 
 
 
def approxlog1(x): 
    """Approximation du logarithme de 1+x pour x dans ]0,1[""" 
    """Approximation of the logarithm of 1+x for x in]0,1["""    
    n = ordre(x) 
    return s(n, x) 
 
 
def approxlog(x): 
    """Cas général - general case""" 
    if x == 1: 
        return 0 
    if x < 1: 
        return approxlog1(x - 1) 
    return -approxlog(1.0 / x)  #cas x>1 
 
 
def main(): 
    """Fonction principale - main function""" 
    for x in [k / 10.0 for k in range(1, 21)]: 
        print(approxlog(x), math.log(x)) 
 
 
if __name__ == '__main__': 
    main()
