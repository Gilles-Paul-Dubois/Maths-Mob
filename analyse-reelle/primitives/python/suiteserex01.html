"""Série arctangente pour des valeurs rationnelles de la variable x"""
"""atan series for rational values of variable x"""
from fractions import Fraction


def u(n, x):  # x doit être une fraction
    """terme général de la série arctangente - general term of atan series"""
    if n % 2 == 0:
        return Fraction(0)
    else:
        r = Fraction(1)
        for k in range(1, n + 1):
            r *= x
        m = (n - 1) / 2
        if m % 2 == 0:
            return r / Fraction(n)
        else:
            return -r / Fraction(n)


def s(n, x):  # x doit être une fraction - should be a fraction
    """Somme partielle de la série - partial sum of the series"""
    s = Fraction(0)
    for k in range(1, n + 1):
        s += u(k, x)
    return s


def Machin(n):
    """Approximation de Machin à l'ordre - Machin's approximation at order n n"""
    return Fraction(4) * ((Fraction(4) * s(n, Fraction(1, 5))) - s(n, Fraction(1, 239)))


def main():
    for i in range(1, 22):
        print(i, float(Machin(i)))


if __name__ == '__main__':
    main()
