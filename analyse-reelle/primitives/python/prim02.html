from fractions import Fraction 
 
 
def pow2(n): 
    """les puissances de 2 entières""" 
    if n == 0: 
        return 1 
    else: 
        return 2 * pow2(n - 1) 
 
 
def coefbinomial(p, n): 
    """les coefficients du binôme""" 
    if p > n: 
        return Fraction(0) 
    if p == 0: 
        return Fraction(1) 
    return coefbinomial(p - 1, n - 1) + coefbinomial(p, n - 1) 
 
 
def coeflinimpcos(k, n):  # cas n=2m+1 
    m = (n - 1) / 2 
    if k % 2 == 0: 
        return Fraction(0) 
    else: 
        return coefbinomial(m - (k - 1) / 2, n) * Fraction(1, pow2(2 * m)) 
 
 
def coeflinpaircos(k, n):  # cas n=2m 
    m = n / 2 
    if k % 2 != 0: 
        return Fraction(0) 
    else: 
        if k == 0: 
            return coefbinomial(m, n) * Fraction(1, 2) * Fraction(1, pow2(2 * m - 1)) 
        else: 
            return coefbinomial(m - k / 2, n) * Fraction(1, pow2(n - 1)) 
 
 
def coeflincos(k, n): 
    """coefficient d'indice k de la linéarisation de cos^n""" 
    if n % 2 == 0: 
        return coeflinpaircos(k, n) 
    else: 
        return coeflinimpcos(k, n) 
 
 
def main(): 
    for n in range(2, 6): 
        print([coeflincos(k, n) for k in range(0, n + 1)]) 
 
 
if __name__ == '__main__': 
    main()
