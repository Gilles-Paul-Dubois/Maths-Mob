import math 
 
"""Série arctangente pour des valeurs rationnelles de la variable x""" 
"""atan series for rational values of the variable x""" 
from fractions import Fraction 
 
 
def u(n):  # x doit être une fraction - x should be a fraction 
    """terme général de la série arctan(1/rac(3)) - general term of the series atan(1/√3) """ 
    x = Fraction(1, 3) 
    r = Fraction(1) 
    for k in range(1, n + 1): 
        r *= x 
    if n % 2 == 0: 
        return r / Fraction(2 * n + 1) 
    else: 
        return -r / Fraction(2 * n + 1) 
 
 
def s(n):  # x doit être une fraction - x should be a fraction 
    """Somme partielle de la série - partial sum of the series""" 
    s = Fraction(0) 
    for k in range(0, n + 1): 
        s += u(k) 
    return s 
 
 
def Madhava(n): 
    """Approximation de Madhava à l'ordre n Madhava's approx n-th order""" 
    return (6 / math.sqrt(3)) * float(s(n)) 
 
 
def main(): 
    """Calcul de pi par la série de Gregory/Madhava/Newton/Sharp - Computation of π using the series""" 
    for i in range(1, 30): 
        print(i, "%.15f" % (Madhava(i))) 
 
 
if __name__ == '__main__': 
    main()
