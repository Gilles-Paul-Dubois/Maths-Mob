from fractions import Fraction 
from functools import reduce 
 
# ************************polynôme à coefficients rationnels- polynomials with rational coefficients********************** 
def ajoute(u, v): 
    """somme de deux listes éventuellement de longueurs différentes - add two lists with different lengths""" 
    if u == []: 
        return v 
    if v == []: 
        return u 
    return [u[0] + v[0]] + ajoute(u[1:], v[1:]) 
 
 
def soustrait(u, v): 
    """différence de deux listes éventuellement de longueurs différentes - subtract two lists with possible different lengths""" 
    if u == []: 
        return [-x for x in v] 
    if v == []: 
        return u 
    return [u[0] - v[0]] + soustrait(u[1:], v[1:]) 
 
 
def facteur(u, v, n): 
    """coefficient d'indice n du produit de convolution - n-th coeff. of convolution product """ 
    r = [u[i] * v[j] for i in range(0, len(u)) for j in range(0, len(v)) if i + j == n] 
    if len(r) == 0: 
        return Fraction(0) 
    if len(r) == 1: 
        return r[0] 
    return reduce(lambda x, y: x + y, r) 
 
 
def multiplie(u, v): 
    """ produit de convolution de deux listes -convolution product of two lists""" 
    return [facteur(u, v, n) for n in range(0, len(u) + len(v))] 
 
 
class Polrat: 
    """ la classe des polynômes à coefficients rationnels - class of rational polynomials""" 
 
    def __init__(self, listrat):  #constructeur - constructor 
        n = 0 
        listrat.reverse() 
        if len(listrat) >= 1: 
            for x in listrat: 
                if x == 0: 
                    n += 1 
                    continue 
                else: 
                    break 
        listrat = listrat[n:] 
        listrat.reverse() 
        self.a = listrat 
 
    def __str__(self):  #représentation externe pour print, etc..- conversion to string 
        return str([x for x in self.a]) 
 
    def __add__(self, other):  #addition 
        return Polrat(ajoute(self.a, other.a)) 
 
    def __sub__(self, other):  #soustraction - subtraction 
        return Polrat(soustrait(self.a, other.a)) 
 
    def __mul__(self, other):  #produit de 2 polynômes - product of polynomials 
        return Polrat(multiplie(self.a, other.a)) 
 
    def degre(self):  #degré d'un polynôme - degree of a polynomial 
        if self.a == []: 
            return -1 
        return len(self.a) - 1 
 
    def coeffdom(self):  #coefficient dominant - leading coefficient 
        d = self.degre() 
        if d < 0: 
            return 0 
        else: 
            return self.a[d] 
 
    def __truediv__(self, other):  # quotient division euclidienne - quotient in Euclidean division 
        m = self.degre() 
        n = other.degre() 
        x = self.coeffdom() 
        y = other.coeffdom() 
        d = m - n 
        if d < 0: 
            return Polrat([]) 
        r = x / y 
        q = [Fraction(0)] * (d + 1) 
        q[d] = r 
        T1 = Polrat(q) 
        P = self - (T1 * other) 
        T2 = P / other 
        return T2 + T1 
 
    def __mod__(self, other):  # reste division euclidienne - remainder in Euclidean division 
        q = self / other 
        return self - other * q 
 
    def decompose(self, other):  #décomposition suivant les puissances croissantes - decomposition according increasing powers 
        m = self.degre() 
        n = other.degre() 
        if m < n: 
            return [self] 
        q = self / other 
        r = self % other 
        return [r] + q.decompose(other)
