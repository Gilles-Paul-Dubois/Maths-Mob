class Pol2Var(): 
    """Modélisation des polynômes à deux variables""" 
 
    def __init__(self, L): 
        """Constructeur à partir d'un dictionnaire""" 
        self.coeffs = L 
 
    def __str__(self): 
        """représentation externe pour print, etc...""" 
        K = self.coeffs.keys() 
        S = "" 
        for k in K: 
            if self.coeffs[k] > 0: 
                S += "+" + str(self.coeffs[k]) + "X^" + str(k[0]) + "Y^" + str(k[1]) 
            else: 
                S += str(self.coeffs[k]) + "X^" + str(k[0]) + "Y^" + str(k[1]) 
        if S == "": 
            S = "0" 
        return S 
 
    def simplif(self): 
        """enlever tous les coefficients nuls""" 
        L = {} 
        for k, v in self.coeffs.items(): 
            if v != 0: 
                L[k] = v 
        return Pol2Var(L) 
 
    def __rmul__(self, x): 
        """multiplication par un scalaire""" 
        if x == 0: 
            return Pol2Var({}) 
        else: 
            L = self.coeffs 
            for k, v in L.items(): 
                L[k] = v * x 
            return Pol2Var(L) 
 
    def __mul__(self, other): 
        """produit d'un polynôme par un autre""" 
        L = {} 
        for k1, v1 in self.coeffs.items(): 
            for k2, v2 in other.coeffs.items(): 
                k = (k1[0] + k2[0], k1[1] + k2[1]) 
                if k in L.keys(): 
                    L[k] += v1 * v2 
                else: 
                    L[k] = v1 * v2 
        P = Pol2Var(L) 
        return P.simplif() 
 
 
    def __add__(self, other): 
        """addition de deux polynomes""" 
        L1 = self.coeffs 
        L2 = other.coeffs 
        K1 = self.coeffs.keys() 
        K2 = other.coeffs.keys() 
        I = [k for k in K1 if k in K2]  # indices communs 
        K1 = [k for k in K1 if k not in I] 
        K2 = [k for k in K2 if k not in I] 
        L = {} 
        for k in I: 
            L[k] = L1[k] + L2[k] 
        for k in K1: 
            L[k] = L1[k] 
        for k in K2: 
            L[k] = L2[k] 
        P = Pol2Var(L) 
        return P.simplif() 
 
    def __sub__(self, other): 
        """soustraction de deux polynomes""" 
        L1 = self.coeffs 
        L2 = other.coeffs 
        K1 = self.coeffs.keys() 
        K2 = other.coeffs.keys() 
        I = [k for k in K1 if k in K2]  # indices communs 
        K1 = [k for k in K1 if k not in I] 
        K2 = [k for k in K2 if k not in I] 
        L = {} 
        for k in I: 
            L[k] = L1[k] - L2[k] 
        for k in K1: 
            L[k] = L1[k] 
        for k in K2: 
            L[k] = -L2[k] 
        P = Pol2Var(L) 
        return P.simplif() 
 
    def __neg__(self): 
        """opposé d'un polynôme""" 
        return Pol2Var({}) - self 
 
    def __pow__(self, n): 
        """puissance entière d'un polynôme""" 
        P = Pol2Var({(0, 0): 1}) 
        if n == 0: 
            return P 
        else: 
            for i in range(1, n + 1): 
                P = P * self 
        return P 
 
 
def main(): 
    P1 = Pol2Var({(0, 0): 1, (1, 0): -1, (0, 1): 1}) 
    P2 = Pol2Var({(0, 0): 1, (0, 1): 1}) 
    print(P1) 
    print(P2) 
    print(2 * P2) 
    print(P1 + P2) 
    print(P1 - P2) 
    print(-P1) 
    print(P1 * P2) 
    print(P1 ** 2) 
 
 
if __name__ == '__main__': 
    main()
