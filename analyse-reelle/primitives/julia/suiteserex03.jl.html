function u(n,x)
    r=1
    for k in 1:n
        r*=x
    end
    n%2==0 && return -r/n
    return r/n
end

s(n,x)=reduce(+,(u(k,x) for k in 1:n))

function ordre(x)
    n,e=1,1
    while true
        e *=(-x)
        if abs(e)/(n*(1+x)) < 0.00001
            return n
        end
        n+=1
    end
end

approxlog1(x)= s(ordre(x),x)

function approxlog(x)
    x==1 && return 0
    x <1 && return approxlog1(x-1)
    return -approxlog(1/x)
end

for x in (k/10 for k in 1:20)
    print(approxlog(x)," : ",log(x),"\n")
end
