using Polynomials

p=Polynomial([1//1, 1//1, 1//1],:x)

function derive(p)
    C=p.coeffs
    K=C[2:end]
    K=[ (i-1)*C[i] for i in 2:length(K)+1]
    return Polynomial(K,:x)
end

function integre(p)
    C=p.coeffs
    K=append!([0//1],C)
    for i in 2:length(K)
        K[i]=K[i]//(i-1)
    end
    return Polynomial(K,:x)
end

println(p)
println(derive(p))
println(integre(p))
