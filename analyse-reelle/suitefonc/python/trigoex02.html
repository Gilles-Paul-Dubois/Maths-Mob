# modélisation d'un espace des séries de fonctions - model for a space of series of functions 
from functools import reduce 
import math 
 
 
class SerieFonc: 
    def __init__(self, f): 
        """constructeur à partir du terme général - constructor from the general term""" 
        self.u = f 
 
    def SomPart(self, n, x): 
        """Somme partielle d'indice n calculée en x - partial sum of index n computed on x""" 
        if n == 0: 
            return self.u(0, x) 
        else: 
            return reduce(lambda x, y: x + y, (self.u(i, x) for i in range(0, n + 1))) 
 
    def __add__(self, other): 
        """Somme de deux séries de fonctions - sum of two series of functions""" 
        return SerieFonc(self.u + other.u) 
 
    def __rmul__(self, k): 
        """Produit par un scalaire - product by a scalar""" 
        return SerieFonc(lambda n, x: k * self.u(n, x)) 
 
    def __mul__(self, other): 
        """produit de convolution - convolution product""" 
 
        def f(n, x): 
            if n == 0: 
                return self.u(0, x) * other.u(0, x) 
            else: 
                return reduce(lambda x, y: x + y, (self.u(i, x) * other.u(n - i, x) for i in range(0, n + 1))) 
 
        return SerieFonc(f) 
 
 
class SerieTrigo(SerieFonc): 
    """Modélisation des séries trigonométriques, classe dérivée de la précédent - Modelisation of trigonometric series by derivation from the previous classe""" 
 
    def __init__(self, a, b): 
        """constructeur au moyen des deux suites des coefficients - constructor from the series of coefficients""" 
 
        def f(n, x): 
            return a(n) * math.cos(n * x) + b(n) * math.sin(n * x) 
 
        self.u = f 
 
 
def test3(): 
    """tester la somme de la seconde série trigonométrique - check the sum of second series""" 
 
    def b3(n): 
        return 0 
 
    def a3(n): 
        if n == 0: 
            return math.pi * math.pi / 3 
        if n % 2 == 0: 
            return 4.0 / (n * n) 
        else: 
            return -4.0 / (n * n) 
 
    T3 = SerieTrigo(a3, b3) 
    for x in [k * 0.1 for k in range(1, 30)]: 
        print(T3.SomPart(1000, x), x * x) 
 
 
def main(): 
    """Fonction principale - main function """ 
    test3()
