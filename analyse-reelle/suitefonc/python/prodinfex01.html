import math 
from functools import reduce 
 
 
class Prodinf: 
    """Classe modélisant les produit infinis dont le facteur général est (1+un(x)) - model for infinite product with general factor (1+un(x))""" 
 
    def __init__(self, f): 
        """constructeur à partir de f(n,x) - constructor from f(n,x)""" 
        self.u = f 
 
    def ProdPart(self, n, x): 
        """Produit partiel d'indice n calculé en x - partial product index n computed at x""" 
        if n == 0: 
            return 1 + self.u(0, x) 
        else: 
            return reduce(lambda x, y: x * y, (1 + self.u(i, x) for i in range(0, n + 1))) 
 
 
def main(): 
    """vérification de la formule du développement du sinus en produit infini - checking of Wallis formula""" 
 
    def v(n, x): 
        if n == 0: 
            return 0 
        else: 
            return -(x * x) / (n * n * math.pi * math.pi) 
 
    P = Prodinf(v) 
    for k in range(1, 11): 
        x = k * 0.1 
        print(P.ProdPart(100, x), math.sin(x) / x) 
 
 
if __name__ == '__main__': 
    main()
