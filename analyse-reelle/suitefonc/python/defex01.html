import math 
# série exponentielle  - exponential series : 1+x+x2/2!+ ... +xn/n!+ ... 
def S(x): 
    """générateur des sommes partielles - generator for partial sums""" 
    n = 0 
    u = 1 
    s = 1 
    yield s 
    while True: 
        n += 1 
        u *= x / n 
        s += u 
        yield s 
 
 
def main(): 
    """fonction principale - main function""" 
    x = 1.0 
    E = S(x) 
    for i in range(0, 10): 
        print(next(E)) 
    print(math.exp(x)) 
 
 
if __name__ == '__main__': 
    main()
