import math 
 
# # exemple de calcul d'une série de fonctions 
## la série de terme général x^n/n! 
"""Dans ce cas précis l'algorithme peut être beaucoup amélioré mais ce schéma (calcul du terme général, 
puis des sommes partielles à partir de ce terme général) fonctionne dans tous les cas""" 
 
 
def u(x, n): 
    """terme général de la série""" 
    r = 1.0 
    for i in range(1, n + 1): 
        r = r * x / i 
    return r 
 
 
def s(x, n): 
    """sommes partielles""" 
    if n == 0: 
        return u(x, 0) 
    if n == 1: 
        return u(x, 0) + u(x, 1) 
    return sum(u(x, i) for i in range(0, n + 1)) 
 
 
def main(): 
    """affichage des 10 premières sommes partielles pour une valeur particulière de la variable""" 
    x = 1 
    for i in range(0, 10): 
        print(s(x, i)) 
 
 
if __name__ == '__main__': 
    main()
