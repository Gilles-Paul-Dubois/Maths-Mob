# Calcul des sommes partielles d'une série entière 
# Computation of partial sums of a power series 
 
def a(n): 
    """coefficients de la série -ici la série géométrique simple - coefficients of the series here the usual geometric progression""" 
    return 1.0 
 
 
def u(x, n): 
    """terme général - general term""" 
    r = 1 
    for i in range(1, n + 1): 
        r *= x 
    return a(n) * r 
 
 
def s(x, n): 
    """Somme partielle - Partial sum""" 
    r = 0 
    for i in range(0, n + 1): 
        r += u(x, i) 
    return r 
 
 
def main(): 
    """fonction principale - main function""" 
    x = 0.5 
    print(s(x, 20)) 
    print(1 / (1 - x))  # juste pour vérifier - just to check 
 
 
if __name__ == '__main__': 
    main()
