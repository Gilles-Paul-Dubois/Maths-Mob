import math 
# série sinus - sine series :  x-x3/3!+ x5/5!... +(-1)nx2n+1/(2n+1)!+ ... 
def S(x): 
    """générateur des sommes partielles - generator for partial sums""" 
    n = 1 
    u = x 
    s = x 
    yield s 
    while True: 
        n += 2 
        u *= x * x / ((n - 1) * n) 
        u = -u 
        s += u 
        yield s 
 
 
def main(): 
    """fonction principale - main function""" 
    x = math.pi / 6 
    E = S(x) 
    for i in range(0, 10): 
        print(next(E)) 
    print(math.sin(x)) 
 
 
if __name__ == '__main__': 
    main()
