struct Prodinf
    v::Function
    p::Function
end

#constructeur à partir du facteur général de la forme 1+v - constructor from general factor 1+v
function Prodinf(v)
     p(x,n)=reduce(*,(1+v(x,k) for k in 0:n))
     return Prodinf(v,p)
end

function main()
    a(n)= n==0 ? 3 : 2*a(n-1)*a(n-1)-1
    v(x,n)= x/a(n)
    P=Prodinf(v)
    print(P.p(1,3)," : ",sqrt(2), "\n")
end

main()
