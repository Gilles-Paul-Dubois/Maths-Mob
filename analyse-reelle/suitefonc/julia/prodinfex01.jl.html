struct Prodinf
    v::Function
    p::Function
end

#constructeur à partir du facteur général de la forme 1+v - constructor from genral factor 1+v
function Prodinf(v)
     p(x,n)=reduce(*,(1+v(x,k) for k in 0:n))
     return Prodinf(v,p)
end

function main()
    v(x,n)= n==0 ? 0 : -x*x/(n*n*pi*pi)
    P=Prodinf(v)
    for x in (k*0.1 for k in 1:20)
        print(P.p(x,100)," : ",sin(x)/x, "\n")
    end
end

main()
