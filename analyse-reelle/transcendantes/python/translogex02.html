import math 
 
"""Calcul de ln(2) en utilisant la définition du log au moyen d'une intégrale par la méthode des trapèzes""" 
"""CComputation of ln(2) using definition of  log as a definite integral  by method of trapezes""" 
 
 
def trapezes(a, b, n, f): 
    """méthode d'approximation dite 'des trapèzes' - approximation method known as 'trapezes'""" 
    S = 0 
    for i in range(0, n): 
        x1 = a + (b - a) * i / float(n) 
        x2 = a + (b - a) * (i + 1) / float(n) 
        S += (f(x1) + f(x2)) / 2.0 * (x2 - x1) 
    return S 
 
 
def fonc(x): 
    return 1.0 / x 
 
 
def main(): 
    """Fonction de tests - testing function""" 
    print(trapezes(1, 2, 20, fonc)) 
    print(math.log(2)) 
 
 
if __name__ == '__main__': 
    main()
