"""Calcul de ln(2) par le développement en série de ln(1+x) pour x=-0.5""" 
"""Computation of ln(2) by the power series expansion for ln(1+x) at x=-0.5""" 
import math 
 
 
def u(n, x): 
    """terme général de la série - general term of the series""" 
    r = 1.0 
    for i in range(1, n + 1): 
        r *= x 
    r /= n 
    if (n % 2) != 0: 
        return r 
    else: 
        return -r 
 
 
def s(n, x): 
    """somme partielle d'ordre n - partial sum with rank n""" 
    s = 0 
    for i in range(1, n + 1): 
        s += u(i, x) 
    return s 
 
 
def main(): 
    x = -0.5 
    print(-s(20, x)) 
    print(math.log(2)) 
 
 
if __name__ == '__main__': 
    main()
