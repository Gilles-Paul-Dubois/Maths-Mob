import math 
from functools import reduce 
 
 
def fact(n): 
    """la factorielle - the factorial""" 
    if n <= 1: 
        return 1 
    return reduce(lambda x, y: x * y, range(1, n + 1)) 
 
 
def u(n): 
    """terme général de la série exp(1) - general term of series exp(1)""" 
    return 1.0 / fact(n) 
 
 
def k(u, v, n): 
    """terme général d'une série produit - general term of a product series""" 
    if n == 0: 
        return u(0) * v(0) 
    return sum((u(i) * v(n - i) for i in range(0, n + 1))) 
 
 
def p(u, v, n): 
    """"somme partielle de la série produit - partial sum of product series""" 
    if n == 0: 
        return k(u, v, 0) 
    return sum((k(u, v, i) for i in range(0, n + 1))) 
 
 
def main(): 
    print(p(u, u, 10))  # somme partielle d'ordre 10 de la série produit exp(1)*exp(1) - partial sum of order 10 for product exp(1)*exp(1) 
    # affichera 7.38899470899 - will print 7.38899470899  
    print(math.exp(2)) 
    # affichera 7.38905609893 - will print 7.38905609893 
 
 
if __name__ == '__main__': 
    main()
