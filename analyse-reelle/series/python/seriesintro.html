# -*- coding: utf-8 -*-
# python version 3.4 
def u(n): 
    """ une suite quelconque - a random sequence""" 
    return 1 if n % 2 == 0 else -1 
 
 
def S(u, J): 
    """Somme des éléments de la suite u lorsque i parcourt une liste d'indice J""" 
    """Sum of the elements of the list  u when i belongs to some set of indices  J"""     
    return sum((u(k) for k in J))  # fait la somme de tous les u(k) pour k dans J - computes the sum of all the u(k) for k belonging to J 
 
 
def main(): 
    """fonction principal - main function e""" 
    print(S(u, [])) 
    print(S(u, [0])) 
    print(S(u, [0, 5, 7, 12, 13])) 
 
 
if __name__ == '__main__': 
    main()
