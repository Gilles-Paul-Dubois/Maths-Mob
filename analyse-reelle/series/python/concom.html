# réarrangement de la série harmonique alternée pour converger vers 1 
# Rearrange the alternating harmonic series for a convergence to limit 1
 
def u(n): 
    """terme général de la série harmonique alternée - general term of the series""" 
    if n % 2 == 0: 
        return 1.0 / (n + 1) 
    else: 
        return -1.0 / (n + 1) 
 
 
def v(n): 
    """suite extraite des termes positifs - subseries of positive terms""" 
    return 1.0 / (2 * n + 1) 
 
 
def w(n): 
    """suite extraite des termes négatifs - subseries of negative terms""" 
    return -1.0 / (2 * (n + 1)) 
 
 
def genp(): 
    """générateur de la permutation p telle que la série de terme général u(p(n)) converge vers 1""" 
    """generator of permutation p """     
    n = 1 
    m = 0 
    s = 1 
    p = 0 
    yield p 
    while True: 
        if s < 1:  # si s est inférieure à 1, ajouter un positif - if s <1 add a positive 
            s += v(n) 
            p = 2 * n 
            n = n + 1 
            yield p 
        else:  #sinon ajouter un négatif else add a negative 
            s += w(m) 
            p = 2 * m + 1 
            m = m + 1 
            yield p 
 
 
def main(): 
    """fonction principale - main function """ 
    P = genp() 
    L = [next(P) for i in range(0, 300)] 
    M = [u(i) for i in L] 
    print(sum(M)) 
 
 
if __name__ == '__main__': 
    main()
