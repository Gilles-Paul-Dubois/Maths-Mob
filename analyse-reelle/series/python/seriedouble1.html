# -*- coding: utf-8 -*-
# python version 3.4 
 
import math 
 
 
def u(m, n): 
    """définition du terme général de la série double""" 
    """definition of the general term of the double series"""     
    return math.sin(m) * math.sin(n) 
 
 
def s(u, m, n): 
    """calcul de la somme partielle du rectangle de dimensions m,n""" 
    """computation of the partial sum of the rectangle of dimensions m,n"""     
    if m == 0 or n == 0:  # cas où une des dimensions est nulle - case where one dimension is null 
        return 0 
    if m == 1 and n == 1:  # un seul terme - a single term 
        return u(0, 0) 
    return sum([sum([u(i, j) for i in range(0, m)]) for j in range(0, n)])  #double sommation 'for,for' - double summation 
 
 
def main(): 
    """fonction principale - main function  """ 
    print(s(u, 0, 0)) 
    print(s(u, 1, 1)) 
    print(s(u, 2, 3)) 
 
 
if __name__ == '__main__': 
    main()
