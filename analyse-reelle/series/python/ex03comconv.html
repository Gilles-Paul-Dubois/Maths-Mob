from functools import reduce


def u(n):
    """terme général de la série harmonique alternée"""
    """general term of alternating harmonic series"""    
    if n % 2 == 0:
        return 1.0 / (n + 1)
    else:
        return -1.0 / (n + 1)


def genp():
    """générateur de la suite des indices permutés"""
    """generator of the sequence of permuted indices"""    
    p = 0
    k = 1
    yield p
    lasteven=p
    lastodd=-1
    last=p
    while True:
        k=k+1
        if last==lasteven:
            for h in range(1,k+1):
                p=lastodd+2*h
                yield(p)
            lastodd=p
            last=p
        else:
            for h in range(1,k+1):
                p=lasteven+2*h
                yield p
            lasteven=p
            last=p

def main():
    """fonction principale - main function"""
    P = genp()
    L = [next(P) for i in range(0, 200)]
    print(L)  # affichage des indices permutés - displaying permuted indices 
    M = [u(k) for k in L]  # série permutée - permuted series 
    S = [reduce(lambda x, y: x + y, M[0:k]) for k in range(2, 200)]  # sommes partielles - partial sums
    print(S)


if __name__ == '__main__':
    main()
