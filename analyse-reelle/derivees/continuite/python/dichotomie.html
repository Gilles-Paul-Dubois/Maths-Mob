def f(x): 
    """On cherche une annulation de cette fonction - searching where this function vanishes""" 
    return x * x - 2 
 
 
def Dichot(f, a, b): 
    """La méthode de dichotomie - Dichotomy method""" 
    x0, x1 = a, b 
    x2 = (x0 + x1) / 2 
    yield x2 
    while True: 
        if f(x2) * f(x0) < 0: 
            x1 = x2 
        else: 
            x0 = x2 
        x2 = (x0 + x1) / 2 
        yield x2 
 
 
def main(): 
    """on applique la méthode de dichotomie entre 1 et 2 - we apply the method between 1 and 2""" 
    D = Dichot(f, 1.0, 2.0) 
    for i in range(1, 10): 
        print(next(D)) 
 
 
if __name__ == '__main__': 
    main()
