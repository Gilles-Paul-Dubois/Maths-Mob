def f(x): 
    """On cherche une annulation de cette fonction""" 
    """Looking for a zero of this function"""     
    return x * x * x - 2 
 
 
def Dichot(f, a, b): 
    """La méthode de dichotomie - the dichotomy method""" 
    x0, x1 = a, b 
    x2 = (x0 + x1) / 2 
    yield x2 
    while True: 
        if f(x2) * f(x0) < 0: 
            x1 = x2 
        else: 
            x0 = x2 
        x2 = (x0 + x1) / 2 
        yield x2 
 
 
def main(): 
    """on applique la méthode de dichotomie avec 10 itérations entre 1 et 2""" 
    """We apply dichotomy with 10 iterations between 1 and 2"""     
    D = Dichot(f, 1.0, 2.0) 
    for i in range(1, 10): 
        r = next(D) 
    print(r * r) 
    print(2 ** (2.0 / 3.0)) 
 
 
if __name__ == '__main__': 
    main()
