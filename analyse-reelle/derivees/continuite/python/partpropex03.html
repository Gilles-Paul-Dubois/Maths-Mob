# # l'algorithme dit de la 'sécante' - The 'secant' algorithm 
 
def func(x): 
    """la fonction dont on cherche l'annulation - function we a looking for a vanishing""" 
    return x * x - x - 1 
 
 
def gensec(x0, x1, f): 
    """générateur de la méthode de la sécante - generator of the method""" 
    a = x0 
    b = x1 
    while True: 
        a, b = b, (a * f(b) - b * f(a)) / (f(b) - f(a)) 
        yield b 
 
 
def main(): 
    """fonction principale - main function""" 
    L = gensec(3.0, 2.0, func) 
    for i in range(0, 6): 
        print(next(L)) 
 
 
if __name__ == "__main__": 
    main();
