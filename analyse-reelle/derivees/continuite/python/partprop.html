def f(x): 
    """On cherche une annulation de cette fonction - We are looking for a value where this function vanishes""" 
    return x * x - 2 
 
 
def Partprop(f, a, b): 
    """La méthode des parties proportionnelles - Method of proportional parts""" 
    x0, x1 = a, b 
    x2 = (x0 * f(x1) - x1 * f(x0)) / (f(x1) - f(x0)) 
    yield x2 
    while True: 
        if f(x2) * f(x0) < 0: 
            x1 = x2 
        else: 
            x0 = x2 
        x2 = (x0 * f(x1) - x1 * f(x0)) / (f(x1) - f(x0)) 
        yield x2 
 
 
def main(): 
    """on applique la méthode des parties proportionnelles entre 1 et 2 - We apply the method between 1 and 2""" 
    D = Partprop(f, 1.0, 2.0) 
    for i in range(1, 10): 
        print(next(D)) 
 
 
if __name__ == '__main__': 
    main()
