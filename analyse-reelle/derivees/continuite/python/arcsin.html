import math
"""exemple d'utilisation de la bibliothèque Python pour les fonctions réciproques des fonctions trigonométriques"""
"""example of use of Python library for inverse trigionometric functions"""
L=[k/10.0 for k in range (0,10)]
S=[math.sin(x) for x in L]
C=[math.cos(x) for x in L]
T=[math.tan(x) for x in L]
print([math.asin(y) for y in S])
print([math.acos(y) for y in C])
print([math.atan(y) for y in T])
