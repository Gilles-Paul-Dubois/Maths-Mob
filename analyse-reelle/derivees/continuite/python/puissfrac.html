# -*- coding: utf-8 -*-

import math 
 
"""Puissances fractionnaires en Python-fonction de la bibliothèque et opérateur""" 
"""Fractional exponents with Python -unction of the library and operator""" 
 
f = 2.0 / 3.0 
 
 
def main(): 
    for x in range(0, 5): 
        print(math.pow(x, f), x ** f) #deux écritures possibles - two possible writings 
 
 
if __name__ == '__main__': 
    main()
