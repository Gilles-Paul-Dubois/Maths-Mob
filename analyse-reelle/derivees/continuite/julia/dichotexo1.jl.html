fonc(x)=(x-0.5)*(x-1/3)*(x-1)

function TrouveChangementSigne(f,n)
	return [(u[1],u[2]) for u in ((k/n,(k+1)/n) for k in -n*n:n*n-1) if f(u[1])*f(u[2]) < 0 || f(u[1])==0 ]
end

function localisation(f,n)
	m,L=0,[]
	while length(L)<n
		m+=1
		L=TrouveChangementSigne(f,m)
	end
	return m,L
end

function TrouveRacine(f,a,b,k)
	c=(a+b)/2
	for j in 0:k
		if f(a)==0 return a	end; if f(c)==0	return c end; if f(b)==0 return b end;		
		c=(a+b)/2
		if f(a)*f(c) < 0
			b=c
		else
			a=c
		end
	end
	return c
end

function Resoudre(f,n,e)
	L=localisation(f,n)
	h=L[1]
	K=L[2]
	k= floor((log(h)-log(e))/log(2)+1)
	for u in K
		println(TrouveRacine(f,u[1],u[2],k))
	end
end

Resoudre(fonc,3,0.001)
