from math import * 
from sympy import * 
 
 
def main(): 
    x = Symbol('x') 
    print(diff(sin(x), x)) #dérivation simple - simple differentiation 
    print(diff(sin(2*x), x, 1)) #dérivation fonction composée - differentiating composite function 
    print(diff(sin(x), x, 2)) #dérivation double - second order diff. 
    print (diff(x*sin(x),x)) #dérivation produit - derivative of a product 
    print (diff(sin(x)/x,x)) #dérivation quotient - derivative of a quotient 
    print(diff(x**3,x,3)) # dérivée d'ordre 3 - third order diff. 
 
 
if __name__ == '__main__': 
    main()
