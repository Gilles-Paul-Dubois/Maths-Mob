def Der(f,h): 
    def df(x): 
        return (f(x+h)-f(x))/h 
    return df 
 
def main(): 
    def F(x): 
        return x*x 
    h=1.0 
    for n in range(0,10): 
        h=h/2 
        f=Der(F,h) 
        for x in range(0,10): 
            print (f(x),":",2*x,"--") 
 
if __name__ == "__main__": 
    main();
