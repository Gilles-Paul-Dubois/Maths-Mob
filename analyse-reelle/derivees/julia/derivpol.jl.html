using Nemo
R,x=PolynomialRing(ZZ,"x")
f=x^4-x^3+3x+21
g=x^2-2

function decomposition(f,g)
    Dec=[]
    while true
        q,r=divrem(f,g)
        push!(Dec,r)
        f=q
        if degree(f) < degree(g)
            push!(Dec,f)
            break
        end
    end
    return reverse(Dec)
end

println(decomposition(f,g))
