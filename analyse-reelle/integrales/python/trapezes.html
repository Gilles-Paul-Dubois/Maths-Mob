import math 
 
 
def trapezes(a, b, n, f): 
    """méthode d'approximation dite 'des trapèzes' - approximation method known as 'trapezes'""" 
    S = 0 
    for i in range(0, n): 
        x1 = a + (b - a) * i / float(n) 
        x2 = a + (b - a) * (i + 1) / float(n) 
        S += (f(x1) + f(x2)) / 2.0 * (x2 - x1) 
    return S 
 
 
def fonc(x): 
    return 4.0 / (1 + (x - 3) * (x - 3)) 
 
 
def main(): 
    """Fonctionde tests - testing function """ 
    t = 4 * (math.atan(5) + math.atan(2)) 
    for i in range(10, 20): 
        s = trapezes(1, 8, i, fonc) 
        print(s, t - s) 
 
 
if __name__ == '__main__': 
    main()
