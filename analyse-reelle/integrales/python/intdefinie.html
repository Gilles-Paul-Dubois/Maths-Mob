#calcul symbolique en python (calcul d'intégrale définie) 
#Symbolic computation with Python (evaluating a definite integral) 
from sympy import * 
 
def main(): 
    x = Symbol('x') 
    print (integrate(cos(x), (x, -pi/2, pi/2))) 
 
if __name__ == '__main__': 
    main()
