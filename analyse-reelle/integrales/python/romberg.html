import math 
 
""" Accélération de convergence -méthode de Richardson Romberg""" 
""" Convergence acceleration - Richardson Romberg method"""  
 
def trapezes(a, b, n, f): 
    """méthode d'approximation dite 'des trapèzes'""" 
    """approximation method know as 'trapezes'"""     
    S = 0 
    for i in range(0, n): 
        x1 = a + (b - a) * i / float(n) 
        x2 = a + (b - a) * (i + 1) / float(n) 
        S += (f(x1) + f(x2)) / 2.0 * (x2 - x1) 
    return S 
 
 
def fonc(x): 
    """la fonction que l'on intègre - the function we want to integrate """ 
    return 4.0 / (1 + (x - 3) * (x - 3)) 
 
 
def pow2(n): 
    """ les puissances successives de 2 - successive powers of 2""" 
    if n == 0: 
        return 1 
    else: 
        return 2 * pow2(n - 1) 
 
 
def main(): 
    """Fonction de tests testing function """ 
    t = 4 * (math.atan(5) + math.atan(2)) 
    u = [trapezes(1, 8, pow2(i), fonc) for i in range(0, 16)] 
    v = [(4 * u[i + 1] - u[i]) / 3.0 for i in range(0, len(u) - 1)] 
    print(t - u[-1]) 
    print(t - v[-1]) 
 
 
if __name__ == '__main__': 
    main()
