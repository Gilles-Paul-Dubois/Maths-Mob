import math 
import random 
 
"""Mesure d'un quart de cercle de rayon unité par la méthode de Monte-Carlo""" 
"""Area of quarter of a circle with radius 1 by Monte-Carlo method""" 
 
 
def rand_point(): 
    """point aléatoire dans le carré de côté 1 - random point in square with unit side""" 
    return random.random(), random.random() 
 
 
def in_circle(P): 
    """Le point est-il dans le cercle unité? - is the point inside the circle""" 
    return math.sqrt(P[0] * P[0] + P[1] * P[1]) <= 1 
 
 
def stat(n): 
    """statistique sur n tirs - statistics about n shots""" 
    success = 0 
    for i in range(0, n): 
        P = rand_point() 
        if in_circle(P): 
            success += 1 
    return success / float(n) 
 
 
def main(): 
    """Fonction de test - test function""" 
    S = 0 
    for i in range(0, 10):  # 10 fois - 10 times 
        s = 4 * stat(10000)  #10000 tirs - 1000 shots 
        S += s 
    print(S / 10) 
 
 
if __name__ == '__main__': 
    main()
