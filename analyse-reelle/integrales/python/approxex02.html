import random 
 
"""Mesure d'un tronc de cone de hauteur unité par la méthode de Monte-Carlo""" 
"""Measurement of a truncated cone of unit height by the Monte-Carlo method""" 
 
 
def rand_point(): 
    """point aléatoire dans le pavé de volume 4 - random point inside the parallelepiped of volume 4""" 
    return random.random() * 2 - 1, random.random() * 2 - 1, random.random() 
 
 
def in_cone(P): 
    """Le point est-il dans le cône ?- Is the point in the cone? """ 
    return (P[0] * P[0] + P[1] * P[1]) <= (1 - P[2]) * (1 - P[2]) 
 
 
def stat(n): 
    """statistique sur n tirs - statistics about n shots""" 
    success = 0 
    for i in range(0, n): 
        P = rand_point() 
        if in_cone(P): 
            success += 1 
    return success / float(n) 
 
 
def main(): 
    """Fonction de test - test function""" 
    S = 0 
    for i in range(0, 10):  # 10 fois - 10 times 
        s = 12 * stat(10000)  #10000 tirs - 1000 shots 
        S += s 
    print(S / 10) 
 
 
if __name__ == '__main__': 
    main()
