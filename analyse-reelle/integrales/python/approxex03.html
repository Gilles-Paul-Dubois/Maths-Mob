"""Mesure d'un tronc de cone de hauteur unité par une méthode déterministe 
Approximation du volume comme réunion de petits cubes""" 
"""Measurement of a truncated cone of unit height by a deterministic method
Approximation of the volume as a union of small cubes""" 
 
def approx(n): 
    step=1.0/n 
    dv=step*step*step 
    V=0 
    for x in (-1+step*i for i in range(0,2*n)): 
        for y in (-1+step*i for i in range(0,2*n)): 
            for z in (step*i for i in range(0,n)): 
                if x*x+y*y <=(1-z)*(1-z): 
                    V+=dv 
    return V 
 
def main(): 
    """Fonction de test - test function """ 
    n=100 
    print (approx(n)*3) 
 
if __name__ == '__main__': 
    main()
