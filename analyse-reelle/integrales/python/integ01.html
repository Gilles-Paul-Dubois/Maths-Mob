# Mesure de la surface d'une ellipse par des approximations par des sommes de carrés 
# Measure of the area of an ellipse using approximations with unions of squares 
import math 
 
 
def IsInEllipse(x, y): 
    """teste si le point de coordonnées (x,y) est à l'intérieur de l'ellipse""" 
    """Checks if point with coordinates (x,y)is inside the ellipse"""    
    return x * x / 4 + y * y <= 1 
 
 
def Quadrillage(n): 
    """Points du quadrillage de maille 1/n""" 
    """Points of the net with step 1/n"""     
    R = [k / float(n) for k in range(-3 * n, +3 * n)] 
    Ry = [k / float(n) for k in range(-2 * n, 2 * n)] 
    return [(x, y) for x in R for y in Ry] 
 
 
def EvalSup(n): 
    """approximation par excès - approx. by excess """ 
    Q = [(x, y) for (x, y) in Quadrillage(n) if 
         IsInEllipse(x, y) or IsInEllipse(x + 1.0 / n, y) or IsInEllipse(x, y + 1.0 / n) or IsInEllipse(x + 1.0 / n, 
                                                                                                        y + 1.0 / n)] 
    return len(Q) * 1.0 / (n * n) 
 
 
def EvalInf(n): 
    """approximation par défaut - approx. by default""" 
    Q = [(x, y) for (x, y) in Quadrillage(n) if 
         IsInEllipse(x, y) and IsInEllipse(x + 1.0 / n, y) and IsInEllipse(x, y + 1.0 / n) and IsInEllipse(x + 1.0 / n, 
                                                                                                           y + 1.0 / n)] 
    return len(Q) * 1.0 / (n * n) 
 
 
def main(): 
    for i in [10, 20, 30, 50, 100]: 
        print(EvalInf(i), EvalSup(i), (EvalInf(i) + EvalSup(i)) / 2, 2 * math.pi) 
 
 
if __name__ == '__main__': 
    main()
