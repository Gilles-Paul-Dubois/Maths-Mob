function trapezes(a,b,n,f)
    S=0
    for i in 0:n-1
        x1=a+(b-a)*i/n
        x2=a+(b-a)*(i+1)/n
        S+=(f(x1)+f(x2))/2*(x2-x1)
    end
    return S
end

fonc(x)=4/(1+(x-3)*(x-3))


function main()
    t = 4 * (atan(5) + atan(2))
    u = [trapezes(1, 8, 2^i, fonc) for i in 0:15]
    v = [(4 * u[i + 1] - u[i]) / 3 for i in 1:15]
    print(t - u[end]) ;print("  :  ");println(t-v[end])
end
 
main()
