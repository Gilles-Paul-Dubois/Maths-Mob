function simpson(a,b,n,f)
    S=0
    for i in 0:n-1
        x1=a+(b-a)*i/n
        x2=a+(b-a)*(i+1)/n
        S+=(f(x1) + 4 * f((x1 + x2) / 2) + f(x2)) * (x2 - x1) / 6
    end
    return S
end

fonc(x)=4/(1+(x-3)*(x-3))

function main()
    t = 4 * (atan(5) + atan(2))
    for i in 10:19
        s = simpson(1, 8, i, fonc)
        print(s);print("  :  ");println(t - s)
    end
end
 
main()
