rand_point()=(2rand()-1,2rand()-1,rand())

in_cone(P)=P[1]*P[1]+P[2]*P[2] <= (1-P[3])*(1-P[3])

function stat(n)
    success=0
    for i in 1:n
        P=rand_point()
        if in_cone(P)
            success+=1
        end
    end
    return success/n
end

function main()
    S=0
    for i in 1:10
        s=12*stat(10000)
        S+=s
    end
    println(S/10)
end

main()
