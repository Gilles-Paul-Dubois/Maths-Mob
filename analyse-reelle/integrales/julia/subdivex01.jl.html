struct Subdivision
    a::Float64
    b::Float64
    n::Int64
    t::Int8
    r::Float64
    q::Float64
    points::Vector{Float64}
end

function Subdivision(a,b,n,t)
    a,b=Float64(a), Float64(b)
    if t==1
        r=(b-a)/n
        q=0
        points=[a+i*r for i in 0:n-1]
        push!(points,b)
    end
    if t==2 #a diff√©rent de 0 
        q=(b/a)^(1/n)
        r=0
        points=[a*q^i for i in 0:n-1]
        push!(points,b)
    end
    if t==3
        r,q=0,0
        points=[a]
        rd= sort!([a+(b-a)*rand() for i in 1:n-1])
        append!(points,rd)
        push!(points,b)
    end
    return Subdivision(a,b,n,t,r,q,points)
end

function Subdivision(L::Vector{Float64})
    a, b, n=L[1],L[end],length(L)-1
    t,r,q = 3,0,0
    points=L
    return Subdivision(a,b,n,t,r,q,points)
end
    
    
    
function pas(S::Subdivision)
    p=S.b-S.a
    for i in 1:S.n-1
        d= S.points[i+1]-S.points[i]
        if d < p
            p=d
        end
    end
    return p
end

function Base.:+(S1::Subdivision,S2::Subdivision)
    a1,a2 =S1.a, S2.a
    b1,b2=S1.b,S2.b
    (a1 != a2 ) || (b1 != b2) && return nothing
    L=S2.points[2:end-1]
    C=copy(S1.points)
    for l in L
        if !(l in C)
            push!(C,l)
        end
    end
    sort!(C)
    return Subdivision(C)
end
