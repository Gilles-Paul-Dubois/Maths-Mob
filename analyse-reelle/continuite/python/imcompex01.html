# ex01 recherche d'un maximum sur un compact à epsilon près 
# ex01 Searching for a maximum on a compact interval with given precision 
 
def func(x): 
    """fonction dont on cherche le maximum - the function for which we want a maximum""" 
    return x * x * x - (3.0 / 2.0) * x * x - (9.0 / 4.0) * x + (11.0 / 8.0) 
 
 
def max(f, a, b, step): 
    M = f(a) 
    x = a 
    c = a 
    while x <= b: 
        if f(x) > M: 
            M = f(x) 
            c = x 
        x += step 
    return M, c 
 
 
def main(): 
    print(func(-1)) 
    print(max(func, -1, 0, 0.01)) 
    print(func(0)) 
 
#la valeur théorique est -0.5 - In theory the value is -0.5 
if __name__ == '__main__': 
    main()
