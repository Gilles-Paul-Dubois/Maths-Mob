import math 
# exercice 02 - exercise 02 
 
def f(x): 
    """On cherche une annulation de cette fonction""" 
    """Looking for a zero of this function"""     
    return math.sin(x) - 0.5 
 
 
def Dichot(f, a, b): 
    """La méthode de dichotomie - the dichotomy mmethod""" 
    x0, x1 = a, b 
    x2 = (x0 + x1) / 2 
    yield x2 
    while True: 
        if f(x2) * f(x0) < 0: 
            x1 = x2 
        else: 
            x0 = x2 
        x2 = (x0 + x1) / 2 
        yield x2 
 
 
def main(): 
    """on applique la méthode de dichotomie entre 0 et π/2 - we apply the method between 0 and π/2""" 
    D = Dichot(f, 0, math.pi / 2) 
    for i in range(1, 35): 
        r = next(D) 
    print(r) 
    print(math.asin(0.5)) 
 
 
if __name__ == '__main__': 
    main()
