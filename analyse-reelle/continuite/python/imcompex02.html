# ex02 recherche d'un minimum sur un compact à epsilon près- searching for a minimum with given precision 
 
def func(x): 
    """fonction dont on cherche le minimum - the function which we study""" 
    return x * x * x - (3.0 / 2.0) * x * x - (9.0 / 4.0) * x + (11.0 / 8.0) 
 
 
def min(f, a, b, step): 
    m = f(a) 
    x = a 
    c = a 
    while x <= b: 
        if f(x) < m: 
            m = f(x) 
            c = x 
        x += step 
    return m, c 
 
 
def main(): 
    print(func(1)) 
    print(min(func, 1, 2, 0.01)) 
    print(func(2)) 
 
#la valeur théorique est 1.5 - the theoretical value is 1.5 
 
if __name__ == '__main__': 
    main()
