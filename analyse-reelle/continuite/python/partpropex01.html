import math 
# exercice 01 (nombre d'or) - exercise 01 (Golden Ratio) 
 
def f(x): 
    """On cherche une annulation de cette fonction - we're looking for a zero of this function""" 
    return x * x - x - 1 
 
 
def Partprop(f, a, b): 
    """La méthode des parties proportionnelles - The algorithm""" 
    x0, x1 = a, b 
    x2 = (x0 * f(x1) - x1 * f(x0)) / (f(x1) - f(x0)) 
    yield x2 
    while True: 
        if f(x2) * f(x0) < 0: 
            x1 = x2 
        else: 
            x0 = x2 
        x2 = (x0 * f(x1) - x1 * f(x0)) / (f(x1) - f(x0)) 
        yield x2 
 
 
def main(): 
    """on applique la méthode des parties proportionnelles entre 1 et2 - Method applied between 1 and 2""" 
    D = Partprop(f, 0, math.pi / 2) 
    for i in range(1, 11): 
        r = next(D) 
    print(r) 
 
 
if __name__ == '__main__': 
    main()
