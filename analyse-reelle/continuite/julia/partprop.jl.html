f(x)=x^2-2
function partprop(f,a,b,c::Channel)
    x0,x1=a,b
    x2 = (x0 * f(x1) - x1 * f(x0)) / (f(x1) - f(x0))
    put!(c,x2)
    while true
        if f(x0)*f(x2)<0
            x1=x2
        else
            x0=x2
        end
        x2 = (x0 * f(x1) - x1 * f(x0)) / (f(x1) - f(x0))
        put!(c,x2)
    end
end

function main()
    D = Channel( (channel_arg) -> partprop(f,1,2, channel_arg) )
    for i in 1:9
        println(take!(D))
    end
end

main()
