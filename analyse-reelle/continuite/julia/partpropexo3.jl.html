func(x)=x*x-x-1

function gensec(x0,x1,f,c::Channel)
    a,b=x0,x1
    while true
        a, b = b, (a * f(b) - b * f(a)) / (f(b) - f(a)) 
        put!(c,b)
    end
end
        
function main()
    S = Channel( (channel_arg) -> gensec(3,2,func, channel_arg) )
    for i in 0:6
        println(take!(S))
    end
end

main()
