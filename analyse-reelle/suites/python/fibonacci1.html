# un exemple de générateur pour une suite récurrente multiple 
# la suite de Fibonacci 
 
def fibonacci(): 
    a, b = 0, 1 
    yield b 
    while True: 
        a, b = b, a + b 
        yield b 
 
 
def main(): 
    F = fibonacci() 
    for i in range(0, 10): 
        print(next(F)) 
 
 
if __name__ == '__main__': 
    main()
