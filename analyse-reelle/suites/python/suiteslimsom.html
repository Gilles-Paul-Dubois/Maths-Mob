# ## convergence de la somme de deux suites convergentes 
# ## convergence of the sum of two convergent sequences 
import math 
 
 
def u(n): 
    """ suite tendant vers 2 - sequence having 2 as limit""" 
    return 2 + math.sin(n + 1) / (n + 1) 
 
 
def v(n): 
    """ suite tendant vers 3 - sequence tending towards 3""" 
    return 3 + math.sin(n + 1) / (n + 1) 
 
 
def main(): 
    for i in [10, 100, 1000, 10000]: 
        print(abs(u(i) - 2), ':', abs(v(i) - 3), ':', abs(u(i) + v(i) - 5), end='\n') 
 
 
if __name__ == '__main__': 
    main()
