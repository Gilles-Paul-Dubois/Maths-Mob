def fact(n): 
    """la factorielle - the factorial """ 
    if n==0: 
        return 1 
    else: 
        r=1; 
        for i in range(1,n+1): 
            r*=i 
        return r 
 
def u(n): 
    """ la suite u - sequence u""" 
    r=0 
    for i in range(0,n+1): 
        r+=1.0/fact(i) 
    return r 
 
def v(n): 
    """la suite v - sequence v""" 
    return u(n) + 1.0/(n*fact(n)) 
 
def main(): 
    """imprimer les termes d'indices 1 Ã  10- print terms with rank from 1 to 10""" 
    for i in range(1,11): 
        print (u(i),v(i)) 
 
 
if __name__ == '__main__': 
    main()
