# la suite (1+1/n)^n (comportement asymptotique) 
# Sequence (1+1/n)^n (asymptotic behaviour) 
import math 
 
 
def u(n): 
    return (1 + 1.0 / n) ** n 
 
 
def main(): 
    for i in [10, 100, 1000, 10000, 20000, 30000, 50000, 100000]: 
        print(i, ":", u(i), end='\n') 
 
 
if __name__ == '__main__': 
    main()
