# ## verification de la convergence de la suite n^(1/n) vers 1 
# ## Checking convergence of n^(1/n) to 1 
 
def u(n): 
    """ le terme d'indice n de la suite - term of index n""" 
    return n ** (1.0 / n) 
 
 
def main(): 
    """fonction principale - main function""" 
    for i in [10, 100, 1000, 10000]: 
        print(u(i)) 
 
 
if __name__ == '__main__': 
    main()
