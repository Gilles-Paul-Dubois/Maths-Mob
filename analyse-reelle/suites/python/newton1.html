# ##Calcul de racine de 2 par la méthode de Newton 
# ##Computation of √2 using Newton's algorithm 
 
from decimal import Decimal, getcontext 
 
 
def u(n): 
    if n == 0: 
        return Decimal(1) 
    return (u(n - 1) + (2 - u(n - 1) * u(n - 1)) / (2 * u(n - 1)))  #appel récursif 
 
 
def main(): 
    """affichage des 10 premiers termes avec 100 décimales""" 
    """display of the first 10 terms with 100 decimals """    
    getcontext().prec = 101 
    for i in range(0, 8): 
        print("i=", i, ":", u(i), end='\n') 
 
 
if __name__ == '__main__': 
    main()
