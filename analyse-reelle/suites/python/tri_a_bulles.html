# ## Mesure de l'efficacité du tri à bulles 
/span># ## Measuring efficiency of bubble sort 
import random 
import time 
 
 
def random_list(n): 
    """genère une liste aléatoire de n entiers entre 0 et n-1""" 
    """generates a random list of n integers between 0 and n-1"""     
    return [random.randint(0, n - 1) for i in range(0, n)] 
 
 
def tri_a_bulles(L): 
    """trie la liste L avec l'algorithme du tri à bulles""" 
    """sorts list L using bubble sort"""     
    en_ordre = False 
    while en_ordre == False: 
        en_ordre = True 
        for i in range(0, len(L) - 1): 
            if L[i] > L[i + 1]: 
                en_ordre = False 
                L[i], L[i + 1] = L[i + 1], L[i] 
 
 
def temps_mis(n): 
    """temps mis pour traiter une liste de longueur n""" 
    """time spent to process a list with length n"""     
    L = random_list(n) 
    start = time.time() 
    tri_a_bulles(L) 
    end = time.time() 
    return end - start 
 
 
def main(): 
    T = [temps_mis(n) for n in [1000 * i for i in range(1, 5)]] 
    K = [T[0] * i * i for i in range(1, 5)] 
    print(T) 
    print(K) 
 
 
if __name__ == '__main__': 
    main()
