def GenPremiers():
    """générateur de nombres premiers - generator of prime numbers"""
    P = [2]
    yield 2  # premier appel - first call
    while True:  # autres appels - next calls
        x = P[-1] + 1  # ajouter 1 au dernier trouvé - add 1 to the last found
        while any([x % i == 0 for i in P]):  # y-a-til un diviseur premier parmi les précédents ? - any divisor among previous ones ?
            x = x + 1  # si oui continuer avec le suivant if yes try next one
        P.append(x)  # sinon l'ajouter à la liste - if not add to the list
        yield (x)  # retourner ce nouveau nombre premier - return this nex prime number

def p(n):
    Premiers = GenPremiers()
    for i in range(0, n + 1):
        x = next(Premiers)
    return x

def main():
    """affiche les 10 premiers nombres premiers - prints the 10 first prime numbers"""
    for i in range(0, 10):
        print(i, ":", p(i))

if __name__ == '__main__':
    main()
