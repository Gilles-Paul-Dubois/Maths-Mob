# ## Accélération de Romberg sur la suite 2^nsin(pi/2^n) 
# ## Romberg acceleration process on sequence 2^nsin(pi/2^n) 
import math 
 
 
def genS(): 
    """périmètre du polygone régulier inscrit""" 
    """perimeter of inscribed polygon"""     
    n = 4 
    u = 2 * math.sqrt(2) 
    yield u 
    while True: 
        u = math.sqrt(2) * n * math.sqrt(1 - math.sqrt(1 - (u / n) * (u / n))) 
        n = n * 2 
        yield u 
 
 
def main(): 
    """programme principal - main function""" 
    S = genS() 
    PI = math.pi 
    S1 = [] 
    for i in range(0, 10): 
        S1.append(next(S)) 
    S2 = [(4 * S1[i + 1] - S1[i]) / 3 for i in range(0, 9)] 
    S3 = [(16 * S2[i + 1] - S2[i]) / 15 for i in range(0, 8)] 
    for i in range(0, 7): 
        print(PI - S1[i], PI - S2[i], PI - S3[i], end='\n') 
 
 
if __name__ == '__main__': 
    main()
