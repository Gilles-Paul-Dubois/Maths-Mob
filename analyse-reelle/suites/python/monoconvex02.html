# ##exercice 02 suites adjacentes 
 
def u(n): 
    x = 1 + 1.0 / n 
    r = 1 
    for i in range(0, n): 
        r *= x 
    return r 
 
 
def v(n): 
    x = 1 - 1.0 / n 
    x = 1.0 / x 
    r = 1 
    for i in range(0, n): 
        r *= x 
    return r 
 
 
def main(): 
    for i in [10, 100, 1000, 10000, 20000]: 
        print(u(i), ';', v(i)) 
 
 
if __name__ == '__main__': 
    main()
