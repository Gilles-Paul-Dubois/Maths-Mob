import math 
# ## exercice 03 suites extraites adjacentes ### 
def genu(): 
    n = 0 
    u = 0 
    yield u 
    while True: 
        n = n + 1 
        u = u + 1.0 / (n * n) 
        yield u 
 
 
def main(): 
    suiteu = genu() 
    n = 1 
    u = next(suiteu)  #u0 
    u = next(suiteu)  #u1 
    while True: 
        n = n + 1  #n=2 
        u = next(suiteu) 
        d = 1.0 / n 
        if d < 0.0001: 
            print(u, ';', u + 1.0 / n) 
            break 
    print(math.pi * math.pi / 6) 
 
 
if __name__ == '__main__': 
    main()
