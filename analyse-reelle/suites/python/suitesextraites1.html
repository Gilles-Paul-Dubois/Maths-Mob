# génération aléatoire de suites extraites d'une suite arithmétique de raison 3""" 
# random generation of subsequences of an arithmetic progression with ration 3 
from random import randint 
 
 
def p(): 
    """la fonction strictement monotone aléatoire - the strictly monotonic random map """ 
    n = randint(1, 5) 
    yield n 
    while True: 
        n += randint(1, 5) 
        yield n 
 
 
inj = p() 
 
 
def u(n): 
    """la suite mère - the mother sequence""" 
    return 3 * n 
 
 
def v(n): 
    """la suite extraite - the subsequence""" 
    global inj 
    p = next(inj) 
    return u(p) 
 
 
def main(): 
    for i in range(0, 10): 
        print(i, u(i), v(i), end='\n') 
 
 
if __name__ == '__main__': 
    main()
