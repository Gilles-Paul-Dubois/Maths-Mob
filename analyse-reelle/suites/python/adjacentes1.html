# ## Suites adjacentes - adjacent sequences 
 
def genv(): 
    """générateur de la suite extraite des termes de rang pair""" 
    """generator of subsequence of terms with even rank"""     
    n = 0 
    u = 1 
    yield u 
    while True: 
        u = u - 1.0 / (n + 2) + 1.0 / (n + 3) 
        n = n + 2 
        yield u 
 
 
def genw(): 
    """générateur de la suite des extraites des termes de rang impair""" 
    """generator of the subsequence of terms with odd rank"""     
    n = 1 
    u = 0.5 
    yield u 
    while True: 
        u = u + 1.0 / (n + 2) - 1.0 / (n + 3) 
        n = n + 2 
        yield u 
 
 
def main(): 
    Suitev = genv() 
    Suitew = genw() 
    for i in range(0, 80000): 
        v = next(Suitev) 
        w = next(Suitew) 
        if i % 1000 == 0: 
            print(']', w, ';', v, '[', end='\n')  #encadrement de la limite - bounds for the limit 
 
 
if __name__ == '__main__': 
    main()
