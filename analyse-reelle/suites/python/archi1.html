# ## Computation of π using Archimedes' method 
import math 
 
 
def S(n): 
    """périmètre du polygone régulier inscrit""" 
    """perimeter of the regular inscribed polygon"""     
    if n == 4: 
        return 2 * math.sqrt(2) 
    else: 
        return math.sqrt(S(n / 2) * T(n)) 
 
 
def T(n): 
    """périmètre du polygone régulier ex-inscrit""" 
    """perimeter of the regular circumscribed polygon"""     
    if n == 4: 
        return 4 
    else: 
        return (2 * S(n / 2) * T(n / 2)) / (S(n / 2) + T(n / 2)) 
 
 
def main(): 
    """programme principal- main program""" 
    E = [] 
    n = 2 
    for i in range(0, 12):  #12 itérations - 12 iterations 
        n = n * 2  #on double le nombre des côtés - double number of sides 
        X = S(n)  #périmètre du polygone régulier inscrit - inscribed regular polygon 
        Y = T(n)  #périmètre du polygone régulier ex-inscrit - circumscribed regular polygon  
        print("i=", i, "n=", n, X, Y, Y - X, end='\n') 
        E.append(Y - X)  #mémorisation de la précision- storing the precision 
    for i in range(1, 12): 
        print(E[i] / E[i - 1])  # affichage des quotients - displaying ratios 
 
 
if __name__ == '__main__': 
    main()
