def GenPremiers(): 
    """générateur de nombres premiers""" 
    P = [2] 
    yield 2  # premier appel 
    while True:  # autres appels 
        x = P[-1] + 1  #ajouter 1 au dernier trouvé 
        while any([x % i == 0 for i in P]):  # y-a-til un diviseur premier parmi les précédents? 
            x = x + 1  #si oui continuer avec le suivant 
        P.append(x)  #sinon l'ajouter à la liste 
        yield (x)  #retourner ce nouveau nombre premier 
 
 
def main(): 
    """affiche les 100 premiers nombres premiers""" 
    Premiers = GenPremiers() 
    for i in range(0, 100): 
        print(i, ":", next(Premiers), end='\n') 
 
 
if __name__ == '__main__': 
    main()
