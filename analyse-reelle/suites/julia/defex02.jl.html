function genpremiers(c::Channel)
    put!(c,2) 
    put!(c,3)
    P, p = [2, 3], 3
    while true
        p += 2 
        if all((p % x != 0 for x in P))
            push!(P, p) 
            put!(c,p) 
        end
    end
end

function p(n)
    Premiers = Channel(genpremiers)
    x=2
    for i in 1:n
        x=take!(Premiers)
    end
    return x
end

function main()
    for i in 1:10
        print(i," : ",p(i),"\n")
    end
end

main()
