function genS(c::Channel)
    u=1
    put!(c,u)
    while true
        u=sqrt(2+u)
        put!(c,u)
    end
end
    
function main()
    S=Channel(genS)
    S1=[]
    for i in 1:10
        push!(S1,take!(S))
    end
    S2=[(4*S1[i+1]-S1[i])/3 for i in 1:9]
    for i in 1:7
        print(2-S1[i]," : ",2-S2[i],"\n")
    end
end

main()
