random_list(n)=[rand(1:n) for i in 1:n]

function tri_a_bulles(L)
    en_ordre=false
    while en_ordre==false
        en_ordre=true
        for i in 1:length(L)-1
            if L[i]>L[i+1]
                en_ordre=false
                L[i], L[i + 1] = L[i + 1], L[i]
            end
        end
    end
end

function test1(n)
    L=random_list(n)
    @timed tri_a_bulles(L)
end

function algotime()
    for n in [100,1000,10000,30000,40000,50000]
        println(test1(n).time*10^8/(n*n))
    end
end

algotime()
