S(n)= n==4 ? 2sqrt(2) : sqrt(S(n÷2)*T(n))
T(n)= n==4 ? 4 : (2S(n÷2)*T(n÷2))/(S(n÷2)+T(n÷2))
function main()
    E,n=[],2
    for i in 1:12
        n=n*2
        X=S(n)
        Y=T(n)
        print("i=",i," ","n=",n," ",X," ",Y," ",X-Y);println("")
        push!(E,Y-X)
    end
    for i in 2:12
        println(E[i]/E[i-1])
    end
end
main()
