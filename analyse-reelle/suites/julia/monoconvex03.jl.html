function genu(c::Channel)
    n,u=0,0
    put!(c,u)
    while true
        n=n+1
        u=u+1/(n*n)
        put!(c,u)
    end
end

function main()
    channel=Channel(genu)
    n=1
    u=take!(channel)
    u=take!(channel)
    while true
        n=n+1
        u=take!(channel)
        d=1/n
        if d<0.0001
            print(u," : ",u+d,"\n")
            break
        end
    end
    print(pi*pi/6)
end

main()
