function genS(c::Channel)
    n,u=4,2*sqrt(2)
    put!(c,u)
    while true
        u = sqrt(2) * n * sqrt(1 - sqrt(1 - (u / n) * (u / n)))
        n = n * 2
        put!(c,u)
    end
end

S = Channel(genS)
suiteu = [take!(S) for i in 1:10]
u(n)=suiteu[n]

function aitken(u)
    delta(u,n)=u(n + 1) - u(n)
    delta2(u,n)=u(n + 2) - 2 * u(n + 1) + u(n)
    function v(n)
        return u(n - 1) - (delta(u, n - 1) * delta(u, n - 1)) / delta2(u, n - 1)
    end
    return v
end

function main()
    v1=aitken(u)
    v2=aitken(v1)
    for i in 3:6
        print(pi - u(i)," ", pi - v1(i)," ", pi - v2(i), '\n')
    end
end

main()
