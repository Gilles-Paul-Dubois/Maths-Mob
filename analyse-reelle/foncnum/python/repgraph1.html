# -*- coding: utf-8 -*-
# python version 3.4 
 
import tkinter as tk  # le module graphique - graphical module 
import math  # module mathématique - math module 
 
"""paramètres du graphique - parameters of the graphic """ 
Width = 600  # largeur de la fenêtre en pixels - width of the window in pixels 
Height = 600  # hauteur de la fenêtre en pixels - height of the windows in pixels 
Unit = 30  # taille en pixels du vecteur unité - size in pixels of unit vector 
Midx = Width / 2  # abscisse du centre du graphique - abscissa of the center 
Midy = Height / 2  # ordonnée du centre du graphique - ordinate of the center 
racine = tk.Tk()  # la fenêtre de l'application - window of the application 
racine.title("Exemple de représentations graphiques - exemples of plotting") 
fond = tk.Canvas(racine, width=Width, height=Height, background='black')  # aire de dessin - area of drawing 
bouton = tk.Button(racine, text="Quitter", command=racine.quit)  # pour quitter l'appli- To quit 
 
 
def f1(x): 
    """une fonction qu'on veut représenter - A function we want to plot""" 
    return x * math.sin(x) 
 
 
def f2(x): 
    """une autre - another one""" 
    return -x * x 
 
 
def convert(x, y): 
    """calcule les coordonnées en pixels sur l'image du point de coordonnées x, y réels""" 
    """Computes the coordinates in pixels on the image of the point with coordinates x,y real numbers"""    
    m = int(x * Unit) + Midx 
    n = Midy - int(y * Unit) 
    return m, n 
 
 
def packall():  # initialisation des composants - initialization of components 
    bouton.pack(side=tk.BOTTOM) 
    fond.pack() 
 
 
def drawtick(x, y, mode): 
    """graduation des axes - graduation of axes""" 
    if mode == "H": 
        fond.create_line(x, y - 2, x, y + 2, fill="white") 
    if mode == "V": 
        fond.create_line(x - 2, y, x + 2, y, fill="white") 
 
 
def drawaxes(): 
    """tracé des axes - drawing of axes""" 
    fond.create_line(0, Midy, Width, Midy, arrow="last", fill="white") 
    fond.create_line(Midx, Height, Midx, 0, arrow="last", fill="white") 
    H = [i * Unit for i in range(0, Width // Unit + 1)] 
    for x in H: 
        drawtick(x, Midy, "H") 
    V = [i * Unit for i in range(0, Height // Unit + 1)] 
    for y in V: 
        drawtick(Midx, y, "V") 
 
 
def drawgraph(f, xmin, xmax, step, color="green"): 
    """Tracé du graphe de la fonction f entre les valeurs xmin et xmax de la variable""" 
    """Plotting of function f between the values xmin and xmax of the variable"""     
    X = [xmin + k * step for k in range(0, int(Width / step)) if xmin + k * step <= xmax] 
    Graphe = [(x, f(x)) for x in X] 
    Points = [convert(P[0], P[1]) for P in Graphe] 
    for i in range(0, len(Points) - 1): 
        P = Points[i] 
        Q = Points[i + 1] 
        x0, y0 = P[0], P[1] 
        x1, y1 = Q[0], Q[1] 
        fond.create_line(x0, y0, x1, y1, fill=color) 
 
 
def main(): 
    """fonction principale - main function""" 
    packall()  # disposition des composants - Placing components 
    drawaxes()  # tracé des axes - drawing axes 
    drawgraph(f1, -5, +5, 0.01, "yellow")  # tracé du graphe de la première fonction 
    drawgraph(f2, -3, 3, 0.01, "red")  # tracé de la seconde 
    racine.mainloop()  # boucle de traitement des évènements souris-clavier- Main loop scanning mouse and keyboard 
 
 
if __name__ == '__main__': 
    main()
