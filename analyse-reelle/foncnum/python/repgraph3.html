"""Etablissement d'un tableau de valeurs""" 
"""Creating a table of values""" 
import math 
 
 
def f(x): 
    """une fonction donn√©e par une formule""" 
    """A function given by a formula"""     
    return x * math.sin(x) 
 
 
def main(): 
    """fonction principale - main function""" 
    for x in [-3 + k * 0.1 for k in range(0, 10)]: 
        print(x, f(x)) 
 
 
if __name__ == '__main__': 
    main()
