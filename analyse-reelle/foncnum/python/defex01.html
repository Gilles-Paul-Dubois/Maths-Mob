# -*- coding: utf-8 -*-

import math  # pour l'exponentielle, le logarithme, la racine carrée - for exponential, logarithm and square root 
 
 
def f(x): 
    """la fonction - the numeric function f""" 
    return math.exp(x) * math.log(1 + abs(x)) / math.sqrt(1 + x * x) 
 
 
def main(): 
    """programme principal - main function""" 
    for x in [k * 0.1 for k in range(0, 11)]: 
        print(x, f(x)) 
 
 
if __name__ == '__main__': 
    main()
