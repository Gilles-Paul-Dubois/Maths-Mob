step = pi / 40
TabX = [step * i for i in 0:21]
TabY = [sin(x) for x in TabX]
TabDa = [(TabY[i + 1] - TabY[i]) / (TabX[i + 1] - TabX[i]) for i in 1:20]
TabDb = [TabY[i] - TabDa[i] * TabX[i] for i in 1:20]

function inter(x)
    for i in 2:20
        if TabX[i] >= x
            return i-1
        end
    end
    return 20
end

function f1(x)
    k=inter(x)
    a=TabDa[k]
    b=TabDb[k]
    return a*x+b
end

f2(x)= x <= pi/2 ? f1(x) : f1(pi-x)
f3(x)= x>=0 ? f2(x) : -f2(-x)
function f4(x)
    if x >=0
        while !(-pi<=x<=pi)
            x=x-2pi
        end
        return f3(x)
    else
        while !(-pi<=x<=pi)
            x=x+2pi
        end
        return f3(x)
    end
end

for i in -10:10
    print(f4(i)," : ",sin(i),'\n')
end
