using Plots
pyplot()
TabX=collect(-3:1:3)
TabA= [x*x-(x-1)*(x-1) for x in TabX[2:end]]
TabB=[TabX[i]*TabX[i]-TabA[i]*TabX[i] for i in 1:6]
inter(x)=x == 3 ? 6 : [i for i in 1:6 if TabX[i]<=x<TabX[i+1]][1]
f(x)=TabA[inter(x)]*x+TabB[inter(x)]
plt=plot(f,-3,3,label="exo2",framestyle=:origin)
display(plt)
