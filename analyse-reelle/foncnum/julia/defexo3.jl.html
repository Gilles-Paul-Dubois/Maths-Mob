using Gtk
using Cairo
win = GtkWindow("Fonctions trigo",600,600)
hbox = GtkBox(:v)  # :h makes a horizontal layout, :v a vertical layout
push!(win, hbox)
c = GtkCanvas(600,500)
push!(hbox,c)
bleu = GtkButton("sinus")
rouge = GtkButton("cosinus")
efface=GtkButton("Effacer")
ferme = GtkButton("Quitter")
push!(hbox, bleu)
push!(hbox, rouge)
push!(hbox, efface)
push!(hbox, ferme)

id1 = signal_connect(rouge, "clicked") do widget
     cosinus()
end
id2 = signal_connect(bleu, "clicked") do widget
     sinus()
end
id3 = signal_connect(ferme, "clicked") do widget
     bye()
end
id4 = signal_connect(efface, "clicked") do widget
     wipe()
end

function wipe()
    ctx = getgc(c)
    h = height(c)
    w = width(c)
    rectangle(ctx, 0, 0, w, h)
    set_source_rgb(ctx, 1, 1, 1)
    fill(ctx)
    draw(c)
end

function cosinus()
    wipe()
    ctx = getgc(c)
    h = height(c)
    w = width(c)
    drawaxes()
    grad()
    step=4pi/100
    L=[(-2pi+k*step,cos(-2pi+k*step)) for k in 0:100]
    drawline(L)
    draw(c)
end

function grad()
    ctx = getgc(c)
    h = height(c)
    w = width(c)
    set_font_size(ctx,20)
    move_to(ctx,w/8-30,h/2)
    show_text(ctx,"-3π/2")    
    move_to(ctx,w/4,h/2)
    show_text(ctx,"-π")
    move_to(ctx,3w/8-30,h/2)
    show_text(ctx,"-π/2") 
    move_to(ctx,5w/8-30,h/2)
    show_text(ctx,"π/2")       
    move_to(ctx,3w/4,h/2)
    show_text(ctx,"π")
    move_to(ctx,7w/8-30,h/2)
    show_text(ctx,"3π/2")     
    move_to(ctx,w/2-20,h/4)
    show_text(ctx,"1")  
    move_to(ctx,w/2-20,3h/4)
    show_text(ctx,"-1")        
end

function drawaxes()
    ctx = getgc(c)
    h = height(c)
    w = width(c)
    move_to(ctx,0,h/2)
    line_to(ctx,w,h/2)
    move_to(ctx,w/2,0)
    line_to(ctx,w/2,h)
    set_source_rgb(ctx,0,0,0)
    stroke(ctx)
end

function drawline(PTS)
    ctx = getgc(c)
    h = height(c)
    w = width(c)    
    convx(x)=floor(x*w/4pi+w/2)
    convy(y)=floor(-y*h/4+h/2)
    pts=[(convx(P[1]),convy(P[2])) for P in PTS]
    start=pts[1]
    move_to(ctx, start[1],start[2])
    for i in 2:length(pts)
        line_to(ctx,pts[i][1],pts[i][2])
    end
    set_source_rgb(ctx,0,0,0)
    stroke(ctx) 
 end

function sinus()
    wipe()
    ctx = getgc(c)
    h = height(c)
    w = width(c)
    drawaxes()
    grad()
    step=4pi/100
    L=[(-2pi+k*step,sin(-2pi+k*step)) for k in 0:100]
    drawline(L)    
    draw(c)
end

function bye()
    Gtk.destroy(win)
end

showall(win)
