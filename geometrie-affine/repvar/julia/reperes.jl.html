#test pour savoir si un syst√®me de vecteur est une base
function base(VS::VectorsSystem{T}) where {T}
    s1, s2 = size(VS, 1), size(VS, 2)
    if s1 != s2
        return false
    end
    return rang(VS) == s1 ? true : false
end


struct OVS{T}
    O::Point{T}
    VS::VectorsSystem{T}
end

function repere(ovs::OVS{T}) where {T}
    if !base(ovs.VS)
        return false
    end
    n = length(ovs.O.C)
    s = size(ovs.VS, 1)
    if n != s
        return false
    end
    return true
end


function coords(B::VectorsSystem{T}, V) where {T}
    if gettype(B) != typeof(V[1])
        return nothing
    end
    n = length(V)
    if size(B, 1) != n
        return nothing
    end
    if !base(B)
        return nothing
    end
    M = matrice(B)
    return transpose(M) \ V
end

function coords(REP::OVS{T},P::Point{T}) where T
    O=REP.O
    V=P-O
    B=REP.VS
    return coords(B,V)
end
