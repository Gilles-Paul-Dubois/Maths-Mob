include("reperes.jl")
#retourne les coordonnées du point de paramètres params de la variété linéaire Var
#returns the coordinates of the point with parameters  params of the linear variety Var
function pointparam(Var::OVS{T},params::Vector{T}) where T
    O=Var.O
    Vs=Var.VS.Vectors
    s= size(Var.VS,1)
    L=[params[i]*Vs[i] for i in 1:s]
    V=reduce(+,L)
    M=O+V
    return M.C
end

function testpointparam()
    P=QP([1,1,3])
    dirs=VSQ([[1//1,2//1,3//1],[0//1,2//1,4//1]])
    VAR=OVS{Rational{Int64}}(P,dirs)
    ps=[2//1,3//1]
    println(pointparam(VAR,ps))
end

testpointparam()
