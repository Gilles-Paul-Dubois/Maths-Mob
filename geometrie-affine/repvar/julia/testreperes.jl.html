include("reperes.jl")

function test_base()
    println("début test_base-------------------------------")
    VS1 = aleavectsystemR(3, 2)
    println(base(VS1))
    VS2 = aleavectsystemR(3, 3)
    println(base(VS2))
    VS3 = VSQ([[1 // 1, 2 // 1], [2 // 1, 4 // 1]])
    println(base(VS3))
    println("fin test_base-------------------------------")
end

function test_repere()
    println("début test_repere-------------------------------")
    ovs1 = OVS(aleaRP(2), aleavectsystemR(3, 3))
    println(repere(ovs1))
    ovs2 = OVS(aleaRP(3), aleavectsystemR(3, 3))
    println(repere(ovs2))
    ovs3 = OVS(aleaQP(2), VSQ([[1 // 1, 2 // 1], [2 // 1, 4 // 1]]))
    println(repere(ovs3))
    println("fin test_repere-------------------------------")
end

function test_coords()
    println("début test coords vect--------------------------")
    println("cas des vecteurs")

function test()
