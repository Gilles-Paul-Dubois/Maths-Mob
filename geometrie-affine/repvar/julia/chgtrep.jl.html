include("reperes.jl")

function changement_origine()
    P=QP([1,2,3])
    Q=QP([2,1,-3])
    u=[1//1,2//1,1//1]
    v=[0//1,1//1,3//1]
    w=[1//1,0//1,2//1]
    S=VSQ([u,v,w])
    REP1=OVS{Rational{Int64}}(P,S)
    REP2=OVS{Rational{Int64}}(Q,S)
    M=QP([7,8,9])
    println(coords(REP1,M))
    println(coords(REP2,M))
end

function changement_base()
    M=QP([7,8,9])
    P=QP([1,2,3])
    u=[1//1,2//1,1//1]
    v=[0//1,1//1,3//1]
    w=[1//1,0//1,2//1]
    S1=VSQ([u,v,w])
    REP1=OVS{Rational{Int64}}(P,S1)
    U,V,W=2u,3v,4w
    S2=VSQ([U,V,W])
    REP2=OVS{Rational{Int64}}(P,S2)
    println(coords(REP1,M))
    println(coords(REP2,M))
end

function test()
    changement_origine()
    changement_base()
end

test()
