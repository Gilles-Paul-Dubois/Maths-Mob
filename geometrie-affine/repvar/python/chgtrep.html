# Changement de repère - change of frame 
from reperes import * 
 
 
def test1(): 
    """changement d'origine - change of origin""" 
    print("Illustration d'un changement d'origine - illustration of such a change") 
    P = PointKn([Fraction(1, 1), Fraction(2, 1), Fraction(3, 1)]) 
    Q = PointKn([Fraction(2, 1), Fraction(1, 1), Fraction(-3, 1)]) 
    u = VecteurKn([Fraction(1, 1), Fraction(2, 1), Fraction(1, 1)]) 
    v = VecteurKn([Fraction(0, 1), Fraction(1, 1), Fraction(3, 1)]) 
    w = VecteurKn([Fraction(1, 1), Fraction(0, 1), Fraction(2, 1)]) 
    B = ([u, v, w]) 
    R1 = Repere(P, B) 
    R2 = Repere(Q, B) 
    M = PointKn([Fraction(7, 1), Fraction(8, 1), Fraction(9, 1)]) 
    C1 = R1.Coord(M) 
    print(C1) 
    C2 = R2.Coord(M) 
    print(C2) 
    C3 = R1.Coord(Q) 
    print(C3) 
 
 
def test2(): 
    """Changement de base - change of basis""" 
    print("Illustration d'un changement de base - illustration of such a change") 
    M = PointKn([Fraction(7, 1), Fraction(8, 1), Fraction(9, 1)]) 
    P = PointKn([Fraction(1, 1), Fraction(2, 1), Fraction(3, 1)]) 
    u1 = VecteurKn([Fraction(1, 1), Fraction(2, 1), Fraction(1, 1)]) 
    v1 = VecteurKn([Fraction(0, 1), Fraction(1, 1), Fraction(3, 1)]) 
    w1 = VecteurKn([Fraction(1, 1), Fraction(0, 1), Fraction(2, 1)]) 
    B1 = ([u1, v1, w1]) 
    R1 = Repere(P, B1) 
    C1 = R1.Coord(M) 
    print(C1) 
    B2 = ([2 * u1, 3 * v1, 4 * w1]) 
    R2 = Repere(P, B2) 
    C2 = R2.Coord(M) 
    print(C2) 
 
 
def main(): 
    test1() 
    test2() 
 
 
if __name__ == '__main__': 
    main()
