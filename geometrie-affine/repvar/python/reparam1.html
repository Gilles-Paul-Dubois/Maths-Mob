# Calcul des coordonnées d'un point d'une variété (représentation paramétrique) 
# Computation of the coordinates of a point of a linear variety  (parametric representation) 
 
from reperes import * 
 
 
def main(): 
    # définition d'un plan - definition of a plane 
    P = PointKn([Fraction(1, 1), Fraction(2, 1), Fraction(3, 1)])  #un point du plan - a point of the plane 
    u = VecteurKn([Fraction(1, 1), Fraction(2, 1), Fraction(1, 1)])  #un premier vecteur directeur - a first direction vector 
    v = VecteurKn([Fraction(0, 1), Fraction(1, 1), Fraction(3, 1)])  #un second vecteur directeur - a second direction vector 
    B = ([u, v])  #base du plan vectoriel directeur - basis for the direction plane 
    R = Repere(P, B) 
    C = [Fraction(1, 3), Fraction(2, 5)]  #Un jeu de deux coefficients - a pair of coefficients 
    Q = R.Point(C)  #le point Q ayant pour paramètres 1/3 et 2/5 - point Q having 1/3 and 2/5 as parameters 
    print(Q) 
 
 
if __name__ == '__main__': 
    main()
