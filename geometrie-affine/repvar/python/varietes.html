from reperes import * 
 
 
class Hyperplan(): 
    """Hyperplan défini par une équation cartésienne - hyperplane under Cartesian form""" 
 
    def __init__(self, R, E): 
        """constructeur prenant en argument un repère et une équation sous forme de liste de coefficients - constructor from a frame and a list of coefficients""" 
        self.Rep = R 
        self.Equ = E 
        self.n = len(R.Base)  # dimension de l'espace - dimension of space 
        # test de compatibilité - compatibility test 
        if len(E) != self.n + 1: 
            raise MyError("Tailles non concordantes") 
 
    def __str__(self): 
        return "+".join(['(' + str(self.Equ[i]) + ')' + "x" + str(i + 1) for i in range(0, self.n)]) + '=' + str( 
            self.Equ[-1]) 
 
    def Contains(self, P): 
        """test de l'appartenance du point P à l'hyperplan - P belongs to the hyperplane ?""" 
        X = self.Rep.Coord(P) 
        return sum([self.Equ[i] * X[i] for i in range(0, self.n)]) == self.Equ[-1] 
 
 
class Variete(): 
    """Variete définie comme intersection d'hyperplans - variety as an intersection of hyperplanes""" 
 
    def __init__(self, L): 
        """Initialisation à partir d'une liste d'hyperplans - constructor from a list of hyperplanes""" 
        self.LH = L 
 
    def Contains(self, P): 
        """test de l'appartenance du point P à la variété - P belongs to the variety ?""" 
        return all([H.Contains(P) for H in self.LH]) 
 
 
def main(): 
    P = PointKn([Fraction(0, 1), Fraction(0, 1), Fraction(0, 1)])  # origine standard - standard origin 
    u = VecteurKn([Fraction(1, 1), Fraction(0, 1), Fraction(0, 1)]) 
    v = VecteurKn([Fraction(0, 1), Fraction(1, 1), Fraction(0, 1)]) 
    w = VecteurKn([Fraction(0, 1), Fraction(0, 1), Fraction(1, 1)]) 
    B = ([u, v, w])  # base canonique - standard basis 
    R = Repere(P, B) 
    Equation = [Fraction(1, 1), Fraction(-2, 1), Fraction(-1, 1), Fraction(3, 1)]  # plan x-2y-z=3 - plane x-2y-z=3 
    H = Hyperplan(R, Equation) 
    print(H) 
    M = PointKn([Fraction(1, 1), Fraction(-2, 1), Fraction(0, 1)]) 
    N = PointKn([Fraction(1, 1), Fraction(-1, 1), Fraction(0, 1)]) 
    print(H.Contains(M)) 
    print(H.Contains(N))
