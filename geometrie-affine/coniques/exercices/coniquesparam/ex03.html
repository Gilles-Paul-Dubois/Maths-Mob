    \( \displaystyle \int_{\frac{\pi}{2}}^{0}\sqrt{a^{2}\cos^{2}(u) + b^{2}\sin^{2}(u)} \times \left( {- 1} \right){du} \)

    \( \displaystyle \int_{0}^{\frac{\pi}{2}}\sqrt{a^{2}\cos^{2}(t) + b^{2}\sin^{2}(t)}{dt} \)

    \( \displaystyle \int_{0}^{\frac{\pi}{2}}a\sqrt{\cos^{2}(t) + \frac{b^{2}}{a^{2}}\sin^{2}(t)}{dt} \)
