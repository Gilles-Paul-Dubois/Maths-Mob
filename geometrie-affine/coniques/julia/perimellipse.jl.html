E(a,b)=sqrt(a*a-b*b)/a

function source(a,b,chnl)
    n=1
    e=E(a,b)
    u=-e^2/4
    s=1+u
    put!(chnl,s)
    while true
        n=n+2
        u=u*((n/(n+1))^2)*e^2
        s+=u/n
        put!(chnl,s)
    end
end

for a in 2:20
    L=0
    for b in 1:a-1
        channel=Channel((channel_arg)->source(a,b,channel_arg))
        for i in 1:20
            L=take!(channel)
        end
        L=2*pi*a*L
        print(L);print("------");print(pi*sqrt(2*(a*a+b*b)));print("------");println(pi*(3(a+b)-sqrt((3a+b)*(a+3b))))

    end
end
