def coeff(n): 
    """calcule le coefficient du terme de rang n=2p pour p >=1""" 
    """computes the coefficientof the term of rank rang n=2p for p >=1"""     
    result = 0.5 
    for i in range(3, n, 2): 
        result *= i 
        result /= i + 1 
    return (result * result) / (n - 1) 
 
 
def terme(a, b, n): 
    """calcule le terme de rank n=2p""" 
    """computes the term of rang n=2p"""     
    c = math.sqrt(a * a - b * b) 
    e = c / a 
    return pow(e, n) * coeff(n) 
 
 
def somme(a, b, n): 
    """ calcule la somme partielle de rang n=2p""" 
    """ computes the partial sum of rank n=2p"""     
    S = 1 
    for k in [2 * i for i in range(1, n // 2 + 1)]: 
        S -= terme(a, b, k) 
    return math.pi * 2 * a * S 
 
 
def approx1(a, b): 
    """première approximation du périmètre de l'ellipse""" 
    """first approximation of the perimeter of the ellipse"""     
    return math.pi * math.sqrt(2 * (a * a + b * b)) 
 
 
def approx2(a, b): 
    """seconde approximation du périmètre de l'ellipse""" 
    """second approximation of the perimeter of the ellipse"""     
    return math.pi * (3 * (a + b) - math.sqrt((3 * a + b) * (a + 3 * b))) 
 
 
def main(): 
    """compare les approximations"""
    """compares the approximations"""     
    for (a, b) in [(x, y) for x in range(2, 21) for y in range(1, x)]: 
        print(somme(a, b, 20), approx1(a, b), approx2(a, b))

    main()
