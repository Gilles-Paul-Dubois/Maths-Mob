tab = [8 1 -4; -4 4 -7 ; 1 8 4]
R(i,j)=tab[i,j]*(1/9)

function main()
    if R(3, 1) != 1 && R(3, 1) != -1
        theta1 = -asin(R(3, 1))
        theta2 = -pi - theta1
        psi1 = atan(R(3, 2) / cos(theta1), R(3, 3) / cos(theta1))
        psi2 = atan(R(3, 2) / cos(theta2), R(3, 3) / cos(theta2))
        phi1 = atan(R(2, 1) / cos(theta1), R(1, 1) / cos(theta1))
        phi2 = atan(R(2, 1) / cos(theta2), R(1, 1) / cos(theta2))
        println(psi1, " ", theta1," " ,phi1)
        println(psi2," ", theta2," ", phi2)
    else
        if R(3, 1) == 1
            phi1 = 0  # une valeur quelconque - any value
            theta1 = pi / 2
            psi1 = phi1 + atan(R(1, 2), R(1, 3))
            println(psi1, " ",theta1, " ",phi1)
        else
            phi1 = 0
            theta1 = -pi / 2
            psi1 = -phi1 + atan(-R(1, 2), -R(1, 3))
            println(psi1," ", theta1," ", phi1)
        end
    end
end

main()
