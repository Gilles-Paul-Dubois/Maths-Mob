import math 
 
"""Implementation de l'algorithme de determination des angles d'Euler"""
"""Implementating the algorithm of determination of Euler angles"""
 
tab = [[8, 1, -4], [-4, 4, -7], [1, 8, 4]] 
 
 
def R(i, j): 
    return tab[i - 1][j - 1] * (1.0 / 9) 
 
 
def main(): 
    if R(3, 1) != 1 and R(3, 1) != -1: 
        theta1 = -math.asin(R(3, 1)) 
        theta2 = -math.pi - theta1 
        psi1 = math.atan2(R(3, 2) / math.cos(theta1), R(3, 3) / math.cos(theta1)) 
        psi2 = math.atan2(R(3, 2) / math.cos(theta2), R(3, 3) / math.cos(theta2)) 
        phi1 = math.atan2(R(2, 1) / math.cos(theta1), R(1, 1) / math.cos(theta1)) 
        phi2 = math.atan2(R(2, 1) / math.cos(theta2), R(1, 1) / math.cos(theta2)) 
        print(psi1, theta1, phi1) 
        print(psi2, theta2, phi2) 
    else: 
        if R(3, 1) == 1: 
            phi1 = 0  # une valeur quelconque - any value 
            theta1 = math.pi / 2 
            psi1 = phi1 + math.atan2(R(1, 2), R(1, 3)) 
            print(psi1, theta1, phi1) 
        else: 
            phi1 = 0 
            theta1 = -math.pi / 2 
            psi1 = -phi1 + math.atan2(-R(1, 2), -R(1, 3)) 
            print(psi1, theta1, phi1) 
 
 
if __name__ == '__main__': 
    main()
