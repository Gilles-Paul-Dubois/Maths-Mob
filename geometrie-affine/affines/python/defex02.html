from affines import * 
 
 
def main(): 
    CZ3 = GenZp(3)  # la classe modélisant Z/3Z - model class for Z/3Z 
    Z3 = [CZ3(x) for x in range(0, 3)]  # le corps Z/3Z 
    P = PointKn([Z3[2], Z3[1], Z3[1]])  # le point P de l’énoncé - point P of ex02  
    V = VecteurKn([Z3[1], Z3[1], Z3[1]])  # le vecteur de l'énoncé - vector V of ex02 
    D = [x * V for x in Z3]  # droite vectorielle - vector line 
    B = [P + v for v in D]  # l'ensemble cherché - the set we are looking for 
    for M in B: 
        print(M) 
 
 
if __name__ == '__main__': 
    main()
