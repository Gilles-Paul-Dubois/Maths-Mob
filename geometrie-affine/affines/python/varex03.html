from affines import * 
 
 
def main(): 
    X = [Fraction(1, p) for p in range(1, 6)] 
    P = PointKn([Fraction(1, 3), 1, Fraction(1, 3)])  # le point P de l'énoncé - point P of ex03 
    V = VecteurKn([1, Fraction(1, 2), Fraction(1, 2)])  # le vecteur v de l'énoncé - vector v of ex03 
    F = [k * V for k in X]  # les 5 vecteurs - the 5 vectors 
    B = [P + x for x in F]  # les cinq points - the 5 points 
    for M in B:  # listage - listing 
        print(M) 
 
 
if __name__ == '__main__': 
    main()
