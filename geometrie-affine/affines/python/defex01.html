from affines import * 
 
 
def main(): 
    CZ3 = GenZp(3)  # la classe mod√©lisant Z/3Z -model class for Z/3Z 
    Z3 = [CZ3(x) for x in range(0, 3)] 
    L = [[x, y, z] for x in Z3 for y in Z3 for z in Z3] 
    A = [PointKn(c) for c in L]  # l'espace affine (Z/3Z)^3 - affine space (Z/3Z)^3 
    H = [P for P in A if (P.C[0] + P.C[1] + P.C[2]).nul()] 
    for P in H: 
        print(P) 
 
 
if __name__ == '__main__': 
    main()
