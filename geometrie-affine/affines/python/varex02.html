# -*- coding: utf-8 -*-

from affines import * 
 
 
def main(): 
    CZ3 = GenZp(3)  # la classe modélisant les éléments Z/3Z - model class for field Z/3Z 
    Z3 = [CZ3(x) for x in range(0, 3)]  # le corps Z/3Z entier - full iield Z/3Z 
    P = PointKn([Z3[1], Z3[2], Z3[1]])  # le point P de l'énoncé - point P of ex02 
    U = VecteurKn([Z3[2], Z3[1], Z3[1]])  # le vecteur u de l'énoncé - vector u of ex02 
    V = VecteurKn([Z3[1], Z3[0], Z3[1]])  # le vecteur V de l'énoncé- vector v of ex02 
    F = [k * U + h * V for k in Z3 for h in Z3]  # le sous-espace directeur (plan) - the direction (plane) 
    B = [P + x for x in F]  # la variété - the variety 
    for M in B:  # listage -listing 
        print(M) 
 
 
if __name__ == '__main__': 
    main()
