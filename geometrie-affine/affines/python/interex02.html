from affines import * 
 
def main(): 
    CZ3=GenZp(3) #la classe modélisant les éléments Z/3Z - model class for field Z/3Z 
    Z3=[CZ3(x) for x in range(0,3)] #le corps Z/3Z entier - all elements of Z/3Z 
    P=PointKn([Z3[1],Z3[2],Z3[1]]) #le point P de l'énoncé - point P of ex02 
    U=VecteurKn ([Z3[2],Z3[1],Z3[1]]) # le vecteur u de l'énoncé - vector u of ex02 
    V=VecteurKn ([Z3[1],Z3[0],Z3[1]]) # le vecteur v de l'énoncé - vector v of ex02 
    F1=[k*U+h*V for k in Z3 for h in Z3] # le plan vectoriel engendré par U et V 
    Q=PointKn([Z3[1],Z3[0],Z3[1]]) #le point Q de l'énoncé - point Q of ex02 
    W=VecteurKn ([Z3[0],Z3[0],Z3[1]]) # le vecteur w de l'énoncé - vector w of ex02 
    F2=[k*W for k in Z3] # droite vectorielle engendrée par w - vector line generated by w 
    B1=[P+x for x in F1] # le plan affine - affine plane 
    B2=[Q+x for x in F2] # la droite affine - afine line 
    INTER=[M for M in B1 if any(M==N for N in B2)] 
    for M in INTER:#liste les points de l'intersection - list of intersection points 
        print (M) 
 
if __name__ == '__main__': 
    main()
