using Nemo
const F = ResidueRing(ZZ, 5) #corps primaire ℤ/5ℤ - primary field Z/5Z

K=[F(i) for i in 0:4]

const NN = nmod
struct Point{T}
    C::Vector{T}
end
Base.show(io::IO, P::Point{T}) where {T} = print(io,P.C)
Base.:+(A::Point{T},V) where {T}=Point{T}(A.C+V)

const FP=Point{NN}
FP(V::Vector{Int64})=Point{NN}(map(x->F(x),V))

Base.:*(k::NN,V::Vector{NN})=[k*v for v in V] #produit d'un  vecteur par un scalaire - product of a vector by a scalar
function isequal(A::Point{NN},B::Point{NN}) #égalité de deux points de FP - equality of 2 points of FP
    s=length(A.C)
    return all((A.C[i]==B.C[i] for i in 1:s))
end

P=FP([1,2,1])
u=[F(2),F(1),F(1)]
D=[P+(x*u) for x in K] #droite affine - affine line
println(D)
