using Nemo
const F = ResidueRing(ZZ, 3) #corps primaire ℤ/3ℤ - primary field Z/3Z
K=[F(i) for i in 0:2] #les 3 éléments de ce corps - the 3 elements of this field
D=[[k,k,k] for k in K]

const NN = nmod
#point sans masse - simple point without mass
struct Point{T}
    C::Vector{T}
end
const FP=Point{NN}
FP(V::Vector{Int64})=Point{NN}(map(x->F(x),V))
Base.show(io::IO, P::Point{T}) where {T} = print(io,P.C)
Base.:+(A::Point{T},V) where {T}=Point{T}(A.C+V)

P=FP([2,1,1])
H=[[P+v] for v in D]
println(H)
