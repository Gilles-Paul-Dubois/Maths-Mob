function test_point()
    println("test_point-------------------------------------")
    println("réels")
    P1=RP([1,2,3])
    println(P1)
    println(gettype(P1))
    P4=RP([1.0,3,6])
    println("différence de deux points")
    println(P1-P4)
    println(typeof(P1-P4))
    println("translaté d'un point")
    println(P1+[1,1,1])
    println("rationnels")
    P2=QP([1//2,2,3])
    println(P2)
    println(gettype(P2))
    P5=QP([1,3,6])
    println("différence de deux points")
    println(P2-P5)
    println(typeof(P2-P5))
    println("translaté d'un point")
    println(P2+[1//2,1,1])
    println("Z/pZ")
    P3=FP([F(2),F(4),F(6)])
    println(P3)
    println(gettype(P3))
    P6=FP([1,3,5])
    println("différence de deux points")
    println(P3-P6)
    println(typeof(P3-P6))
    println("translaté d'un point")
    println(P3+[F(2),F(4),F(6)])
    println("fin de test_point-------------------------------")
end
