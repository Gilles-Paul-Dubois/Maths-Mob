#point sans masse - point without mass
struct Point{T}
    C::Vector{T}
end

gettype(P::Point{T}) where {T}= T
Base.show(io::IO, P::Point{T}) where {T} = print(io,P.C)
const QP=Point{Rational{Int64}}
Base.:+(A::Point{T},V) where {T}=Point{T}(A.C+V)
Base.:*(k::Rational{Int64},V::Vector{Rational{Int64}})=[k*v for v in V] #produit d'un vecteur par un scalaire - external operator scalar×vector --> vector
X=[1//1,1//2,1//3,1//4,1//5]
P=QP([1//3,1//1,1//3])
v=[1//1,1//2,1//2]
D=[P+k*v for k in X]
println(D)
