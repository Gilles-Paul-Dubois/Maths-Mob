#calcule un système de vecteurs directeurs de la variété engendrée par le système de points S
#computes a system of directing vectors for the variety generated by the system of points S
function directeurs(S::PointsSystem{P}) where {P}
    W=S.Pts
    n=length(W)
    S1=[W[i]-W[1] for i in 2:n]
    T=typeof(S1[1][1])
    return VectorsSystem{T}(S1)
end

#calcul de la dimension de la variété engendrée par le système de points S
#computation of the dimension of the variety generated  by the system SS
function rang(S::PointsSystem{P}) where {P}
    VS=directeurs(S)
    return rang(VS)
end
