import numpy as np 
""" transposition de matrices avec numpy - transposing matrices with numpy""" 
 
 
def main(): 
 
    A=np.arange(6).reshape(2,3) 
    print (A) 
    B=A.transpose()#première syntaxe - first syntax 
    print (B) 
    C=np.ones((2,3))# 2 lignes, 3 colonnes seulement des 1 - 2 rows, 3 columns only 1,s 
    print (C) 
    C=np.transpose(C)#autre syntaxe possible - other possible syntax 
    print (C.shape) #le type - the type 
 
 
if __name__ == '__main__': 
    main()
