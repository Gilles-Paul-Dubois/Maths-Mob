from numpy import matrix 
from scipy.linalg import inv 
 
# matrice de passage calculée par inversion - transition matrix computed by inversion 
 
def main(): 
    V1=[2,3,1,2] 
    V2=[1,2,1,-1] 
    V3=[0,0,3,2] 
    V4=[0,0,4,3] 
    #matrices réelles - real matrices 
    A=matrix([V1,V2,V3,V4]).transpose() # 4 lignes, 4 colonnes - 4 rows, 4 columns 
 
    P=inv(A) 
    print (A) #affichage - display 
    print (P) 
 
# des erreurs d'arrondis laissent apparaître des valeurs nulles comme 5.55111512e-17 - a few rounding errors 
if __name__ == '__main__': 
    main()
