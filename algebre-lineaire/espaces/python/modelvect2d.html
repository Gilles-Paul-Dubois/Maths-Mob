# -*- coding: utf-8 -*- 
#python version 3.4 
 
class Vector2D: 
    """modélisation vecteurs 2D par une classe python - Python class for modeling 2D vectors """ 
    def __init__(self,a,b): 
        """constructeur - constructor""" 
        self.x=float(a) 
        self.y=float(b) 
    def __str__(self): 
        """représentation externe- conversion to string""" 
        return "("+str(self.x)+","+str(self.y)+")" 
    def __add__(self,other): 
        """addition de deux vecteurs - addition""" 
        return Vector2D(self.x+other.x,self.y+other.y) 
    def __neg__(self): 
        """opposé d'un vecteur - opposite""" 
        return Vector2D(-self.x,-self.y) 
    def __sub__(self,other): 
        """différence de deux vecteurs -difference""" 
        return Vector2D(self.x-other.x,self.y-other.y) 
    def __rmul__(self,k): 
        """loi externe - external multiplication""" 
        return Vector2D(k*self.x,k*self.y) 
 
     
def main(): 
    """fonction principale - main function""" 
    u=Vector2D(1,2) 
    v=Vector2D(2,3) 
    print (u+v) 
    print (u-v) 
    print (-u) 
    print (2*u) 
     
if __name__=='__main__': 
    main()
