""" Vérification expérimentale des propriétés d'un sous-espace vectoriel""" 
""" Experimental checking of properties of a subspace""" 
from numpy import array 
from random import randint 
 
# ensemble aléatoire de vecteurs du 'plan' P x+y+z=0 
# Random subset of the plane P x+y+z=0
L = [array([x,y,-x-y]) for x in range(-10,11) for y in range(-10,11)] 
 
# test d'appartenance de u au plan P 
# Checking u member of P
def IsInP (u): 
    return u[0]+u[1]+u[2]==0 
 
""" Liste aléatoire de 20 vecteurs de la forme au+bv avec u dans L, v dans L et a et b aléatoires entiers """ 
""" Random list of 20 vectors of the form au+bv with u in L, v in L and a , b random integers """ 
AL=[ randint(-10,10)*L[randint(0,400)]+randint(-10,10)*L[randint(0,400)] for i in range(0,20)] 
 
 
def main(): 
    """ vérifie si tous les vecteurs de AL sont dans P""" 
    """ Checks if all vectors of AL are in P"""    
    global AL 
    for u in AL: 
        print (u) 
    print (all([IsInP(u) for u in AL])) 
 
if __name__=='__main__': 
    main()
