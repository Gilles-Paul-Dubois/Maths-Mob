""" Calculs de combinaisons linéaires""" 
""" Computations of linear combinations""" 
from numpy import array 
from functools import reduce 
 
 
def delta (i,j): 
    """symbole de Kronecker - Kronecker symbol""" 
    if i==j: 
        return 1 
    return 0 
 
def canonique(i,n): 
    """vecteur (0,..0,1,0, ..0) de R^n""" 
    """vector (0,..0,1,0, ..0) of R^n"""     
    return array([ delta(i-1,j) for j in range(0,n)]) 
 
def combinaison(coeffs, vecteurs): 
    """combinaison linéaire d'un système de vecteurs""" 
    """linear combination of a system of vectors"""     
    L= [coeffs[i]*vecteurs[i] for i in range(0,len(coeffs))] 
    def somme(x,y): 
        return x+y 
    return reduce(somme,L) 
 
def main(): 
    """fonction principale - main function""" 
    coeffs=[1,2,3,4,5,6] 
    vecteurs=[canonique(i,6) for i in range(1,7)] 
    print (combinaison(coeffs,vecteurs)) 
 
if __name__=='__main__': 
    main()
