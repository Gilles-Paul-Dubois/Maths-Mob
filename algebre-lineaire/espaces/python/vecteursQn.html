# -*- coding: utf-8 -*- 
#python version 3.4 
 
from fractions import Fraction 
 
class VectornD: 
    """modélisation vecteurs de dimension n à coefficients rationnels  par une classe python""" 
    """Modeling vectors of n dimensions with rational coefficients by a Python class"""     
    def __init__(self, m, L): 
        """constructeur - constructor""" 
        self.n=m 
        self.coeffs=L 
 
    def __str__(self): 
        """représentation externe - conversion to string""" 
        L=[str(x) for x in self.coeffs] 
        return "("+",".join(L)+")" 
 
    def __add__(self,other): 
        """addition de deux vecteurs - addition""" 
        L=[] 
        for i in range(0,self.n): 
            L.append(self.coeffs[i]+other.coeffs[i]) 
        return VectornD(self.n,L) 
    def __neg__(self): 
        """opposé d'un vecteur - opposite""" 
        L=[] 
        for i in range(0,self.n): 
            L.append(-self.coeffs[i]) 
        return VectornD(self.n,L) 
 
    def __sub__(self, other): 
        """différence de deux vecteurs - difference""" 
        L = [] 
        for i in range(0, self.n): 
            L.append(self.coeffs[i] - other.coeffs[i]) 
        return VectornD(self.n, L) 
 
    def __rmul__(self,k): 
        """loi externe - external multiplication""" 
        L=[] 
        for i in range(0,self.n): 
            L.append(self.coeffs[i]*k) 
        return VectornD(self.n,L) 
 
     
def main(): 
    """fonction principale - main function""" 
    #exemples avec la dimension 2 - example with dimension 2 plane 
    f1=Fraction(1,2) 
    f2=Fraction (2,3) 
    f3=Fraction(-2,5) 
    f4=Fraction (7,-3) 
    u=VectornD(2,[f1,f2]) 
    v=VectornD(2,[f3,f4]) 
    print (u+v) 
    print (u-v) 
    print (-u) 
    print (2*u) 
    #exemple avec la dimension 3 - exemple with dimension 3, 3D space 
    w=VectornD(3,[f1,f2,f3]) 
    print(2*w) 
     
if __name__=='__main__': 
    main()
