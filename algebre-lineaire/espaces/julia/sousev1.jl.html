L=[[x,y,-x-y] for x in -10:10 for y in -10:10]

IsInP(u)=reduce(+,u)==0

u=L[rand(1:400)]
v=L[rand(1:400)]

AL=[rand(-10:10)*u+rand(-10:10)*v for i in 1:20]

function main()
    print("u=");println(u)
    print("v=");println(v)
    for w in AL
        println(w)
    end
    println(all(IsInP(w) for w in AL))
end

main()
