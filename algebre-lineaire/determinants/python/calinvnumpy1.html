# exemple de calcul d'inverse avec la librairie scipy 
# example of computation of an inverse with  scipy library 
 
from numpy import matrix 
from scipy import linalg 
 
A = matrix('[3 1 1 1; 1 3 1 1; 1 1 3 1; 1 1 1 3]') 
 
 
def main(): 
    print(linalg.inv(A)) 
 
 
if __name__ == '__main__': 
    main()
