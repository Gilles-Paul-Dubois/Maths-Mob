A=[0 0 2 1;0 0 -4 0;1 1 7 0;-1 0 -17 0]

delrowcol(A,i,j)=A[1:end .!=i,1:end .!=j] #suppression de la ième ligne et de la jème colonne - delete i-th row and j-th column

#développement suivant la première ligne - expansion along the first row
function det(A)
    if size(A,1)==1
        return A[1,1]
    end
    s=size(A,1)
    L=[(-1)^(1+j)*A[1,j]*det(delrowcol(A,1,j)) for j in 1:s]
    return reduce(+,L)
end

println(A)
println(det(A))
