using Polynomials

function legendre(chnl)
    n=0
    P=Polynomial([1//1])
    put!(chnl,P)
    Q=Polynomial((0,1//1))
    put!(chnl,Q)
    while true
        n=n+1
        T=P
        P=Q
        Q=Polynomial([0,(2n+1)//(n+1)])*Q- n//(n+1)*T
        put!(chnl,Q)
    end
n=0
    
end

my_channel = Channel( (channel_arg) -> legendre(channel_arg) )

function main()
    for i in 1:5
        println(take!(my_channel))
    end
end

main()
