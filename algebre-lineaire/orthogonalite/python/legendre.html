""" Program which computes the sequence of Legendre polynomials"""

    n=0 
    P=poly1d([1]) 
    yield P 
    n=1

        P,Q=Q,poly1d([(2*n+1.0)/(n+1.0),0])*Q-poly1d([n/(n+1.0)])*P 
        n=n+1         
        yield Q
