using LinearAlgebra
#surcharge de * pour le produit scalaire - oveloading * for dot product
Base.:*(v1,v2)=reduce(+,map(*,v1,v2))
#norme d'un vecteur - norm of a vector
norme(v)=sqrt(v*v)
#normalisation - normalization
normalize(v)=(1/norme(v))*v

function aleamatrice(n)
    R=zeros(n,n)
    for i in 1:n, j in 1:n
            R[i,j]=rand(-n:n)
        end
    return R
end

c(p,q,A)= A[p,p]==0 && A[q,p]==0 ? 1 : A[p,p]/sqrt(A[p,p]*A[p,p]+A[q,p]*A[q,p])
s(p,q,A)= A[p,p]==0 && A[q,p]==0 ? 0 : A[q,p]/sqrt(A[p,p]*A[p,p]+A[q,p]*A[q,p])

function G(q,p,A)
    n=size(A,1)
    G=Matrix(1.0I,n,n)
    G[p,p]=c(p,q,A) 
    G[p,q]=s(p,q,A) 
    G[q,p]= -s(p,q,A) 
    G[q,q]= c(p,q,A)  
    return G
end

function q(A)
    n=size(A,1)
    q=Matrix(1.0I,n,n)
    for i in 1:n, j in 1:i-1
        q=G(i,j,A)*q
        A=G(i,j,A)*A
    end
    return q
end


function main()
    n=4
    A=aleamatrice(n)
    println(A)
    Q=q(A)
    R=Q*A
    for i in 1:n, j in 1:i-1
        R[i,j]=0
    end    
    Q=transpose(Q)
    println(Q)
    println(R)
    println(Q*R)

end

main()
