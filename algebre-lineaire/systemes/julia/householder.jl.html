using LinearAlgebra
#surcharge de * pour le produit scalaire - overloading * for dot product
Base.:*(v1,v2)=reduce(+,map(*,v1,v2))
#norme d'un vecteur - norm of a vector
norme(v)=sqrt(v*v)
#normalisation - normalization
normalize(v)=(1/norme(v))*v

function aleamatrice(n)
    R=zeros(n,n)
    for i in 1:n, j in 1:n
            R[i,j]=rand(Float64)
        end
    return R
end

function H1(A)
    n=size(A,1)
    X=A[:,1]
    N1=norm(X)
    Y=repeat([0.0],n)
    W=X-Y
    W=normalize(W)
    W=hcat(W)
    TENS=W*transpose(W)
    IDn=Matrix(1.0I,n,n)
    H=IDn-2TENS
    return H
    
end

function q(A)
    n=size(A,1)
    if n==1
        return Matrix(I,1,1)
    end
    R1=H1(A)*A
    R2=R1[2:end,2:end]
    Q1=q(R2)
    Q2=Matrix(1.0I,n,n)
    for i in 2:n, j in 2:n
        Q2[i,j]=Q1[i-1,j-1]
    end
    return Q2*H1(A)
end

function main()
    n=3
    A=aleamatrice(n)
    println(A)
    Q=q(A)
    println(Q)
    R=Q*A
    println(R)
    Q=transpose(Q)
    println(Q*R)

end

main()
