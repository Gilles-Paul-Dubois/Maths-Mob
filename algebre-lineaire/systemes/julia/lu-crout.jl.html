using LinearAlgebra
A = [1//1 2//1 3//1; 4//1 5//1 6//1;7//1 8//1 8//1]

zerosQ(n)=Matrix{Rational{Int64}}(zeros(n,n))
unitQ(n)=Matrix{Rational{Int64}}(Matrix(I, n, n))

function Perm(i,j,n)
    R=unitQ(n)
    L=R[i,:]
    R[i,:]=R[j,:]
    R[j,:]=L
    return transpose(R)
end


function LU(M)
    n=size(M,1)
    L=zerosQ(n)
    U=unitQ(n)
    PERM=unitQ(n)
    for k in 1:n
        for i in k:n
            m=[L[i,h]*U[h,k] for h in 1:k]
            if length(M)==0
                s=0//1
            else
                s=reduce(+,m)
            end
            L[i,k]=M[i,k]-s
        end
        j=k
        while L[j,k]==0
            j+=1
        end
        p=j
        if p != k
            PERM=Perm(p,k,n)*PERM
            L=M[p,:]
            M[p,:]=M[k,:]
            M[k,:]=L
        end
        for j in k+1:n
            m=[L[k,h]*U[h,j] for h in 1:k]
            if length(m)==0
                s=0//1
            else
                s=reduce(+,m)
            end
            U[k,j]=(M[k,j]-s)/L[k,k]
        end
    end
    return L,U,PERM
end
