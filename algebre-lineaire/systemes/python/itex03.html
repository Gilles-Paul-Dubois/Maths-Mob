# détermination du point fixe de ln(x)+2 
 
import math 
 
def IterFonc(): 
    x=1 
    yield x 
    while True: 
        x=math.log(x)+2 
        yield x 
         
def main(): 
    I=IterFonc() 
    for i in range (0,20): 
        x=next(I) 
    print (x) 
    print (math.log(x)+2) 
 
main()
