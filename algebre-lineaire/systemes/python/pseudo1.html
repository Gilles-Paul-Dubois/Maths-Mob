from numpy import matrix 
from scipy.linalg import pinv 
# calcul de pseudo-solution - computation of pseudo-solution 
#au moyen de la pseudo-inverse de Moore-Penrose 
 #using the Moore-Penrose pseudo-inverse  
# la matrice du système sur-déterminé - the matrix of the system 
A= matrix([[1,1],[1,-1],[1,2]]) 
print (A) 
#le second membre - the second member 
B=matrix([[1],[-1],[4]]) 
print (B) 
#calcul de l'inverse de Moore-Penrose - Computation of the Moore-Penrose inverse 
PIA=pinv(A) 
print (PIA) 
#application au second membre pour la pseudo-solution<- applying to the second memeber for the solution 
print (PIA*B)
