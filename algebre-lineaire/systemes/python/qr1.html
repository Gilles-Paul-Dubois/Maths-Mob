# -*- coding: utf-8 -*-

from scipy.linalg import qr 
import numpy as np 
 
 
def main(): 
    n = 3 
    A = np.random.rand(n, n)  # matrice aléatoire - random matrix 
    Q, R = qr(A)  # décomposition QR de A - QR decomposition of A 
    print(A) 
    print('--------------') 
    # résultats - results 
    print(Q) 
    print('--------------') 
    print(R) 
    print('--------------') 
    Q = np.matrix(Q) 
    R = np.matrix(R) 
    print(Q * R) 
 
 
if __name__ == '__main__': 
    main()
