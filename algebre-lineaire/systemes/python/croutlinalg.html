from numpy import * 
from scipy.linalg import * 
 
 
def main(): 
    A = array([[1, 2, 3], [4, 5, 6], [7, 8, 8]]) 
    P, L, U = lu(A) 
    print(L) 
    print(U) 
    print(dot(L, U))  # tout simplement le produit des deux matrices L et U - simply the product of L and U 
    print(dot(P, dot(L, U)))  # doit tout remettre en ordre - should restore everything in good order 
 
 
if __name__ == '__main__': 
    main()
