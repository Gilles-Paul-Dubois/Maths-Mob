from scipy import linalg
from random import randint 
import numpy as np 
from functools import reduce 

### Méthode des itérations successives de Seidel 
### Seidel method of successive iterations 
 
def DiagDom(n): 
    """Genère une matrice aléatoire à diagonale dominante""" 
    """Generates ua diagonally dominant random matrix"""     
    M = np.random.rand(n, n) #matrice aléatoire - random matrix 
    for i in range (0,n): 
        L1=[abs(M[i,j]) for j in range(0,n) if j !=i] 
        L2=[abs(M[j,i]) for j in range(0,n) if j !=i] 
        s= reduce(lambda x,y:x+y,L1+L2) 
        M[i,i]=s+randint(5,10) 
    return np.matrix(M) 
 
def MatD(A): 
    """Matrice D de A - D matrix for A""" 
    D=A.copy() 
    n=len(D) 
    for i in range(0,n): 
        for j in range(0,i): 
            if j<i: 
                D[i,j]=0 
    return np.matrix(D) 
 
def MatM(A): 
    """Matrice extraite de A triangulaire inférieure - Upper triangular matrix extracted from A""" 
    M=A.copy() 
    for i in range(0,len(A)): 
        for j in range(i,len(A)): 
            M[i,j]=0 
    return M 
 
def Seidel(D,M,B,X): 
    """itérateur de Gauss-Seidel - Gauss Seidel iterator""" 
    yield X 
    while True: 
        X=D*B-D*M*X 
        yield X 
 
 
def SeidTest(k): 
    n=4 
    A=DiagDom(n) # matrice du système - matrix of the system 
    B= np.matrix(np.random.rand(n, 1)) #matrice aléatoire second membre - random second member 
    D=MatD(A) 
    D=linalg.inv(D) 
    M=MatM(A) 
    X=np.matrix( [[0] for i in range(0,n)]) 
    Seid=Seidel(D,M,B,X) 
    for i in range(0,k):# k itérations par Seidel - k steps with Seidel 
        S1=next(Seid) 
    S2=linalg.solve(A,B)# solution 'exacte' - 'exact' solution 
    return linalg.norm(S1-S2)     
 
def main(): 
    for k in range(5,41,5): 
        s=0 
        for i in range(0,100): 
            s+= SeidTest(k) 
        print (k,s/100) 
     
if __name__== "__main__":
