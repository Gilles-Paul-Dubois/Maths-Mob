from numpy import matrix 
from scipy.linalg import solve 
# résolution de systèmes - system resolution 
# un exemple mal conditionné - an ill-conditioned example 
 
# la matrice du système - the matrix of the system 
A= matrix([[1,1,1,1],[1,3,1,1],[1,1,3,1],[1,1,1,3]]) 
#le second membre - the second memeber 
B=matrix([[1],[2],[3],[3]]) 
print (A) 
print ("*********************") 
print (B) 
print ("*********************") 
S=solve(A,B) #non destructeur - non destructive 
print ("*********************") 
print (S) 
print ("*********************") 
print (A) 
print ("*********************") 
print (B) 
print ("*********************") 
S=solve(A,B,overwrite_a = True, overwrite_b = True)# peut éventuellement détruire A et B - can eventually destroy A and B
