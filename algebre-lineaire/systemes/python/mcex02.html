import numpy 
### Conditionnement des matrices de Hilbert### 
### Conditioning of Hilbert matrices### 
 
def Hilbert(n): 
    """Matrice de Hilbert d'ordre n - Hilbert matrix of order n""" 
    H=numpy.eye(n) # matrice unité d'ordre n - unity matrix of order n 
    for i in range(0,n): 
        for j in range(0,n): 
            H[i,j]=1.0/(i+j+1)# définition des matrices de Hilbert - definition of Hilbert matrices 
    return H 
 
def main(): 
    for n in range(1,10): 
        H=Hilbert(n) 
        print (numpy.linalg.cond(H)) # afficher le conditionnement pour la norme L2 - print conditioning for norm L2 
 
 
if __name__=="__main__": 
    main()
