# détermination du point fixe de exp(-x) 
 
import math 
 
def IterExp(): 
    x=0 
    yield x 
    while True: 
        x=math.exp(-x) 
        yield x 
         
def main(): 
    I=IterExp() 
    for i in range (0,20): 
        x=next(I) 
    print (x) 
    print (math.exp(-x)) 
 
main()
