from numpy import matrix 
from scipy.linalg import solve 
# résolution de systèmes - solving systems 
# un exemple mal conditionné - an example of ill-conditioning 
 
# la matrice du système - the matrix of the system 
A= matrix([[0.12065,0.98775],[0.12032,0.98755]]) 
#les seconds membres - second members 
B1=matrix([[2.01045],[2.00555]]) 
B2=matrix([[2.01145],[2.00555]]) 
S1=solve(A,B1) 
S2=solve(A,B2) 
print (S1) 
print ("*********************") 
print (S2)
