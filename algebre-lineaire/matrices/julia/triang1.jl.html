using LinearAlgebra

M=rand(3,3) # matrice aléatoire de réels dans l'intervalle [[0,1]-3 lignes - 3 colonnes
println(M)
# quelques procédés de fabrication de matrices triangulaires à partir de M
T1=LowerTriangular(M)
println(T1)
TU1=UnitLowerTriangular(M)
println(TU1)
T2=UpperTriangular(M)
println(T2)
TU2=UnitUpperTriangular(M)
println(TU2)
