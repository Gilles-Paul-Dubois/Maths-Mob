using SparseArrays
using LinearAlgebra
A = sparse([1, 1, 2, 3], [1, 3, 2, 3], [0, 1, 2, 0])
println(A)
#les valeurs manquantes sont supposées nulles - missing values supposed to be null
#s'il en reste on peut encore purger - If there are some left we can continue to purge
B=dropzeros(A)
println(B)
#retrouver les paramètres du constructeur (positions et valeurs) - retrieve parameters of the constructor (positions and values)
println(findnz(B))
#positions des nuls - positions of the zeros
println(findall(iszero, B))
#l'unité de R^5 - unity matrix of ℝ5
I5=Matrix(1.0I, 5, 5)
#affichage pour contrôle - display for checking
println(I5)
#que nous 'raréfions' - which we 'rarify"
I5P=sparse(I5)
#on affiche - display
println(I5P)
