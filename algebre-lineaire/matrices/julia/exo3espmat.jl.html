#classe paramétrée pour représenter tous les corps Z/pZ
#parameterized class to represent all fields Z/pZ

macro Z(p)
    quote
        struct F_p'
            x::Int
            function F_p(n::Int)#constructeur interne - internal costructor
                r=n % $p
                if r>= 0
                    return new(r) # appel à new possible unqiuement dans le constructeur 'interne'- call to new
                else
                    return new($p+r) # à cause de la définition du reste en julia ...- because of remainder def in Julia
                end
            end
        end
    end
end

Base.:+(k::F_p,h::F_p) = F_p(k.x+h.x)
Base.:*(k::F_p,h::F_p) = F_p(k.x*h.x)
Base.:-(k::F_p)=F_p(-k.x)
Base.:-(k::F_p,h::F_p)=F_p(k.x-h.x)
Base.string(k::F_p)=string(k.x)

#macro pour construire l'espace vectoriel puissance n d'un corps quelconque, Z/pZ, Q, R, C, etc..
#macro to build nth-power of any field Z/pZ, Q, R, C, etc..
# n détermine la dimension et T le corps via le type - n fixes the dimension and T the field via the type
macro E(n, T)
    quote
        struct V
            C::Vector{$T} #unique donnée membre un tableau à une dimension - signe memebr data one dimensional array
            function V(W) #constructeur 'interne' - internal constructor
                if length(W) != $n #le paramètre du constructeur doit avoir la longueur voulue - the parameter of the constructor should have the desired length
                    throw(IncorrectVector("dimension du tableau incorrecte"))
                end
                #Voyons si le transtypage est possible - Let's see if trans-typing is possible
                for x in W
                    if typeof(x)!=$T
                        try
                            y=$T(x)
                        catch err
                            throw(IncorrectVector("Type non convertible"))
                        end
                     end
                end
                k(x)= typeof(x)==$T ? x : $T(x)#transtyper si besoin est - change type if needed
                W=[k(x) for x in W]
                return new(W) #appel à new possible depuis le constructeur interne - call to new possible from this place
            end
        end
    end
end

Base.getindex(v::V,i::Int64)= v.C[i] #surcharge des crochets- overloading brackets
Base.:+(v1::V,v2::V)=V(map(+,v1.C,v2.C)) #addition des vecteurs - addition of vectors
Base.:-(v::V)=V([-x for x in v.C]) #opposé - opposite
Base.:-(v1::V,v2::V)=V(map(-,v1.C,v2.C)) #différence - difference
function Base.:*(λ,v::V) #loi externe -produit d'un vecteur par un scalaire - external multiplication
    T=typeof(v[1])
    try
        if typeof(λ)!=T
            λ=T(λ)
        end
    catch err
        throw(IncorrectVector("Type non convertible"))
    end
    return V(map(x->λ*x,v.C))
end
Base.string(v::V)=map(y->y.x,v.C)
nul(v::V)= all((nul(c) for c in v.C))

function Base.:*(λ::F_p,M::Matrix{F_p})#produit d'une matrice par un scalaire- product of a matrix by a scalar
    m,n=size(M,1),size(M,2)
    R=copy(M)
    for i in 1:m,j in 1:n
        R[i,j]=λ*M[i,j]
    end
    return R
end
