#checking distributivity of product over sum

function aleamat(m,n)
    A=zeros(Int64,m,n)
    for i in 1:m
        for j in 1:n
            A[i,j]=rand(-9:+9)
        end
    end
    return A
end


for m in 3:4, n in 2:4, p in 2:5
    A=aleamat(m,n)
    B=aleamat(n,p)
    C=aleamat(n,p)

    P1=A*(B+C)
    P2=A*B+A*C

    print(P1==P2); print("-")
end
