from scipy import sparse 
from numpy.random import rand 
 
""" Conversion d'une matrice creuse en matrice dense""" 
""" Conversion of a sparse matrix into a dense one"""
 
A = sparse.lil_matrix((20, 20)) 
A[0, :5] = rand(5) #5 coefficients 
A[1, 5:7] = A[0, :2] #2 nouveaux - 2 new ones 
A.setdiag(rand(20)) #19 autres - 19 others 
### A est creuse aléatoire - A is random sparse 
B=A.todense() 
 
def main(): 
    print(B) # affichage type:'creux' 26 coefficients sur 400 non nuls- display 'sparse' type 26 coefficients non zeros on a total of 400 
 
 
if __name__ == '__main__': 
    main()
