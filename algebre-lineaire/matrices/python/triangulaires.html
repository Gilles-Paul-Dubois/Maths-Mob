from random import randint 
from numpy import array, matrix, eye 
 
 
def TriangleSup(n): 
    """Triangulaire supérieure aléatoire""" 
    T = eye(n) 
    for i in range(0, n): 
        for j in range(i, n): 
            T[i, j] = randint(1, 9) 
    return T 
 
 
def TriangleInf(n): 
    """Triangulaire inférieure aléatoire""" 
    T = eye(n) 
    for i in range(0, n): 
        for j in range(0, i + 1): 
            T[i, j] = randint(1, 9) 
    return T 
 
 
def main(): 
    print(TriangleSup(10)) 
    print(TriangleInf(10)) 
 
 
if __name__ == '__main__': 
    main()
