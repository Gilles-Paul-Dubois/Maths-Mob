### Produit de matrices à coefficients dans Z/pZ###
 
def GenZp(p): 
    """génère la classe paramétrée Z/pZ - generates parameterized class Z/pZ""" 
 
    class C(object): 
        car = p 
 
        def __init__(self, m): 
            """initialisation à partir d'un entier - constructor""" 
            self.n = m % self.car 
 
        def __str__(self): 
            """représentation externe pour print etc...- conversion to string""" 
            return str(self.n) 
 
        def nul(self): 
            """test de nullité - nullity test""" 
            return self.n == 0 
 
        ### les opérateurs unaires - unary operators 
        def __neg__(self): 
            """opposé - opposite""" 
            return C(-self.n) 
 
        def __invert__(self): 
            """inverse""" 
            if not self.nul(): 
                y = 1 
                while (self.n * y) % self.car != 1: 
                    y += 1 
            return C(y) 
 
        ### les opérations binaires - binary operators 
        def __add__(self, other): 
            """addition""" 
            return C(self.n + other.n) 
 
        def __rmul__(self, other): 
            """multiplication""" 
            return C(self.n * other.n) 
 
        def __sub__(self, other): 
            """soustraction - subtraction""" 
            return C(self.n - other.n) 
 
        def __div__(self, other): 
            return self * ~(other) 
 
        def __pow__(self, k): 
            """puissances - powers""" 
            if k >= 0: 
                return C(self.n ** k) 
            else: 
                return ~self ** (-k) 
 
    return C 
 
 
def GenVectnZp(p, m): 
    """génère l'espace Z/pZ x .... x Z/pZ (n facteurs) - generates n-th power of Z/pZ""" 
 
    class Vpn: 
        n = m 
 
        def __init__(self, L): 
            Zp = GenZp(p) 
            self.V = [Zp(x) for x in L] 
 
        def __str__(self): 
            L = [str(x) for x in self.V] 
            return '(' + ",".join(L) + ')' 
 
        def nul(self): 
            """test de nullité - nullity test""" 
            return all([x.nul() for x in self.V]) 
 
        def __add__(self, other): 
            """addition des vecteurs - addition of vectors""" 
            L = [self.V[i].n + other.V[i].n for i in range(0, self.n)] 
            return Vpn(L) 
 
        def __neg__(self): 
            """opposé d'un vecteur""" 
            return Vpn([-x.n for x in self.V]) 
 
        def __sub__(self, other): 
            """différence de deux vecteurs - difference of 2 vectors""" 
            return self + (-other) 
 
        def __rmul__(self, k): 
            """ produit par un scalaire - external product """ 
            L = [k.n * x.n for x in self.V] 
            return Vpn(L) 
 
    return Vpn 
 
 
""" Objet Matrice à coefficients dans Z/pZ - The matrix object with coefs in Z/pZ """ 
 
 
class Matrice(): 
    """Modélisation d'une matrice à coefficients dans Z/pZ - Model for such matrix""" 
 
    def __init__(self, L, p): 
        """constructeur à partir d'une liste de lignes - constructor from list of rows""" 
        self.carac = p; 
        self.coeffs = L  # les coefficients - the coefficients 
        self.lignes = len(L)  # le nombre de lignes - the number of rows 
        self.colonnes = L[0].n  # le nombre de colonnes - the number of columns 
 
    def __str__(self): 
        """représentation externe - converts to string""" 
        return '\n'.join([str(v) for v in self.coeffs]) 
 
    def GetCoef(self, i, j): 
        """Obtenir un coefficient - get one coefficient""" 
        return self.coeffs[i].V[j] 
 
    def SetCoef(self, i, j, v): 
        """Changer un coefficient - modify one coefficient""" 
        self.coeffs[i].V[j] = v 
 
    def __rmul__(self, alpha): 
        """ multiplication par un scalaire - external multiplication""" 
        return Matrice([alpha * X for X in self.coeffs], self.carac) 
 
    def __add__(self, other): 
        """ Somme de deux matrices - Sum of 2 matrices """ 
        L = [self.coeffs[i] + other.coeffs[i] for i in range(0, self.lignes)] 
        return Matrice(L, self.carac) 
 
    def __mul__(self, other): 
        """ Produit d'une matrice par une autre - Product of a matrix by another one """ 
        Zp = GenZp(self.carac) 
        m, n, p = self.lignes, self.colonnes, other.colonnes 
        Resultat = MatriceNulle(self.lignes, other.colonnes, self.carac) 
        for u in range(0, m): 
            for v in range(0, p): 
                s = Zp(0) 
                for w in range(0, n): 
                    s = s + (self.GetCoef(u, w) * other.GetCoef(w, v)) 
                Resultat.SetCoef(u, v, s) 
        return Resultat 
 
 
p = 5 
Zp = GenZp(p) 
m = 3 
Vpm = GenVectnZp(p, m) 
 
 
def VecteurNul(n): 
    return Vpm([0] * n) 
 
 
def MatriceNulle(m, n, p): 
    return Matrice([VecteurNul(n) for i in range(0, m)], p)
