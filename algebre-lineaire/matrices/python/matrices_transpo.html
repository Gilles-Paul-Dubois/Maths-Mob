from random import randint 
from numpy import array, matrix 
 
def AleaLine(n): 
    """ligne aléatoire de taille n à coefficients entiers - random row of size n""" 
    return [randint(-9,+9) for i in range(0,n)] 
 
def AleaMatrice(m,n): 
    """ Matrice aléatoire de type m,n - random matrix of type (m,n)""" 
    L= [AleaLine(n) for j in range(0,m)] 
    return matrix(L) 
 
def CalTranspo(): 
    """ Calcul de la transposée - computation of the transpose""" 
    m=randint(2,5) 
    n=randint(2,5) 
    A=AleaMatrice(m,n) 
    B=A.transpose() 
    print ("A=",A) 
    print ("t(A)=",B) 
 
def main():
