from random import randint 
from numpy import array, matrix 
 
# symbole de Kronecker - Kronecker's symbol 
def delta(i, j): 
    if i == j: 
        return 1 
    else: 
        return 0 
 
 
# Vecteur de la base canonique - Vector of the canonical basis 
def Can(i, n): 
    return [delta(i, j) for j in range(0, n)] 
 
 
# produit par un entier aléatoire - product by a random integer 
def AleaParCan(i, n): 
    return randint(0, 9) * array(Can(i, n)) 
 
 
# tableau de n tels vecteurs - Array of such vectors 
def AleaTab(n): 
    return [AleaParCan(i, n) for i in range(0, n)] 
 
 
def main(): 
    # matrice diagonale construite avec - Diagonal matrice built with 
    M = matrix(AleaTab(5)) 
    print(M) 
 
 
if __name__ == '__main__': 
    main()
