from random import randint 
from numpy import array, matrix 
 
def AleaLine(n): 
    """ligne aléatoire de taille n à coefficients entiers - random row with size n and integral coefficients""" 
    return [randint(-9,+9) for i in range(0,n)] 
 
def AleaMatrice(m,n): 
    """ Matrice aléatoitre de type m,n - Random matrix of type (m,n)""" 
    L= [AleaLine(n) for j in range(0,m)] 
    return matrix(L) 
 
def VerifAssoc(): 
    """ Vérification associativité - checking associativity""" 
    m=randint(2,5) 
    n=randint(2,5) 
    p=randint(2,5) 
    q=randint(2,5) 
    A=AleaMatrice(m,n) 
    B=AleaMatrice(n,p) 
    C=AleaMatrice(p,q) 
    print ("A=",A) 
    print ("B=",B) 
    print ("C=",C) 
    print ("(AxB)xC=",(A*B)*C) 
    print ("Ax(BxC)=",A*(B*C)) 
 
def main(): 
    VerifAssoc() 
 
 
if __name__=='__main__': 
    main()
