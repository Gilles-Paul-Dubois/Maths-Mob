# -*- coding: utf-8 -*- 
#python version 3.4 
 
from numpy import matrix 
 
# espaces vectoriels de matrices 
# vector spaces of matrices 
 
#matrices réelles - real matrices 
A=matrix([[1,-1,1,2],[4,4,3,5],[7,8,5,-6]]) # 3 lignes, 4 colonnes - 3 rows, 4 columns 
B=matrix([[-1,0,1,-2],[4,2,3,6],[0,1,5,6]]) # idem 
 
#matrices complexes - complex matrices 
C=matrix([[1,1+2.j],[1.j, -1.j]]) # carrée (2,2) - square of order 2 
 
print (0.5*A) # produit par un réel - product by a real 
print (A+B) # somme de deux matrices - sum of two matrices/span> 
print (1.j*C) # produit par un complexe - product by a complex
