from scipy import sparse 
from numpy import array 
""" création d'une matrice creuse format COO""" 
""" creation of a sparse matrix with format COO"""
I = array([0,3,1,0]) #indices i - i indexes 
J = array([0,3,1,2]) # indices j - j indexes 
V = array([4,5,7,9]) # valeurs - values 
A = sparse.coo_matrix((V,(I,J)),(4,4)) 
 
def main(): 
    print(A) # affichage type:'creux' 26 coefficients sur 400 non nuls - Display 'sparse' type 26 coefficents non zeros on a total of 400 
    print(A.todense()) # affichage dense - 'dense' display 
 
 
if __name__ == '__main__': 
    main()
