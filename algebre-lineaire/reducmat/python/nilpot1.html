import numpy as np 
""" Détection des matrices nilpotentes - dtecting nilpotent matrices""" 
 
def IsNull (A): 
    """Teste si une matrice est nulle - cheks for null matrix""" 
    return all([ A[i,j]==0 for i in range(0,len(A)) for j in range(0, len(A))]) 
     
def Nilpotente(A): 
    """teste si une puissance de A s'annule - checks if some power of A is null""" 
    for n in range(1,len(A)+1): 
        if IsNull(A**n): 
            return True,n 
    return False 
 
def main (): 
    N=np.matrix([[3,9,-9],[2,0,0],[3,3,-3]]) 
    print (Nilpotente(N)) 
 
if __name__=='__main__': 
    main()
