using LinearAlgebra
using AbstractAlgebra
import PolynomialRoots

A=[3 1 0 0;-4 -1 0 0;7 1 2 1;-17 -6 -1 0]

function nul(M)
    s= size(M,1)
    return all(M[i,j]==0 for i in 1:s for j in 1:s)
end

function nilpotente(M)
    s=size(M,1)
    for p in 1:s
        if nul(M^p)
            return true,p
        end
    end
    return false,0
end

R, x = PolynomialRing(ZZ, "x")
M=matrix(R,A)
B=x*one(M)-M
P=det(B)
println(P)
C=[c for c in coefficients(P)]
println(PolynomialRoots.roots(C))
B=M-one(M)
println(nilpotente(B))
